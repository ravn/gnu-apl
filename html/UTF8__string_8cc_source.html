32 /* from RFC 3629 / STD 63:
33 
34  Char. number range | UTF-8 octet sequence
35  (hexadecimal) | (binary)
36  --------------------+---------------------------------------------
37  0000 0000-0000 007F | 0xxxxxxx
38  0000 0080-0000 07FF | 110xxxxx 10xxxxxx
39  0000 0800-0000 FFFF | 1110xxxx 10xxxxxx 10xxxxxx
40  0001 0000-0010 FFFF | 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
41 
42  RFC 2279, the predecessor of RFC 3629, also allowed these:
43 
44  0000 0000-0000 007F 0xxxxxxx
45  0000 0080-0000 07FF 110xxxxx 10xxxxxx
46  0000 0800-0000 FFFF 1110xxxx 10xxxxxx 10xxxxxx
47  0001 0000-001F FFFF 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
48  0020 0000-03FF FFFF 111110xx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
49  0400 0000-7FFF FFFF 1111110x 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
50 
51  In order to be more general, we follow RFC 2279 and also allow 5-byte and
52  6-byte encodings
53 */

55 //-----------------------------------------------------------------------------
56 UTF8_string::UTF8_string(const UCS_string & ucs)
57 {
58  Log(LOG_char_conversion)
59  CERR << "UTF8_string::UTF8_string(ucs = " << ucs << ")" << endl;
60 
61  loop(i, ucs.size())
62  {
63  int uni = ucs[i];
64  if (uni < 0x80) // 1-byte unicode (ASCII)
65  {
66  *this += uni;
67  }
68  else if (uni < 0x800) // 2-byte unicode
69  {
70  const uint8_t b1 = uni & 0x3F; uni >>= 6;
71  *this += uni | 0xC0;
72  *this += b1 | 0x80;
73  }
74  else if (uni < 0x10000) // 3-byte unicode
75  {
76  const uint8_t b2 = uni & 0x3F; uni >>= 6;
77  const uint8_t b1 = uni & 0x3F; uni >>= 6;
78  *this += uni | 0xE0;
79  *this += b1 | 0x80;
80  *this += b2 | 0x80;
81  }
82  else if (uni < 0x200000) // 4-byte unicode
83  {
84  const uint8_t b3 = uni & 0x3F; uni >>= 6;
85  const uint8_t b2 = uni & 0x3F; uni >>= 6;
86  const uint8_t b1 = uni & 0x3F; uni >>= 6;
87  *this += uni | 0xF0;
88  *this += b1 | 0x80;
89  *this += b2 | 0x80;
90  *this += b3 | 0x80;
91  }
92  else if (uni < 0x4000000) // 5-byte unicode
93  {
94  const uint8_t b4 = uni & 0x3F; uni >>= 6;
95  const uint8_t b3 = uni & 0x3F; uni >>= 6;
96  const uint8_t b2 = uni & 0x3F; uni >>= 6;
97  const uint8_t b1 = uni & 0x3F; uni >>= 6;
98  *this += uni | 0xF8;
99  *this += b1 | 0x80;
100  *this += b2 | 0x80;
101  *this += b3 | 0x80;
102  *this += b4 | 0x80;
103  }
104  else if (size_t(uni) < 0x80000000) // 6-byte unicode
105  {
106  const uint8_t b5 = uni & 0x3F; uni >>= 6;
107  const uint8_t b4 = uni & 0x3F; uni >>= 6;
108  const uint8_t b3 = uni & 0x3F; uni >>= 6;
109  const uint8_t b2 = uni & 0x3F; uni >>= 6;
110  const uint8_t b1 = uni & 0x3F; uni >>= 6;
111  *this += uni | 0xFC;
112  *this += b1 | 0x80;
113  *this += b2 | 0x80;
114  *this += b3 | 0x80;
115  *this += b4 | 0x80;
116  *this += b5 | 0x80;
117  }
118  else
119  {
120  CERR << "Bad Unicode: " << UNI(uni) << endl
121  << "The offending ucs string is:";
122  loop(ii, ucs.size()) CERR << " " << HEX(ucs[ii]);
123  CERR << endl;
124 
125  BACKTRACE
126  Assert(0 && "Error in UTF8_string::UTF8_string(ucs)");
127  }
128  }
129 
130  Log(LOG_char_conversion)
131  CERR << "UTF8_string::UTF8_string(): utf = " << *this << endl;
132 }

156 Unicode
157 UTF8_string::toUni(const UTF8 * string, int & len, bool verbose)
158 {
159 const uint32_t b0 = *string++;
160  if (b0 < 0x80) { len = 1; return Unicode(b0); }
161 
162 uint32_t bx = b0; // the "significant" bits in b0
163  if ((b0 & 0xE0) == 0xC0) { len = 2; bx &= 0x1F; }
164  else if ((b0 & 0xF0) == 0xE0) { len = 3; bx &= 0x0F; }
165  else if ((b0 & 0xF8) == 0xF0) { len = 4; bx &= 0x0E; }
166  else if ((b0 & 0xFC) == 0xF8) { len = 5; bx &= 0x0E; }
167  else if ((b0 & 0xFE) == 0xFC) { len = 6; bx &= 0x0E; }
168  else if (verbose)
169  {
170  CERR << "Bad UTF8 sequence: " << HEX(b0);
171  loop(j, 6)
172  {
173  const uint32_t bx = string[j];
174  if (bx & 0x80) CERR << " " << HEX(bx);
175  else break;
176  }
177 
178  CERR << " at " LOC << endl;
179 
180  BACKTRACE
181  Assert(0 && "Internal error in UTF8_string::toUni()");
182  }
183  else
184  {
185  len = 0;
186  return Invalid_Unicode;
187  }
188 
189 uint32_t uni = 0;
190  loop(l, len - 1)
191  {
192  const UTF8 subc = *string++;
193  if ((subc & 0xC0) != 0x80)
194  {
195  CERR << "Bad UTF8 sequence: " << HEX(b0) << "... at " LOC << endl;
196  Assert(0 && "Internal error in UTF8_string::toUni()");
197  }
198 
199  bx <<= 6;
200  uni <<= 6;
201  uni |= subc & 0x3F;
202  }
203 
204  return Unicode(bx | uni);
205 }
206 //-----------------------------------------------------------------------------
207 Unicode
208 UTF8_string::getc(istream & in)
209 {
210 const uint32_t b0 = in.get() & 0xFF;
211  if (!in.good()) return Invalid_Unicode;
212  if (b0 < 0x80) { return Unicode(b0); }
213 
214 uint32_t bx = b0; // the "significant" bits in b0
215 int len;
216 
217  if ((b0 & 0xE0) == 0xC0) { len = 1; bx &= 0x1F; }
218  else if ((b0 & 0xF0) == 0xE0) { len = 2; bx &= 0x0F; }
219  else if ((b0 & 0xF8) == 0xF0) { len = 3; bx &= 0x0E; }
220  else
221  {
222  CERR << "Bad UTF8 sequence: " << HEX(b0) << "... at " LOC << endl;
223  BACKTRACE
224  return Invalid_Unicode;
225  }
226 
227 char cc[4];
228  in.get(cc, len + 1); // read subsequent characters + terminating 0
229  if (!in.good()) return Invalid_Unicode;
230 
231 uint32_t uni = 0;
232  loop(l, len)
233  {
234  const UTF8 subc = cc[l];
235  if ((subc & 0xC0) != 0x80)
236  {
237  CERR << "Bad UTF8 sequence: " << HEX(b0) << "... at " LOC << endl;
238  return Invalid_Unicode;
239  }
240 
241  bx <<= 6;
242  uni <<= 6;
243  uni |= subc & 0x3F;
244  }
245 
246  return Unicode(bx | uni);
247 }

251 {
252  if (path == 0) return false; // no path provided
253

277 int
278 UTF8_string::un_HTML(int in_HTML)
279 {
280 int dest = 0;
281 bool got_tag = false;
282  loop(src, size())
283  {
284  const char cc = at(src);
285  if (in_HTML == 2) // inside HTML tag, i.e. < seen
286  {
287  if (cc == '>') in_HTML = 1; // in HTML but not in HTML tag
288  continue;
289  }
290 
291  if (cc == '<') // start of HTML tag
292  {
293  in_HTML = 2; // now inside HTML tag
294  got_tag = true;
295  continue;
296  }
297 
298  if (cc != '&') // unless HTML-escaped char
299  {
300  at(dest++) = cc;
301  continue;
302  }
303 
304  // at this point cc == '&' which is the start of an HTML-escaped
305  // character. This can be:
306  //
307  // &NN; with decimal digits NN..., or
308  // &#XX; with hexadecimal digits XX..., or (incomplete list)
309  // &gt; for >, or
310  // &lt; for <
311  //
312  // There exist more HTML escapes, but the function producing them
313  // (i.e. UCS_string::to_HTML()) does not emit them.
314  //
315  const int rest = size() - src;
316  if (rest > 3 && at(src + 1) == '#' &&
317  strchr("0123456789", at(src + 2)))
318  {
319  src += 2; // skip "&#"
320  int val = 0;
321  char * end = 0;
322  if (at(src) == 'x') // hex value
323  {
324  ++src; // skip "x"
325  val = strtoll(c_str() + src, &end, 16);
326  }
327  else // decimal
328  {
329  val = strtoll(c_str() + src, &end, 10);
330  }
331  at(dest++) = val;
332  src = end - c_str(); // skip hex or decimal digits
333  // ";" skipped by loop()
334  }
335  else if (rest > 3 && at(src + 1) == 'g' &&
336  at(src + 2) == 't' &&
337  at(src + 3) == ';')
338  {
339  at(dest++) = '>';
340  src += 3; // skip "gt;"
341  }
342  else if (rest > 3 && at(src + 1) == 'l' &&
343  at(src + 2) == 't' &&
344  at(src + 3) == ';')
345  {
346  at(dest++) = '<';
347  src += 3; // skip "lt;"
348  }
349  else if (rest > 4 && at(src + 1) == 'n' &&
350  at(src + 2) == 'b' &&
351  at(src + 3) == 's' &&
352  at(src + 4) == 'p' &&
353  at(src + 5) == ';')
354  {
355  at(dest++) = ' ';
356  src += 5; // skip "nbsp;"
357  }
358  else
359  {
360  at(dest++) = cc;
361  }
362  }
363 
364  if (got_tag) dest = 0;
365  resize(dest);
366  return in_HTML;
367 }
368 //-----------------------------------------------------------------------------
369 bool
370 UTF8_string::round_0_1()
371 {
372  if (back() >= '5') // round up
373  {
374  // rounding up of the last digit creates a carry
375  //
376  for (int j = size() - 2; j >= 0; --j)
377  {
378  if (at(j) == '9') // propagate carry
379  {
380  at(j) = '0';
381  continue; // carry survived
382  }
383 
384  // eat carry
385  //
386  ++at(j);
387  pop_back(); // discard last digit.
388  return false; // no carry
389  }
390  }
391 
392  // if carry survived then digits were 0.999... and were then rounded
393  // up to 1.000...
394  //
395  at(0) = '1';
396  pop_back();
397  return true; // 1.0 â 0.1
398 }

30 class ArrayIterator
31 {
32 public:
34  ArrayIterator(const Shape & shape)
35  : ref_B(shape),
36  current_offset(0),
37  rank(shape.get_rank()),
38  done(false)
39  {
40  ShapeItem _weight = 1;
41  loop(r, rank)
42  {
43  const ShapeItem sB = shape.get_transposed_shape_item(r);
44  _twc & twc_r = twc[r];
45 
46  twc_r.current = 0;
47  twc_r.to = sB;
48  twc_r.weight = _weight;
49 
50  if (0 == sB) done = true; // empty array
51  _weight *= sB;
52  }
53  }
54 
56  void operator++()
57  {
58  ++current_offset;
59  loop(r, rank)
60  {
61  _twc & twc_r = twc[r];
62  ++twc_r.current;
63 
64  if (twc_r.current < twc_r.to) return;
65 
66  // end of the axis reached: reset this axis
67  // and increment next axis via loop()
68  //
69  twc_r.current = 0;
70  }
71  done = true;
72  }
73 
75  bool more() const { return !done; }
76 
78  ShapeItem get_offset(Rank r) const
79  { return twc[rank - r - 1].current; }
80 
82  ShapeItem operator()() const
83  { return current_offset; }
84 
86  Shape get_offsets() const
87  {
88  Shape ret;
89  loop(r, rank) ret.add_shape_item(twc[rank - r - 1].current);
90  return ret;
91  }
92 
94  ShapeItem multiply(const Shape & w) const
95  { Assert1(rank == w.get_rank());
96  ShapeItem ret = 0;
97  loop(r, rank) ret += twc[rank - r - 1].current * w.get_shape_item(r);
98  return ret; }
99 
100 protected:
102  const Shape & ref_B;
103 
105  struct _twc twc[MAX_RANK];
106 
108  ShapeItem current_offset;
109 
111  const uRank rank;
112 
114  bool done;
115 };
116 //-----------------------------------------------------------------------------
121 class PermutedArrayIterator : public ArrayIterator
123 {
124 public:
126  PermutedArrayIterator(const Shape & shape, const Shape & perm)
127  : ArrayIterator(shape),
128  permutation(perm),
129  weight(shape.reverse_scan())
130  { Assert(rank == perm.get_rank()); }
131 
133  void operator ++()
134  {
135  if (done) return;
136 
137  loop(upr, rank) // upr: un-permuted rank
138  {
139  const Axis r = permutation.get_shape_item(rank - upr - 1);
140  _twc & twc_r = twc[rank - r - 1];
141  ++twc_r.current;
142  current_offset += twc_r.weight;
143 
144  if (twc_r.current < twc_r.to) return;
145 
146  // end of the axis reached: reset this axis
147  // and increment the next axis via loop()
148  //
149  current_offset -= twc_r.weight * twc_r.current;
150  twc_r.current = 0;
151  }
152 
153  done = true;
154  }
155 
156 protected:
158  const Shape permutation;
159 
161  const Shape weight;
162 };

30 class AxisIterator
31 {
32 public:
34  AxisIterator()
35  {}
36 
38  AxisIterator(ShapeItem len, ShapeItem weight, bool _wrap)
39  : axis_length(len),
40  axis_weight(weight),
41  shape_offset(0),
42  ravel_offset(0),
43  wrap(_wrap)
44  {}
45 
52  bool next()
53  {
54  ravel_offset += axis_weight;
55 
56  // NOTE: increment of scalar shape_offset needed for more() below.
57  if (++shape_offset < axis_length || is_scalar_iterator()) return false;
58 
59  // carry has occurred.
60  if (wrap) shape_offset = ravel_offset = 0;
61  return true; // carry: increment next (if any) iterator
62  }
63 
65  bool more() const
66  { return is_scalar_iterator() ? shape_offset == 0 // before next()
67  : shape_offset < axis_length; }
68 
70  ShapeItem get_shape_offset() const
71  { return shape_offset; }
72 
74  ShapeItem get_ravel_offset() const
75  { return ravel_offset; }
76 
78  bool is_scalar_iterator() const
79  { return axis_weight == 0; }
80 
81 protected:
83  ShapeItem axis_length;
84 
87  ShapeItem axis_weight;
88 
90  ShapeItem shape_offset;
91 
93  ShapeItem ravel_offset;
94 
99  bool wrap;
100 };

103 class ArrayIterator
104 {
105 public:
107  ArrayIterator(const Shape & shape)
108  : rank(shape.get_rank())
109  {
110  ShapeItem weight = 1;
111  for (Axis a = rank - 1; a > 0; --a)
112  {
113  // axis_iterators[pos == 0] is the one with the highest weight
114  const ShapeItem len = shape.get_shape_item(a);
115  new (axis_iterators + a) AxisIterator(len, weight, true);
116  weight *= len;
117  }
118 
119  if (rank == 0) // scalar
120  {
121  new (axis_iterators) AxisIterator(1, 0, false);
122  }
123  else
124  {
125  // the final iterator (which shall not wrap.)
126  const ShapeItem len = shape.get_shape_item(0);
127  new (axis_iterators) AxisIterator(len, weight, false);
128  }
129  }
130 
132  bool more() const
133  {
134  return axis_iterators[0].more();
135  }
136 
138  void operator++()
139  {
140  if (rank == 0) // scalar
141  {
142  axis_iterators->next();
143  return;
144  }
145 
146  for (Axis a = rank - 1; a >= 0; --a)
147  {
148  if (!axis_iterators[a].next()) break; // axis_iterators[0]
149  }
150  }
151 
153  ShapeItem get_shape_offset(Axis a) const
154  { return axis_iterators[a].get_shape_offset(); }
155 
157  ShapeItem get_ravel_offset() const
158  {
159  ShapeItem ret = 0;
160  loop(a, rank) ret += axis_iterators[a].get_ravel_offset();
161  return ret;
162  }
163 
165  Shape get_shape_offsets()
166  {
167  Shape ret;
168  loop(r, rank)
169  ret.add_shape_item(axis_iterators[r].get_shape_offset());
170  return ret;
171  }
172 
174  ShapeItem multiply(const Shape & weights) const
175  {
176  Assert1(rank == weights.get_rank());
177  ShapeItem ret = 0;
178  for (Axis a = rank - 1; a >= 0; --a)
179  ret += axis_iterators[a].get_shape_offset()
180  * weights.get_shape_item(a);
181  return ret;
182  }
183 
184 protected:
186  uRank rank;
187 
194  AxisIterator axis_iterators[MAX_RANK];
195 };

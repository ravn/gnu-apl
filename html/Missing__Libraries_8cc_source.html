21 /* This file contains eval_XXX() functions that are (conditionally) compiled
22  only if their corresponding real functions (with the same name) could not
23  be compiled because some library or header file could not be found by
24  ./configure, i.e. before compliling GNU APL.
25 
26  All such functions in this file add some more information to )MORE and then
27  raise a SYNTAX ERROR.
28  */

44 //=============================================================================
45 Token
46 missing_files(const char * qfun, // the function, e.g. "⎕RE"
47  const char ** libs, // the required libraries
48  const char ** hdrs, // the required header files
49  const char ** pkgs) // the proposed packages
50 {
51 UCS_string & more = MORE_ERROR() <<
52 "Bad luck. The system function " << qfun <<
53 " has raised a SYNTAX ERROR even though the\n"
54 "syntax used was correct. The real reason for the SYNTAX ERROR was that a\n"
55 "library or header file on which " << qfun << " depends:\n"
56 "\n"
57 " ⋆ could not be found by ./configure, and/or\n"
58 " ⋆ was explitly disabled by a ./configure argument\n"
59 "\n"
60 "just before GNU APL was compiled.\n";
61 
62  if (libs && libs[0])
63  {
64  const char * numerus1 = libs[1] ? "ies " : "y ";
65  const char * numerus2 = libs[1] ? " were:" : " was:";
66  more << "\nThe possibly missing (or disabled) librar" << numerus1
67  << "needed by " << qfun << numerus2;
68  for (int j = 0; libs[j]; ++j) more << " " << libs[j];
69  more << "\nTo locate installed versions of it, run e.g.:\n\n";
70  for (int j = 0; libs[j]; ++j)
71  more << " )HOST find /usr -name '" << libs[j]
72  << "*' 2>/dev/null\n";
73  }
74 
75  if (hdrs && hdrs[0])
76  {
77  const char * numerus = hdrs[1] ? "s were" : " was";
78  more << "\nThe possibly missing header file" << numerus << ":";
79  for (int j = 0; hdrs[j]; ++j) more << " " << hdrs[j];
80  more << "\nTo locate it, run e.g.:\n\n";
81  for (int j = 0; hdrs[j]; ++j)
82  more << " )HOST find /usr -name '" << hdrs[j]
83  << "' 2>/dev/null\n";
84  }
85 
86  more <<
87 "\n"
88 "This GNU APL interpreter instance was probably configured like this:\n"
89 "\n"
90 " " CONFIGURE_ARGS "\n\n";
91 
92  if (pkgs && pkgs[0])
93  {
94  more <<
95 "If the problem was caused by a missing library or header file, then (on a\n"
96 " standard GNU/Linux/Debian system) it can usually be installed with the\n"
97 "following command (as root and in a shell):\n"
98 "\n"
99 " apt install";
100  for (int j = 0; pkgs[j]; ++j) more << " " << pkgs[j];
101  more <<
102 "\n"
103 "\n"
104 "and after that, reconfigure, recompile, and reinstall GNU APL:\n\n"
105 " ./configure # if the ./configure options were incorrect, or else\n"
106 " ./configure " CONFIGURE_ARGS "\n"
107 " make\n"
108 " sudo make install\n"
109 "\n"
110 "in the top-level directory of the GNU APL package.\n";
111  }
112 
113  SYNTAX_ERROR;
114  return Token();
115 }

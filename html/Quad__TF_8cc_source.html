40 Token
41 Quad_TF::eval_AB(Value_P A, Value_P B)
42 {
43  // A should be an integer scalar or 1-element_vector with value 1 or 2
44  //
45  if (A->get_rank() > 0) RANK_ERROR;
46  if (A->element_count() != 1) LENGTH_ERROR;
47 
48 const APL_Integer mode = A->get_ravel(0).get_int_value();
49 const UCS_string symbol_name(*B.get());
50 
51 Value_P Z;
52 
53  if (mode == 1)
54  {
55  const bool inverse = is_inverse(symbol_name);
56  if (inverse) Z = tf1_inv(symbol_name);
57  else Z = tf1(symbol_name);
58  }
59  else if (mode == 2)
60  {
61  const bool inverse = is_inverse(symbol_name);
62  if (!inverse) return tf2(symbol_name);
63  Z = Value_P(tf2_inverse(symbol_name), LOC);
64  }
65  else if (mode == 3)
66  {
67  Z = tf3(symbol_name);
68  }
69  else
70  {
71  DOMAIN_ERROR;
72  }
73 
74  if (!Z) DOMAIN_ERROR;
75 
76  Z->check_value(LOC);
77  return Token(TOK_APL_VALUE1, Z);
78 }

82 {
83  // a forward âTF contains only a function or variable name.
84  // an inverse âTF contains at least one non-symbol character
85  //
86  loop(n, maybe_name.size())
87  {
88  if (n == 0 && Avec::is_quad(maybe_name[n])) continue;
89  if (Avec::is_symbol_char(maybe_name[n])) continue;
90 
91  return true; // inverse âTF
92  }
93 
94  // all chars in val were symbol characters. Therefore it is a forward âTF
95  return false;

128 {

130 UCS_string ucs(is_char_array ? UNI_ASCII_C : UNI_ASCII_N);
131 
132  ucs.append(var_name);
133  ucs.append(UNI_ASCII_SPACE);
134  ucs.append_number(val->get_rank()); // rank
135 
136  loop(r, val->get_rank())
137  {
138  ucs.append(UNI_ASCII_SPACE);
139  ucs.append_number(val->get_shape_item(r)); // shape
140  }
141 
142 const ShapeItem ec = val->element_count();
143 
144  if (is_char_array)
145  {
146  ucs.append(UNI_ASCII_SPACE);
147 
148  loop(e, ec)
149  {
150  const Cell & cell = val->get_ravel(e);
151  if (!cell.is_character_cell())
152  {
153  return Str0(LOC);
154  }
155 
156  ucs.append(cell.get_char_value());
157  }
158  }
159  else // number
160  {
161  loop(e, ec)
162  {
163  ucs.append(UNI_ASCII_SPACE);

169  ucs.append_number(cell.get_int_value());
170  if (ucs[sign] == '-') ucs[sign] = UNI_OVERBAR;
171  }
172  else if (cell.is_near_real())
173  {
174  PrintContext pctx(PR_APL_MIN, Quad_PP_TF, MAX_Quad_PW);
175  bool scaled = true;
176  UCS_string ucs1(cell.get_real_value(), scaled, pctx);
177  ucs.append(ucs1);
178  }
179  else if (cell.is_numeric())
180  {
181  PrintContext pctx(PR_APL_MIN, Quad_PP_TF, MAX_Quad_PW);
182  bool scaled = true;
183  UCS_string ucs1(cell.get_real_value(), scaled, pctx);
184  ucs.append(ucs1);
185 
186  ucs.append(UNI_ASCII_J);
187 
188  ucs1 = UCS_string(cell.get_imag_value(), scaled, pctx);
189  ucs.append(ucs1);
190  }
191  else
192  {
193  MORE_ERROR() << "Non-number in 1 âTF N record";
194  return Value_P();
195  }
196  }
197  }
198 
199  return Value_P(ucs, LOC);
200 }

229 Value_P
230 Quad_TF::tf1_inv(const UCS_string & ravel)
231 {
232 const int len = ravel.size();
233 
234  if (len < 2)
235  {
236  MORE_ERROR() << "1 âTF record too short";
237  return Value_P();
238  }
239 
240  // mode should be 'F', 'N', or 'C'.
241 const Unicode mode = ravel[0];
242  if (mode != 'C' && mode != 'F' && mode != 'N')
243  {
244  MORE_ERROR() << "1 âTF record type not F, N, or C";
245  return Value_P();
246  }
247 
248 UCS_string name(ravel[1]);
249  if (!Avec::is_quad(name[0]) && !Avec::is_first_symbol_char(name[0]))
250  {
251  MORE_ERROR() << "Bad variable name in 1 âTF record";
252  return Value_P();
253  }
254 
255 ShapeItem idx = 2;
256  while (idx < len && Avec::is_symbol_char(ravel[idx]))
257  name.append(ravel[idx++]);
258 
259  if (Avec::is_quad(name[0]) && name.size() == 1)
260  {
261  MORE_ERROR() << "Bad variable name in 1 âTF record";
262  return Value_P();
263  }
264 
265  if (ravel[idx++] != UNI_ASCII_SPACE)
266  {
267  MORE_ERROR() << "missing space (before rank) in 1 âTF record";
268  return Value_P();
269  }
270 
271 ShapeItem idx0 = idx;
272 Rank rank = 0;
273  while (idx < len && Avec::is_digit(ravel[idx]))
274  rank = 10 * rank + ravel[idx++] - UNI_ASCII_0;
275 
276  if (rank == 0 && idx0 == idx)
277  {
278  MORE_ERROR() << "missing rank in 1 âTF record";
279  return Value_P();
280  }
281 
282  if (ravel[idx++] != UNI_ASCII_SPACE)
283  {
284  MORE_ERROR() << "missing space (after rank) in 1 âTF record";
285  return Value_P();
286  }
287 
288  if (rank > MAX_RANK)
289  {
290  MORE_ERROR() << "max. rank exceeded in 1 âTF record";
291  return Value_P();
292  }
293 
294 Shape shape;
295  loop(r, rank)
296  {
297  idx0 = idx;
298  ShapeItem sh = 0;
299  while (idx < len && Avec::is_digit(ravel[idx]))
300  sh = 10 * sh + ravel[idx++] - UNI_ASCII_0;
301  if (sh == 0 && idx0 == idx) // no shape
302  {
303  MORE_ERROR() << "too few shape items in 1 âTF record";
304  return Value_P();
305  }
306 
307  if (ravel[idx++] != UNI_ASCII_SPACE)
308  {
309  MORE_ERROR() << "missing space (in shape) in 1 âTF record";
310  return Value_P();
311  }
312  shape.add_shape_item(sh);
313  }
314 
315 Symbol * symbol = 0;
316 NameClass nc = NC_UNUSED_USER_NAME;
317 NamedObject * sym_or_fun = Workspace::lookup_existing_name(name);
318  if (sym_or_fun) // existing name
319  {
320  symbol = sym_or_fun->get_symbol();
321  nc = sym_or_fun->get_nc();
322  if (symbol && symbol->is_readonly())
323  {
324  MORE_ERROR() << "symbol cannot be modified in 1 âTF record";
325  return Value_P();
326  }
327  }
328 
329 const int data_chars = len - idx;
330 
331  if (mode == UNI_ASCII_F) // function
332  {
333  if (rank != 2)
334  {
335  MORE_ERROR() << "function text is not a matrix in 1 âTF record";
336  return Value_P();
337  }
338 
339  if (nc != NC_UNUSED_USER_NAME && nc != NC_FUNCTION && nc != NC_OPERATOR)
340  {
341  MORE_ERROR() << "symbol is an existing variable in 1 âTF F record";
342  return Value_P();
343  }
344 
345  if (data_chars != shape.get_volume())
346  {
347  MORE_ERROR() << "item count mismatch in 1 âTF N record";
348  return Value_P();
349  }
350 
351  Value_P new_val(shape, LOC);
352  loop(d, data_chars)
353  new (&new_val->get_ravel(d)) CharCell(ravel[idx + d]);
354  new_val->check_value(LOC);
355 
356  Token t = Quad_FX::fun->eval_B(new_val);
357  }
358  else if (mode == UNI_ASCII_C) // char array
359  {
360  if (data_chars != shape.get_volume())
361  {
362  MORE_ERROR() << "item count mismatch in 1 âTF C record";
363  return Value_P();
364  }
365  if (nc != NC_UNUSED_USER_NAME && nc != NC_VARIABLE)
366  {
367  MORE_ERROR() <<
368  "symbol is existing and not a variable in 1 âTF C record";
369  return Value_P();
370  }
371 
372  Value_P new_val(shape, LOC);
373  loop(d, data_chars)
374  new (&new_val->get_ravel(d)) CharCell(ravel[idx + d]);
375 
376  new_val->check_value(LOC);
377 
378  if (symbol == 0) symbol = Workspace::lookup_symbol(name);
379  symbol->assign(new_val, false, LOC);
380  }
381  else if (mode == UNI_ASCII_N) // numeric array
382  {
383  if (nc != NC_UNUSED_USER_NAME && nc != NC_VARIABLE)
384  {
385  MORE_ERROR() << "symbol cannot be assigned in 1 âTF N record";
386  return Value_P();
387  }
388 
389  UCS_string data(ravel, idx, len - idx);
390  Tokenizer tokenizer(PM_EXECUTE, LOC, false);
391  Token_string tos;
392  if (tokenizer.tokenize(data, tos) != E_NO_ERROR)
393  {
394  MORE_ERROR() << "tokenization failed in 1 âTF N record";
395  return Value_P();
396  }
397  if (size_t(tos.size()) != size_t(shape.get_volume()))
398  {
399  MORE_ERROR() << "item count mismatch in 1 âTF N record";
400  return Value_P();
401  }
402 
403  // check that all token are numeric...
404  //
405  loop(t, tos.size())
406  {
407  const TokenTag tag = tos[t].get_tag();
408  if (tag == TOK_INTEGER) continue;
409  if (tag == TOK_REAL) continue;
410  if (tag == TOK_COMPLEX) continue;
411  MORE_ERROR() << "Non-number in 1 âTF N record";
412  return Value_P();
413  }
414 
415  // at this point, we have a valid inverse 1 âTF.
416  //
417  Value_P new_val(shape, LOC);
418 
419  loop(t, tos.size())
420  {
421  const TokenTag tag = tos[t].get_tag();
422  if (tag == TOK_INTEGER)
423  new (&new_val->get_ravel(t)) IntCell(tos[t].get_int_val());
424  else if (tag == TOK_REAL)
425  new (&new_val->get_ravel(t)) FloatCell(tos[t].get_flt_val());
426  else if (tag == TOK_COMPLEX)
427  new (&new_val->get_ravel(t)) ComplexCell(tos[t].get_cpx_real(),
428  tos[t].get_cpx_imag());
429  else Assert(0); // since checked above
430  }
431 
432  new_val->check_value(LOC);
433 
434  if (!symbol) symbol = Workspace::lookup_symbol(name);
435 
436  symbol->assign(new_val, false, LOC);
437  }
438  else Assert(0); // since checked above
439 
440  return Value_P(name, LOC);
441 }

516 Quad_TF::tf2_inverse(const UCS_string & ravel)
517 {

520

522

528  }

532  return UCS_string();
533  }
534 
535  // tos is now ( VAR â ... ) or: ( âFX val ...
536  // make it VAR â ( ... ) or: leave as is
537  //

540  {

544  }
545  }
546  tos.push_back(Token(TOK_R_PARENT, int64_t(0)));

548  // simplify tos as much as possible. We replace Aâ´B by reshaped B
549  // and glue values together.

552 
553  Log(LOG_Quad_TF)
554  {

556  tos.print(CERR, false);
557  CERR << endl;
558  }
559 
560  if (tos.size() < 2)
561  {
562  Log(LOG_Quad_TF)
563  CERR << "short tos in tf2_inverse()" << tos.size() << endl;
564  return UCS_string(); // too short for an inverse 2âTF
565  }
566 
567  // it could happen that some system variable âXX which is not known by
568  // GNU APL is read. This case is parsed as â XX â ...
569  // Issue a warning in that case
570  //

573  {

576  << " in 2âTF / )IN (assignment ignored)" << endl;
577  return new_var_or_fun;
578  }
579 
580  /* we expect either VAR â VALUE
581  or: ( âFX fun-text )
582 
583  Try VAR â VALUE first.
584  */
585  if (tos[0].get_Class() == TC_SYMBOL && tos[1].get_Class() == TC_ASSIGN)
586  {
587  // at this point, we expect SYM â VALUE.
588  //
589  if (tos.size() != 3) return UCS_string();
590  if (tos[2].get_Class() != TC_VALUE) return UCS_string();

593  Log(LOG_Quad_TF) CERR << "valid inverse 2 âTF" << endl;
594  return tos[0].get_sym_ptr()->get_name(); // valid 2âTF
595  }
596 
597  // try dyadic âFX (native function)
598  //
599  // Then tos[5] is: ( VALUE âFX VALUE )
600  //
601  if (tos.size() == 5 &&
602  tos[1].get_Class() == TC_VALUE &&
603  tos[2].get_tag() == TOK_Quad_FX &&
604  tos[3].get_Class() == TC_VALUE)
605  {
606  Value_P fname = tos[1].get_apl_val();
607  Value_P so_path = tos[3].get_apl_val();
608 
609  const Token tok = Quad_FX::fun->eval_AB(so_path, fname);
610  if (tok.get_Class() == TC_VALUE) // âFX successful

613  Log(LOG_Quad_TF)

616  }
617 
618  Log(LOG_Quad_TF)
619  CERR << "invalid inverse 2 âTF (native function):" << endl;
620  return UCS_string();
621  }
622 
623  /* monadic âFX. at this point we should have:
624 
625  tos[4] = ( âFX Value )
626  */
627 
628  Log(LOG_Quad_TF)
629  CERR << "inverse 2 âTF (monadic âFX):" << endl;
630  if (tos.size() != 4) return UCS_string();
631  if (tos[1].get_tag() != TOK_Quad_FX) return UCS_string();
632  if (tos[2].get_Class() != TC_VALUE) return UCS_string();
633 
634 static const int eprops[] = { 0, 0, 0, 0 };
635 const Token tok = Quad_FX::do_quad_FX(eprops, tos[2].get_apl_val(),
636  UTF8_string("2 âTF"), true);
637 
638  if (tok.get_Class() != TC_VALUE)

640  Log(LOG_Quad_TF)
641  CERR << "Quad_FX::do_quad_FX() failed in tf2_inverse()" << endl;
642  return UCS_string(); // error in âFX
643  }
644 
645  // âFX succeeded (and the token returned is the name of the function)
646  //
647  return UCS_string(*tok.get_apl_val().get());
648 }

39 //-----------------------------------------------------------------------------
40 Token
41 Quad_TF::eval_AB(Value_P A, Value_P B) const
42 {
43  // A should be an integer scalar or 1-element_vector with value 1 or 2
44  //
45  if (A->get_rank() > 0) RANK_ERROR;
46  if (A->element_count() != 1) LENGTH_ERROR;
47 
48 const APL_Integer mode = A->get_cfirst().get_int_value();
49 const UCS_string symbol_name(*B.get());
50 
51 Value_P Z;
52 
53  if (mode == 1)
54  {
55  const bool inverse = is_inverse(symbol_name);
56  if (inverse) Z = tf1_inv(symbol_name);
57  else Z = tf1(symbol_name);
58  }
59  else if (mode == 2)
60  {
61  const bool inverse = is_inverse(symbol_name);
62  if (!inverse) return tf2(symbol_name);
63  Z = Value_P(tf2_inverse(symbol_name), LOC);
64  }
65  else if (mode == 3)
66  {
67  Z = tf3(symbol_name);
68  }
69  else
70  {
71  DOMAIN_ERROR;
72  }
73 
74  if (!Z) DOMAIN_ERROR;
75 
76  Z->check_value(LOC);
77  return Token(TOK_APL_VALUE1, Z);
78 }
79 //-----------------------------------------------------------------------------
80 bool
81 Quad_TF::is_inverse(const UCS_string & maybe_name)
82 {
83  // a forward ⎕TF contains only a function or variable name.
84  // an inverse ⎕TF contains at least one non-symbol character
85  //
86  loop(n, maybe_name.size())
87  {
88  if (n == 0 && Avec::is_quad(maybe_name[n])) continue;
89  if (Avec::is_symbol_char(maybe_name[n])) continue;
90 
91  return true; // inverse ⎕TF
92  }
93 
94  // all chars in val were symbol characters. Therefore it is a forward ⎕TF
95  return false;
96 }

98 Value_P
99 Quad_TF::tf1(const UCS_string & name)
100 {
101 const NamedObject * obj = Workspace::lookup_existing_name(name);
102  if (obj == 0) return Str0(LOC);
103 
104 const Function * function = obj->get_function();
105 
106  if (function)
107  {
108  if (obj->is_user_defined()) return tf1(name, *function);
109 
110  // quad function or primitive: return ''
111  return Str0(LOC);
112  }
113 
114 
115  if (const Symbol * symbol = obj->get_symbol())
116  {
117  Value_P value = symbol->get_apl_value();
118  if (+value) return tf1(name, value);
119 
120  /* not a variable: fall through */
121  }
122 
123  return Str0(LOC);
124 }
125 //-----------------------------------------------------------------------------
126 Value_P
127 Quad_TF::tf1(const UCS_string & var_name, Value_P val)
128 {
129 const bool is_char_array = val->get_cfirst().is_character_cell();
130 UCS_string ucs(is_char_array ? UNI_C : UNI_N);
131 
132  ucs.append(var_name);
133  ucs.append(UNI_SPACE);
134  ucs.append_number(val->get_rank()); // rank
135 
136  loop(r, val->get_rank())
137  {
138  ucs.append(UNI_SPACE);
139  ucs.append_number(val->get_shape_item(r)); // shape
140  }
141 
142 const ShapeItem ec = val->element_count();
143 
144  if (is_char_array)
145  {
146  ucs.append(UNI_SPACE);
147 
148  loop(e, ec)
149  {
150  const Cell & cell = val->get_cravel(e);
151  if (!cell.is_character_cell())
152  {
153  return Str0(LOC);
154  }
155 
156  ucs.append(cell.get_char_value());
157  }
158  }
159  else // number
160  {
161  loop(e, ec)
162  {
163  ucs.append(UNI_SPACE);
164 
165  const Cell & cell = val->get_cravel(e);
166  if (cell.is_integer_cell())
167  {
168  const int sign = ucs.size();
169  ucs.append_number(cell.get_int_value());
170  if (ucs[sign] == '-') ucs[sign] = UNI_OVERBAR;
171  }
172  else if (cell.is_near_real())
173  {
174  PrintContext pctx(PR_APL_MIN, Quad_PP_TF, MAX_Quad_PW);
175  bool scaled = true;
176  UCS_string ucs1(cell.get_real_value(), scaled, pctx);
177  ucs.append(ucs1);
178  }
179  else if (cell.is_numeric())
180  {
181  PrintContext pctx(PR_APL_MIN, Quad_PP_TF, MAX_Quad_PW);
182  bool scaled = true;
183  UCS_string ucs1(cell.get_real_value(), scaled, pctx);
184  ucs.append(ucs1);
185 
186  ucs.append(UNI_J);
187 
188  ucs1 = UCS_string(cell.get_imag_value(), scaled, pctx);
189  ucs.append(ucs1);
190  }
191  else
192  {
193  MORE_ERROR() << "Non-number in 1 ⎕TF N record";
194  return Value_P();
195  }
196  }
197  }
198 
199  return Value_P(ucs, LOC);
200 }

211  ucs.append(fun_name);
212  ucs.append(UNI_SPACE);
213  ucs.append_number(2); // rank
214  ucs.append(UNI_SPACE);
215  ucs.append_number(lines.size()); // rows
216  ucs.append(UNI_SPACE);

218  ucs.append(UNI_SPACE);
219 
220  loop(l, lines.size())
221  {
222  ucs.append(lines[l]);
223  loop(c, max_len - lines[l].size()) ucs.append(UNI_SPACE);
224  }
225 
226  return Value_P(ucs, LOC);
227 }
228 //-----------------------------------------------------------------------------
229 Value_P
230 Quad_TF::tf1_inv(const UCS_string & ravel)
231 {
232 const int len = ravel.size();
233 
234  if (len < 2)
235  {
236  MORE_ERROR() << "1 ⎕TF record too short";
237  return Value_P();
238  }
239 
240  // mode should be 'F', 'N', or 'C'.
241 const Unicode mode = ravel[0];
242  if (mode != 'C' && mode != 'F' && mode != 'N')
243  {
244  MORE_ERROR() << "1 ⎕TF record type not F, N, or C";
245  return Value_P();
246  }
247 
248 UCS_string name(ravel[1]);
249  if (!Avec::is_quad(name[0]) && !Avec::is_first_symbol_char(name[0]))
250  {
251  MORE_ERROR() << "Bad variable name in 1 ⎕TF record";
252  return Value_P();
253  }
254 
255 ShapeItem idx = 2;
256  while (idx < len && Avec::is_symbol_char(ravel[idx]))
257  name.append(ravel[idx++]);
258 
259  if (Avec::is_quad(name[0]) && name.size() == 1)
260  {
261  MORE_ERROR() << "Bad variable name in 1 ⎕TF record";
262  return Value_P();
263  }
264 
265  if (ravel[idx++] != UNI_SPACE)
266  {
267  MORE_ERROR() << "missing space (before rank) in 1 ⎕TF record";
268  return Value_P();
269  }
270 
271 ShapeItem idx0 = idx;
272 Rank rank = 0;
273  while (idx < len && Avec::is_digit(ravel[idx]))
274  rank = 10 * rank + ravel[idx++] - UNI_0;
275 
276  if (rank == 0 && idx0 == idx)
277  {
278  MORE_ERROR() << "missing rank in 1 ⎕TF record";
279  return Value_P();
280  }
281 
282  if (ravel[idx++] != UNI_SPACE)
283  {
284  MORE_ERROR() << "missing space (after rank) in 1 ⎕TF record";
285  return Value_P();
286  }
287 
288  if (rank > MAX_RANK)
289  {
290  MORE_ERROR() << "max. rank exceeded in 1 ⎕TF record";
291  return Value_P();
292  }
293 
294 Shape shape;
295  loop(r, rank)
296  {
297  idx0 = idx;
298  ShapeItem sh = 0;
299  while (idx < len && Avec::is_digit(ravel[idx]))
300  sh = 10 * sh + ravel[idx++] - UNI_0;
301  if (sh == 0 && idx0 == idx) // no shape
302  {
303  MORE_ERROR() << "too few shape items in 1 ⎕TF record";
304  return Value_P();
305  }
306 
307  if (ravel[idx++] != UNI_SPACE)
308  {
309  MORE_ERROR() << "missing space (in shape) in 1 ⎕TF record";
310  return Value_P();
311  }
312  shape.add_shape_item(sh);
313  }
314 
315 const Symbol * symbol = 0;
316 NameClass nc = NC_UNUSED_USER_NAME;
317 const NamedObject * sym_or_fun = Workspace::lookup_existing_name(name);
318  if (sym_or_fun) // existing name
319  {
320  symbol = sym_or_fun->get_symbol();
321  nc = sym_or_fun->get_nc();
322  if (symbol && symbol->is_readonly())
323  {
324  MORE_ERROR() << "symbol cannot be modified in 1 ⎕TF record";
325  return Value_P();
326  }
327  }
328 
329 const int data_chars = len - idx;
330 
331  if (mode == UNI_F) // function record
332  {
333  if (rank != 2)
334  {
335  MORE_ERROR() << "function text is not a matrix in 1 ⎕TF record";
336  return Value_P();
337  }
338 
339  if (nc != NC_UNUSED_USER_NAME &&
340  nc != NC_FUNCTION &&
341  nc != NC_OPERATOR)
342  {
343  MORE_ERROR() << "symbol is an existing variable in 1 ⎕TF F record";
344  return Value_P();
345  }
346 
347  if (data_chars != shape.get_volume())
348  {
349  MORE_ERROR() << "item count mismatch in 1 ⎕TF N record";
350  return Value_P();
351  }
352 
353  Value_P new_val(shape, LOC);
354  loop(d, data_chars) new_val->next_ravel_Char(ravel[idx + d]);
355  new_val->check_value(LOC);
356 
357  Token t = Quad_FX::fun->eval_B(new_val);
358  }
359  else if (mode == UNI_C) // char array
360  {
361  if (data_chars != shape.get_volume())
362  {
363  MORE_ERROR() << "item count mismatch in 1 ⎕TF C record";
364  return Value_P();
365  }
366  if (nc != NC_UNUSED_USER_NAME && nc != NC_VARIABLE)
367  {
368  MORE_ERROR() <<
369  "symbol is existing and not a variable in 1 ⎕TF C record";
370  return Value_P();
371  }
372 
373  Value_P new_val(shape, LOC);
374  loop(d, data_chars) new_val->next_ravel_Char(ravel[idx + d]);
375 
376  new_val->check_value(LOC);
377 
378  if (symbol == 0) symbol = Workspace::lookup_symbol(name);
379  const_cast<Symbol *>(symbol)->assign(new_val, false, LOC);
380  }
381  else if (mode == UNI_N) // numeric array
382  {
383  if (nc != NC_UNUSED_USER_NAME && nc != NC_VARIABLE)
384  {
385  MORE_ERROR() << "symbol cannot be assigned in 1 ⎕TF N record";
386  return Value_P();
387  }
388 
389  UCS_string data(ravel, idx, len - idx);
390  Tokenizer tokenizer(PM_EXECUTE, LOC, false);
391  Token_string tos;
392  if (tokenizer.tokenize(data, tos) != E_NO_ERROR)
393  {
394  MORE_ERROR() << "tokenization failed in 1 ⎕TF N record";
395  return Value_P();
396  }
397  if (size_t(tos.size()) != size_t(shape.get_volume()))
398  {
399  MORE_ERROR() << "item count mismatch in 1 ⎕TF N record";
400  return Value_P();
401  }
402 
403  // check that all token are numeric...
404  //
405  loop(t, tos.size())
406  {
407  const TokenTag tag = tos[t].get_tag();
408  if (tag == TOK_INTEGER) continue;
409  if (tag == TOK_REAL) continue;
410  if (tag == TOK_COMPLEX) continue;
411  MORE_ERROR() << "Non-number in 1 ⎕TF N record";
412  return Value_P();
413  }
414 
415  // at this point, we have a valid inverse 1 ⎕TF.
416  //
417  Value_P ZZ(shape, LOC);
418 
419  loop(t, tos.size())
420  {
421  const Token & tok = tos[t];
422  const TokenTag tag = tok.get_tag();
423  if (tag == TOK_INTEGER)
424  ZZ->next_ravel_Int(tok.get_int_val());
425  else if (tag == TOK_REAL)
426  ZZ->next_ravel_Float(tok.get_flt_val());
427  else if (tag == TOK_COMPLEX)
428  ZZ->next_ravel_Complex(tok.get_cpx_real(),
429  tok.get_cpx_imag());
430  else Assert(0); // since checked above
431  }
432 
433  ZZ->check_value(LOC);
434 
435  if (!symbol) symbol = Workspace::lookup_symbol(name);
436 
437  const_cast<Symbol *>(symbol)->assign(ZZ, false, LOC);
438  }
439  else Assert(0); // since checked above
440 
441  return Value_P(name, LOC);
442 }
443 //-----------------------------------------------------------------------------
444 Token
445 Quad_TF::tf2(const UCS_string & name)
446 {
447 const NamedObject * obj = Workspace::lookup_existing_name(name);
448  if (obj == 0)
449  {
450  Log(LOG_Quad_TF) CERR << "bad name in tf2(" << name << ")" << endl;
451  return Token(TOK_APL_VALUE1, Str0(LOC));
452  }
453 
454 const Function * function = obj->get_function();
455 
456  if (function)
457  {
458  if (obj->is_user_defined())
459  {
460  UCS_string ucs = tf2_fun(name, *function);
461  Value_P Z(ucs, LOC);
462  Z->check_value(LOC);
463  return Token(TOK_APL_VALUE1, Z);
464  }
465 
466  // quad function or primitive: return ''
467  return Token(TOK_APL_VALUE1, Str0(LOC));
468  }
469 
470 const Symbol * symbol = obj->get_symbol();
471  if (symbol)
472  {
473  Value_P value = symbol->get_apl_value();
474  if (+value) return tf2_var(name, value);
475 
476  /* not a variable: fall through */
477  }
478 
479  Log(LOG_Quad_TF) CERR << "error in tf2(" << name << ")" << endl;
480  return Token(TOK_APL_VALUE1, Str0(LOC));
481 }

483 Token
484 Quad_TF::tf2_var(const UCS_string & var_name, Value_P value)
485 {
486  Log(LOG_Quad_TF) CERR << "tf2_var(" << var_name << ")" << endl;
487 
488 UCS_string ucs_value;
489  if (value->is_scalar() && !value->is_simple_scalar())
490  {
491  const Cell & cell = value->get_cfirst();
492  Assert(cell.is_pointer_cell());
493  tf2_value(0, ucs_value, cell.get_pointer_value().getref(), 1);
494  }
495  else
496  {
497  tf2_value(0, ucs_value, value.getref(), 0);
498  }
499 
500  Assert(ucs_value[0] == UNI_L_PARENT);
501  Assert(ucs_value[ucs_value.size() - 1] == UNI_R_PARENT);
502 
503 UCS_string ucs(var_name);
504  ucs.append(UNI_LEFT_ARROW);
505 
506  // copy ucs_value except the outer parrentheses
507  for (ShapeItem v = 1; v < (ucs_value.size() - 1); ++v)
508  ucs.append(ucs_value[v]);
509 
510  Log(LOG_Quad_TF) CERR << "success in tf2_var(): " << ucs << endl;
511 Value_P Z(ucs, LOC);
512  Z->check_value(LOC);
513  return Token(TOK_APL_VALUE1, Z);
514 }
515 //-----------------------------------------------------------------------------
516 UCS_string
517 Quad_TF::tf2_inverse(const UCS_string & ravel)
518 {
519 Token_string tos;
520  tos.push_back(Token(TOK_L_PARENT, int64_t(0)));
521 
522  Log(LOG_Quad_TF) CERR << "inverse ⎕TF2: " << ravel << endl;
523 
524  try
525  {
526  UCS_string ucs1 = no_UCS(ravel);
527  const Parser parser(PM_EXECUTE, LOC, false);
528  parser.parse(ucs1, tos);
529  }
530  catch(...)
531  {
532  Log(LOG_Quad_TF) CERR << "parse error in tf2_inverse()" << endl;
533  return UCS_string();
534  }
535 
536  // tos is now ( VAR ← ... ) or: ( ⎕FX val ...
537  // make it VAR ← ( ... ) or: leave as is
538  //
539  {
540  if (tos.size() >= 3 && tos[1].get_tag() != TOK_Quad_FX)
541  {
542  tos[0] = tos[1]; // move VAR
543  tos[1] = tos[2]; // move ←
544  tos[2] = Token(TOK_L_PARENT, int64_t(0));
545  }
546  }
547  tos.push_back(Token(TOK_R_PARENT, int64_t(0)));
548 
549  // simplify tos as much as possible. We replace A⍴B by reshaped B
550  // and glue values together.
551  //
552  tf2_reduce(tos);
553 
554  Log(LOG_Quad_TF)
555  {
556  CERR << "inverse ⎕TF2: tos[" << tos.size() << "] after reduce() is: ";
557  tos.print(CERR, false);
558  CERR << endl;
559  }
560 
561  if (tos.size() < 2)
562  {
563  Log(LOG_Quad_TF)
564  CERR << "short tos in tf2_inverse()" << tos.size() << endl;
565  return UCS_string(); // too short for an inverse 2⎕TF
566  }
567 
568  // it could happen that some system variable ⎕XX which is not known by
569  // GNU APL is read. This case is parsed as ⎕ XX ← ...
570  // Issue a warning in that case
571  //
572  if (tos[0].get_tag() == TOK_Quad_Quad &&
573  tos[1].get_Class() == TC_SYMBOL)
574  {
575  UCS_string new_var_or_fun = tos[1].get_sym_ptr()->get_name();
576  CERR << "*** Unknown system variable ⎕" << new_var_or_fun
577  << " in 2⎕TF / )IN (assignment ignored)" << endl;
578  return new_var_or_fun;
579  }
580 
581  /* we expect either VAR ← VALUE
582  or: ( ⎕FX fun-text )
583 
584  Try VAR ← VALUE first.
585  */
586  if (tos[0].get_Class() == TC_SYMBOL && tos[1].get_Class() == TC_ASSIGN)
587  {
588  // at this point, we expect SYM ← VALUE.
589  //
590  if (tos.size() != 3) return UCS_string();
591  if (tos[2].get_Class() != TC_VALUE) return UCS_string();
592 
593  tos[0].get_sym_ptr()->assign(tos[2].get_apl_val(), true, LOC);
594  Log(LOG_Quad_TF) CERR << "valid inverse 2 ⎕TF" << endl;
595  return tos[0].get_sym_ptr()->get_name(); // valid 2⎕TF
596  }
597 
598  // try dyadic ⎕FX (native function)
599  //
600  // Then tos[5] is: ( VALUE ⎕FX VALUE )
601  //
602  if (tos.size() == 5 &&
603  tos[1].get_Class() == TC_VALUE &&
604  tos[2].get_tag() == TOK_Quad_FX &&
605  tos[3].get_Class() == TC_VALUE)
606  {
607  Value_P fname = tos[1].get_apl_val();
608  Value_P so_path = tos[3].get_apl_val();
609 
610  const Token tok = Quad_FX::fun->eval_AB(so_path, fname);
611  if (tok.get_Class() == TC_VALUE) // ⎕FX successful
612  {
613  Value_P val = tok.get_apl_val();
614  Log(LOG_Quad_TF)
615  CERR << "valid inverse 2 ⎕TF (native function):" << endl;
616  return UCS_string(*val.get());
617  }
618 
619  Log(LOG_Quad_TF)
620  CERR << "invalid inverse 2 ⎕TF (native function):" << endl;
621  return UCS_string();
622  }
623 
624  /* monadic ⎕FX. at this point we should have:
625 
626  tos[4] = ( ⎕FX Value )
627  */
628 
629  Log(LOG_Quad_TF)
630  CERR << "inverse 2 ⎕TF (monadic ⎕FX):" << endl;
631  if (tos.size() != 4) return UCS_string();
632  if (tos[1].get_tag() != TOK_Quad_FX) return UCS_string();
633  if (tos[2].get_Class() != TC_VALUE) return UCS_string();
634 
635 static const int eprops[] = { 0, 0, 0, 0 };
636 const Token tok = Quad_FX::do_quad_FX(eprops, tos[2].get_apl_val(),
637  UTF8_string("2 ⎕TF"), true);
638 
639  if (tok.get_Class() != TC_VALUE)
640  {
641  Log(LOG_Quad_TF)
642  CERR << "Quad_FX::do_quad_FX() failed in tf2_inverse()" << endl;
643  return UCS_string(); // error in ⎕FX
644  }
645 
646  // ⎕FX succeeded (and the token returned is the name of the function)
647  //
648  return UCS_string(*tok.get_apl_val().get());
649 }

652 Quad_TF::tf2_shape(UCS_string & ucs, const Shape & shape, ShapeItem nesting)
653 {
654  ucs.append(UNI_L_PARENT);
655  loop(n, nesting) ucs.append(UNI_SUBSET); // ⊂...
656 
657  // scalars are ''⍴SCALAR but ''⍴ has no effect and can be omitted
658 
659  if (shape.get_rank()) // non-scalar
660  {

662  {
663  if (r) ucs.append(UNI_SPACE);
664  ucs.append_number(shape.get_shape_item(r));
665  }
666 
667  ucs.append(UNI_RHO);
668  }

965  // 0 1 2 3 4 5
966  //

987  .get_int_value();

989  loop(j, 6) tos[s + j].clear(LOC);
990

1006  }
1007 
1008  return skipped > 0;
1009 }

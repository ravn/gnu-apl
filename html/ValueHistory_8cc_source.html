69 void
70 VH_entry::print_history(ostream & out, const Value * val, const char * loc)
71 {
72  // search backwards for events of val.
73  //
74 std::vector<const VH_entry *> var_events;
75 int cidx = VH_entry::idx;
76 
77  loop(e, VALUEHISTORY_SIZE)
78  {
79  --cidx;
80  if (cidx < 0) cidx = VALUEHISTORY_SIZE - 1;
81 
82  const VH_entry * entry = VH_entry::history + cidx;
83 
84  if (entry->event == VHE_None) break; // end of history
85 
86  if (entry->event == VHE_Error)
87  {
88  // add error event to every value history
89  //
90  var_events.push_back(entry);
91  continue;
92  }
93 
94  if (entry->val != val) continue; // some other var
95 
96  var_events.push_back(entry);
97 
98  if (entry->event == VHE_Create) break; // create event found
99  }
100 
101  if (VALUEHISTORY_SIZE == 0)
102  {
103  out << "value history disabled" << endl;
104  }
105  else
106  {
107  out << endl << "value " << voidP(val)
108  << " has " << var_events.size()
109  << " events in its history";
110  if (loc) out << " (at " << loc << ")";
111  out << ":" << endl;
112  }
113 
114 int flags = 0;
115 const VH_entry * previous = 0;
116  loop(e, var_events.size())
117  {
118  const VH_entry * vev = var_events[var_events.size() - e - 1];
119  vev->print(flags, out, val, previous);
120  previous = vev;
121  }
122  out << endl;
123 }

125 static UCS_string
126 flags_name(ValueFlags flags)
127 {
128 UCS_string ret;
129 
130  if (flags & VF_marked) ret.append(UNI_ASCII_M);
131  if (flags & VF_complete) ret.append(UNI_ASCII_C);
132 
133  while (ret.size() < 4) ret.append(UNI_ASCII_SPACE);
134  return ret;
135 }

138 VH_entry::print(int & flags, ostream & out, const Value * val,
139  const VH_entry * previous) const
140 {
141 const ValueFlags flags_before = ValueFlags(flags);
142 
143  if (previous == 0 ||
144  previous->testcase_file == 0 ||
145  previous->testcase_line != testcase_line ||
146  strcmp(previous->testcase_file, testcase_file))
147  {
148  if (testcase_file) out << " FILE: " << testcase_file
149  << ":" << testcase_line << endl;
150  }
151 
152  switch(event)
153  {
154  case VHE_Create:
155  out << " VHE_Create " << flags_before
156  << " ";
157  break;
158 
159  case VHE_Unroll:
160  out << " VHE_Unroll " << flags_before
161  << " ";
162  break;
163 
164  case VHE_Check:
165  flags |= VF_complete;
166  out << " VHE_Check " << flags_before << " ";
167  if (flags_before != flags) out << ValueFlags(flags)
168  << " ";
169  else out << " ";
170  break;
171 
172  case VHE_SetFlag:
173  flags |= iarg;
174  out << " Set " << flags_name(ValueFlags(iarg))
175  << " " << flags_before << " ";
176  if (flags_before != flags) out << ValueFlags(flags)
177  << " ";
178  else out << " ";
179  break;
180 
181  case VHE_ClearFlag:
182  flags &= ~iarg;
183  out << " Clear " << flags_name(ValueFlags(iarg))
184  << " " << flags_before << " ";
185  if (flags_before != flags) out << ValueFlags(flags)
186  << " ";
187  else out << " ";
188  break;
189 
190  case VHE_Erase:
191  out << " VHE_Erase " << flags_before << " ";
192  break;
193 
194  case VHE_Destruct:
195  out << " VHE_Destruct " << setw(26) << iarg;
196  break;
197 
198  case VHE_Error:
199  out << " " << setw(38)
200  << Error::error_name(ErrorCode(iarg)) << " ";
201  break;
202 
203  case VHE_PtrNew:
204  out << " VHE_PtrNew " << setw(26) << iarg;
205  break;
206

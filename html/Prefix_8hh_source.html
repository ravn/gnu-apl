241  void set_action(const Token & result)
242  {
243  switch(result.get_Class())
244  {
245  case TC_VALUE:
246  case TC_VOID:
247  case TC_END:
248  case TC_FUN2:
249  action = RA_CONTINUE;
250  return;
251 
252  case TC_RETURN:
253  // result was TOK_RETURN_EXEC, TOK_RETURN_STATS,
254  // TOK_RETURN_VOID, or TOK_RETURN_SYMBOL. The current
255  // context is xomplete and may or may not have produced
256  // a value.
257  //
258  action = RA_RETURN;
259  return;
260 
261  case TC_SI_LEAVE:
262  //
263  // result was TOK_SI_PUSHED or TOK_ERROR
264  if (result.get_tag() == TOK_ERROR) action = RA_RETURN;
265  else action = RA_SI_PUSHED;
266  return;
267 
268  default: CERR << "CLASS = " << result.get_Class()
269  << " at " << LOC << endl;
270  TODO;
271  }

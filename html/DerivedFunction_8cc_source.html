107 Token
108 DerivedFunction::eval_B(Value_P B)
109 {
110  Log(LOG_FunOperX)
111  {
112  print(CERR << "entering DerivedFunction");
113  CERR << "::eval_B() , this = "
114  << voidP(this) << endl;
115  }
116 
117  if (left_fun.get_tag() == TOK_VOID) // function bound to axis
118  {
119  return oper->eval_XB(axis, B);
120  }
121 
122  if (right_fun.get_tag() != TOK_VOID) // dyadic operator
123  {
124  return oper->eval_LRB(left_fun, right_fun, B);
125  }
126  else // monadic operator
127  {
128  if (!!axis) return oper->eval_LXB(left_fun, axis, B);
129  else return oper->eval_LB(left_fun, B);
130  }
131 }

153 Token
154 DerivedFunction::eval_AB(Value_P A, Value_P B)
155 {
156  Log(LOG_FunOperX)
157  {
158  print(CERR << "entering DerivedFunction");
159  CERR << "::eval_AB()" << endl;
160  }
161 
162  if (left_fun.get_tag() == TOK_VOID) // function bound to axis
163  {
164  return oper->eval_AXB(A, axis, B);
165  }
166 
167  if (right_fun.get_tag() != TOK_VOID) // dyadic operator
168  {
169  if (!axis) return oper->eval_ALRB(A, left_fun, right_fun, B);
170  else return oper->eval_ALRXB(A, left_fun, right_fun, axis, B);
171  }
172  else // monadic operator
173  {
174  if (!axis) return oper->eval_ALB(A, left_fun, B);
175  else return oper->eval_ALXB(A, left_fun, axis, B);
176  }
177 }

198 ostream &
199 DerivedFunction::print(ostream & out) const
200 {
201  out << "(";
202  if (left_fun.is_function()) left_fun.get_function()->print(out);
203  else out << "VAL";
204  out << " ";
205 
206  oper->print(out);
207  if (!!axis) out << "[]";
208 
209  if (right_fun.get_tag() != TOK_VOID) // dyadic operator
210  {
211  out << " ";
212  if (right_fun.is_function()) right_fun.get_function()->print(out);
213  else out << "VAL";
214  }
215 
216  return out << ")";
217 }

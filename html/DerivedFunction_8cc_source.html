127 Token
128 DerivedFunction::eval_B(Value_P B) const
129 {
130  Log(LOG_FunOperX)
131  {
132  print(CERR << "entering DerivedFunction");
133  CERR << "::eval_B() , this = "
134  << voidP(this) << endl;
135  }
136 
137  if (left_arg.get_tag() == TOK_VOID) // function bound to axis
138  {
139  return oper->eval_XB(axis, B);
140  }
141 
142  if (right_fun.get_tag() != TOK_VOID) // dyadic operator
143  {
144  Token & left = const_cast<Token &>(left_arg);
145  Token & right = const_cast<Token &>(right_fun);
146  return oper->eval_LRB(left, right, B);
147  }
148  else // monadic operator
149  {
150  if (left_arg.is_function()) // normal operator
151  {
152  Token & left = const_cast<Token &>(left_arg);
153  if (!axis) return oper->eval_LB(left, B);
154  else return oper->eval_LXB(left, axis, B);
155  }
156  else // operator with left value operand
157  {
158  Value_P A = left_arg.get_apl_val();
159  if (!axis) return oper->eval_AB(A, B);
160  else return oper->eval_AXB(A, axis, B);
161  }
162  }
163 }

95 bool
96 Plot_window_properties::update(int verbosity)
97 {
98  // this function is called for 2 reasons:
99  //
100  // 1. when constructor Plot_window_properties::Plot_window_properties()
101  // is called in plot_main(), i.e. when the the plot window is createdÂ·
102  // for the first time, and
103  //
104  // 2. when the user resizes the plot window.
105  //
106  // In both cases the caller has changed (at least) pa_width and pa_height,
107  // and this function recomputes all variables that depend on them.
108  //
109 
110  window_width = pa_border_L + origin_X + pa_width + pa_border_R;
111  window_height = pa_border_T + origin_Y + pa_height + pa_border_B;
112 
113  min_X = plot_data.get_min_X();
114  max_X = plot_data.get_max_X();
115  min_Y = plot_data.get_min_Y();
116  max_Y = plot_data.get_max_Y();
117  min_Z = plot_data.get_min_Z();
118  max_Z = plot_data.get_max_Z();
119 
120  // the user may override the range derived from the data
121  //
122  if (rangeX_min_valid()) min_X = rangeX_min;
123  if (rangeX_max_valid()) max_X = rangeX_max;
124  if (rangeY_min_valid()) min_Y = rangeY_min;
125  if (rangeY_max_valid()) max_Y = rangeY_max;
126  if (rangeZ_min_valid()) min_Z = rangeZ_min;
127  if (rangeZ_max_valid()) max_Z = rangeZ_max;
128 
129  if (min_X >= max_X)
130  {
131  MORE_ERROR() << "empty X range in A âPLOT B";
132  return true;
133  }
134 
135  if (min_Y >= max_Y)
136  {
137  MORE_ERROR() << "empty Y range " << min_Y << ":" << max_Y
138  << " in A âPLOT B";
139  return true;
140  }
141 
142  if (min_Z >= max_Z)
143  {
144  MORE_ERROR() << "empty Z range in A âPLOT B";
145  return true;
146  }
147 
148 double delta_X = max_X - min_X;
149 double delta_Y = max_Y - min_Y;
150 double delta_Z = max_Z - min_Z;
151 const double orig_len = sqrt(origin_X*origin_X + origin_Y*origin_Y);
152 
153  // first approximation for value â pixel factor
154  //
155  scale_X = pa_width / delta_X; // pixels per X-value
156  scale_Y = pa_height / delta_Y; // pixels per Y-value
157  scale_Z = orig_len ? orig_len / delta_Z : 1.0; // pixels per Z-value
158 
159  if (verbosity & SHOW_DRAW)
160  CERR << setw(20) << "delta_X (1): " << delta_X << " (Xmax-Xmin)" << endl
161  << setw(20) << "delta_Y (1): " << delta_Y << " (Ymax-Ymin)" << endl
162  << setw(20) << "delta_Z (1): " << delta_Z << " (Zmax-Zmin)" << endl
163  << setw(20) << "scale_X (1): " << scale_X << " pixels/X" << endl
164  << setw(20) << "scale_Y (1): " << scale_Y << " pixels/Y" << endl
165  << setw(20) << "scale_Z (1): " << scale_Z << " pixels/Z" << endl;
166 
167 const double tile_X_raw = gridX_pixels / scale_X;
168 const double tile_Y_raw = gridY_pixels / scale_Y;
169 const double tile_Z_raw = gridZ_pixels / scale_Z;
170 
171  if (verbosity & SHOW_DRAW)
172  CERR << setw(20) << "tile_X_raw: " << tile_X_raw
173  << "(âX between grid lines)" << endl
174  << setw(20) << "tile_Y_raw: " << tile_Y_raw
175  << "(âY between grid lines)" << endl
176  << setw(20) << "tile_Z_raw: " << tile_Z_raw
177  << "(âZ between grid lines)" << endl;
178 
179  tile_X = round_up_125(tile_X_raw);
180  tile_Y = round_up_125(tile_Y_raw);
181  tile_Z = round_up_125(tile_Z_raw);
182 
183 const int min_Xi = floor(min_X / tile_X);
184 const int min_Yi = floor(min_Y / tile_Y);
185 const int min_Zi = floor(min_Z / tile_Z);
186 const int max_Xi = ceil(max_X / tile_X);
187 const int max_Yi = ceil(max_Y / tile_Y);
188 const int max_Zi = ceil(max_Z / tile_Z);
189  gridX_last = 0.5 + max_Xi - min_Xi;
190  gridY_last = 0.5 + max_Yi - min_Yi;
191  gridZ_last = 0.5 + max_Zi - min_Zi;
192  min_X = tile_X * floor(min_X / tile_X);
193  min_Y = tile_Y * floor(min_Y / tile_Y);
194  min_Z = tile_Z * floor(min_Z / tile_Z);
195  max_X = tile_X * ceil(max_X / tile_X);
196  max_Y = tile_Y * ceil(max_Y / tile_Y);
197  max_Z = tile_Z * ceil(max_Z / tile_Z);
198 
199  delta_X = max_X - min_X;
200  delta_Y = max_Y - min_Y;
201  delta_Z = max_Z - min_Z;
202 
203  // final value â pixel factor
204  //
205  scale_X = pa_width / delta_X;
206  scale_Y = pa_height / delta_Y;
207  scale_Z = orig_len ? orig_len / delta_Z : 1.0;
208 
209  if (verbosity & SHOW_DRAW)
210  CERR << setw(20) << "min_X (2): " << min_X << endl
211  << setw(20) << "max_X (2): " << max_X << endl
212  << setw(20) << "min_Y (2): " << min_Y << endl
213  << setw(20) << "max_Y (2): " << max_Y << endl
214  << setw(20) << "min_Z (2): " << min_Z << endl
215  << setw(20) << "max_Z (2): " << max_Z << endl
216  << setw(20) << "delta_X (2): " << delta_X << " (Xmax-Xmin)" << endl
217  << setw(20) << "delta_Y (2): " << delta_Y << " (Ymax-Ymin)" << endl
218  << setw(20) << "delta_Z (2): " << delta_Z << " (Zmax-Zmin)" << endl
219  << setw(20) << "scale_X (2): " << scale_X << " pixels/X" << endl
220  << setw(20) << "scale_Y (2): " << scale_Y << " pixels/Y" << endl
221  << setw(20) << "scale_Z (2): " << scale_Z << " pixels/Z" << endl
222  << setw(20) << "tile_X (2): " << tile_X
223  << "(âX between grid lines)" << endl
224  << setw(20) << "tile_Y (2): " << tile_Y
225  << "(âY between grid lines)" << endl
226  << setw(20) << "tile_Z (2): " << tile_Z
227  << "(âZ between grid lines)" << endl;
228 
229  return false; // OK
230 }

256 const char *
257 Plot_window_properties::set_attribute(const char * att_and_val)
258 {
259  /*
260  called with leading whitespaces in att_and_val removed.
261 
262  att_and_val is a string specifying one of:
263 
264  a window attribute, like: origin_X: 100 , or
265  a line attribute, like: point_color-1: #000000 , or
266  a color gradient, like: color_level-50: #00FF00
267 
268  The value (right of ':') can be
269 
270  an integer value, like: 100
271  a color, like: #000000 or #000
272  a string, like: "Window Title"
273 
274  return 0 on success or else an appropriate error string.
275  */
276 
277  // find demarcation between attribute name and attribute value
279 const char * colon = strchr(att_and_val, ':');
280  if (colon == 0) return "Expecting 'attribute: value' but no : found";
281 
282  // skip leading whitespace before the attribute value
283  //
284 const char * value = colon + 1;
285  while (*value && (*value <= ' ')) ++value;
286 
287  // check for line number resp. level
288  //Â·
289 const char * minus = strchr(att_and_val, '-');
290  if (minus && minus < colon) // line attribute or color gradient
291  {
292  // it is not unlikely that the user wants to use the same attributes
293  // for different plots that may differ in the number of lines. We
294  // therefore silently ignore such over-specified attribute rather
295  // than returning an error string (which then raises a DOMAIN error).
296  //Â·
297  if (!strncmp(att_and_val, "color_level-", 12))
298  {
299  const int level = strtoll(minus + 1, 0, 10);
300  if (level < 0) return "negative color level";
301  if (level > 100) return "color level > 100%";
302  const char * error = 0;
303  const Color color = Plot_data::Color_from_str(value, error);
304  if (error) return error;
305 
306  level_color grad = { level, color };
307 
308  // insert grad into gradient, but keeping it sorted by level
309  //
310  loop(g, gradient.size())
311  {
312  if (gradient[g].level == grad.level)
313  return "duplicate gradient level";
314 
315  if (gradient[g].level > grad.level)
316  {
317  gradient.insert(gradient.begin() + g, grad);
318  return 0;
319  }
320  }
321  gradient.push_back(grad);
322  return 0; // OK
323  }
324 
325  const int line = strtoll(minus + 1, 0, 10) - Workspace::get_IO();
326  if (line < 0) return "line number â¤ âIO";
327  if (line >= line_count) return "line number too large";
328 
329 # define ldef(ty, na, val, _descr) \
330  if (!strncmp(#na "-", att_and_val, minus - att_and_val)) \
331  { const char * error = 0; \
332  line_properties[line]->set_ ## na(Plot_data::ty ## \
333  _from_str(value, error)); \
334  return error; \
335  }
336 # include "Quad_PLOT.def"
337 
338  // nothing found
339  //
340  return "Bad or unknown line attribute";
341  }
342  else // window attribute
343  {
344 # define gdef(ty, na, _val, _descr) \
345  if (!strncmp(#na, att_and_val, colon - att_and_val)) \
346  { const char * error = 0; \
347  set_ ## na(Plot_data::ty ## _from_str(value, error)); \
348  return error; \
349  }
350 # include "Quad_PLOT.def"
351  }
352 
353  return "Bad or unknown window attribute";
354 }
355 //-----------------------------------------------------------------------------
356 double
357 Plot_window_properties::round_up_125(double val)
358 {
359 int expo = 0;
360  while (val >= 10) { val /= 10; ++expo; }
361  while (val < 1) { val *= 10; --expo; }
362 
363  // at this point 1 â¤ val < 10.
364  //
365  if (expo > -18 && expo < 18) // use integer arithmetic for 10^expo
366  {
367  const int abs_expo = expo < 0 ? -expo : expo;
368  uint64_t expo_val = 1;
369  loop(a, abs_expo) expo_val *= 10;
370 
371  if (val <= 2.0) return expo < 0 ? 2.0/expo_val : 2.0*expo_val;
372  if (val <= 5.0) return expo < 0 ? 5.0/expo_val : 5.0*expo_val;
373  return expo < 0 ? 10.0/expo_val : 10.0*expo_val;
374  }
375  else // use float arithmetic for 10^expo
376  {
377  const double expo_val = pow(10, expo);
378  if (val <= 2.0) return expo < 0 ? 2.0/expo_val : 2.0*expo_val;
379  if (val <= 5.0) return expo < 0 ? 5.0/expo_val : 5.0*expo_val;
380  return expo < 0 ? 10.0/expo_val : 10.0*expo_val;
381  }
382 }
383 //-----------------------------------------------------------------------------
384 uint32_t
385 Plot_window_properties::get_color(double alpha) const
386 {
387  Assert(alpha >= 0.0);
388  Assert(alpha <= 1.0);
389 
390  if (gradient.size() == 0) // no gradient specified: use gray-scale
391  return uint32_t(255*alpha) << 16
392  | uint32_t(255*alpha) << 8
393  | uint32_t(255*alpha);
394 
395  if (gradient.size() == 1) // only one gradient specified: use it
396  return gradient[0].rgb;
397 
398  loop(g, gradient.size())
399  {
400  const double level_g = 0.01*gradient[g].level;
401  if (alpha == level_g) return gradient[g].rgb;
402  if (alpha < level_g) // level_g is the first level above alpha
403  {
404  if (g == 0) return gradient[0].rgb; // use first gradient
405 
406  // interpolate
407  //
408  const double l0 = 0.01*gradient[g-1].level; // previous level
409  const double beta1 = (alpha - l0) / (level_g - l0);
410  Assert(beta1 >= 0.0);
411  Assert(beta1 <= 1.0);
412  const double beta0 = 1.0 - beta1;
413  const uint32_t rgb0 = gradient[g-1].rgb;
414  const uint32_t rgb1 = gradient[g].rgb;
415  const uint32_t red = beta1*(rgb1 >> 16 & 0xFF)
416  + beta0*(rgb0 >> 16 & 0xFF);
417  const uint32_t grn = beta1*(rgb1 >> 8 & 0xFF)
418  + beta0*(rgb0 >> 8 & 0xFF);
419  const uint32_t blu = beta1*(rgb1 & 0xFF)
420  + beta0*(rgb0 & 0xFF);
421  return red << 16 | grn << 8 | blu;
422  }
423  }
424 
425  // alpha is above last gradient: use last gradient
426  //
427  return gradient.back().rgb;
428 }

430 Pixel_XY
431 Plot_window_properties::valXYZ2pixelXY(double X, double Y, double Z) const
432 {
433 const double phi = atan2(get_origin_Y(), get_origin_X());
434 const Pixel_Z pz = valZ2pixel(Z - get_min_Z());
435 const Pixel_X px = valX2pixel(X - get_min_X()) + get_origin_X() - pz*cos(phi);
436 const Pixel_X py = valY2pixel(Y - get_min_Y()) + pz*sin(phi);
437 
438  return Pixel_XY(px, py);
439 }

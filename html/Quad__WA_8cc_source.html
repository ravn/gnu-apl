47 void
48 Quad_WA::init(bool log_startup)
49 {
50  initial_sbrk = top_of_memory();
51 
52 rlimit rl;
53 
54 #ifndef RLIMIT_AS // BSD does not define RLIMIT_AS
55 # define RLIMIT_AS RLIMIT_DATA
56 #endif
57 
58  getrlimit(RLIMIT_AS, &rl);
59  initial_rlimit = rl.rlim_cur;
60 
61  if (log_startup)
62  {
63  CERR << "initial RLIMIT_AS (aka. virtual memory) is: ";
64  if (initial_rlimit == ~rlim_t(0)) CERR << "'unlimited'" << endl;
65  else CERR << initial_rlimit << endl;
66  }
67 
68  // if the user has set a memory rlimit (and hopefully knowing what she is
69  // doing) then leave it as is; otherwise set the limit to 80 % of the
70  // avaiable memory
71  //
72  total_memory = get_free_memory();
73  if (log_startup)
74  CERR << "estimated available memory: " << total_memory
75  << " bytes (" << (total_memory/1000000) << " MB)" << endl;
76 
77  if (rl.rlim_cur == RLIM_INFINITY)
78  {
79  // the user has not set a memory limit, so we may do that. However,
80  // limits > 2GB are treated as RLIM_INFINITY, so setting larger
81  // limits may have undesirable effects
82  //
83  if (total_memory < 0x8000000)
84  {
85  rl.rlim_cur = total_memory;
86  setrlimit(rl.rlim_cur, &rl);
87  getrlimit(RLIMIT_AS, &rl);
88  if (log_startup)
89  CERR << "decreasing RLIMIT_AS to: " << rl.rlim_cur
90  << " bytes (" << int64_t(rl.rlim_cur/1000000)
91  << " MB)" << endl;
92  }
93  }
94  else if (rl.rlim_cur > total_memory)
95  {
96  // the user has set a memory limit, but it is above the available
97  // memory
98  //
99  CERR <<
100 "*** Warning: the process memory limit (RLIMIT_AS) of " << rl.rlim_cur << endl
101 << " is more than the estimated total_memory of " << total_memory << "." << endl
102 << " This could cause improper WS FULL handling." << endl;
103  }
104 }

106 uint64_t
107 Quad_WA::get_free_memory()
108 {
109 uint64_t result = 0xC0000000; // assume ~3 GB on error
110 
111  if (FILE * pm = fopen("/proc/meminfo", "r"))
112  {
113  uint64_t mem_ACF[] = { 0, 0, 0 }; // available / cached / free
114  const char * tags_ACF[] = { "MemAvailable:", "Cached:", "MemFree:" };
115  for (;;)
116  {
117  char buffer[2000];
118  if (fgets(buffer, sizeof(buffer) - 1, pm) == 0) break;
119  buffer[sizeof(buffer) - 1] = 0;
120 
121  for (size_t t = 0; t < sizeof(tags_ACF) / sizeof(*tags_ACF); ++t)
122  {
123  const char * tag = tags_ACF[t];
124  const size_t tag_len = strlen(tag);
125  if (!strncmp(buffer, tag, tag_len))
126  {
127  mem_ACF[t] = 1024 * strtoll(buffer + tag_len, 0, 10);
128  }
129  }
130  }
131  fclose(pm);
132 
133  result = mem_ACF[0] ? mem_ACF[0] // MemAvailable:
134  : (mem_ACF[1] + mem_ACF[2]); // Cached: + MemFree:
135  }
136 
137  // limit size on 32-bit machines to 2.95 < 3 GB
138  if (sizeof(const void *) == 4 && result > 0xB0000000) result = 0xB0000000;
139 
140  return result;
141 }

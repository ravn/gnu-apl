160 //-----------------------------------------------------------------------------
161 DLX_Node::Col_Type
162 DLX_Node::get_col_type(const Cell & cell)
163 {
164  if (cell.is_character_cell())
165  {
166  // blank and '0' shall mean 0, '1' means 1, and '2' means 2
167  //
168  switch(cell.get_char_value())
169  {
170  case ' ':
171  case '0': return Col_NULL;
172  case '1': return Col_PRIMARY;
173  case '2': return Col_SECONDARY;
174  default: return Col_INVALID;
175  }
176  }
177  if (cell.is_numeric())
178  {
179  switch(cell.get_int_value())
180  {
181  case 0: return Col_NULL;
182  case 1: return Col_PRIMARY;
183  case 2: return Col_SECONDARY;
184  default: return Col_INVALID;
185  }
186  }
187 
188  return Col_INVALID;
189 }

210  void cover(ShapeItem j)
211  {
212  ++cover_count;
213 
214  DLX_Header_Node & c = headers[j];
215  Assert(c.row == -1);
216  Assert(c.col == j);
217  c.remove_lr();
218  if (c.col_type == Col_PRIMARY) --primary_count;
219 
220  for (DLX_Node * i = c.down; i != &c; i = i->down)
221  {
222  Assert(i->col == j);
223  for (DLX_Node * j = i->right; j != i; j = j->right)
224  {
225  j->remove_ud();
226  --headers[j->col].count;
227  }
228  }
229  }
230 
232  void uncover(ShapeItem j)
233  {
234  DLX_Header_Node & c = headers[j];
235  Assert(c.row == -1);
236  Assert(c.col == j);
237  for (DLX_Node * i = c.up; i != &c; i = i->up)
238  {
239  Assert(i->col == j);
240  for (DLX_Node * j = i->left; j != i; j = j->left)
241  {
242  j->insert_ud();
243  headers[j->col].count++;
244  }
245  }
246  c.insert_lr();
247  if (c.col_type == Col_PRIMARY) ++primary_count;
248  c.check();
249  }

309 //-----------------------------------------------------------------------------
310 DLX_Root_Node::DLX_Root_Node(ShapeItem rs, ShapeItem cs, ShapeItem max_sol,
311  const Value & B)
312  : DLX_Node(-1, -1, this, this, this, this),
313  max_solutions(max_sol),
314  rows(rs),
315  cols(cs),
316  primary_count(0),
317  secondary_count(0),
318  solution_count(0),
319  level(0),
320  pick_count(0),
321  cover_count(0),
322  first_0(0)
323 {
324  // count the number of non-zero elements in the matrix
325  //
326 const ShapeItem ec_B = B.element_count();
327 const int qio = Workspace::get_IO();
328 ShapeItem ones = 0;
329 const Cell * b = &B.get_ravel(0);
330  loop(e, ec_B)
331  {
332  const Cell & cell = *b++;
333  const Col_Type ct = get_col_type(cell);
334  if (ct == Col_INVALID)
335  {
336  const ShapeItem row = qio + e/B.get_cols();
337  const ShapeItem col = qio + e%B.get_cols();
338  UCS_string & more = MORE_ERROR() << "Bad B[" << row << ";"
339  << col << "]: ";
340  if (cell.is_integer_cell())
341  more << cell.get_int_value();
342  else if (cell.is_character_cell())
343  more << "'" << cell.get_char_value() << "'";
344  else
345  more << "neither integer nor character";
346  more << " in âDLX B";
347  DOMAIN_ERROR;
348  }
349  if (ct != Col_UNKNOWN) ++ones;
350  else if (!first_0) first_0 = &cell;
351  }
352 
353  Log(LOG_Quad_DLX) CERR << "Matrix has " << ones << " ones" << endl;
354 
355  // set up column headers. std::vector.push_back() may move the headers so
356  // we first append all headers and initialize then
357  //
358  headers.reserve(cols);
359  loop (c, cols) headers.push_back(DLX_Header_Node());
360  loop (c, cols)
361  {
362  if (c) new (&headers[c]) DLX_Header_Node(c, &headers[c - 1], this);
363  else new (&headers[c]) DLX_Header_Node(c, this, this);
364  }
365 
366  // set up non-header nodes. std::vector.push_back() may move the headers so
367  // we first append all nodes and initialize then.
368  //
369  b = &B.get_ravel(0);
370  nodes.reserve(ones);
371  loop(o, ones) nodes.push_back(DLX_Node());
372 
373  DLX_Node * n = &nodes[0];
374  loop (r, rows)
375  {
376  DLX_Node * const lm = n; // leftmost item in this row
377  DLX_Node * rm = n; // rightmost item in this row
378  loop (c, cols)
379  {
380  DLX_Header_Node & hdr = headers[c];
381  const Col_Type ct = get_col_type(*b++);
382  if (ct < Col_PRIMARY) continue; // most likely: not '1' or '2'
383 
384  if (hdr.col_type == Col_UNKNOWN) // first '1' or '2' in this col
385  {
386  hdr.col_type = ct;
387  if (ct == Col_PRIMARY) ++primary_count;
388  else ++secondary_count;
389  }
390  else if (ct != hdr.col_type) // col type mismatch
391  {
392  MORE_ERROR() <<
393  "the columns of B in âDLX B must either be primary ("
394  "containing 1s) or secondary (containing 2s), but not both";
395  DOMAIN_ERROR;
396  }
397 
398 
399  if (rm == n)
400  rm = new (n++) DLX_Node(r, c, hdr.up, &hdr, rm, lm);
401  else
402  rm = new (n++) DLX_Node(r, c, hdr.up, &hdr, rm, lm);
403  ++headers[c].count;
404  }
405  }
406 
407  // check that all columns have a 1 or 2
408  //
409  loop (c, cols)
410  {
411  if (headers[c].col_type == Col_UNKNOWN)
412  {
413  MORE_ERROR() << "âDLX B with empty column B[;"
414  << (c + qio) << "]";
415  DOMAIN_ERROR;
416  }
417  }
418 
419  Log(LOG_Quad_DLX) { loop(n, ones) nodes[n].print(CERR); }
420 }

495 DLX_Root_Node::deep_check() const
496 {
497  for (const DLX_Node * x = right; x != this; x = x->right)
498  {
499  const DLX_Header_Node & hdr =

501  Assert(hdr.row == -1);
502  hdr.check();

512 //-----------------------------------------------------------------------------
513 void
514 DLX_Root_Node::solve()
515 {
516 new_level:
517  Assert(level <= cols);
518 
519  deep_check();
520 
521  if (LOG_Quad_DLX || attention_is_raised())
522  {
523  CERR << "âDLX[" << level << "]";
524  loop(s, level)
525  CERR << " " << (headers[s].item_r->row + Workspace::get_IO());
526  CERR << endl;
527  clear_attention_raised(LOC);
528  }
529 
530  if (interrupt_is_raised())
531  {
532  clear_interrupt_raised(LOC);
533  return;
534  }
535 
536  // the problem is solved if there are no primary columns left
537  //
538 DLX_Node * item = right;
539 
540  Log(LOG_Quad_DLX) item->print_RC(CERR << "At header item ") << endl;
541 
542  if (primary_count == 0)
543  {
544  all_solutions.push_back(level);
545  loop(s, level)
546  all_solutions.push_back(headers[s].item_r->row);
547 
548  Log(LOG_Quad_DLX)
549  {
550  CERR << "!!!!! solution " << solution_count << ": rows are";
551  loop(s, level) CERR << " " << setw(2) << headers[s].item_r->row;
552  CERR << endl;
553  }
554 
555  ++solution_count;
556  if (max_solutions && solution_count >= max_solutions) return;
557 
558  goto level_done;
559  }
560 
561  if (item != this) // matrix not empty: select shortest column
562  {
563  ShapeItem col_size = 2*cols;
564  for (DLX_Node * i = right; i != this; i = i->right)
565  {
566  if (headers[i->col].col_type != Col_PRIMARY) continue;
567  const ShapeItem i_size = headers[i->col].count;
568  if (i_size < col_size)
569  {
570  item = i;
571  col_size = i_size;
572  }
573  }
574 
575  if (col_size == 0)
576  {
577  Log(LOG_Quad_DLX) CERR << "column " << item-> col
578  << " is empty" << endl;
579  goto level_done; // empty column: no solution
580  }
581  }
582 
583  Assert(item->row == -1);
584  Log(LOG_Quad_DLX) indent(CERR) << "Choose and cover column c="
585  << item->col << endl;
586  cover(item->col);
587 
588  headers[level].item_r = item;
589 
590 rloop: // running â
591  {
592  DLX_Node * r = headers[level].item_r->down;
593 
594  Log(LOG_Quad_DLX) r->print_RC(indent(CERR) << "rloop â at r= ") << endl;
595 
596  for (DLX_Node * j = r->right; j != r; j = j->right)
597  {
598  Log(LOG_Quad_DLX)
599  {
600  indent(CERR) << "Covering column j=" << j->col << endl;
601  }
602  cover(j->col);
603  }
604 
605  Log(LOG_Quad_DLX)
606  {
607  r->print_RC(indent(CERR) << "Picking item ") << endl;
608  }
609  headers[level].item_r = r;
610  ++level;
611  ++pick_count;
612  goto new_level;
613  }
614 
615 level_done:
616  if (level > 0)
617  {
618  --level;
619  DLX_Node * r = headers[level].item_r;
620  Log(LOG_Quad_DLX)
621  {
622  r->print_RC(indent(CERR) << "Backtracking") << endl;
623  }
624 
625  for (DLX_Node * j = r->left; j != r; j = j->left)
626  {
627  Log(LOG_Quad_DLX)
628  {
629  indent(CERR) << "Uncovering column j=" << j->col << endl;
630  }
631  uncover(j->col);
632  }
633 
634  r = r->down;
635  if (r->row == -1) // end of rloop reached
636  {
637  Log(LOG_Quad_DLX)
638  {
639  indent(CERR) << "Uncovering column c=" << r->col << endl;
640  }
641  uncover(r->col);
642  goto level_done;
643  }
644 
645  goto rloop;
646  }
647 }
648 //-----------------------------------------------------------------------------
649 Token
650 DLX_Root_Node::preset(const Cell * steps, ShapeItem step_count, const Cell * cB)
651 {
652 const int qio = Workspace::get_IO();
653  loop(a, step_count)
654  {
655  const APL_Integer row = steps++->get_int_value() - qio;
656  if (row < 0 || row >= rows)
657  {
658  MORE_ERROR() << "bad row: " << row;
659  DOMAIN_ERROR;
660  }
661 
662  bool found_a = false;
663 
664  // find a '1' or '2' item in row
665  //
666  loop(b, nodes.size())
667  {
668  const DLX_Node & n = nodes[b];
669  if (n.row == row) // found '1' or '2'
670  {
671  found_a = true;
672  Log(LOG_Quad_DLX)
673  n.print_RC(CERR << "Covering column " << n.col
674  << " of rightmost item ") << endl;
675  cover(n.col);
676 
677  for (DLX_Node * j = n.right; j != &n; j = j->right)
678  {
679  Log(LOG_Quad_DLX)
680  {
681  CERR << "Covering column j=" << j->col << endl;
682  }
683  cover(j->col);
684  }
685 
686  break; // next row a
687  }
688  }
689  Assert(found_a);
690  }
691 
692  // construct return value
693  //
694 ShapeItem cols_Z = 0;
695  for (DLX_Node * h = right; h != this; h = h->right) ++cols_Z;
696 
697  // set al matrix elements to 0
698  //
699 const Shape shape_Z(rows, cols_Z);
700 Value_P Z(shape_Z, LOC);
701  loop(z, (rows*cols_Z))
702  {
703  if (first_0) Z->next_ravel()->init(*first_0, Z.getref(), LOC);
704  else if (cB->is_integer_cell()) new (Z->next_ravel()) IntCell(0);
705  else new (Z->next_ravel()) CharCell(UNI_ASCII_0);
706  }
707 
708 ShapeItem c = 0; // the column in the reduced matrix
709  for (DLX_Node * h = right; h != this; h = h->right)
710  {
711  for (DLX_Node * j = h->down; j != h; j = j->down)
712  {
713  Assert(h->col == j->col);
714  const Cell & src = cB[h->col + cols*j->row];
715  Z->get_ravel(c + cols_Z*j->row).init(src, Z.getref(), LOC);
716  }
717  ++c;
718  }
719 
720  Z->check_value(LOC);
721  return Token(TOK_APL_VALUE1, Z);
722 }

724 Token
725 Quad_DLX::eval_AB(Value_P A, Value_P B)
726 {
727  if (A->get_rank() > 1) RANK_ERROR;
728  if (A->element_count() < 1) LENGTH_ERROR;
729 
730  if (B->get_rank() != 2) RANK_ERROR;
731 const APL_Integer a0 = A->get_ravel(0).get_int_value();
732 
733  if (a0 < -4) DOMAIN_ERROR;
734  if (a0 == -4) // perform some dance steps with the constraints matrix
735  {
736  const ShapeItem rows = B->get_rows();
737  const ShapeItem cols = B->get_cols();
738  DLX_Root_Node root(rows, cols, 0, B.getref());
739  return root.preset(&A->get_ravel(1), A->element_count() - 1,
740  &B->get_ravel(0));
741  }
742 
743  if (A->element_count() != 1) LENGTH_ERROR;
744  return do_DLX(a0, B.getref());
745 }
746 //-----------------------------------------------------------------------------
747 Token
748 Quad_DLX::eval_B(Value_P B)
749 {
750  if (B->get_rank() != 2) RANK_ERROR;
751  return do_DLX(0, B.getref());
752 };
753 //-----------------------------------------------------------------------------
754 Token
755 Quad_DLX::do_DLX(ShapeItem how, const Value & B)
756 {
757 
758 const ShapeItem rows = B.get_rows();
759 const ShapeItem cols = B.get_cols();
760 
761 ShapeItem result_count = how;
762  if (how == 0) result_count = 1; // first solution
763  else if (how == -1) result_count = 0; // all solution
764  else if (how == -2) result_count = 1; // statistics: first solution
765  else if (how == -3) result_count = 0; // statistics: all solutions
766 
767 DLX_Root_Node root(rows, cols, result_count, B);
768  root.solve();
769 
770  if (how <= -2) // return statistics
771  {
772  Value_P Z(3, LOC);
773  new (Z->next_ravel()) IntCell(root.get_solution_count());
774  new (Z->next_ravel()) IntCell(root.get_pick_count());
775  new (Z->next_ravel()) IntCell(root.get_cover_count());
776  Z->check_value(LOC);
777  return Token(TOK_APL_VALUE1, Z);
778  }

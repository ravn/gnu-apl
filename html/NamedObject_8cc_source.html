28 NameClass
29 NamedObject::get_nc() const
30 {
31  if (id == ID_USER_SYMBOL || // this named object is a user defined object
32  id == ID_ALPHA || // ⍺
33  id == ID_ALPHA_U || // ⍶
34  id == ID_LAMBDA || // λ
35  id == ID_OMEGA || // ⍵
36  id == ID_OMEGA_U || // ⍹
37  id == ID_CHI) // χ
38  {
39  const Symbol * sym = get_symbol();
40  if (sym)
41  {
42  const ValueStackItem * tos = sym->top_of_stack();
43  if (tos) return tos->name_class;
44  }
45 
46  return NC_UNUSED_USER_NAME;
47  }
48 
49  // Distinguished name.
50  //
51  Assert(Avec::is_quad(get_name()[0]) || get_name()[0] == UNI_QUOTE_Quad);
52 
53 int len;
54 Token tok = Workspace::get_quad(get_name(), len);
55  if (len == 1) return NC_INVALID;
56 
57  if (tok.get_Class() == TC_SYMBOL) return NC_VARIABLE;
58  if (tok.get_Class() == TC_FUN2) return NC_FUNCTION;
59  if (tok.get_Class() == TC_FUN1) return NC_FUNCTION;
60  if (tok.get_Class() == TC_FUN0) return NC_FUNCTION;
61 
62  return NC_INVALID;
63 }

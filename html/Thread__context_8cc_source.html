93 //-----------------------------------------------------------------------------
94 void
95 Thread_context::PF_lock_unlock_pool(Thread_context & tctx)
96 {
97  Log(LOG_Parallel)
98  {
99  PRINT_LOCKED(CERR << "worker #" << tctx.get_N()
100  << " will now block itself on its pool_sema" << endl)
101  }
102 
103  tctx.do_join = false;
104  tctx.PF_join();
105 
106  tctx.blocked = true;
107  sem_wait(tctx.pool_sema);
108  tctx.blocked = false;
109 
110  Log(LOG_Parallel)
111  {
112  PRINT_LOCKED(CERR << "thread #" << tctx.get_N()
113  << " was unblocked from pool_sema" << endl)
114  }
115 }

141 void
142 Thread_context::bind_to_cpu(CPU_Number core, bool logit)
143 {
144 #if ! HAVE_AFFINITY_NP
145 
146  CPU = CPU_0;
147 
148 #else
149 
150  CPU = core;
151 
152  Log(LOG_Parallel || logit)
153  {
154  PRINT_LOCKED(CERR << "Binding thread #" << N
155  << " to core " << core << endl;);
156  }
157 
158 cpu_set_t cpus;
159  CPU_ZERO(&cpus);
160 
161  if (active_core_count == CCNT_1)
162  {
163  // there is only one core in total (which is the master).
164  // restore its affinity to all cores.
165  //
166  loop(a, Parallel::get_max_core_count())
167  CPU_SET(Parallel::get_CPU(a), &cpus);
168  }
169  else
170  {
171  CPU_SET(CPU, &cpus);
172  }
173 
174 const int err = pthread_setaffinity_np(thread, sizeof(cpu_set_t), &cpus);
175  if (err)
176  {
177  cerr << "pthread_setaffinity_np() failed with error "
178  << err << endl;
179  }
180 #endif // HAVE_AFFINITY_NP
181 }

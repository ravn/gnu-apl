113 //-----------------------------------------------------------------------------
114 void
115 Thread_context::PF_lock_unlock_pool(Thread_context & tctx)
116 {
117  Log(LOG_Parallel)
118  {
119  PRINT_LOCKED(CERR << "worker #" << tctx.get_N()
120  << " will now block itself on its pool_sema" << endl)
121  }
122 
123  tctx.do_join = false;
124  tctx.PF_join();
125 
126  tctx.blocked = true;
127  sem_wait(&tctx.pool_sema);
128  tctx.blocked = false;
129 
130  Log(LOG_Parallel)
131  {
132  PRINT_LOCKED(CERR << "thread #" << tctx.get_N()
133  << " was unblocked from pool_sema" << endl)
134  }
135 }

169 //-----------------------------------------------------------------------------
170 void
171 Thread_context::bind_to_cpu(CPU_Number core, bool logit)
172 {
173 #if ! HAVE_AFFINITY_NP
174 
175  CPU = CPU_0;
176 
177 #else
178 
179  CPU = core;
180 
181  Log(LOG_Parallel || logit)
182  {
183  PRINT_LOCKED(CERR << "Binding thread #" << N
184  << " to core " << core << endl;);
185  }
186 
187 cpu_set_t cpus;
188  CPU_ZERO(&cpus);
189 
190  if (active_core_count == CCNT_1)
191  {
192  // there is only one core in total (which is the master).
193  // restore its affinity to all cores.
194  //
195  loop(a, CPU_pool::get_count())
196  CPU_SET(CPU_pool::get_CPU(a), &cpus);
197  }
198  else
199  {
200  CPU_SET(CPU, &cpus);
201  }
202 
203 const int err = pthread_setaffinity_np(thread, sizeof(cpu_set_t), &cpus);
204  if (err)
205  {
206  cerr << "pthread_setaffinity_np() failed with error "
207  << err << endl;
208  }
209 #endif // HAVE_AFFINITY_NP
210 }

256 static void
257 draw_arrow(cairo_t * cr, Pixel_XY O, Pixel_XY A, const Color color)
258 {
259  // draw an arrow for an axis that starts at origin O and ends at
260  // A (where the arrow starts).
261 
262  // 1. create a unit vector U of length 1 parallel to the axis O-A
263  //
264 const int dx = A.x - O.x;
265 const int dy = A.y - O.y;
266 const double len = sqrt(dx*dx + dy*dy);
267 const double Ux = dx/len;
268 const double Uy = dy/len;
269 
270  // arrow dimensions (pixel)
271  //
272  enum { SHAFT = 20, WING = 5, TIP = 15 };
273 
274  // draw the shaft
275  //
276 const Pixel_XY S(A.x + SHAFT*Ux, A.y + SHAFT*Uy);
277  cairo_move_to(cr, A.x, A.y);
278  cairo_line_to(cr, S.x, S.y);
279  cairo_stroke(cr);
280 
281 const Pixel_XY W(S.x - WING*Ux, S.y - WING*Uy);
282 const Pixel_XY W1(W.x - WING*Uy, W.y + WING*Ux);
283 const Pixel_XY W2(W.x + WING*Uy, W.y - WING*Ux);
284 const Pixel_XY T(S.x + TIP*Ux, S.y + TIP*Uy);
285  cairo_move_to(cr, S.x, S.y);
286  cairo_line_to(cr, W1.x, W1.y);
287  cairo_line_to(cr, T.x, T.y);
288  cairo_line_to(cr, W2.x, W2.y);
289  cairo_close_path(cr);
290  cairo_fill(cr);
291 }

293 const char *
294 format_tick(double val)
295 {
296 static char cc[40];
297 char * cp = cc;
298  if (val == 0)
299  {
300  *cp++ = '0';
301  *cp++ = 0;
302  return cc;
303  }
304 
305  if (val < 0)
306  {
307  *cp++ = '-';
308  val = -val;
309  }
310 
311  // at this point,. val > 0
312  //
313  if (val < 1 && val >= 0.1) // 100-999 milli
314  {
315  snprintf(cc, sizeof(cc), "0.%03d", int(val*1000));
316  size_t cc_len = strlen(cc);
317  if (cc[--cc_len] == '0') cc[cc_len] = 0; // skip trailing 0
318  return cc;
319  }
320 
321 const char * unit = 0;
322  if (val >= 1e3)
323  {
324  if (val >= 1e15) ;
325  else if (val >= 1e12) { unit = "T"; val *= 1e-12; }
326  else if (val >= 1e9) { unit = "G"; val *= 1e-9; }
327  else if (val >= 1e6) { unit = "M"; val *= 1e-6; }
328  else { unit = "k"; val *= 1e-3; }
329  }
330  else
331  {
332 
333  if (val >= 1e0) { unit = ""; }
334  else if (val >= 1e-3) { unit = "m"; val *= 1e3; }
335  else if (val >= 1e-6) { unit = "Î¼"; val *= 1e6; }
336  else if (val >= 1e-9) { unit = "n"; val *= 1e9; }
337  else if (val >= 1e-12) { unit = "p"; val *= 1e12; }
338  }
339 
340  if (unit == 0) // very large or very small
341  {
342  snprintf(cp, sizeof(cc) - 1, "%.2E", val);
343  return cc;
344  }
345 
346  // at this point: 1 â¤ val < 1000
347  //
348  if (val < 10) snprintf(cp, sizeof(cc) - 1, "%.2f%s", val, unit);
349  else if (val < 100) snprintf(cp, sizeof(cc) - 1, "%.1f%s", val, unit);
350  else snprintf(cp, sizeof(cc) - 1, "%.0f%s", val, unit);
351  return cc;
352 }

480 // draw a 3D triangle where one point P0 is at visual height H0 and the
481 // other two points P1 and P2 are the same at visual height H12.
482 void
483 draw_triangle(cairo_t * cr, const Plot_context & pctx, int verbosity,
484  Pixel_XY P0, double H0, Pixel_XY P1, Pixel_XY P2, double H12)
485 {
486  // draw a triangle with P1 and P2 at the same level H12
487  //
488  Assert(H0 >= 0.0); Assert(H0 <= 1.0);
489  Assert(H12 >= 0.0); Assert(H12 <= 1.0);
490 
491  if (verbosity & SHOW_DRAW)
492  CERR << " â2: P0(" << P0.x << ":" << P0.y << ") @H0=" << H0
493  << " P1(" << P1.x << ":" << P1.y << ") @H12=" << H12
494  << " P2(" << P2.x << ":" << P2.y << ") @H12=" << H12 << endl;
495 
496  // every line is ~1 pixel, so the max y should suffice for steps
497  //
498 int steps = P0.y;
499  if (steps < P1.y) steps = P1.y;
500  if (steps < P2.y) steps = P2.y;
501 const double dH = (H12 - H0) / steps;
502 
503  loop(s, steps + 1)
504  {
505  const double alpha = H0 + s*dH;
506  const double beta = (1.0 * s)/steps; // line lenght
507  const Color line_color = pctx.w_props.get_color(alpha);
508  const Pixel_XY P0_P1(P0.x + beta * (P1.x - P0.x),
509  P0.y + beta * (P1.y - P0.y));
510  const Pixel_XY P0_P2(P0.x + beta * (P2.x - P0.x),
511  P0.y + beta * (P2.y - P0.y));
512  draw_line(cr, line_color, 1, 1, P0_P1, P0_P2);
513  }
514 }
515 //-------------------------------------------------------------------------------
516 // draw a 3D triangle where the points P0, P1, and P2 are at visual
517 // heights H0, H1, and H2 respectively. This is done by splitting the
518 // triangle into two triangles that each have 2 points at the same height.
519 void
520 draw_triangle(cairo_t * cr, const Plot_context & pctx, int verbosity,
521  Pixel_XY P0, double H0, Pixel_XY P1, double H1,
522  Pixel_XY P2, double H2)
523 {
524  // draw a triangle with P0, P1 and P2 at levels H0, H1, and H2
525 
526  Assert(H0 >= 0); Assert(H0 <= 1.0);
527  Assert(H1 >= 0); Assert(H1 <= 1.0);
528  Assert(H2 >= 0); Assert(H2 <= 1.0);
529 
530  if (verbosity & SHOW_DRAW)
531  CERR << "\nâ1: P0(" << P0.x << ":" << P0.y << ")@H=" << H0
532  << " P1(" << P1.x << ":" << P1.y << ")@H=" << H1
533  << " P2(" << P2.x << ":" << P2.y << ")@H=" << H2 << endl;
534 
535  if (H0 < H1) pv_swap(P0, H0, P1, H1); // then H0 >= H1
536  if (H0 < H2) pv_swap(P0, H0, P2, H2); // then H0 >= H2
537  if (H1 < H2) pv_swap(P1, H1, P2, H2); // then H1 >= H2
538 
539  // here H0 >= H1 >= H2
540  //
541  Assert(H0 >= H1);
542  Assert(H1 >= H2);
543 
544 const Plot_window_properties & w_props = pctx.w_props;
545 const vector<level_color> & color_steps = w_props.get_gradient();
546  if (color_steps.size() == 0)
547  {
548  CERR << "*** no color_steps" << endl;
549  return;
550  }
551 
552  if (H0 == H1) // P0 and P1 have the same height
553  {
554  draw_triangle(cr, pctx, verbosity, P2, H2, P0, P1, H0);
555  }
556  else if (H1 == H2) // P1 and P2 have the same height
557  {
558  draw_triangle(cr, pctx, verbosity, P0, H0, P1, P2, H1);
559  }
560  else // P2 lies below P1
561  {
562  const double alpha = (H0 - H1) / (H0 - H2); // alpha â 1 as P1 â P2
563  Assert(alpha >= 0.0);
564  Assert(alpha <= 1.0);
565

926  */

929

107 static void
108 draw_line(cairo_t * cr, Color line_color, int line_style, int line_width,
109  const Pixel_XY & P0, const Pixel_XY & P1)
110 {
111 double dashes_1[] = { }; // ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
112 double dashes_2[] = { 5.0 }; // ‚ï¥‚ï¥‚ï¥‚ï¥‚ï¥‚ï¥‚ï¥‚ï¥
113 double dashes_3[] = { 10.0, 5.0 }; // ‚îÄ‚ï¥‚îÄ‚ï¥‚îÄ‚ï¥‚îÄ‚ï¥
114 
115  enum
116  {
117  num_dashes_1 = sizeof(dashes_1) / sizeof(double),
118  num_dashes_2 = sizeof(dashes_2) / sizeof(double),
119  num_dashes_3 = sizeof(dashes_3) / sizeof(double),
120  };
121 
122  cairo_set_RGB_source(cr, line_color);
123  if (line_style == 3) cairo_set_dash(cr, dashes_3, num_dashes_3, 0);
124  else if (line_style == 2) cairo_set_dash(cr, dashes_2, num_dashes_2, 0);
125  else cairo_set_dash(cr, dashes_1, num_dashes_1, 0);
126 
127  cairo_set_line_width(cr, 1.0*line_width); // 1 pixel = 2 cairo
128  cairo_move_to(cr, P0.x, P0.y);
129  cairo_line_to(cr, P1.x, P1.y);
130  cairo_stroke(cr);
131 
132  if (line_style != 1) cairo_set_dash(cr, dashes_1, num_dashes_1, 0);
133 }

143 static void
144 draw_delta(cairo_t * cr, Pixel_XY P0, bool up, Color color, int size)
145 {
146 const double l = 0.51*size; // ‚àÜ center to top vertex
147 const double m = 0.866025404*l; // ‚àÜ center to base
148 const double s = 0.5*l; // ‚àÜ base middle to left/right vertex
149 
150  cairo_set_RGB_source(cr, color);
151 
152  if (up) // ‚ñ≤
153  {
154  cairo_move_to(cr, P0.x, P0.y + l); // top vertex
155  cairo_line_to(cr, P0.x + m, P0.y - s); // right vertex
156  cairo_line_to(cr, P0.x - m, P0.y - s); // left vertex
157  cairo_close_path(cr); // back to top
158  }
159  else // ‚ñº
160  {
161  cairo_move_to(cr, P0.x, P0.y - l); // bottom vertex
162  cairo_line_to(cr, P0.x + m, P0.y + s); // right vertex
163  cairo_line_to(cr, P0.x - m, P0.y + s); // left vertex
164  cairo_close_path(cr); // back to top
165  }
166 
167  cairo_fill(cr);
168 }

171 draw_quad(cairo_t * cr, Pixel_XY P0, bool caro, Color color, int size)
172 {
173  cairo_set_RGB_source(cr, color);
174 
175  if (caro) // ‚óÜ
176  {
177  const double dlta = 0.5 * size;
178  cairo_move_to(cr, P0.x, P0.y + dlta); // top vertex
179  cairo_line_to(cr, P0.x + dlta, P0.y); // left vertex
180  cairo_line_to(cr, P0.x, P0.y - dlta); // top vertex
181  cairo_line_to(cr, P0.x - dlta, P0.y); // left vertex
182  cairo_close_path(cr); // back to top
183  }
184  else // ‚ñ†
185  {
186  const double dlta = 0.35*size;
187  cairo_rectangle(cr, P0.x - dlta, P0.y - dlta, 2*dlta, 2*dlta);
188  }
189 
190  cairo_fill(cr);
191 }

194 draw_cross(cairo_t * cr, Pixel_XY P0, bool plus, Color color,
195  double size, int size2)
196 {
197  size *= 0.98;
198  cairo_set_RGB_source(cr, color);

200 const double half = 0.5*size;
201  if (size2 == 0) size2 = 2; // default line thickness
202  cairo_set_line_width(cr, 1.0*size2); // 1 pixel = 2 cairo
203  if (plus) // üû§
204  {
205  cairo_move_to(cr, P0.x - half, P0.y);
206  cairo_line_to(cr, P0.x + half, P0.y);
207  cairo_move_to(cr, P0.x, P0.y + half);
208  cairo_line_to(cr, P0.x, P0.y - half);
209  }
210  else // üû´
211  {
212  const double dlta = 0.707106781*half;
213  cairo_move_to(cr, P0.x - dlta, P0.y + dlta);
214  cairo_line_to(cr, P0.x + dlta, P0.y - dlta);
215  cairo_move_to(cr, P0.x + dlta, P0.y + dlta);
216  cairo_line_to(cr, P0.x - dlta, P0.y - dlta);
217  }
218 
219  cairo_stroke(cr);
220 }

222 static void
223 draw_point(cairo_t * cr, Pixel_XY P, int point_style,
224  const Color outer_color, int outer_dia,
225  const Color inner_color, int inner_dia)
226 {
227 const bool es = ! (point_style & 1); // even style
228  switch(point_style)
229  {
230  case 0: return;
231  case 1: draw_circle(cr, P, outer_color, outer_dia); break; // ‚óè
232  case 2: // ‚ñ≤
233  case 3: draw_delta( cr, P, es, outer_color, outer_dia); break; // ‚ñº
234  case 4: // ‚óÜ
235  case 5: draw_quad( cr, P, es, outer_color, outer_dia); break; // ‚ñ†
236  case 6: // üû§
237  case 7: draw_cross( cr, P, es, outer_color, outer_dia,
238  inner_dia); return; // üû´
239  default: MORE_ERROR() << "Invalid point style: "
240  << point_style; return;
241  }
242 
243  // at this point, point_style understands (though may not have) inner_dia.
244  //
245  if (inner_dia) switch(point_style)
246  {
247  case 1: draw_circle(cr, P, inner_color, inner_dia); return; // ‚óè
248  case 2: // ‚ñ≤
249  case 3: draw_delta( cr, P, es, inner_color, inner_dia); return; // ‚ñº
250  case 4: // ‚óÜ
251  case 5: draw_quad( cr, P, es, inner_color, inner_dia); return; // ‚ñ†
252  }
253 }

261 static void
262 draw_arrow(cairo_t * cr, Pixel_XY O, Pixel_XY A, const Color color)
263 {
264  // draw an arrow for an axis that starts at origin O and ends at
265  // A (where the arrow starts).
266 
267  // 1. create a unit vector U of length 1 parallel to the axis O-A
268  //
269 const int dx = A.x - O.x;
270 const int dy = A.y - O.y;
271 const double len = sqrt(dx*dx + dy*dy);
272 const double Ux = dx/len;
273 const double Uy = dy/len;
274 
275  // arrow dimensions (pixel)
276  //
277  enum { SHAFT = 20, WING = 5, TIP = 15 };
278 
279  // draw the shaft
280  //
281 const Pixel_XY S(A.x + SHAFT*Ux, A.y + SHAFT*Uy);
282  cairo_move_to(cr, A.x, A.y);
283  cairo_line_to(cr, S.x, S.y);
284  cairo_stroke(cr);
285 
286 const Pixel_XY W(S.x - WING*Ux, S.y - WING*Uy);
287 const Pixel_XY W1(W.x - WING*Uy, W.y + WING*Ux);
288 const Pixel_XY W2(W.x + WING*Uy, W.y - WING*Ux);
289 const Pixel_XY T(S.x + TIP*Ux, S.y + TIP*Uy);
290  cairo_move_to(cr, S.x, S.y);
291  cairo_line_to(cr, W1.x, W1.y);
292  cairo_line_to(cr, T.x, T.y);
293  cairo_line_to(cr, W2.x, W2.y);
294  cairo_close_path(cr);
295  cairo_fill(cr);
296 }

298 const char *
299 format_tick(double val)
300 {
301 static char cc[40];
302 char * cp = cc;
303  if (val == 0)
304  {
305  *cp++ = '0';
306  *cp++ = 0;
307  return cc;
308  }
309 
310  if (val < 0)
311  {
312  *cp++ = '-';
313  val = -val;
314  }
315 
316  // at this point,. val > 0
317  //
318  if (val < 1 && val >= 0.1) // 100-999 milli
319  {
320  snprintf(cc, sizeof(cc), "0.%03d", int(val*1000));
321  size_t cc_len = strlen(cc);
322  if (cc[--cc_len] == '0') cc[cc_len] = 0; // skip trailing 0
323  return cc;
324  }
325 
326 const char * unit = 0;
327  if (val >= 1e3)
328  {
329  if (val >= 1e15) ;
330  else if (val >= 1e12) { unit = "T"; val *= 1e-12; }
331  else if (val >= 1e9) { unit = "G"; val *= 1e-9; }
332  else if (val >= 1e6) { unit = "M"; val *= 1e-6; }
333  else { unit = "k"; val *= 1e-3; }
334  }
335  else
336  {
337 
338  if (val >= 1e0) { unit = ""; }
339  else if (val >= 1e-3) { unit = "m"; val *= 1e3; }
340  else if (val >= 1e-6) { unit = "Œº"; val *= 1e6; }
341  else if (val >= 1e-9) { unit = "n"; val *= 1e9; }
342  else if (val >= 1e-12) { unit = "p"; val *= 1e12; }
343  }
344 
345  if (unit == 0) // very large or very small
346  {
347  snprintf(cp, sizeof(cc) - 1, "%.2E", val);
348  return cc;
349  }
350 
351  // at this point: 1 ‚â§ val < 1000
352  //
353  if (val < 10) snprintf(cp, sizeof(cc) - 1, "%.2f%s", val, unit);
354  else if (val < 100) snprintf(cp, sizeof(cc) - 1, "%.1f%s", val, unit);
355  else snprintf(cp, sizeof(cc) - 1, "%.0f%s", val, unit);
356  return cc;
357 }

393 draw_legend(cairo_t * cr, const Plot_context & pctx, bool surface_plot)
394 {
395 const Plot_window_properties & w_props = pctx.w_props;
396 const int lx = w_props.get_legend_lX();
397  if (lx <= 0) return; // no legend
398 
399 Plot_line_properties const * const * l_props = w_props.get_line_properties();
400 
401 const int line_count = surface_plot ? 1 : w_props.get_line_count();
402 
403  // compute the length of the longest legend string
404  //
405  cairo_select_font_face(cr, FONT_NAME, CAIRO_FONT_SLANT_NORMAL,
406  CAIRO_FONT_WEIGHT_NORMAL);
407  cairo_set_font_size(cr, FONT_SIZE);
408 
409 double ly = FONT_SIZE;
410 double longest_len = 0.0;
411  for (int l = 0; l < line_count; ++l)
412  {
413  double lx = 0.0;
414  cairo_string_size(lx, ly, cr,
415  l_props[l]->get_legend_name().c_str(),
416  FONT_NAME, FONT_SIZE);
417  if (longest_len < lx) longest_len = lx;
418  }
419 
420 const Color canvas_color = w_props.get_canvas_color();
421 
422  /*
423  ‚îå‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îê
424  every legend looks like: ‚îÇ ---o--- legend_name ‚îÇ -- y0
425  ‚îî‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îò
426  | | | |
427  x0 x1 x2 xt
428  */
429 
430 const Pixel_XY origin = w_props.get_origin(surface_plot);
431 const int x0 = origin.x + w_props.get_legend_X();
432 const int y0 = origin.y - w_props.get_pa_height() + w_props.get_legend_Y();
433 
434 const int x1 = x0 + 30; // point o in --o--
435 const int x2 = x1 + 30; // end of --o--
436 const int xt = x2 + 10; // text after --o--
437 const int xe = xt + longest_len; // end of legend_name
438 
439 const int dy = w_props.get_legend_dY();
440 
441  // draw legend background
442  {
443  // clear the background with a 10 px border around the items
444  //
445  const double ly2 = 0.5*ly;
446 
447  enum { BORDER = 10 }; // border around legend block
448  const double X0 = x0 - BORDER;
449  const double X1 = xe + BORDER;
450  const double Y0 = y0 - ly2 - BORDER;
451  const double Y1 = y0 + ly2 + dy*(line_count - 1) + BORDER;
452 
453  cairo_set_RGB_source(cr, canvas_color);
454  cairo_rectangle(cr, X0, Y0, X1 - X0, Y1 - Y0);
455  cairo_fill_preserve(cr);
456  cairo_set_RGB_source(cr, 0x000000);
457  cairo_set_line_width(cr, 2);
458  cairo_stroke(cr);
459  }
460  for (int l = 0; l < line_count; ++l)
461  {
462  const Plot_line_properties & lp = *l_props[l];
463  const Color line_color = lp.get_line_color();
464  const int line_style = lp.get_line_style();
465  const int line_width = lp.get_line_width();
466  const int point_style = lp.get_point_style();
467  const Color point_color = lp.get_point_color();
468  const int point_size = lp.get_point_size();
469  const int point_size2 = lp.get_point_size2();
470 
471  const Pixel_Y y1 = y0 + l*dy;
472 
473  draw_line(cr, line_color, line_style, line_width,
474  Pixel_XY(x0, y1), Pixel_XY(x2, y1));
475  draw_point(cr, Pixel_XY(x1, y1), point_style, point_color,
476  point_size, canvas_color, point_size2);
477  draw_text(cr, lp.get_legend_name().c_str(), Pixel_XY(xt, y1 + 5));
478  }
479 }

487 //----------------------------------------------------------------------------
488 // draw a 3D triangle where one point P0 is at visual height H0 and the
489 // other two points P1 and P2 are the same at visual height H12.
490 void
491 draw_triangle(cairo_t * cr, const Plot_context & pctx, int verbosity,
492  Pixel_XY P0, double H0, Pixel_XY P1, Pixel_XY P2, double H12)
493 {
494  // draw a triangle with P1 and P2 at the same level H12
495  //
496  Assert(H0 >= 0.0); Assert(H0 <= 1.0);
497  Assert(H12 >= 0.0); Assert(H12 <= 1.0);
498 
499  if (verbosity & SHOW_DRAW)
500  CERR << " ‚àÜ2: P0(" << P0.x << ":" << P0.y << ") @H0=" << H0
501  << " P1(" << P1.x << ":" << P1.y << ") @H12=" << H12
502  << " P2(" << P2.x << ":" << P2.y << ") @H12=" << H12 << endl;
503 
504  // every line is ~1 pixel, so the max y should suffice for steps
505  //
506 int steps = P0.y;
507  if (steps < P1.y) steps = P1.y;
508  if (steps < P2.y) steps = P2.y;
509 const double dH = (H12 - H0) / steps;
510 
511  loop(s, steps + 1)
512  {
513  const double alpha = H0 + s*dH;
514  const double beta = (1.0 * s)/steps; // line lenght
515  const Color line_color = pctx.w_props.get_color(alpha);
516  const Pixel_XY P0_P1(P0.x + beta * (P1.x - P0.x),
517  P0.y + beta * (P1.y - P0.y));
518  const Pixel_XY P0_P2(P0.x + beta * (P2.x - P0.x),
519  P0.y + beta * (P2.y - P0.y));
520  draw_line(cr, line_color, 1, 1, P0_P1, P0_P2);
521  }
522 }
523 //----------------------------------------------------------------------------
524 // draw a 3D triangle where the points P0, P1, and P2 are at visual
525 // heights H0, H1, and H2 respectively. This is done by splitting the
526 // triangle into two triangles that each have 2 points at the same height.
527 void
528 draw_triangle(cairo_t * cr, const Plot_context & pctx, int verbosity,
529  Pixel_XY P0, double H0, Pixel_XY P1, double H1,
530  Pixel_XY P2, double H2)
531 {
532  // draw a triangle with P0, P1 and P2 at levels H0, H1, and H2
533 
534  Assert(H0 >= 0); Assert(H0 <= 1.0);
535  Assert(H1 >= 0); Assert(H1 <= 1.0);
536  Assert(H2 >= 0); Assert(H2 <= 1.0);
537 
538  if (verbosity & SHOW_DRAW)
539  CERR << "\n‚àÜ1: P0(" << P0.x << ":" << P0.y << ")@H=" << H0
540  << " P1(" << P1.x << ":" << P1.y << ")@H=" << H1
541  << " P2(" << P2.x << ":" << P2.y << ")@H=" << H2 << endl;
542 
543  if (H0 < H1) pv_swap(P0, H0, P1, H1); // then H0 >= H1
544  if (H0 < H2) pv_swap(P0, H0, P2, H2); // then H0 >= H2
545  if (H1 < H2) pv_swap(P1, H1, P2, H2); // then H1 >= H2
546 
547  // here H0 >= H1 >= H2
548  //
549  Assert(H0 >= H1);
550  Assert(H1 >= H2);
551 
552 const Plot_window_properties & w_props = pctx.w_props;
553 const vector<level_color> & color_steps = w_props.get_gradient();
554  if (color_steps.size() == 0)
555  {
556  CERR << "*** no color_steps" << endl;
557  return;
558  }
559 
560  if (H0 == H1) // P0 and P1 have the same height
561  {
562  draw_triangle(cr, pctx, verbosity, P2, H2, P0, P1, H0);
563  }
564  else if (H1 == H2) // P1 and P2 have the same height
565  {
566  draw_triangle(cr, pctx, verbosity, P0, H0, P1, P2, H1);
567  }
568  else // P2 lies below P1
569  {

576  P0.y + alpha*(P2.y - P0.y));

578  draw_triangle(cr, pctx, verbosity, P2, H2, P1, P, H1);
579  }

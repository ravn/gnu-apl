32 /*
33  A base class for dynamically allocated objects. It remembers where an object
34  was allocated and maintains a doubly linked ring of all allocated objects.
35 
36  The doubly linked list has a statically allocated anchor that must not
37  be removed. The existance of the anchor guarantees that the list is never
38  empty (which simplifies programming).
39  */
41 class DynamicObject
42 {
43 public:
45  DynamicObject(const char * loc, DynamicObject * anchor)
46  : alloc_loc(loc)
47  {
48  Log(LOG_delete) print_new(CERR, loc);
49 
50  next = anchor->next;
51  anchor->next = this;
52 
53  prev = anchor;
54  next->prev = this;
55  }
56 
58  DynamicObject(const char * loc)

60  next(this),
61  prev(this)
62  {}

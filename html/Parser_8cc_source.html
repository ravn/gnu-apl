40 //-----------------------------------------------------------------------------
41 ErrorCode
42 Parser::parse(const UCS_string & input, Token_string & tos) const
43 {
44  // convert input characters into token
45  //
46 Token_string tos1;
47 
48  {
49  Tokenizer tokenizer(pmode, LOC, macro);
50  ErrorCode ec = tokenizer.tokenize(input, tos1);
51  if (ec != E_NO_ERROR) return ec;
52  }
53 
54  // special case: single token (to speed up ⍎)
55  //
56  if (tos1.size() == 1)
57  {
58  const ErrorCode err = parse_statement(tos1);
59  if (err == E_NO_ERROR) tos.push_back(tos1[0]);
60  return err;
61  }
62 
63  return parse(tos1, tos);
64 }
65 //-----------------------------------------------------------------------------
66 ErrorCode
67 Parser::parse(const Token_string & input, Token_string & tos) const
68 {
69  Log(LOG_parse)
70  {
71  CERR << "parse 1 [" << input.size() << "]: ";
72  input.print(CERR, true);
73  CERR << endl;
74  }
75 
76  // split input line into statements (separated by ◊)
77  //
78 std::vector<Token_string *> statements;
79  {
80  Token_string * stat = new Token_string();
81  loop(idx, input.size())
82  {
83  const Token & tok = input[idx];
84  if (tok.get_tag() == TOK_DIAMOND)
85  {
86  statements.push_back(stat);
87  stat = new Token_string();
88  }
89  else
90  {
91  stat->push_back(tok);
92  }
93  }
94  statements.push_back(stat);
95  }
96 
97  loop(s, statements.size())
98  {
99  Token_string * stat = statements[s];
100  if (const ErrorCode err = parse_statement(*stat))
101  {
102  while (size_t(s) < statements.size())
103  {
104  stat = statements[s++];
105  delete stat;
106  }
107  return err;
108  }
109 
110  if (s) tos.push_back(Token(TOK_DIAMOND));
111 
112  loop(t, stat->size())
113  {
114  tos.push_back(Token());
115  tos[tos.size() - 1].move_1((*stat)[t], LOC);
116  }
117  delete stat;
118  }
119 
120  return E_NO_ERROR;
121 }
122 //-----------------------------------------------------------------------------
123 ErrorCode
124 Parser::parse_statement(Token_string & tos)
125 {
126  // 1. convert (X) into X and ((X...)) into (X...)
127  //
128  remove_nongrouping_parantheses(tos);
129  remove_void_token(tos);
130 
131  Log(LOG_parse)
132  {
133  CERR << "parse 2 [" << tos.size() << "]: ";
134  tos.print(CERR, true);
135  }
136 
137  // 2. convert groups like '(' val val...')' into single APL values
138  // A group is defined as 2 or more values enclosed in '(' and ')'.
139  // For example: (1 2 3) or (4 5 (6 7) 8 9)
140  //
141  for (int progress = true; progress;)
142  {
143  progress = collect_groups(tos);
144  if (progress) remove_void_token(tos);
145  }
146 
147  Log(LOG_parse)
148  {
149  CERR << "parse 3 [" << tos.size() << "]: ";
150  tos.print(CERR, true);
151  }
152 
153  // 3. convert vectors like 1 2 3 or '1' 2 '3' into single APL values
154  //
155  collect_constants(tos);
156  remove_void_token(tos);
157 
158  // special case: single APL value (to speed up ⍎)
159  //
160  if (tos.size() == 1)
161  {
162  Log(LOG_parse) CERR << "parse 3a: single value " << tos[0] << endl;;
163  return E_NO_ERROR;
164  }
165 
166  Log(LOG_parse)
167  {
168  CERR << "parse 4 [" << tos.size() << "]: ";
169  tos.print(CERR, true);
170  }
171 
172  // 4. mark symbol left of ← as LSYMB
173  //
174  mark_lsymb(tos);
175  Log(LOG_parse)
176  {
177  CERR << "parse 5 [" << tos.size() << "]: ";
178  tos.print(CERR, true);
179  }
180 
181  // 5. replace bitwise functons ⊤∧, ⊤∨, ⊤⍲, and ⊤⍱ by their bitwise variant
182  //
183  replace_bitwise_functions(tos);
184  remove_void_token(tos);
185  Log(LOG_parse)
186  {
187  CERR << "parse 6 [" << tos.size() << "]: ";
188  tos.print(CERR, true);
189  }
190 
191  // 6. update distances between (), [], and {}
192  //
193  {
194  const ErrorCode ec = match_par_bra(tos, false);
195  if (ec != E_NO_ERROR)
196  {
197  loop(t, tos.size()) tos[t].clear(LOC);
198  return ec;
199  }
200  }
201 
202  return E_NO_ERROR;
203 }
204 //-----------------------------------------------------------------------------
205 void
206 Parser::collect_constants(Token_string & tos)
207 {
208  Log(LOG_collect_constants)
209  {
210  CERR << "collect_constants [" << tos.size() << " token] in: ";
211  tos.print(CERR, true);
212  }
213 
214  // convert several items in vector notation into a single APL value
215  //
216  loop (t, tos.size())
217  {
218  size_t to;
219  switch(tos[t].get_tag())
220  {
221  case TOK_APL_VALUE1:
222  case TOK_APL_VALUE3:
223  case TOK_CHARACTER:
224  case TOK_COMPLEX:
225  case TOK_INTEGER:
226  case TOK_REAL:
227  break; // continue below
228 
229  default: continue; // nex token
230  }
231 
232  // at this point, t is the first item. Collect subsequenct items.
233  //
234  for (to = t + 1; to < tos.size(); ++to)
235  {
236 #if 1
237  // Note: ISO 13751 gives an example with 1 2 3[2] ↔ 2
238  //
239  // In contrast, the binding rules stated in IBM's apl2lrm.pdf
240  // would give 3[2] ↔ RANK ERROR.
241  //
242  // We follow apl2lrm; to enable IBM behavior write #if 0 above.
243  //
244 
245  // if tos[to + 1] is [ then [ binds stronger than
246  // vector notation and we stop collecting.
247  //
248  if (size_t(to + 1) < tos.size() &&
249  tos[to + 1].get_tag() == TOK_L_BRACK) break;
250 #endif
251  const TokenTag tag = tos[to].get_tag();
252 
253  if (tag == TOK_CHARACTER) continue;
254  if (tag == TOK_INTEGER) continue;
255  if (tag == TOK_REAL) continue;
256  if (tag == TOK_COMPLEX) continue;
257  if (tag == TOK_APL_VALUE1) continue;
258  if (tag == TOK_APL_VALUE3) continue;
259 
260  // no more values: stop collecting
261  //
262  break;
263  }
264 
265  create_value(tos, t, to - t);
266  }
267 
268  Log(LOG_collect_constants)
269  {
270  CERR << "collect_constants [" << tos.size() << " token] out: ";
271  tos.print(CERR, true);
272  }
273 }
274 //-----------------------------------------------------------------------------
275 bool
276 Parser::collect_groups(Token_string & tos)
277 {
278  Log(LOG_collect_constants)
279  {
280  CERR << "collect_groups [" << tos.size() << " token] in: ";
281  tos.print(CERR, true);
282  }
283 
284 int opening = -1;
285  loop(t, tos.size())
286  {
287  switch(tos[t].get_tag())
288  {
289  case TOK_CHARACTER:
290  case TOK_INTEGER:
291  case TOK_REAL:
292  case TOK_COMPLEX:
293  case TOK_APL_VALUE1:
294  case TOK_APL_VALUE3:
295  continue;
296 
297  case TOK_L_PARENT:
298  // we remember the last opening '(' so that we know where
299  // the group started when we see a ')'. This causes the
300  // deepest ( ... ) to be grouped first.
301  opening = t;
302  continue;
303 
304  case TOK_R_PARENT:
305  if (opening == -1) continue; // ')' without '('
306  if (opening == t - 1) SYNTAX_ERROR; // '(' ')'
307 
308  tos[opening].clear(LOC); // invalidate '('
309  tos[t].clear(LOC); // invalidate ')'
310  create_value(tos, opening + 1, t - opening - 1);
311 
312  // we removed parantheses, so we close the value.
313  //
314  if (tos[opening + 1].get_tag() == TOK_APL_VALUE3)
315  tos[opening + 1].ChangeTag(TOK_APL_VALUE1);
316 
317  Log(LOG_collect_constants)
318  {
319  CERR << "collect_groups [" << tos.size()
320  << " token] out: ";
321  tos.print(CERR, true);
322  }
323  return true;
324 
325  default: // nothing group'able
326  opening = -1;
327  continue;
328  }
329  }
330 
331  return false;
332 }

334 int
335 Parser::find_closing_bracket(const Token_string & tos, int pos)
336 {
337  Assert(tos[pos].get_tag() == TOK_L_BRACK);
338 
339 int others = 0;
340 
341  for (size_t p = pos + 1; p < tos.size(); ++p)
342  {
343  Log(LOG_find_closing)
344  CERR << "find_closing_bracket() sees " << tos[p] << endl;
345 
346  if (tos[p].get_tag() == TOK_R_BRACK)
347  {
348  if (others == 0) return p;
349  --others;
350  }
351  else if (tos[p].get_tag() == TOK_R_BRACK) ++others;
352  }
353 
354  SYNTAX_ERROR;
355 }

425 //-----------------------------------------------------------------------------
426 void
427 Parser::remove_nongrouping_parantheses(Token_string & tos)
428 {
429  //
430  // 1. replace ((X...)) by: (X...)
431  // 2. replace (X) by: X for a single token X,
432  //
433  for (bool progress = true; progress; progress = false)
434  {
435  loop(t, int(tos.size()) - 2)
436  {
437  if (tos[t].get_Class() != TC_L_PARENT) continue;
438 
439  const int closing = find_closing_parent(tos, t);
440 
441  // check for case 1.
442  //
443  if (tos[t + 1].get_Class() == TC_L_PARENT)
444  {
445  // tos[t] is (( ...
446  //
447  const int closing_1 = find_closing_parent(tos, t + 1);
448  if (closing == (closing_1 + 1))
449  {
450  // tos[closing_1] is )) ...
451  // We have case 1. but not for example ((...)(...))
452  // remove redundant tos[t] and tos[closing] because
453  // ((...)) are not "not separating"
454  //
455  progress = true;
456  tos[t].clear(LOC);
457  tos[closing].clear(LOC);
458  continue;
459  }
460  }
461 
462  // check for case 2.
463  //
464  if (closing != (t + 2)) continue;
465 
466 
467  // case 2. We have tos[t] = ( X ) ...
468  //
469  // (X) : "not grouping" if X is a scalar.
470  // If X is non-scalar, enclose it
471  //
472  progress = true;
473  tos[t + 2].move_1(tos[t + 1], LOC);
474 
475  // we "remember" the nongrouping parantheses to disambiguate
476  // e,g, SYM/xxx from (SYM)/xxx
477  //
478  if (tos[t + 2].get_tag() == TOK_SYMBOL)
479  tos[t + 2].ChangeTag(TOK_P_SYMB);
480  tos[t + 1].clear(LOC);
481  tos[t].clear(LOC);
482  ++t; // skip tos[t + 1]
483  }
484  }
485 }
486 //-----------------------------------------------------------------------------
487 void
488 Parser::replace_bitwise_functions(Token_string & tos)
489 {
490  // replace bitwise functions (⊤∧ ⊤∨ ⊤⍲ ⊤⍱ ⊤= or ⊤≠) by their
491  // own token
492  //
493  loop(t, int(tos.size()) - 1)
494  {
495  if (tos[t].get_tag() != TOK_F12_ENCODE) continue;
496 
497  switch(tos[t+1].get_tag())
498  {
499  case TOK_F2_AND:
500  tos[t] = Token(TOK_F2_AND_B, &Bif_F2_AND_B::_fun);
501  tos[++t].clear(LOC);
502  continue;
503 
504  case TOK_F2_OR:
505  tos[t] = Token(TOK_F2_OR_B, &Bif_F2_OR_B::_fun);
506  tos[++t].clear(LOC);
507  continue;
508 
509  case TOK_F2_NAND:
510  tos[t] = Token(TOK_F2_NAND_B,&Bif_F2_NAND_B::_fun);
511  tos[++t].clear(LOC);
512  continue;
513 
514  case TOK_F2_NOR:
515  tos[t] = Token(TOK_F2_NOR_B, &Bif_F2_NOR_B::_fun);
516  tos[++t].clear(LOC);
517  continue;
518 
519  case TOK_F2_EQUAL:
520  tos[t] = Token(TOK_F2_EQUAL_B, &Bif_F2_EQUAL_B::_fun);
521  tos[++t].clear(LOC);
522  continue;
523 
524  case TOK_F2_UNEQ:
525  tos[t] = Token(TOK_F2_UNEQ_B, &Bif_F2_UNEQ_B::_fun);
526  tos[++t].clear(LOC);
527  continue;
528 
529  default: break;
530  }
531  }
532 }
533 //-----------------------------------------------------------------------------
534 bool
535 Parser::check_if_value(const Token_string & tos, int pos)
536 {
537  // figure if tos[pos] (the token left of /. ⌿. \. or ⍀) is the end of a
538  // function (and then return false) or the end of a value (and then
539  // return true).
540  //
541  switch(tos[pos].get_Class())
542  {
543  case TC_ASSIGN: // e.g. ←/ (actually syntax error)
544  case TC_R_ARROW: // e.g. →/ (actually syntax error)
545  case TC_L_BRACK: // e.g. [/ (actually syntax error)
546  case TC_END: // e.g. / (actually syntax error)
547  case TC_L_PARENT: // e.g. (/ (actually syntax error)
548  case TC_VALUE: // e.g. 5/
549  case TC_RETURN: // e.g. / (actually syntax error)
550  case TC_OPER2: // e.g. ./
551  return true; // tos[pos] is at the end of a value
552 
553  case TC_R_BRACK: // e.g. +[1]/2 or 2/[1]2
554  {
555  const int pos1 = find_opening_bracket(tos, pos);
556  if (pos1 == 0) return true; // this is a syntax error
557  return check_if_value(tos, pos1 - 1);
558  }
559 
560  case TC_R_PARENT: // e.g. (2+3)/2 or 1 2 3 (2+3)/2
561  {
562  if (pos == 0) return true; // (actually syntax error)
563  return check_if_value(tos, pos - 1);
564  }
565 
566  case TC_SYMBOL: // e.g. A/2 or FOO/2
567  if (tos[pos].get_tag() == TOK_ALPHA) return true;
568  if (tos[pos].get_tag() == TOK_CHI) return true;
569  if (tos[pos].get_tag() == TOK_OMEGA) return true;
570  if (tos[pos].get_tag() == TOK_ALPHA_U) return false;
571  if (tos[pos].get_tag() == TOK_OMEGA_U) return false;
572  return (tos[pos].get_tag() == TOK_P_SYMB); // if value
573 
574  default: break;
575  }
576 
577  return false; // tos[pos] is at the end of a function
578 }
579 //-----------------------------------------------------------------------------
580 void
581 Parser::remove_void_token(Token_string & tos)
582 {
583 size_t dst = 0;
584 
585  loop(src, tos.size())
586  {
587  if (tos[src].get_tag() == TOK_VOID) continue;
588  if (src != ShapeItem(dst)) tos[dst].move_1(tos[src], LOC);
589  ++dst;
590  }
591 
592  tos.resize(dst);
593 }
594 //-----------------------------------------------------------------------------
595 ErrorCode
596 Parser::match_par_bra(Token_string & tos, bool backwards)
597 {
598 std::vector<ShapeItem> stack;
599  loop(s, tos.size())
600  {
601  const ShapeItem t = backwards ? (tos.size() - 1) - s : s;
602  ErrorCode ec;
603  TokenClass tc_peer;
604  switch(tos[t].get_Class())
605  {
606  // for [ ( or { push the position onto stack
607  case TC_L_BRACK: if (tos[t].get_tag() != TOK_L_BRACK) continue;
608  case TC_L_PARENT:
609  case TC_L_CURLY: stack.push_back(t);
610  continue;
611 
612  case TC_R_BRACK: ec = E_UNBALANCED_BRACKET;
613  tc_peer = TC_L_BRACK;
614  break;
615 
616  case TC_R_PARENT: ec = E_UNBALANCED_PARENT;
617  tc_peer = TC_L_PARENT;
618  break;
619 
620  case TC_R_CURLY: ec = E_UNBALANCED_CURLY;
621  tc_peer = TC_L_CURLY;
622  break;
623 
624  default: continue;
625  }
626 
627  // at this point, a closing ), ], or } was detected
628  //
629  if (stack.size() == 0) return ec;
630 
631  const ShapeItem t1 = stack.back();
632  stack.pop_back();
633 
634  if (tos[t1].get_Class() != tc_peer) return ec;
635 
636  const ShapeItem diff = (t > t1) ? t - t1 : t1 - t;
637  tos[t].set_int_val2(diff);
638  tos[t1].set_int_val2(diff);
639  }
640 
641  // if there are unmatched items: return syntax error of the outer token
642  //
643  if (stack.size())
644  {
645  const TokenClass outer = tos[stack[0]].get_Class();
646  if (outer == TC_L_BRACK) return E_UNBALANCED_BRACKET;
647  if (outer == TC_L_PARENT) return E_UNBALANCED_PARENT;
648  return E_UNBALANCED_CURLY;
649  }
650 
651  return E_NO_ERROR;
652 }

674 //-----------------------------------------------------------------------------
675 void
676 Parser::create_scalar_value(Token & output)
677 {
678  switch(output.get_tag())
679  {
680  case TOK_CHARACTER:
681  {
682  Value_P scalar(LOC);
683 
684  new (scalar->next_ravel()) CharCell(output.get_char_val());
685  scalar->check_value(LOC);
686  Token tok(TOK_APL_VALUE3, scalar);
687  output.move_1(tok, LOC);
688  }
689  return;
690 
691  case TOK_INTEGER:
692  {
693  Value_P scalar(LOC);
694 
695  new (scalar->next_ravel()) IntCell(output.get_int_val());
696  scalar->check_value(LOC);
697  Token tok(TOK_APL_VALUE3, scalar);
698  output.move_1(tok, LOC);
699  }
700  return;
701 
702  case TOK_REAL:
703  {
704  Value_P scalar(LOC);
705 
706  new (scalar->next_ravel()) FloatCell(output.get_flt_val());
707  scalar->check_value(LOC);
708  Token tok(TOK_APL_VALUE3, scalar);
709  output.move_1(tok, LOC);
710  }
711  return;
712 
713  case TOK_COMPLEX:
714  {
715  Value_P scalar(LOC);
716 
717  new (scalar->next_ravel()) ComplexCell(output.get_cpx_real(),
718  output.get_cpx_imag());

792 //-----------------------------------------------------------------------------
793 void
794 Parser::mark_lsymb(Token_string & tos)
795 {
796  loop(ass, tos.size())
797  {
798  if (tos[ass].get_Class() != TC_ASSIGN) continue;
799 
800  // found ←. move backwards. Before that we handle the special case of
801  // vector specification, i.e. (SYM SYM ... SYM) ← value
802  //
803  if (ass >= 3 && tos[ass - 1].get_Class() == TC_R_PARENT &&
804  tos[ass - 2].get_Class() == TC_SYMBOL &&
805  tos[ass - 3].get_Class() == TC_SYMBOL)
806  {
807  // first make sure that this is really a vector specification.
808  //
809  const int syms_to = ass - 2;
810  int syms_from = ass - 3;
811  bool is_vector_spec = true;
812  for (int a1 = ass - 4; a1 >= 0; --a1)
813  {
814  if (tos[a1].get_Class() == TC_L_PARENT)
815  {
816  syms_from = a1 + 1;
817  break;
818  }
819  if (tos[a1].get_Class() == TC_SYMBOL) continue;
820  is_vector_spec = false;
821  break;
822  }
823 
824  // if this is a vector specification, then mark all symbols
825  // inside ( ... ) as TOK_LSYMB2.
826  //
827  if (is_vector_spec)
828  {
829  for (int a1 = syms_from; a1 <= syms_to; ++a1)
830  tos[a1].ChangeTag(TOK_LSYMB2);
831  }
832 
833  continue;
834  }
835 
836  int bracks = 0;
837  for (int prev = ass - 1; prev >= 0; --prev)
838  {
839  switch(tos[prev].get_Class())
840  {
841  case TC_R_BRACK: ++bracks; break;
842  case TC_L_BRACK: --bracks; break;
843  case TC_SYMBOL: if (bracks) break; // inside [ ... ]
844  if (tos[prev].get_tag() == TOK_SYMBOL)
845  {
846  tos[prev].ChangeTag(TOK_LSYMB);
847  }
848  // it could be that a system variable
849  // is assigned, e.g. ⎕SVE←0. In that case
850  // (actually always) we simply stop here.
851  //
852 
853  /* fall-through */
854 
855  case TC_END: prev = 0; // stop prev loop
856  break;
857 
858  default: break;
859  }
860  }
861  }
862 }

74 //-----------------------------------------------------------------------------
75 void
76 Nabla::edit()
77 {
78 const char * error = start();
79  if (error)
80  {
81  Log(LOG_verbose_error) if (Workspace::more_error().size() == 0)
82  {
83  UERR << "Bad â-open '" << first_command
84  << "' : '" << error << "'" << endl;
85  }
86  throw_edit_error(error);
87  }
88 
89  // editor loop
90  //
91 int control_D_count = 0;
92  Log(LOG_nabla) UERR << "Nabla(" << fun_header << ")..." << endl;
93  while (!do_close)
94  {
95  const UCS_string prompt = current_line.print_prompt(0);
96  bool eof = false;
97  UCS_string line;
98  if (uprefs.raw_cin)
99  {
100  LineHistory lh(10);
101  InputMux::get_line(LIM_Nabla, prompt, line, eof, lh);
102  }
103  else
104  {
105  LineHistory lh(*this);
106  InputMux::get_line(LIM_Nabla, prompt, line, eof, lh);
107  }
108 
109  if (eof) // end-of-input (^D) pressed
110  {
111  ++control_D_count;
112  if (control_D_count < 5)
113  {
114  COUT << "^D" << endl;
115  continue;
116  }
117  COUT << endl << " *** end of input" << endl;
118  Command::cmd_OFF(5);
119  }
120 
121  if (const char * loc = parse_oper(line, false))
122  {
123  UERR << "??? " << loc << endl;
124  continue;
125  }
126 
127  if (const char * loc = execute_oper())
128  {
129  UERR << "â-command failed: " << loc << endl;
130  Output::set_color_mode(Output::COLM_INPUT);
131  }
132  }
133 
134  Log(LOG_nabla)
135  {
136  UERR << "done: '" << fun_header << "'" << endl;
137  loop(l, lines.size()) UERR << lines[l].text << endl;
138  }
139 
140 UCS_string fun_text;
141  loop(l, lines.size())
142  {
143  fun_text.append(lines[l].text);
144  fun_text.append(UNI_ASCII_LF);
145  }
146 
147  // maybe copy function into the history
148  //
149  if (!uprefs.raw_cin && !InputFile::running_script() &&
150  ((uprefs.nabla_to_history == /* always */ 2) ||
151  ((uprefs.nabla_to_history == /* if changed */ 1) && modified)))
152  {
153  // create a history entry that can be re-entered and replace
154  // the last history line (which contained some âfoo ...)
155  //
156  {
157  UCS_string line_0(" ");
158  line_0.append(UNI_NABLA);
159  line_0.append(lines[0].text);
160  line_0.remove_trailing_whitespaces();
161  LineInput::replace_history_line(line_0);
162  }
163 
164  for (size_t l = 1; l < lines.size(); ++l)
165  {
166  UCS_string line_l("[");
167  line_l.append_number(l);
168  line_l.append_UTF8("] ");
169  while (line_l.size() < 6) line_l.append(UNI_ASCII_SPACE);
170  line_l.append(lines[l].text);
171  line_l.remove_trailing_whitespaces();
172 
173  LineInput::add_history_line(line_l);
174  }
175 
176  UCS_string line_N(" ");
177  line_N.append(UNI_NABLA);
178  LineInput::add_history_line(line_N);
179  }
180 
181 int error_line = 0;
182 UCS_string creator(InputFile::current_filename());
183  creator.append(UNI_ASCII_COLON);
184  creator.append_number(defn_line_no);
185 UTF8_string creator_utf8(creator);
186 
187 UserFunction * ufun = UserFunction::fix(fun_text, error_line, false,
188  LOC, creator_utf8, false);
189 
190  if (ufun == 0) throw_edit_error(LOC);
191 
192  if (locked)
193  {
194  const int exec_properties[4] = { 1, 1, 1, 1 };
195  ufun->set_exec_properties(exec_properties);
196  }
197 
198  // set stop and trace vectors
199  //
200 std::vector<Function_Line> stop_vec;
201 std::vector<Function_Line> trace_vec;
202  loop(l, lines.size())
203  {
204  if (lines[l].stop_flag) stop_vec.push_back(Function_Line(l));
205  if (lines[l].trace_flag) trace_vec.push_back(Function_Line(l));
206  }
207 
208  ufun->set_trace_stop(stop_vec, true);
209  ufun->set_trace_stop(trace_vec, false);
210 }

212 const char *
213 Nabla::start()
214 {
215  // cmd should be something like:
216  //
217  // âFUN
218  // âFUN[â]
219  // âFUN[â]â
220  // etc.
221  //
222 UCS_string::iterator c(first_command.begin());
223 
224  // skip leading spaces
225  //
226  while (c.more() && Avec::is_white(c.get())) c.next();
227 
228  // skip leading nabla.
229  //
230  if (c.next() != UNI_NABLA) return "Bad â-command (no â)";
231 
232  // skip leading spaces
233  //
234  while (c.more() && Avec::is_white(c.get())) c.next();
235 
236  // function header.
237  //
238  while (c.more() && c.get() != UNI_ASCII_L_BRACK)
239  fun_header.append(c.next());
240 
241  // at this point there could be an axis specification [X] that
242  // could be confused with an operation like [â]. We take the first char
243  // after the [ to decide if there is an axis or an operation.
244  //
245  if (c.get() == UNI_ASCII_L_BRACK) // [
246  {
247  for (int off = 1; ; ++off)
248  {
249  if (Avec::is_white(c.get(off))) continue;
250  if (Avec::is_first_symbol_char(c.get(off))) // axis
251  {
252  fun_header.append(c.next()); // copy the [
253  while (c.more() && c.get() != UNI_ASCII_L_BRACK)
254  fun_header.append(c.next());
255  }
256  break;
257  }
258  }
259 
260 UserFunction_header hdr(fun_header, false);
261  if (hdr.get_error()) return "Bad function header";
262 
263  fun_symbol = Workspace::lookup_symbol(hdr.get_name());
264  Assert(fun_symbol);
265 
266  if (fun_header.size() == 0) return "no function name";
267 
268  // optional operation
269  //
270  if (c.get() == UNI_ASCII_L_BRACK)
271  {
272  UCS_string oper;
273  Unicode cc;
274  do
275  {
276  if (!c.more()) return "no ] in â-command";
277  oper.append(cc = c.next());
278  } while (cc != UNI_ASCII_R_BRACK);
279 
280  const char * loc = parse_oper(oper, true);
281  if (loc) return loc; // error
282  }
283 
284  switch(fun_symbol->get_nc())
285  {
286  case NC_UNUSED_USER_NAME: // open a new function
287  function_existed = false;
288  modified = true;
289  {
290  // a new function must not have a command
291  //
292  if (ecmd != ECMD_NOP) return "â-command in new function";
293 
294  const char * open_loc = open_new_function();
295  if (open_loc) return open_loc;
296  }
297  break;
298 
299  case NC_FUNCTION:
300  case NC_OPERATOR: // open an existing function
301  function_existed = true;
302  if (InputFile::running_script()) // script
303  {
304  const char * open_loc = open_new_function();
305  if (open_loc) return open_loc;
306  break; // continue below
307  }
308 
309  // interactive
310  {
311  if (hdr.has_vars())
312  {
313  // an existing function was opened with a header that
314  // contains more than the function name.
315  //
316  return "attempt to â-open existing function with "
317  "new function header";
318  }
319 
320  const char * open_loc = open_existing_function();
321  if (open_loc) return open_loc;
322  }
323  break;
324 
325  default:
326  return "attempt to â-open a variable at " LOC;
327  }
328 
329  // at this point the (new or existing) function was successfully opened.
330  // That means that at least the header is present (lines.size() > 0)
331 
332  // immediate close (only show command is allowed here),
333  // e.g. âfun[â]â
334  //
335  if (c.get() == UNI_NABLA)
336  {
337  if (ecmd != ECMD_SHOW) return "illegal command between â ... â";
338  if (const char * loc = execute_oper())
339  UERR << "execute_oper() failed at " << loc << endl;
340  do_close = true;
341  return 0;
342  }
343 
344  if (const char * loc = execute_oper())
345  UERR << "execute_oper() failed at " << loc << endl;
346 
347  return 0; // no error
348 }
349 //-----------------------------------------------------------------------------
350 const char *
351 Nabla::parse_oper(UCS_string & oper, bool initial)
352 {
353  Log(LOG_nabla)
354  UERR << "parsing oper '" << oper << "'" << endl;
355 
356  // skip trailing spaces
357  //
358  while (oper.size() > 0 && oper.back() <= ' ') oper.pop_back();
359  if (oper.size() > 0 && oper.back() == UNI_NABLA)
360  {
361  do_close = true;
362  oper.pop_back();
363  while (oper.size() > 0 && oper.back() <= ' ') oper.pop_back();
364  }
365  else if (oper.size() > 0 && oper.back() == UNI_DEL_TILDE)
366  {
367  do_close = true;
368  locked = true;
369  oper.pop_back();
370  while (oper.size() > 0 && oper.back() <= ' ') oper.pop_back();
371  }
372 
373  current_text.clear();
374  ecmd = ECMD_NOP;
375 
376  if (oper.size() == 0 && do_close) return 0;
377 
378 UCS_string::iterator c(oper.begin());
379 Unicode cc = c.next();
380 UCS_string text = oper;
381  while (cc == ' ') cc = c.next(); // skip leading whitespace
382 
383  // we expect one of the following:
384  //
385  // [â] [nâ] [âm] [nâm] [ân-m] (show)
386  // [nâ] [âm] [nâm] [ân-m] [ân1 n2 ...] (delete)
387  // [â] (escape)
388  // [n] (goto)
389  // text (override text)
390 
391  if (cc != UNI_ASCII_L_BRACK) // override text
392  {
393  ecmd = ECMD_EDIT;
394  edit_from = current_line;
395  current_text = text;
396 // for (; cc != Invalid_Unicode; cc = c.next()) current_text.append(cc);
397  return 0;
398  }
399 
400  // a loop over multiple commands, like
401  // [2â4] [â5]
402  //
403  // only the last command is executed; the previous commands are discarded/
404  //
405 command_loop:
406 
407  // at this point, [ was seen and skipped
408 
409  ecmd = ECMD_NOP;
410  edit_from.clear();
411  edit_to.clear();
412  got_minus = false;
413 
414  // optional edit_from
415  //
416  if (Avec::is_digit(c.get()) || c.get() == UNI_ASCII_FULLSTOP)
417  edit_from = parse_lineno(c);
418 
419  // operation, which is one of:
420  //
421  // [â show
422  // [] edit
423  // [â delete
424  // [â abandon
425  switch (c.get())
426  {
427  case UNI_Quad_Quad:
428  case UNI_Quad_Quad1: ecmd = ECMD_SHOW; c.next(); break;
429  case UNI_ASCII_R_BRACK: ecmd = ECMD_EDIT; break;
430  case UNI_DELTA: ecmd = ECMD_DELETE; c.next(); break;
431  case UNI_RIGHT_ARROW: ecmd = ECMD_ESCAPE; c.next(); break;
432  case Invalid_Unicode: return "Bad â-command";
433 
434  default: UERR << "Bad edit op '" << c.get() << "'" << endl;
435  return "Bad â-command";
436  }
437 
438  // don't allow delete or escape in the first command
439  if (initial)
440  {
441  if (ecmd == ECMD_DELETE) return "Bad initial â-command â";
442  if (ecmd == ECMD_ESCAPE) return "Bad initial â-command â";
443  }
444 
445 again:
446  // optional edit_to
447  //
448  if (Avec::is_digit(c.get())) edit_to = parse_lineno(c);
449 
450  if (c.get() == UNI_ASCII_MINUS) // range
451  {
452  if (got_minus) return "error: second - in â-range";
453  got_minus = true;
454  edit_from = edit_to;
455  c.next(); // eat the -
456  goto again;
457  }
458 
459  if (c.next() != UNI_ASCII_R_BRACK) return "missing ] in â-range";
460 
461  // at this point we have parsed an editor command, like:
462  //
463  // [from â to]
464  // [from â to]
465  // [from]
466 
467  while (Avec::is_white(c.get())) c.next();
468 
469  if (c.get() == UNI_ASCII_L_BRACK) // another command: ignore previous
470  {
471  c.next(); // eat the [
472  goto command_loop;
473  }
474 
475  // copy the rest to current_text. Set do_close if â or â« is seen
476  // unless inside strings.
477  //
478  for (;;)
479  {
480  switch(cc = c.next())
481  {
482  case Invalid_Unicode: // regular end of input
483  return 0;
484 
485  case UNI_NABLA: // â
486  do_close = true;
487  return 0;
488 
489  case UNI_DEL_TILDE: // â«
490  locked = true;
491  do_close = true;
492  return 0;
493 
494  case UNI_ASCII_DOUBLE_QUOTE: // "
495  current_text.append(cc);
496  for (;;)
497  {
498  cc = c.next();
499  if (cc == Invalid_Unicode) // premature end of input
500  {
501  current_text.append(UNI_ASCII_DOUBLE_QUOTE);
502  return 0;
503  }
504 
505  current_text.append(cc);
506  if (cc == UNI_ASCII_DOUBLE_QUOTE) break; // string end
507  if (cc == UNI_ASCII_BACKSLASH) // \x
508  {
509  cc = c.next();
510  if (cc == Invalid_Unicode) // premature input end
511  {
512  current_text.append(UNI_ASCII_BACKSLASH);
513  current_text.append(UNI_ASCII_DOUBLE_QUOTE);
514  return 0;
515  }
516  current_text.append(cc);
517  }
518  }
519  break;
520 
521  case UNI_SINGLE_QUOTE: // '
522  current_text.append(cc);
523  for (;;)
524  {
525  // no need to care for ''. Since we only copy, we can
526  // handle ' ... '' ... ' like two adjacent strings
527  // instead of a string containing a (doubled) quote.
528  //
529  cc = c.next();
530  if (cc == Invalid_Unicode) // premature end of input
531  {
532  current_text.append(UNI_SINGLE_QUOTE);
533  return 0;
534  }
535 
536  current_text.append(cc);
537  if (cc == UNI_SINGLE_QUOTE) break; // string end
538  }
539  break;
540 
541  default:
542  current_text.append(cc);
543  }
544  }
545 
546  return 0;
547 }
548 //-----------------------------------------------------------------------------
549 LineLabel
550 Nabla::parse_lineno(UCS_string::iterator & c)
551 {
552 LineLabel ret(0);
553 
554  while (Avec::is_digit(c.get()))
555  {
556  ret.ln_major *= 10;
557  ret.ln_major += c.next() - UNI_ASCII_0;
558  }
559 
560  if (c.get() == UNI_ASCII_FULLSTOP)
561  {
562  c.next(); // eat the .
563  while (Avec::is_digit(c.get())) ret.ln_minor.append(c.next());
564  }
565 
566  return ret;
567 }

579 //-----------------------------------------------------------------------------
580 const char *
581 Nabla::open_existing_function()
582 {
583  Log(LOG_nabla)
584  UERR << "opening existing function '" << fun_symbol->get_name()
585  << "'" << endl;
586 
587  if (const char * why = fun_symbol->cant_be_defined()) return why;
588 
589  // this function must only be called when editing functions interactively
590  //
591  if (InputFile::running_script())
592  return "â-edit existing function from a script";
593 
594 Function * function = fun_symbol->get_function();
595  Assert(function);
596 
597  if (function->get_exec_properties()[0])
598  return "function is locked";
599 
600  if (function->is_native())
601  return "function is native";
602 
603  if (function->is_lambda())
604  return "function is a lambda";
605 
606  if (Workspace::is_called(fun_symbol->get_name()))
607  return "function is used, pendent or suspended";
608 
609 const UserFunction * ufun = function->get_ufun1();
610  if (ufun == 0)
611  return "function is not editable at " LOC;
612 
613 const UCS_string ftxt = function->canonical(false);
614  Log(LOG_nabla) UERR << "existing function is:\n" << ftxt << endl;
615 
616 UCS_string_vector tlines;
617  ftxt.to_vector(tlines);
618 
619  Assert(tlines.size());
620  fun_header = tlines[0];
621  loop(t, tlines.size())
622  {
623  FunLine fl(t, tlines[t]);
624 
625  // set stop and trace flags
626  //
627  loop(st, ufun->get_stop_lines().size())
628  {
629  if (t == ufun->get_stop_lines()[st]) // stop set
630  {
631  fl.stop_flag = true;
632  break;
633  }
634  }
635 
636  loop(tr, ufun->get_trace_lines().size())
637  {
638  if (t == ufun->get_trace_lines()[tr]) // trace set
639  {
640  fl.trace_flag = true;
641  break;
642  }
643  }
644 
645  lines.push_back(fl);
646  }
647 
648  current_line = LineLabel(tlines.size());
649 
650  return 0;
651 }

654 Nabla::execute_oper()
655 {
656  if (ecmd == ECMD_NOP)
657  {
658  Log(LOG_nabla)
659  UERR << "Nabla::execute_oper(NOP)" << endl;
660  return 0;
661  }
662 
663 const bool have_from = edit_from.ln_major != -1;
664 const bool have_to = edit_to.ln_major != -1;
665 
666  if (lines.size())
667  {
668  if (!have_from && ecmd != ECMD_DELETE) edit_from = lines[0].label;
669  if (!have_to) edit_to = lines[lines.size() - 1].label;
670  }
671 
672  if (ecmd == ECMD_SHOW) return execute_show();
673  if (ecmd == ECMD_DELETE) return execute_delete();
674  if (ecmd == ECMD_EDIT) return execute_edit();
675  if (ecmd == ECMD_ESCAPE) return execute_escape();
676 
677  UERR << "edit command " << ecmd
678  << " from " << edit_from << " to " << edit_to << endl;
679  FIXME;
680 
681  return LOC;
682 }
683 //-----------------------------------------------------------------------------
684 const char *
685 Nabla::execute_show()
686 {
687  Log(LOG_nabla)
688  UERR << "Nabla::execute_oper(SHOW) from " << edit_from
689  << " to " << edit_to << " line-count " << lines.size() << endl;
690 
691 int idx_from = find_line(edit_from);
692 int idx_to = find_line(edit_to);
693 
694 const LineLabel user_edit_to = edit_to;
695 
696  if (idx_from == -1) edit_from = lines[idx_from = 0].label;
697  if (idx_to == -1) edit_to = lines[idx_to = lines.size() - 1].label;
698 
699  Log(LOG_nabla)
700  UERR << "Nabla::execute_oper(SHOW) from "
701  << edit_from << " to " << edit_to << endl;
702 
703  if (idx_from == 0) // then print header line
704  COUT << " â" << endl;
705  for (int e = idx_from; e <= idx_to; ++e) lines[e].print(COUT);
706  if (idx_to == int(lines.size() - 1)) // then print last line
707  COUT << " â" << endl;
708 
709  if (user_edit_to.valid()) // eg. [â42] or [2â42]
710  {
711  current_line = user_edit_to;
712  if (line_exists(current_line)) current_line.next();
713  if (line_exists(current_line))
714  {
715  // user_edit_to and its next line exist. Increase minor length
716  // That is, if both [8] and [9] exist then use [8.1]
717  //
718  current_line = user_edit_to;
719  current_line.insert();
720  }
721  }
722  else // eg. [42â] or [â]
723  {
724  current_line = lines.back().label;
725  current_line.next();
726  }
727 
728  Log(LOG_nabla)
729  UERR << "Nabla::execute_oper(SHOW) done with current_line '"
730  << current_line << "'" << endl;
731 
732  return 0;

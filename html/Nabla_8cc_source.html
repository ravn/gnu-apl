60 Nabla::throw_edit_error(const char * why)
61 {

63  << " " << first_command << endl
64  << " " << UCS_string(first_command.size() - 1, UNI_SPACE)
65  << "^" << endl;
66 
67  if (Workspace::more_error().size() == 0)

74 //-----------------------------------------------------------------------------
75 void
76 Nabla::edit()
77 {
78 const char * error = start();
79  if (error)
80  {
81  Log(LOG_verbose_error) if (Workspace::more_error().size() == 0)
82  {
83  UERR << "Bad ∇-open '" << first_command
84  << "' : '" << error << "'" << endl;
85  }
86  throw_edit_error(error);
87  }
88 
89  // editor loop
90  //
91 int control_D_count = 0;
92  Log(LOG_nabla) UERR << "Nabla(" << fun_header << ")..." << endl;
93 try_again:
94  while (!do_close)
95  {
96  const UCS_string prompt = current_line.print_prompt(0);
97  bool eof = false;
98  UCS_string line;
99  if (uprefs.raw_cin)
100  {
101  LineHistory lh(10);
102  InputMux::get_line(LIM_Nabla, prompt, line, eof, lh);
103  }
104  else
105  {
106  LineHistory lh(*this);
107  InputMux::get_line(LIM_Nabla, prompt, line, eof, lh);
108  }
109 
110  if (eof) // end-of-input (^D) pressed
111  {
112  ++control_D_count;
113  if (control_D_count < 5)
114  {
115  COUT << "^D" << endl;
116  continue;
117  }
118  COUT << endl << " *** end of input" << endl;
119  Command::cmd_OFF(5);
120  }
121 
122  if (const char * loc = parse_oper(line, false))
123  {
124  UERR << "??? " << loc << endl;
125  continue;
126  }
127 
128  if (const char * loc = execute_oper())
129  {
130  UERR << "∇-command failed: " << loc << endl;
131  Output::set_color_mode(Output::COLM_INPUT);
132  }
133  }
134 
135  Log(LOG_nabla)
136  {
137  UERR << "done: '" << fun_header << "'" << endl;
138  loop(l, lines.size()) UERR << lines[l].text << endl;
139  }
140 
141 UCS_string fun_text;
142  loop(l, lines.size())
143  {
144  fun_text.append(lines[l].text);
145  fun_text.append(UNI_LF);
146  }
147 
148  // maybe copy function into the history
149  //
150  if (!uprefs.raw_cin && !InputFile::running_script() &&
151  ((uprefs.nabla_to_history == /* always */ 2) ||
152  ((uprefs.nabla_to_history == /* if changed */ 1) && modified)))
153  {
154  // create a history entry that can be re-entered and replace
155  // the last history line (which contained some ∇foo ...)
156  //
157  {
158  UCS_string line_0(" ");
159  line_0.append(UNI_NABLA);
160  line_0.append(lines[0].text);
161  line_0.remove_trailing_whitespaces();
162  LineInput::replace_history_line(line_0);
163  }
164 
165  for (size_t l = 1; l < lines.size(); ++l)
166  {
167  UCS_string line_l("[");
168  line_l.append_number(l);
169  line_l.append_UTF8("] ");
170  while (line_l.size() < 6) line_l.append(UNI_SPACE);
171  line_l.append(lines[l].text);
172  line_l.remove_trailing_whitespaces();
173 
174  LineInput::add_history_line(line_l);
175  }
176 
177  UCS_string line_N(" ");
178  line_N.append(UNI_NABLA);
179  LineInput::add_history_line(line_N);
180  }
181 
182 int error_line = 0;
183 UCS_string creator(InputFile::current_filename());
184  creator.append(UNI_COLON);
185  creator.append_number(defn_line_no);
186 UTF8_string creator_utf8(creator);
187 
188 UserFunction * ufun = UserFunction::fix(fun_text, error_line, false,
189  LOC, creator_utf8, false);
190 
191  if (ufun == 0) // UserFunction::fix() failed
192  {
193  if (InputFile::running_script())
194  {
195  // the ∇-editor runs from a script, therefore warning the user
196  // interactively and asking to fix the fault makles no sense. We
197  // therefore exit with DEFN_ERROR, so that the scrip does not hang
198  // in a endless try again loop.
199  //
200  UTF8_string more_utf8(MORE_ERROR());
201  throw_edit_error(more_utf8.c_str());
202  }
203 
204  COUT << "Fatal error in defined function. Please correct "
205  "(or else delete with [∆" << error_line <<"])\n"
206  " the offending line indicated above.\n";
207  do_close = false;
208  COUT << MORE_ERROR();
209  goto try_again;
210 
211  }
212 
213  if (locked)
214  {
215  const int exec_properties[4] = { 1, 1, 1, 1 };
216  ufun->set_exec_properties(exec_properties);
217  }
218 
219  // set stop and trace vectors
220  //
221 std::vector<Function_Line> stop_vec;
222 std::vector<Function_Line> trace_vec;
223  loop(l, lines.size())
224  {
225  if (lines[l].stop_flag) stop_vec.push_back(Function_Line(l));
226  if (lines[l].trace_flag) trace_vec.push_back(Function_Line(l));
227  }
228 
229  ufun->set_trace_stop(stop_vec, true);
230  ufun->set_trace_stop(trace_vec, false);
231 }
232 //-----------------------------------------------------------------------------
233 const char *
234 Nabla::start()
235 {
236  // cmd should be something like:
237  //
238  // ∇FUN
239  // ∇FUN[⎕]
240  // ∇FUN[⎕]∇
241  // etc.
242  //
243 UCS_string::iterator c(first_command.begin());
244 
245  // skip leading spaces
246  //
247  while (c.more() && Avec::is_white(c.get())) c.next();
248 
249  // skip leading nabla.
250  //
251  if (c.next() != UNI_NABLA) return "Bad ∇-command (no ∇)";
252 
253  // skip leading spaces
254  //
255  while (c.more() && Avec::is_white(c.get())) c.next();
256 
257  // function header.
258  //
259  while (c.more() && c.get() != UNI_L_BRACK)
260  fun_header.append(c.next());
261 
262  // at this point there could be an axis specification [X] that
263  // could be confused with an operation like [⎕]. We take the first char
264  // after the [ to decide if there is an axis or an operation.
265  //
266  if (c.get() == UNI_L_BRACK) // [
267  {
268  for (int off = 1; ; ++off)
269  {
270  if (Avec::is_white(c.get(off))) continue;
271  if (Avec::is_first_symbol_char(c.get(off))) // axis
272  {
273  fun_header.append(c.next()); // copy the [
274  while (c.more() && c.get() != UNI_L_BRACK)
275  fun_header.append(c.next());
276  }
277  break;
278  }
279  }
280 
281 UserFunction_header hdr(fun_header, false);
282  if (hdr.get_error())
283  {
284  static char cc[200];
285  snprintf(cc, sizeof(cc), "Bad function header (%s)",
286  hdr.get_error_info());
287  cc[sizeof(cc) - 1] = 0;
288  return cc;
289  }
290 
291  fun_symbol = Workspace::lookup_symbol(hdr.get_name());
292  Assert(fun_symbol);
293 
294  if (fun_header.size() == 0) return "no function name";
295 
296  // optional operation
297  //
298  if (c.get() == UNI_L_BRACK)
299  {
300  UCS_string oper;
301  Unicode cc;
302  do
303  {
304  if (!c.more()) return "no ] in ∇-command";
305  oper.append(cc = c.next());
306  } while (cc != UNI_R_BRACK);
307 
308  const char * loc = parse_oper(oper, true);
309  if (loc) return loc; // error
310  }
311 
312  switch(fun_symbol->get_nc())
313  {
314  case NC_UNUSED_USER_NAME: // open a new function
315  function_existed = false;
316  modified = true;
317  {
318  // a new function must not have a command
319  //
320  if (ecmd != ECMD_NOP) return "∇-command in new function";
321 
322  const char * open_loc = open_new_function();
323  if (open_loc) return open_loc;
324  }
325  break;
326 
327  case NC_FUNCTION:
328  case NC_OPERATOR: // open an existing function
329  function_existed = true;
330  if (InputFile::running_script()) // script
331  {
332  const char * open_loc = open_new_function();
333  if (open_loc) return open_loc;
334  break; // continue below
335  }
336 
337  // interactive
338  {
339  if (hdr.has_vars())
340  {
341  // an existing function was opened with a header that
342  // contains more than the function name.
343  //
344  return "attempt to ∇-open existing function with "
345  "new function header";
346  }
347 
348  const char * open_loc = open_existing_function();
349  if (open_loc) return open_loc;
350  }
351  break;
352 
353  default:
354  return "attempt to ∇-open a variable at " LOC;
355  }
356 
357  // at this point the (new or existing) function was successfully opened.
358  // That means that at least the header is present (lines.size() > 0)
359 
360  // immediate close (only show command is allowed here),
361  // e.g. ∇fun[⎕]∇
362  //
363  if (c.get() == UNI_NABLA)
364  {
365  if (ecmd != ECMD_SHOW) return "illegal command between ∇ ... ∇";
366  if (const char * loc = execute_oper())
367  UERR << "execute_oper() failed at " << loc << endl;
368  do_close = true;
369  return 0;
370  }
371 
372  if (const char * loc = execute_oper())
373  UERR << "execute_oper() failed at " << loc << endl;
374 
375  return 0; // no error
376 }
377 //-----------------------------------------------------------------------------
378 const char *
379 Nabla::parse_oper(UCS_string & oper, bool initial)
380 {
381  Log(LOG_nabla)
382  UERR << "parsing oper '" << oper << "'" << endl;
383 
384  // skip trailing spaces
385  //
386  while (oper.size() > 0 && oper.back() <= ' ') oper.pop_back();
387  if (oper.size() > 0 && oper.back() == UNI_NABLA)
388  {
389  do_close = true;
390  oper.pop_back();
391  while (oper.size() > 0 && oper.back() <= ' ') oper.pop_back();
392  }
393  else if (oper.size() > 0 && oper.back() == UNI_DEL_TILDE)
394  {
395  do_close = true;
396  locked = true;
397  oper.pop_back();
398  while (oper.size() > 0 && oper.back() <= ' ') oper.pop_back();
399  }
400 
401  current_text.clear();
402  ecmd = ECMD_NOP;
403 
404  if (oper.size() == 0 && do_close) return 0;
405 
406 UCS_string::iterator c(oper.begin());
407 Unicode cc = c.next();
408 UCS_string text = oper;
409  while (cc == ' ') cc = c.next(); // skip leading whitespace
410 
411  // we expect one of the following:
412  //
413  // [⎕] [n⎕] [⎕m] [n⎕m] [⎕n-m] (show)
414  // [n∆] [∆m] [n∆m] [∆n-m] [∆n1 n2 ...] (delete)
415  // [→] (escape)
416  // [n] (goto)
417  // text (override text)
418 
419  if (cc != UNI_L_BRACK) // override text
420  {
421  ecmd = ECMD_EDIT;
422  edit_from = current_line;
423  current_text = text;
424 // for (; cc != Invalid_Unicode; cc = c.next()) current_text.append(cc);
425  return 0;
426  }
427 
428  // a loop over multiple commands, like
429  // [2⎕4] [∆5]
430  //
431  // only the last command is executed; the previous commands are discarded/
432  //
433 command_loop:
434 
435  // at this point, [ was seen and skipped
436 
437  ecmd = ECMD_NOP;
438  edit_from.clear();
439  edit_to.clear();
440  got_minus = false;
441 
442  // optional edit_from
443  //
444  if (Avec::is_digit(c.get()) || c.get() == UNI_FULLSTOP)
445  edit_from = parse_lineno(c);
446 
447  // operation, which is one of:
448  //
449  // [⎕ show
450  // [] edit
451  // [∆ delete
452  // [→ abandon
453  switch (c.get())
454  {
455  case UNI_Quad_Quad:
456  case UNI_Quad_Quad1: ecmd = ECMD_SHOW; c.next(); break;
457  case UNI_R_BRACK: ecmd = ECMD_EDIT; break;
458  case UNI_DELTA: ecmd = ECMD_DELETE; c.next(); break;
459  case UNI_RIGHT_ARROW: ecmd = ECMD_ESCAPE; c.next(); break;
460  case Invalid_Unicode: return "Bad ∇-command";
461 
462  default: UERR << "Bad edit op '" << c.get() << "'" << endl;
463  return "Bad ∇-command";
464  }
465 
466  // don't allow delete or escape in the first command
467  if (initial)
468  {
469  if (ecmd == ECMD_DELETE) return "Bad initial ∇-command ∆";
470  if (ecmd == ECMD_ESCAPE) return "Bad initial ∇-command →";
471  }
472 
473 again:
474  // optional edit_to
475  //
476  if (Avec::is_digit(c.get())) edit_to = parse_lineno(c);
477 
478  if (c.get() == UNI_MINUS) // range
479  {
480  if (got_minus) return "error: second - in ∇-range";
481  got_minus = true;
482  edit_from = edit_to;
483  c.next(); // eat the -
484  goto again;
485  }
486 
487  if (c.next() != UNI_R_BRACK) return "missing ] in ∇-range";
488 
489  // at this point we have parsed an editor command, like:
490  //
491  // [from ⎕ to]
492  // [from ∆ to]
493  // [from]
494 
495  while (Avec::is_white(c.get())) c.next();
496 
497  if (c.get() == UNI_L_BRACK) // another command: ignore previous
498  {
499  c.next(); // eat the [
500  goto command_loop;
501  }
502 
503  // copy the rest to current_text. Set do_close if ∇ or ⍫ is seen
504  // unless inside strings.
505  //
506  for (;;)
507  {
508  switch(cc = c.next())
509  {
510  case Invalid_Unicode: // regular end of input
511  return 0;
512 
513  case UNI_NABLA: // ∇
514  do_close = true;
515  return 0;
516 
517  case UNI_DEL_TILDE: // ⍫
518  locked = true;
519  do_close = true;
520  return 0;
521 
522  case UNI_DOUBLE_QUOTE: // "
523  current_text.append(cc);
524  for (;;)
525  {
526  cc = c.next();
527  if (cc == Invalid_Unicode) // premature end of input
528  {
529  current_text.append(UNI_DOUBLE_QUOTE);
530  return 0;
531  }
532 
533  current_text.append(cc);
534  if (cc == UNI_DOUBLE_QUOTE) break; // string end
535  if (cc == UNI_BACKSLASH) // \x
536  {
537  cc = c.next();
538  if (cc == Invalid_Unicode) // premature input end
539  {
540  current_text.append(UNI_BACKSLASH);
541  current_text.append(UNI_DOUBLE_QUOTE);
542  return 0;
543  }
544  current_text.append(cc);
545  }
546  }
547  break;
548 
549  case UNI_SINGLE_QUOTE: // '
550  current_text.append(cc);
551  for (;;)
552  {
553  // no need to care for ''. Since we only copy, we can
554  // handle ' ... '' ... ' like two adjacent strings
555  // instead of a string containing a (doubled) quote.
556  //
557  cc = c.next();
558  if (cc == Invalid_Unicode) // premature end of input
559  {
560  current_text.append(UNI_SINGLE_QUOTE);
561  return 0;
562  }
563 
564  current_text.append(cc);
565  if (cc == UNI_SINGLE_QUOTE) break; // string end
566  }
567  break;
568 
569  default:
570  current_text.append(cc);
571  }
572  }
573 
574  return 0;
575 }
576 //-----------------------------------------------------------------------------
577 LineLabel
578 Nabla::parse_lineno(UCS_string::iterator & c)
579 {
580 LineLabel ret(0);
581 
582  while (Avec::is_digit(c.get()))
583  {
584  ret.ln_major *= 10;
585  ret.ln_major += c.next() - UNI_0;
586  }
587 
588  if (c.get() == UNI_FULLSTOP)
589  {
590  c.next(); // eat the .
591  while (Avec::is_digit(c.get())) ret.ln_minor.append(c.next());
592  }
593 
594  return ret;
595 }

607 //-----------------------------------------------------------------------------
608 const char *
609 Nabla::open_existing_function()
610 {
611  Log(LOG_nabla)
612  UERR << "opening existing function '" << fun_symbol->get_name()
613  << "'" << endl;
614 
615  if (const char * why = fun_symbol->cant_be_defined()) return why;
616 
617  // this function must only be called when editing functions interactively
618  //
619  if (InputFile::running_script())
620  return "∇-edit existing function from a script";
621 
622 Function_P function = fun_symbol->get_function();
623  Assert(function);
624 
625  if (function->get_exec_properties()[0])
626  return "function is locked";
627 
628  if (function->is_native())
629  return "function is native";
630 
631  if (function->is_lambda())
632  return "function is a lambda";
633 
634  if (Workspace::is_called(fun_symbol->get_name()))
635  return "function is used, pendent or suspended";
636 
637 const UserFunction * ufun = function->get_ufun1();
638  if (ufun == 0)
639  return "function is not editable at " LOC;
640 
641 const UCS_string ftxt = function->canonical(false);
642  Log(LOG_nabla) UERR << "existing function is:\n" << ftxt << endl;
643 
644 UCS_string_vector tlines;
645  ftxt.to_vector(tlines);
646 
647  Assert(tlines.size());
648  fun_header = tlines[0];
649  loop(t, tlines.size())
650  {
651  FunLine fl(t, tlines[t]);
652 
653  // set stop and trace flags
654  //
655  loop(st, ufun->get_stop_lines().size())
656  {
657  if (t == ufun->get_stop_lines()[st]) // stop set
658  {
659  fl.stop_flag = true;
660  break;
661  }
662  }
663 
664  loop(tr, ufun->get_trace_lines().size())
665  {
666  if (t == ufun->get_trace_lines()[tr]) // trace set
667  {
668  fl.trace_flag = true;
669  break;
670  }
671  }
672 
673  lines.push_back(fl);
674  }
675 
676  current_line = LineLabel(tlines.size());
677 
678  return 0;
679 }
680 //-----------------------------------------------------------------------------
681 const char *
682 Nabla::execute_oper()
683 {
684  if (ecmd == ECMD_NOP)
685  {
686  Log(LOG_nabla)
687  UERR << "Nabla::execute_oper(NOP)" << endl;
688  return 0;
689  }
690 
691 const bool have_from = edit_from.ln_major != -1;
692 const bool have_to = edit_to.ln_major != -1;
693 
694  if (lines.size())
695  {
696  if (!have_from && ecmd != ECMD_DELETE) edit_from = lines[0].label;
697  if (!have_to) edit_to = lines[lines.size() - 1].label;
698  }
699 
700  if (ecmd == ECMD_SHOW) return execute_show();
701  if (ecmd == ECMD_DELETE) return execute_delete();
702  if (ecmd == ECMD_EDIT) return execute_edit();
703  if (ecmd == ECMD_ESCAPE) return execute_escape();
704 
705  UERR << "edit command " << ecmd
706  << " from " << edit_from << " to " << edit_to << endl;
707  FIXME;
708 
709  return LOC;
710 }
711 //-----------------------------------------------------------------------------
712 const char *
713 Nabla::execute_show()
714 {
715  Log(LOG_nabla)
716  UERR << "Nabla::execute_oper(SHOW) from " << edit_from
717  << " to " << edit_to << " line-count " << lines.size() << endl;
718 
719 int idx_from = find_line(edit_from);
720 int idx_to = find_line(edit_to);
721 
722 const LineLabel user_edit_to = edit_to;
723 
724  if (idx_from == -1) edit_from = lines[idx_from = 0].label;
725  if (idx_to == -1) edit_to = lines[idx_to = lines.size() - 1].label;
726 
727  Log(LOG_nabla)
728  UERR << "Nabla::execute_oper(SHOW) from "
729  << edit_from << " to " << edit_to << endl;
730 
731  if (idx_from == 0) // then print header line
732  COUT << " ∇" << endl;
733  for (int e = idx_from; e <= idx_to; ++e) lines[e].print(COUT);
734  if (idx_to == int(lines.size() - 1)) // then print last line
735  COUT << " ∇" << endl;
736 
737  if (user_edit_to.valid()) // eg. [⎕42] or [2⎕42]
738  {
739  current_line = user_edit_to;
740  if (line_exists(current_line)) current_line.next();
741  if (line_exists(current_line))
742  {

855 {

857 Token_string in;
858

860  current_text.remove_leading_and_trailing_whitespaces();

865  ec = E_NO_ERROR;
866  Workspace::more_error().clear();
867  }
868

870  {
871  CERR << "SYNTAX ERROR";
872  if (Workspace::more_error().size())
873  {
874  CERR << "+" << endl << Workspace::more_error();
875  }

877  return 0;
878  }
879 
880  modified = true;
881

883 
884  Assert(lines.size() > 0);
885  if (idx_from == -1) // new line
886  {

888  //

890  for (size_t i = 0; i < lines.size(); ++i)

898  }
899  else

901  // replace line
902  //
903  lines[idx_from].text = current_text;
904  }
905 
906  current_line.next();
907 
908  return 0;
909 }

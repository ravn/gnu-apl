75 //-----------------------------------------------------------------------------
76 ErrorCode
77 ComplexCell::bif_near_int64_t(Cell * Z) const
78 {
79  if (!is_near_int64_t()) return E_DOMAIN_ERROR;
80 
81  // return an int Cell if the imaginary part is 0 and the number is not too
82  // large
83  //
84  if (value.cval[1] < INTEGER_TOLERANCE &&
85  value.cval[1] > -INTEGER_TOLERANCE &&
86  value.cval[0] < BIG_INT64_F &&
87  value.cval[0] > -BIG_INT64_F) return zv(Z, round(value.cval[0]));
88 
89  return zv(Z, round(value.cval[0]), round(value.cval[1]));
90 }
91 //-----------------------------------------------------------------------------
92 ErrorCode
93 ComplexCell::bif_within_quad_CT(Cell * Z) const
94 {
95 const double val_r = value.cval[0];
96  if (val_r > LARGE_INT) return E_DOMAIN_ERROR;
97  if (val_r < SMALL_INT) return E_DOMAIN_ERROR;
98 
99 const double val_i = value.cval[1];
100  if (val_i > LARGE_INT) return E_DOMAIN_ERROR;
101  if (val_i < SMALL_INT) return E_DOMAIN_ERROR;
102 
103 const double max_diff_r = Workspace::get_CT() * val_r; // scale ⎕CT
104 const double max_diff_i = Workspace::get_CT() * val_i; // scale ⎕CT
105 
106 const APL_Float val_dn_r = floor(val_r);
107 const APL_Float val_up_r = ceil(val_r);
108 const APL_Float val_dn_i = floor(val_i);
109 const APL_Float val_up_i = ceil(val_i);
110 
111 double z_r;
112  if (val_r < (val_dn_r + max_diff_r)) z_r = val_dn_r;
113  else if (val_r < (val_up_r - max_diff_r)) z_r = val_up_r;
114  else return E_DOMAIN_ERROR;
115 
116 double z_i;
117  if (val_i < (val_dn_i + max_diff_i)) z_i = val_dn_i;
118  else if (val_i < (val_up_i - max_diff_i)) z_i = val_up_i;
119  else return E_DOMAIN_ERROR;
120 
121  return zv(Z, z_r, z_i);
122 }

124 bool
125 ComplexCell::is_near_zero() const
126 {
127  if (value.cval[0] >= INTEGER_TOLERANCE) return false;
128  if (value.cval[0] <= -INTEGER_TOLERANCE) return false;
129  if (value.cval[1] >= INTEGER_TOLERANCE) return false;
130  if (value.cval[1] <= -INTEGER_TOLERANCE) return false;
131  return true;
132 }
133 //-----------------------------------------------------------------------------
134 bool
135 ComplexCell::is_near_one() const
136 {
137  if (value.cval[0] > (1.0 + INTEGER_TOLERANCE)) return false;
138  if (value.cval[0] < (1.0 - INTEGER_TOLERANCE)) return false;
139  if (value.cval[1] > INTEGER_TOLERANCE) return false;
140  if (value.cval[1] < -INTEGER_TOLERANCE) return false;
141  return true;
142 }
143 //-----------------------------------------------------------------------------
144 bool
145 ComplexCell::is_near_real() const
146 {
147 const APL_Float B2 = REAL_TOLERANCE*REAL_TOLERANCE;
148 const APL_Float I2 = value.cval[1] * value.cval[1];
149 
150  if (I2 < B2) return true; // I is absolutely small
151 
152 const APL_Float R2 = value.cval[0] * value.cval[0];
153  return (I2 < R2*B2); // I is relatively small
154 }

209 ErrorCode
210 ComplexCell::bif_reciprocal(Cell * Z) const
211 {
212  if (is_near_zero()) return E_DOMAIN_ERROR;
213 
214  if (is_near_real())
215  {
216  const APL_Float z = 1.0/value.cval[0];
217  if (!isfinite(z)) return E_DOMAIN_ERROR;
218  return FloatCell::zv(Z, z);
219  }
220 
221 const APL_Float denom = mag2();
222 const APL_Float r = value.cval[0]/denom;
223  if (!isfinite(r)) return E_DOMAIN_ERROR;
224 
225 const APL_Float i = value.cval[1]/denom;
226  if (!isfinite(i)) return E_DOMAIN_ERROR;
227 
228 const APL_Complex z(r, -i);
229  return zv(Z, z);
230 }

259 //-----------------------------------------------------------------------------
260 ErrorCode
261 ComplexCell::bif_ceiling(Cell * Z) const
262 {
263 const APL_Float cr = ceil(value.cval[0]); // cr ≥ value.cval[0]
264 const APL_Float Dr = cr - value.cval[0]; // Dr ≥ 0
265 const APL_Float ci = ceil(value.cval[1]); // ci ≥ value.cval[1]
266 const APL_Float Di = ci - value.cval[1]; // Di ≥ 0
267 const APL_Float D = Dr + Di; // 0 ≤ D < 2
268 
269  // ISO: if D is tolerantly less than 1 return fr + 0J1×fi
270  // IBM: if D is less than 1 return fr + 0J1×fi
271  // However, ISO examples follow IBM (and so do we)
272  //
273 // if (D < 1.0 + Workspace::get_CT()) return zv(Z, cr, ci); // ISO
274  if (D < 1.0) return zv(Z, cr, ci); // IBM and examples in ISO
275 
276  if (Di > Dr) return zv(Z, cr, ci - 1.0);
277  else return zv(Z, cr - 1.0, ci);
278 }
279 //-----------------------------------------------------------------------------
280 ErrorCode
281 ComplexCell::bif_floor(Cell * Z) const
282 {
283 APL_Float fr = floor(value.cval[0]); // fr ≤ value.cval[0]
284 APL_Float Dr = value.cval[0] - fr; // 0 ≤ Dr < 1
285 APL_Float fi = floor(value.cval[1]); // fi ≤ value.cval[1]
286 APL_Float Di = value.cval[1] - fi; // 0 ≤ Di < 1
287 const double qct = Workspace::get_CT(); // ⎕CT
288 const double limit = 1.0 - qct;
289 
290  if (Dr > limit)
291  {
292  fr = fr + 1.0;
293  Dr = 0.0;
294  }
295 
296  if (Di > limit)
297  {
298  fi = fi + 1.0;
299  Di = 0.0;
300  }
301 
302  if ((Dr + Di) < limit)
303  {
304  return zv(Z, fr, fi);
305  }
306 
307  if (Dr < (Di - qct))
308  {
309  return zv(Z, fr, fi + 1.0);
310  }
311  else
312  {
313  return zv(Z, fr + 1.0, fi);
314  }
315 }

367 ErrorCode
368 ComplexCell::bif_divide(Cell * Z, const Cell * A) const
369 {
370  if (cval().real() == 0.0 && cval().imag() == 0.0) // A ÷ 0
371  {
372  if (A->get_real_value() != 0.0) return E_DOMAIN_ERROR;
373  if (A->get_imag_value() != 0.0) return E_DOMAIN_ERROR;
374  return IntCell::zv(Z, 1); // 0÷0 is 1 in APL

378  return E_NO_ERROR;

380 //-----------------------------------------------------------------------------
381 ErrorCode
382 ComplexCell::bif_residue(Cell * Z, const Cell * A) const
383 {
384  if (!A->is_numeric()) return E_DOMAIN_ERROR;
385 
386 const APL_Complex mod = A->get_complex_value();
387 const APL_Complex b = get_complex_value();
388 
389  // if A is zero , return B
390  //
391  if (mod.real() == 0.0 && mod.imag() == 0.0)
392  return zv(Z, b);
393 
394  // IBM: if B is zero , return 0
395  //
396  if (b.real() == 0.0 && b.imag() == 0.0)
397  return IntCell::z0(Z);
398 
399  // compliant implementation: B-A×⌊B÷A+A=0
400  // The b=0 case may still exist due to ⎕CT
401  //
402  // // op Z before Z after
403  new (Z) IntCell(0); // Z←0 any 0
404  Z->bif_equal(Z, A); // Z←A=Z 0 A=0
405  Z->bif_add(Z, A); // Z←A+Z A=0 A+A=0
406  Z->bif_divide(Z, this); // Z←B÷Z A+A=0 B÷A+A=0
407  Z->bif_floor(Z); // Z←⌊Z B÷A+A=0 ⌊B÷A+A=0
408  Z->bif_multiply(Z, A); // Z←A×Z ⌊B÷A+A=0 A×⌊B÷A+A=0
409  Z->bif_subtract(Z, this); // Z←A-Z A×⌊B÷A+A=0 B-A×⌊B÷A+A=0
410  return E_NO_ERROR;
411 }

413 ErrorCode
414 ComplexCell::bif_maximum(Cell * Z, const Cell * A) const
415 {
416  // maximum of complex numbers gives DOMAN ERROR if one of the cells
417  // is not near-real.
418  //
419  if (!is_near_real()) return E_DOMAIN_ERROR;
420  if (!A->is_near_real()) return E_DOMAIN_ERROR;
421 
422 const APL_Float breal = get_real_value();
423 
424  if (A->is_integer_cell())
425  {
426  const APL_Integer a = A->get_int_value();
427  return a >= breal ? IntCell::zv(Z, a) : FloatCell::zv(Z, breal);
428  }
429 
430  // both A and B are near-real, therefore they must be numeric and have no
431  // non-0 imag part
432  //
433 const APL_Float a = A->get_real_value();
434  return FloatCell::zv(Z, a >= breal ? a : breal);
435 }

460 //-----------------------------------------------------------------------------
461 ErrorCode
462 ComplexCell::bif_equal(Cell * Z, const Cell * A) const
463 {
464 const double qct = Workspace::get_CT();
465  if (A->is_complex_cell())
466  {
467  const APL_Complex diff = A->get_complex_value() - get_complex_value();
468  if (diff.real() > qct) return IntCell::zv(Z, 0);
469  if (diff.real() < -qct) return IntCell::zv(Z, 0);
470  if (diff.imag() > qct) return IntCell::zv(Z, 0);
471  if (diff.imag() < -qct) return IntCell::zv(Z, 0);
472  return IntCell::zv(Z, 1);
473  }
474 
475  if (A->is_numeric())
476  {
477  if (get_imag_value() > qct) return IntCell::zv(Z, 0);
478  if (get_imag_value() < -qct) return IntCell::zv(Z, 0);
479 
480  const APL_Float diff = A->get_real_value() - get_real_value();
481  if (diff > qct) return IntCell::zv(Z, 0);
482  if (diff < -qct) return IntCell::zv(Z, 0);
483  return IntCell::zv(Z, 1);
484  }
485 
486  return IntCell::zv(Z, 1);
487 }
488 //-----------------------------------------------------------------------------
489 ErrorCode
490 ComplexCell::bif_power(Cell * Z, const Cell * A) const
491 {
492  // some A to the complex B-th power
493  //
494  if (!A->is_numeric()) return E_DOMAIN_ERROR;
495 
496 const APL_Float ar = A->get_real_value();
497 const APL_Float ai = A->get_imag_value();
498 
499  // 1. A == 0
500  //
501  if (ar == 0.0 && ai == 0.0)
502  {
503  if (cval().real() == 0.0) return IntCell::z1(Z); // 0⋆0 is 1
504  if (cval().imag() > 0.0) return IntCell::z0(Z); // 0⋆N is 0
505  return E_DOMAIN_ERROR; // 0⋆¯N = 1÷0
506  }
507 
508  // 2. complex result
509  {
510  const APL_Complex a(ar, ai);
511  const APL_Complex z = complex_power(a, cval());
512  if (!isfinite(z.real())) return E_DOMAIN_ERROR;
513  if (!isfinite(z.imag())) return E_DOMAIN_ERROR;
514 
515  new (Z) ComplexCell(z);
516  return E_NO_ERROR;
517  }
518 }
519 //-----------------------------------------------------------------------------
520 ErrorCode
521 ComplexCell::bif_logarithm(Cell * Z, const Cell * A) const
522 {
523  // ISO p. 88
524  //
525  if (!A->is_numeric()) return E_DOMAIN_ERROR;
526 
527  if (get_real_value() == A->get_real_value() &&
528  get_imag_value() == A->get_imag_value()) return IntCell::z1(Z);
529 
530 
531  if (value.cval[0] == 0.0 && value.cval[1] == 0.0) return E_DOMAIN_ERROR;
532  if (A->is_near_one()) return E_DOMAIN_ERROR;
533 
534  if (A->is_real_cell())
535  {
536  const APL_Float lg_a = log(A->get_real_value());
537  const APL_Complex lg_z = log(cval());
538  const APL_Complex z(lg_z.real() / lg_a, lg_z.imag() / lg_a);
539  if (!isfinite(z.real())) return E_DOMAIN_ERROR;
540  if (!isfinite(z.imag())) return E_DOMAIN_ERROR;
541  return zv(Z, z);
542  }
543 
544  if (A->is_complex_cell())
545  {
546  const APL_Complex z = log(cval()) / log(A->get_complex_value());
547  if (!isfinite(z.real())) return E_DOMAIN_ERROR;
548  if (!isfinite(z.imag())) return E_DOMAIN_ERROR;
549  return zv(Z, z);
550  }
551 
552  return E_DOMAIN_ERROR;
553 }

567 //-----------------------------------------------------------------------------
568 ErrorCode
569 ComplexCell::bif_circle_fun_inverse(Cell * Z, const Cell * A) const
570 {
571  if (!A->is_near_int()) return E_DOMAIN_ERROR;
572 const APL_Integer fun = A->get_checked_near_int();
573 
574  new (Z) FloatCell(0); // prepare for DOMAIN ERROR
575 
576 ErrorCode ret = E_DOMAIN_ERROR;
577  switch(fun)
578  {
579  case 1: case -1:
580  case 2: case -2:
581  case 3: case -3:
582  case 4: case -4:
583  case 5: case -5:
584  case 6: case -6:
585  case 7: case -7:
586  ret = do_bif_circle_fun(Z, -fun, cval());
587  if (!Z->is_finite()) return E_DOMAIN_ERROR;
588  return ret;
589 
590  case -10: // +A (conjugate) is self-inverse
591  ret = do_bif_circle_fun(Z, fun, cval());
592  if (!Z->is_finite()) return E_DOMAIN_ERROR;
593  return ret;
594 
595  default: return E_DOMAIN_ERROR;
596  }
597 
598  // not reached
599  return E_DOMAIN_ERROR;
600 }
601 //-----------------------------------------------------------------------------
602 ErrorCode
603 ComplexCell::do_bif_circle_fun(Cell * Z, int fun, APL_Complex b)
604 {
605 const APL_Complex one(1.0, 0.0);
606  switch(fun)
607  {
608  case -12: {
609  ComplexCell cb(-b.imag(), b.real());
610  cb.bif_exponential(Z);
611  }
612  return E_NO_ERROR;
613 
614  case -11: return zv(Z, -b.imag(), b.real());
615 
616  case -10: return zv(Z, b.real(), -b.imag());
617 
618  case -9: return zv(Z, b);
619  case -8: // ¯8○Z ←→ -8○Z
620  do_bif_circle_fun(Z, 8, b);
621  return zv(Z, -Z->get_complex_value());
622 
623  case -7: // arctanh(z) = 0.5 (ln(1.0 + z) - ln(1.0 - z))
624  {
625  const APL_Complex b1 = ONE() + b;
626  const APL_Complex b_1 = ONE() - b;
627  const APL_Complex log_b1 = log(b1);
628  const APL_Complex log_b_1 = log(b_1);
629  const APL_Complex diff = log_b1 - log_b_1;
630  const APL_Complex half(diff.real()*0.5, diff.imag()*0.5);
631  return zv(Z, half);
632  }
633 
634  case -6: // arccosh(z) = ln(z + sqrt(z + 1) sqrt(z - 1))
635  {
636  const APL_Complex b1 = b + ONE();
637  const APL_Complex b_1 = b - ONE();
638  const APL_Complex root1 = complex_sqrt(b1);
639  const APL_Complex root_1 = complex_sqrt(b_1);
640  const APL_Complex prod = root1 * root_1;
641  const APL_Complex sum = b + prod;
642  const APL_Complex loga = log(sum);
643  return zv(Z, loga);
644  }
645 
646  case -5: // arcsinh(z) = ln(z + sqrt(z^2 + 1))
647  {
648  const APL_Complex b2 = b*b;
649  const APL_Complex b2_1 = b2 + ONE();
650  const APL_Complex root = complex_sqrt(b2_1);
651  const APL_Complex sum = b + root;
652  const APL_Complex loga = log(sum);
653  return zv(Z, loga);
654  }
655 
656  case -4: if (b.real() >= 0.0 ||
657  (b.real() > -1.0 && Cell::is_near_zero(b.imag()))
658  ) return zv(Z, complex_sqrt(b*b - one));
659  else return zv(Z, -complex_sqrt(b*b - one));
660 
661  case -3: // arctan(z) = i/2 (ln(1 - iz) - ln(1 + iz))
662  {
663  const APL_Complex iz = APL_Complex(- b.imag(), b.real());
664  const APL_Complex piz = ONE() + iz;
665  const APL_Complex niz = ONE() - iz;
666  const APL_Complex log_piz = log(piz);
667  const APL_Complex log_niz = log(niz);
668  const APL_Complex diff = log_niz - log_piz;
669  const APL_Complex prod = APL_Complex(0, 0.5) * diff;
670  return zv(Z, prod);
671  }
672 
673  case -2: // arccos(z) = -i (ln( z + sqrt(z^2 - 1)))
674  {
675  const APL_Complex b2 = b*b;
676  const APL_Complex diff = b2 - ONE();

678  const APL_Complex sum = b + root;
679  const APL_Complex loga = log(sum);
680  const APL_Complex prod = MINUS_i() * loga;
681  return zv(Z, prod);
682  }
683 
684  case -1: // arcsin(z) = -i (ln(iz + sqrt(1 - z^2)))
685  {
686  const APL_Complex b2 = b*b;
687  const APL_Complex diff = ONE() - b2;
688  const APL_Complex root = complex_sqrt(diff);
689  const APL_Complex sum = APL_Complex(-b.imag(), b.real())
690  + root;
691  const APL_Complex loga = log(sum);
692  const APL_Complex prod = MINUS_i() * loga;
693  return zv(Z, prod);
694  }
695 
696  case 0: return zv(Z, complex_sqrt(one - b*b));
697 
698  case 1: return zv(Z, sin(b));
699 
700  case 2: return zv(Z, cos(b));
701 
702  case 3: return zv(Z, tan(b));
703 
704  case 4: return zv(Z, complex_sqrt(one + b*b));
705 
706  case 5: return zv(Z, sinh(b));
707 
708  case 6: return zv(Z, cosh(b));
709 
710  case 7: return zv(Z, tanh(b));
711 
712  case 8: { const APL_Complex b2 = b*b;
713  const APL_Complex square = // (¯1 - R⋆2)
714  APL_Complex(-1, 0) - b2;
715  const APL_Complex root = complex_sqrt(square);
716  if (b.real() > 0.0)
717  {
718  if (b.imag() > 0.0) return zv(Z, root);
719  else return zv(Z, -root);
720  }
721  else if (b.real() == 0.0)
722  {
723  if (b.imag() > 1.0) return zv(Z, root);
724  else return zv(Z, -root);
725  }
726  else // b.real() < 0,0;
727  {
728  if (b.imag() >= 0.0) return zv(Z, root);
729  else return zv(Z, -root);
730  }
731  }
732 
733  case 9: return FloatCell::zv(Z, b.real());
734 
735  case 10: return FloatCell::zv(Z, sqrt(mag2(b)));
736 
737  case 11: return FloatCell::zv(Z, b.imag());
738 
739  case 12: return FloatCell::zv(Z, atan2(b.imag(), b.real()));
740  }
741 
742  // invalid fun
743  //
744  return E_DOMAIN_ERROR;
745 }

747 APL_Complex
748 ComplexCell::gamma(APL_Float x, const APL_Float & y)
749 {
750 const APL_Complex pi(M_PI, 0);
751  if (x < 0.5)
752  return pi / (sin(pi * APL_Complex(x, y)) * gamma(1.0 - x, -y));
753 
754  // coefficients for lanczos approximation of the gamma function.
755  //
756 #define p0 APL_Complex( 1.000000000190015 )
757 #define p1 APL_Complex( 76.18009172947146 )
758 #define p2 APL_Complex(-86.50532032941677 )
759 #define p3 APL_Complex( 24.01409824083091 )
760 #define p4 APL_Complex( -1.231739572450155 )
761 #define p5 APL_Complex( 1.208650973866179E-3)
762 #define p6 APL_Complex( -5.395239384953E-6 )
763 
764  errno = 0;
765  feclearexcept(FE_ALL_EXCEPT);
766 
767 const APL_Complex z(x, y);
768 const APL_Complex z1(x + 5.5, y);
769 const APL_Complex z2(x + 0.5, y);
770 
771 const APL_Complex ret( (complex_sqrt(APL_Complex(2*M_PI)) / z)
772  * (p0 +
773  p1 / APL_Complex(x + 1.0, y) +
774  p2 / APL_Complex(x + 2.0, y) +
775  p3 / APL_Complex(x + 3.0, y) +
776  p4 / APL_Complex(x + 4.0, y) +
777  p5 / APL_Complex(x + 5.0, y) +
778  p6 / APL_Complex(x + 6.0, y))
779  * complex_power(z1, z2)
780  * complex_exponent(-z1)
781  );
782 
783  if (!isfinite(ret.real())) DOMAIN_ERROR;
784  if (!isfinite(ret.imag())) DOMAIN_ERROR;
785  return ret;
786 
787 #undef p0
788 #undef p1
789 #undef p2
790 #undef p3
791 #undef p4
792 #undef p5
793 #undef p6
794 }

856 Comp_result
857 ComplexCell::compare(const Cell & other) const
858 {
859  if (other.is_character_cell()) return COMP_GT; // numeric > char
860  if (other.is_pointer_cell()) return COMP_LT; // numeric < nested
861  if (!other.is_numeric()) DOMAIN_ERROR;
862 
863 const double qct = Workspace::get_CT();
864  if (equal(other, qct)) return COMP_EQ;
865 
866 APL_Float areal = other.get_real_value();
867 APL_Float breal = get_real_value();
868 
869  // we compare complex numbers by their real part unless the real parsts
870  // are equal. In that case we compare the imag parts. The reason for
871  // comparing this way is compatibility with the comparison of real numbers
872  //
873  if (tolerantly_equal(areal, breal, qct))
874  {
875  areal = other.get_imag_value();
876  breal = get_imag_value();
877  }
878 
879  return (breal < areal) ? COMP_LT : COMP_GT;

882 PrintBuffer
883 ComplexCell::character_representation(const PrintContext & pctx) const
884 {
885  if (pctx.get_PP() < MAX_Quad_PP)
886  {
887  // 10⋆get_PP()
888  //
889  APL_Float ten_to_PP = 1.0;
890  loop(p, pctx.get_PP()) ten_to_PP = ten_to_PP * 10.0;
891 
892  // lrm p. 13: In J notation, the real or imaginary part is not
893  // displayed if it is less than the other by more than ⎕PP orders
894  // of magnitude (unless ⎕PP is at its maximum).
895  //
896  const APL_Float pos_real = value.cval[0] < 0.0
897  ? -value.cval[0] : value.cval[0];
898  const APL_Float pos_imag = value.cval[1] < 0.0
899  ? -value.cval[1] : value.cval[1];
900 
901  if (pos_real >= pos_imag) // pos_real dominates pos_imag
902  {
903  if (pos_real > pos_imag*ten_to_PP)
904  {
905  const FloatCell real_cell(value.cval[0]);
906  return real_cell.character_representation(pctx);
907  }
908  }
909  else // pos_imag dominates pos_real
910  {
911  if (pos_imag > pos_real*ten_to_PP)
912  {
913  const FloatCell imag_cell(value.cval[1]);
914  PrintBuffer ret = imag_cell.character_representation(pctx);
915  ret.pad_l(UNI_ASCII_J, 1);
916  ret.pad_l(UNI_ASCII_0, 1);
917 
918  ret.get_info().flags |= CT_COMPLEX;
919  ret.get_info().imag_len = 1 + ret.get_info().real_len;
920  ret.get_info().int_len = 1;
921  ret.get_info().fract_len = 0;
922  ret.get_info().real_len = 1;
923  return ret;
924  }
925  }
926  }
927 
928 bool scaled_real = pctx.get_scaled(); // may be changed by print function
929 UCS_string ucs(value.cval[0], scaled_real, pctx);
930 
931 ColInfo info;
932  info.flags |= CT_COMPLEX;
933  if (scaled_real) info.flags |= real_has_E;
934 int int_fract = ucs.size();
935  info.real_len = ucs.size();
936  info.int_len = ucs.size();
937  loop(u, ucs.size())
938  {
939  if (ucs[u] == UNI_ASCII_FULLSTOP)
940  {
941  info.int_len = u;
942  if (!scaled_real) break;
943  continue;
944  }
945 
946  if (ucs[u] == UNI_ASCII_E)
947  {
948  if (info.int_len > u) info.int_len = u;
949  int_fract = u;
950  break;
951  }
952  }
953  info.fract_len = int_fract - info.int_len;
954 
955  if (!is_near_real())
956  {
957  ucs.append(UNI_ASCII_J);
958  bool scaled_imag = pctx.get_scaled(); // may be changed by UCS_string()
959  const UCS_string ucs_i(value.cval[1], scaled_imag, pctx);
960 
961  ucs.append(ucs_i);
962 
963  info.imag_len = ucs.size() - info.real_len;
964  if (scaled_imag) info.flags |= imag_has_E;
965  }
966 
967  return PrintBuffer(ucs, info);
968 }

206 Token
207 Quad_PLOT::eval_AB(Value_P A, Value_P B)
208 {
209  if (B->get_rank() > 3) RANK_ERROR;
210  if (B->element_count() < 2) LENGTH_ERROR;
211 
212  // plot window with default attributes
213  //
214 Plot_data * data = setup_data(B.get());
215  if (data == 0) DOMAIN_ERROR;
216 
217 Plot_window_properties * w_props = new Plot_window_properties(data, verbosity);
218  if (w_props == 0)
219  {
220  delete data;
221  WS_FULL;
222  }
223 
224  Log(LOG_Quad_PLOT)
225  CERR << "wprops = " << w_props << " created." << endl;
226 
227  // from here on 'data' is owned by 'w_props' (whose destructor
228  // will delete it).
229  //
230  if (A->get_rank() > 1) { delete w_props; RANK_ERROR; }
231 
232 const ShapeItem len_A = A->element_count();
233  if (len_A < 1) { delete w_props; LENGTH_ERROR; }
234 
235 const APL_Integer qio = Workspace::get_IO();
236 
237  loop(a, len_A)
238  {
239  const Cell & cell_A = A->get_ravel(a);
240  if (!cell_A.is_pointer_cell())
241  {
242  MORE_ERROR() << "A[" << (a + qio)
243  << "] is not a string in A ⎕PLOT B";
244  delete w_props;
245  DOMAIN_ERROR;
246  }
247 
248  const Value * attr = cell_A.get_pointer_value().get();
249  if (!attr->is_char_string())
250  {
251  MORE_ERROR() << "A[" << (a + qio)
252  << "] is not a string in A ⎕PLOT B";
253  delete w_props;
254  DOMAIN_ERROR;
255  }
256 
257  UCS_string ucs = attr->get_UCS_ravel();
258  ucs.remove_leading_and_trailing_whitespaces();
259  if (ucs.size() == 0) continue;
260  if (Avec::is_comment(ucs[0])) continue;
261  UTF8_string utf(ucs);
262  const char * error = w_props->set_attribute(utf.c_str());
263  if (error)
264  {
265  MORE_ERROR() << error << " in ⎕PLOT attribute '" << ucs << "'";
266  delete w_props;
267  DOMAIN_ERROR;
268  }
269  }
270 
271  if (w_props->update(verbosity)) { delete w_props; DOMAIN_ERROR; }
272 
273  // do_plot_data takes ownership of w_props and will delete w_props
274  //
275  return Token(TOK_APL_VALUE1, do_plot_data(w_props, data));
276 }

278 Token
279 Quad_PLOT::eval_B(Value_P B)
280 {
281  if (B->get_rank() == 0 && !B->get_ravel(0).is_pointer_cell())
282  {
283  // scalar argument: plot window control
284  union
285  {
286  APL_Integer B0; // APL
287  const Plot_window_properties * w_props; // gtk
288  const void * vp; // plot_stop() result
289  pthread_t thread; // xcb
290  } u;
291 
292  u.B0 = B->get_ravel(0).get_int_value();
293  if (u.B0 == 0) // reset plot verbosity
294  {
295  verbosity = 0;
296  CERR << "⎕PLOT verbosity turned off" << endl;
297  return Token(TOK_APL_VALUE1, Idx0(LOC));
298  }
299 
300  if (u.B0 == -1) // enable SHOW_EVENTS
301  {
302  verbosity |= SHOW_EVENTS;
303  CERR << "⎕PLOT will show X events " << endl;
304  return Token(TOK_APL_VALUE1, Idx0(LOC));
305  }
306 
307  if (u.B0 == -2) // enable SHOW_DATA
308  {
309  verbosity |= SHOW_DATA;
310  CERR << "⎕PLOT will show APL data " << endl;
311  return Token(TOK_APL_VALUE1, Idx0(LOC));
312  }
313 
314  if (u.B0 == -3) // close all windows
315  {
316  while (plot_threads.size())
317  {
318  plot_threads.pop_back();
319  }
320  }
321 
322  if (u.B0 == -4) // enable SHOW_DRAW
323  {
324  verbosity |= SHOW_DRAW;
325  CERR << "⎕PLOT will show rendering details " << endl;
326  return Token(TOK_APL_VALUE1, Idx0(LOC));
327  }
328 
329 #if HAVE_GTK3
330 
331  u.vp = plot_stop(u.w_props);
332  return Token(TOK_APL_VALUE1, IntScalar(u.B0, LOC));
333 
334 #else // XCB
335 
336  bool found = false;
337  sem_wait(plot_threads_sema);
338  loop(pt, Quad_PLOT::plot_threads.size())
339  {
340  if (Quad_PLOT::plot_threads[pt] != u.thread) continue;
341 
342  plot_threads[pt] = plot_threads[plot_threads.size() - 1];
343  plot_threads.pop_back();
344  found = true;
345  break;
346  }
347  sem_post(plot_threads_sema);
348  return Token(TOK_APL_VALUE1, IntScalar(found ? u.B0 : 0, LOC));
349 
350 #endif // GTK vs. XCB
351  }
352 
353  if (B->get_rank() == 1 && B->element_count() == 0)
354  {
355  help();
356  return Token(TOK_APL_VALUE1, Idx0(LOC));
357  }
358 
359  if (B->get_rank() > 3) RANK_ERROR;
360 
361  // plot window with default attributes
362  //
363 Plot_data * data = setup_data(B.get());
364  if (data == 0) DOMAIN_ERROR;
365 
366 Plot_window_properties * w_props = new Plot_window_properties(data, verbosity);
367  if (w_props == 0)
368  {
369  delete data;
370  WS_FULL;
371  }
372 
373  return Token(TOK_APL_VALUE1, do_plot_data(w_props, data));
374 }

376 Plot_data *
377 Quad_PLOT::setup_data(const Value * B)
378 {
379  // check data. We expect B to be either:
380  // 1a. a numeric vector (for a single plot line), or
381  // 1b. a numeric matrix (for one plot line for each row of the matrix), or
382  // 2a. a scalar of a nested numeric vector (for a single plot line), or
383  // 2b. a vector of nested numeric vectors (for one plot line per item)
384  // 3. a 3-dimensional real vector for surface plots
385  //
386  if (B->is_scalar()) // 2a. → 1a.
387  {
388  if (!B->get_ravel(0).is_pointer_cell()) DOMAIN_ERROR;
389  B = B->get_ravel(0).get_pointer_value().get();
390  }
391 
392 const APL_Integer qio = Workspace::get_IO();
393 Plot_data * data = 0;
394  if (B->get_rank() == 3) // case 3.
395  {
396  const ShapeItem planes = B->get_shape_item(0); // (X), Y, and (Z)
397  const ShapeItem rows = B->get_shape_item(1);
398  const ShapeItem cols = B->get_shape_item(2);
399  if (planes < 1 || planes > 3) LENGTH_ERROR;
400  if (rows < 1) LENGTH_ERROR;
401  if (cols < 1) LENGTH_ERROR;
402  const ShapeItem data_points = rows * cols;
403 
404  if (planes == 3) // X, Y, and Z
405  {
406  double * X = new double[3*data_points];
407  double * Y = X + data_points;
408  double * Z = Y + data_points;
409  if (!X) WS_FULL;
410 
411  data = new Plot_data(rows);
412  loop(r, rows)
413  {
414  loop(c, cols)
415  {
416  const ShapeItem p = c + r*cols;
417  const Cell & cX = B->get_ravel(p);
418  const Cell & cY = B->get_ravel(p + data_points);
419  const Cell & cZ = B->get_ravel(p + 2*data_points);
420 
421  if (!(cX.is_integer_cell() ||
422  cX.is_real_cell())) DOMAIN_ERROR;
423  if (!(cY.is_integer_cell() ||
424  cY.is_real_cell())) DOMAIN_ERROR;
425  if (!(cZ.is_integer_cell() ||
426  cZ.is_real_cell())) DOMAIN_ERROR;
427 
428  X[p] = cX.get_real_value();
429  Y[p] = cY.get_real_value();
430  Z[p] = cZ.get_real_value();
431  }
432  const double * pX = X + r*cols;
433  const double * pY = Y + r*cols;
434  const double * pZ = Z + r*cols;
435  const Plot_data_row * pdr = new Plot_data_row(pX, pY, pZ,
436  r, cols);
437  data->add_row(pdr);
438  }
439  }
440  else if (planes == 2) // X and Y, but no Z
441  {
442  double * X = new double[2*data_points];
443  double * Y = X + data_points;
444  if (!X) WS_FULL;
445 
446  data = new Plot_data(rows);
447  loop(r, rows)
448  {
449  loop(c, cols)
450  {
451  const ShapeItem p = c + r*cols;
452  const Cell & cX = B->get_ravel(p);
453  const Cell & cY = B->get_ravel(p + data_points);
454 
455  if (!(cX.is_integer_cell() ||
456  cX.is_real_cell())) DOMAIN_ERROR;
457  if (!(cY.is_integer_cell() ||
458  cY.is_real_cell())) DOMAIN_ERROR;
459 
460  X[p] = cX.get_real_value();
461  Y[p] = cY.get_real_value();
462  }
463  const double * pX = X + r*cols;
464  const double * pY = Y + r*cols;
465  const Plot_data_row * pdr = new Plot_data_row(pX, pY, 0,
466  r, cols);
467  data->add_row(pdr);
468  }
469  }
470  else // Y, but no X or Z
471  {
472  double * Y = new double[data_points];
473  if (!Y) WS_FULL;
474 
475  data = new Plot_data(rows);
476  loop(r, rows)
477  {
478  loop(c, cols)
479  {
480  const ShapeItem p = c + r*cols;
481  const Cell & cY = B->get_ravel(p);
482 
483  if (!(cY.is_integer_cell() ||
484  cY.is_real_cell())) DOMAIN_ERROR;
485 
486  Y[p] = cY.get_real_value();
487  }
488  const double * pY = Y + r*cols;
489  const Plot_data_row * pdr = new Plot_data_row(0, pY, 0,
490  r, cols);
491  data->add_row(pdr);
492  }
493  }
494  data->surface = true;
495  }
496  else if (B->get_ravel(0).is_pointer_cell()) // 2b.
497  {
498  ShapeItem data_points = 0;
499  if (B->get_rank() > 1) RANK_ERROR;
500  const ShapeItem rows = B->get_cols();
501  loop(r, rows)
502  {
503  const Value * vrow = B->get_ravel(r).get_pointer_value().get();
504  if (vrow->get_rank() > 1) RANK_ERROR;
505  const ShapeItem row_len = vrow->element_count();
506  data_points += row_len;
507  loop(rb, row_len)
508  {
509  if (!vrow->get_ravel(rb).is_numeric()) DOMAIN_ERROR;
510  }
511  }
512  double * X = new double[3*data_points];
513  double * Y = X + data_points;
514  double * Z = Y + data_points;
515  if (!X) WS_FULL;
516 
517  ShapeItem idx = 0;
518  loop(r, rows)
519  {
520  const Value * vrow = B->get_ravel(r).get_pointer_value().get();
521  loop(v, vrow->element_count())
522  {
523  const Cell & cB = vrow->get_ravel(v);
524  if (cB.is_complex_cell())
525  {
526  X[idx] = cB.get_real_value();
527  Z[idx] = Y[idx] = cB.get_imag_value();
528  }
529  else
530  {
531  X[idx] = qio + v;
532  Z[idx] = Y[idx] = cB.get_real_value();
533  }
534  ++idx;
535  }
536  }
537  data = new Plot_data(rows);
538  idx = 0;
539  loop(r, rows)
540  {
541  const double * pX = X + idx;
542  const double * pY = Y + idx;
543  const double * pZ = Z + idx;
544  const Value * vrow = B->get_ravel(r).get_pointer_value().get();
545  const ShapeItem row_len = vrow->element_count();
546  const Plot_data_row * pdr = new Plot_data_row(pX, pY, pZ, r,
547  row_len);
548  data->add_row(pdr);
549  idx += row_len;
550  }
551  }
552  else
553  {
554  const ShapeItem cols_B = B->get_cols();
555  const ShapeItem rows_B = B->get_rows();
556  const ShapeItem len_B = rows_B * cols_B;
557 
558  loop(b, len_B)
559  {
560  if (!B->get_ravel(b).is_numeric()) return 0;
561  }
562 
563  // split B into X=real B, Y=imag Y
564 
565  double * X = new double[3*len_B];
566  double * Y = X + len_B;
567  double * Z = Y + len_B;
568  if (!X) WS_FULL;
569 
570  loop(b, len_B)
571  {
572  const Cell & cB = B->get_ravel(b);
573  if (cB.is_complex_cell())
574  {
575  X[b] = cB.get_real_value();
576  Z[b] = Y[b] = cB.get_imag_value();
577  }
578  else
579  {
580  X[b] = qio + b % cols_B;
581  Z[b] = Y[b] = cB.get_real_value();
582  }
583  }
584 
585  data = new Plot_data(rows_B);
586  loop(r, rows_B)
587  {
588  const double * pX = X + r*cols_B;
589  const double * pY = Y + r*cols_B;
590  const double * pZ = Z + r*cols_B;
591  const Plot_data_row * pdr = new Plot_data_row(pX, pY, pZ, r, cols_B);
592  data->add_row(pdr);
593  }
594 
595  }
596  return data;
597 }
598 //-----------------------------------------------------------------------------
599 Value_P
600 Quad_PLOT::do_plot_data(Plot_window_properties * w_props,
601  const Plot_data * data)
602 {
603  w_props->set_verbosity(verbosity);
604  verbosity > 0 && w_props->print(CERR);
605 
606 union
607 {
608  pthread_t thread; // xcb
609  const Plot_window_properties * w_props; // gtk
610  APL_Integer ret; // APL
611 } u;
612  u.ret = 0;
613 
614 #if HAVE_GTK3 // GTK
615  u.w_props = w_props;
616  plot_main(w_props);
617 #else // XCB
618  pthread_create(&u.thread, 0, plot_main, w_props);
619 #endif
620 
621  sem_wait(plot_threads_sema);
622  plot_threads.push_back(u.thread);
623  sem_post(plot_threads_sema);
624 
625  sem_wait(plot_window_sema); // blocks until window shown
626  return IntScalar(u.ret, LOC);
627 }

629 static UCS_string
630 fill_14(const std::string & name)
631 {
632 UTF8_string name_utf(name.c_str());
633 UCS_string ret(name_utf);
634  while (ret.size() < 14) ret += UNI_ASCII_SPACE;
635  return ret;
636 }

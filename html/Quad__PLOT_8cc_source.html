306 Color
307 Plot_data::Color_from_str(const char * str, const char * & error)
308 {
309  error = 0; // assume no error
310 
311 uint32_t r, g, b;
312  if (3 == sscanf(str, " %u %u %u", &r, &b, &g))
313  return (r & 0xFF) << 16 | (g & 0xFF) << 8 | (b & 0xFF);
314 
315  if (const char * h = strchr(str, '#'))
316  {
317  if (strlen(h) == 4) // #RGB
318  {
319  const int v = strtol(str + 1, 0, 16);
320  return (0x11*(v >> 8 & 0x0F)) << 16
321  | (0x11*(v >> 4 & 0x0F)) << 8
322  | (0x11*(v & 0x0F) << 0);
323  }
324  else if (strlen(h) == 7) // #RRGGBB
325  {
326  return strtol(str + 1, 0, 16);
327  }
328  }
329 
330  error = "Bad color format";
331  return 0;

625 bool
626 Plot_window_properties::update(int verbosity)
627 {
628  window_width = pa_border_L + pa_width + pa_border_R;
629  window_height = pa_border_T + pa_height + pa_border_B;
630 
631  min_X = plot_data.get_min_X();
632  max_X = plot_data.get_max_X();
633  min_Y = plot_data.get_min_Y();
634  max_Y = plot_data.get_max_Y();
635 
636  // the user may override the range derived from the data
637  //
638  if (rangeX_min_valid()) min_X = rangeX_min;
639  if (rangeX_max_valid()) max_X = rangeX_max;
640  if (rangeY_min_valid()) min_Y = rangeY_min;
641  if (rangeY_max_valid()) max_Y = rangeY_max;
642 
643  if (min_X >= max_X)
644  {
645  MORE_ERROR() << "empty X range in A âPLOT B";
646  return true;
647  }
648 
649  if (min_Y >= max_Y)
650  {
651  MORE_ERROR() << "empty Y range in A âPLOT B";
652  return true;
653  }
654 
655 double delta_X = max_X - min_X;
656 double delta_Y = max_Y - min_Y;
657 
658  // first approximation for value â pixel factor
659  //
660  scale_X = pa_width / delta_X; // pixels per X-value
661  scale_Y = pa_height / delta_Y; // pixels per Y-value
662 
663  verbosity > 1 &&
664  CERR << setw(20) << "delta_X (1): " << delta_X << " (Xmax-Xmin)" << endl
665  << setw(20) << "delta_Y (1): " << delta_Y << " Ymax-Ymin" << endl
666  << setw(20) << "scale_X (1): " << scale_X << " pixels/X" << endl
667  << setw(20) << "scale_Y (1): " << scale_Y << " pixels/Y" << endl;
668 
669 const double tile_X_raw = gridX_pixels / scale_X;
670 const double tile_Y_raw = gridY_pixels / scale_Y;
671 
672  verbosity > 1 &&
673  CERR << setw(20) << "tile_X_raw: " << tile_X_raw
674  << "(âX between grid lines)" << endl
675  << setw(20) << "tile_Y_raw: " << tile_Y_raw
676  << "(âY between grid lines)" << endl;
677 
678  tile_X = round_up_125(tile_X_raw);
679  tile_Y = round_up_125(tile_Y_raw);
680 
681 const int min_Xi = floor(min_X / tile_X);
682 const int min_Yi = floor(min_Y / tile_Y);
683 const int max_Xi = ceil(max_X / tile_X);
684 const int max_Yi = ceil(max_Y / tile_Y);
685  gridX_last = 0.5 + max_Xi - min_Xi;
686  gridY_last = 0.5 + max_Yi - min_Yi;
687 
688  min_X = tile_X * floor(min_X / tile_X);
689  min_Y = tile_Y * floor(min_Y / tile_Y);
690  max_X = tile_X * ceil(max_X / tile_X);
691  max_Y = tile_Y * ceil(max_Y / tile_Y);
692 
693  delta_X = max_X - min_X;
694  delta_Y = max_Y - min_Y;
695 
696  // final value â pixel factor
697  //
698  scale_X = pa_width / delta_X;
699  scale_Y = pa_height / delta_Y;
700 
701  verbosity > 1 &&
702  CERR << setw(20) << "min_X (2): " << min_X << endl
703  << setw(20) << "max_X (2): " << max_X << endl
704  << setw(20) << "min_Y (2): " << min_Y << endl
705  << setw(20) << "max_Y (2): " << max_Y << endl
706  << setw(20) << "delta_X (2): " << delta_X << " (Xmax-Xmin)" << endl
707  << setw(20) << "delta_Y (2): " << delta_Y << " (Ymax-Ymin)" << endl
708  << setw(20) << "scale_X (2): " << scale_X << " pixels/X" << endl
709  << setw(20) << "scale_Y (2): " << scale_Y << " pixels/Y" << endl

712  << setw(20) << "tile (2): " << tile_Y
713  << "(âY between grid lines)" << endl;
714 
715  return false; // OK

736 {
737  while (*att_and_val && (*att_and_val <= ' ')) ++ att_and_val;
738 const char * colon = strchr(att_and_val, ':');
739  if (colon == 0) return "Expecting 'attribute: value' but no : found";
740 
741 const char * value = colon + 1;
742  while (*value && (*value <= ' ')) ++value;
743 
744 const char * minus = strchr(att_and_val, '-');
745  if (minus && minus < colon) // line attribute
746  {
747  // it is not unlikely that the user wants to use the same attributes
748  // for different plots that may differ in the number of lines. We
749  // therefore silently ignore such over-specified attribute rather
750  // than returning an error string (which then raises a DOMAIN error).
751  //
752  const int line = strtol(minus + 1, 0, 10) - Workspace::get_IO();
753  if (line < 0) return 0;
754  if (line >= line_count) return 0;
755 
756 #define ldef(ty, na, val, _descr) \
757  if (!strncmp(#na "-", att_and_val, minus - att_and_val)) \
758  { const char * error = 0; \
759  line_properties[line]->set_ ## na(Plot_data::ty ## \
760  _from_str(value, error)); \
761  return error; \
762  }
763 #include "Quad_PLOT.def"
764  }
765  else // window attribute
766  {
767 #define gdef(ty, na, _val, _descr) \
768  if (!strncmp(#na, att_and_val, colon - att_and_val)) \
769  { const char * error = 0; \
770  set_ ## na(Plot_data::ty ## _from_str(value, error)); \
771  return error; \

784  while (val < 1) { val *= 10; --expo; }
785 
786  // at this point 1 â¤ val < 10.
787  //
788  if (expo > -18 && expo < 18) // use integer arithmetic for 10^expo
789  {

791  uint64_t expo_val = 1;
792  loop(a, abs_expo) expo_val *= 10;
793 
794  if (val <= 2.0) return expo < 0 ? 2.0/expo_val : 2.0*expo_val;
795  if (val <= 5.0) return expo < 0 ? 5.0/expo_val : 5.0*expo_val;
796  return expo < 0 ? 10.0/expo_val : 10.0*expo_val;
797  }
798  else // use float arithmetic for 10^expo
799  {
800  const double expo_val = pow(10, expo);
801  if (val <= 2.0) return expo < 0 ? 2.0/expo_val : 2.0*expo_val;
802  if (val <= 5.0) return expo < 0 ? 5.0/expo_val : 5.0*expo_val;

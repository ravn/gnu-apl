285 template<>
286 inline void
287 Bif_F12_DOMINO::Matrix<false>::col1_norm(norm_result & result) const
288 {
289 double sum = 0;
290 
291  loop(y, M)
292  {
293  const double r = real(y, 0);
294  sum += r*r;
295  }
296  result.norm2_real = sum;
297  result.norm2_imag = 0;
298  result.norm_real = sqrt(sum);
299  result.norm_imag = 0;
300 
301  result.norm__2_real = 2.0 / result.norm2_real;
302  result.norm__2_imag = 0.0;
303 }
304 //-----------------------------------------------------------------------------
306 template<>
307 inline void
308 Bif_F12_DOMINO::Matrix<true>::col1_norm(norm_result & result) const
309 {
310 double sum_real = 0;
311 double sum_imag = 0;
312 
313  loop(y, M)
314  {
315  const double r = real(y, 0);
316  const double i = imag(y, 0);
317  sum_real += r*r - i*i;
318  sum_imag += 2*r*i;
319  }
320 
321 const complex<double> sum(sum_real, sum_imag);
322  result.norm2_real = sum.real();
323  result.norm2_imag = sum.imag();
324 const complex<double> root = sqrt(sum);
325  result.norm_real = root.real();
326  result.norm_imag = root.imag();
327 
328 const complex<double> _2__sum = 2.0 / sum;
329  result.norm__2_real = _2__sum.real();
330  result.norm__2_imag = _2__sum.imag();
331 }

409 template<>
410 inline void Bif_F12_DOMINO::Matrix<false>::imbed(const Matrix<false>& S)
411 {
412  matrix_assert( M == N); // Qi is quadratic (NÃN)
413  matrix_assert(S.M == S.N); // S is quadratic (MÃM)
414  matrix_assert(S.M <= M); // S is smaller than Qi
415 
416 const ShapeItem IN = M - S.M; // the number of rows and columns from ID N
417 
418  loop(y, M)
419  loop(x, N)
420  {
421  if (y < IN || x < IN) // upper row or left col: init from ID N
422  real(x, y) = real(y, x) = 0.0;
423  else // lower row and right col: init from S
424  real(y, x) = S.real(y - IN, x - IN);
425  }
426 
427  // upper diagonal
428  //
429  loop(y, IN) real(y,y) = 1.0;
430 }
431 //-----------------------------------------------------------------------------
433 template<>
434 inline void Bif_F12_DOMINO::Matrix<true>::imbed(const Matrix<true>& S)
435 {
436  matrix_assert( M == N); // Qi is quadratic (NÃN)
437  matrix_assert(S.M == S.N); // S is quadratic (MÃM)
438  matrix_assert(S.M <= M); // S is smaller than Qi
439 
440 const ShapeItem IN = M - S.M; // the number of rows and columns from ID N
441 
442  loop(y, M)
443  {
444  if (y < IN) // upper row: init from identity matrix
445  {
446  loop(x, N) real(y, x) = imag(y, x) = 0.0;
447  real(y, y) = 1.0; // diagonal
448  }
449  else // lower row: maybe init from S
450  {
451  loop(x, N)
452  {
453  if (x < IN) // left columns: init from identity matrix
454  real(y, x) = imag(y, x) = 0.0;
455  else // right columns: init from S
456  {
457  real(y, x) = S.real(y - IN, x - IN);
458  imag(y, x) = S.imag(y - IN, x - IN);
459  }
460  }
461  }
462  }

467 template<>
468 inline void
469 Bif_F12_DOMINO::Matrix<false>::init_outer_product(const norm_result & scale,
470  const Matrix<false> & src)
471 {
472  // the resulting matrix is symmetric, so we can take advantage of that
473  // by computing every value only once,
474  //
475  matrix_assert(M == N);
476 
477  // off-diagonal elements...
478  //
479  for (ShapeItem y = 1; y < M; ++y) // for every row below the first
480  loop(x, y) // for every column left of the diagonal
481  real(x, y) = real(y, x)
482  = scale.norm__2_real * src.real(y, 0) * src.real(x, 0);
483 
484  // diagonal elements...
485  //
486  loop(y, M)
487  real(y, y) = scale.norm__2_real * src.real(y, 0) * src.real(y, 0);
488 
489 }
490 //-----------------------------------------------------------------------------
493 template<>
494 inline void
495 Bif_F12_DOMINO::Matrix<true>::init_outer_product(const norm_result & scale,
496  const Matrix<true> & src)
497 {
498  // the resulting matrix is symmetric, so we can take advantage of that
499  // by computing every value only once,
500  //
501  matrix_assert(M == N);
502 
503 const complex<double> sc(scale.norm__2_real, scale.norm__2_imag);
504 
505  // off-diagonal elements...
506  //
507  for (ShapeItem y = 1; y < M; ++y) // for every row below the first
508  loop(x, y) // for every column left of the diagonal
509  {
510  const complex<double> sx(src.real(x, 0), src.imag(x, 0));
511  const complex<double> sy(src.real(y, 0), src.imag(y, 0));
512  const complex<double> prod = sc*sx*sy;
513  real(x, y) = real(y, x) = prod.real();
514  imag(x, y) = imag(y, x) = prod.imag();
515  }
516 
517  // diagonal elements...
518  //
519  loop(y, M)
520  {
521  const complex<double> sd(src.real(y, 0), src.imag(y, 0));
522  const complex<double> prod = sc*sd*sd;
523  real(y, y) = prod.real();
524  imag(y, y) = prod.imag();
525  }
526 }

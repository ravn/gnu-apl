87 //-----------------------------------------------------------------------------
88 void
89 Quad_SVx::start_AP(AP_num ap)
90 {
91  if (!uprefs.system_do_svars) // something went wrong
92  {
93  if (uprefs.user_do_svars) // user wanted APs
94  {
95  // user wanted APs, but something went wrong
96  //
97  CERR << "*** Not starting AP << ap because the connecton to "
98  " APserver has failed earlier." << endl;
99  }
100  else // user gave --noSV
101  {
102  CERR << "*** Not starting AP " << ap
103  << " because --noSV (or equivalent) was given." << endl
104  << " That conflicts with the use of âSVxxx functions "
105  "and variables." << endl;
106  }
107  ATTENTION;
108  }
109 
110 const char * dirs[] = { "", "/APs" };
111  enum { dircount = sizeof(dirs) / sizeof(*dirs) };
112 
113 const AP_num own_ID = ProcessorID::get_own_ID();
114 const AP_num par_ID = ProcessorID::get_parent_ID();
115 
116 const char * verbose = "";
117  Log(LOG_shared_variables) verbose = " -v";
118 
119 char filename[APL_PATH_MAX + 1];
120 
121 bool found_executable = false;
122  for (int d = 0; d < dircount; ++d)
123  {
124  const int slen = snprintf(filename, APL_PATH_MAX,
125  "%s%s/AP%u --id %u --par %u --gra %u --auto%s",
126  LibPaths::get_APL_bin_path(), dirs[d], ap, ap,
127  own_ID, par_ID, verbose);
128  if (slen >= APL_PATH_MAX) filename[APL_PATH_MAX] = 0;
129 
130  found_executable = is_executable(filename);
131  if (found_executable) break;
132  }
133 
134  if (!found_executable)
135  {
136  CERR << "No binary found for AP " << ap << " (interpreter path = "
137  << LibPaths::get_APL_bin_path() << ")" << endl;
138  return;
139  }
140 
141 FILE * fp = popen(filename, "r");
142  if (fp == 0)
143  {
144  CERR << "popen(" << filename << " failed: " << strerror(errno)
145  << endl;
146  return;
147  }
148 
149  for (int cc; (cc = getc(fp)) != EOF;) CERR << char(cc);
150  CERR << endl;
151 
152  pclose(fp);
153 
154  usleep(100000); // give new AP time to register with APserver
155 }

157 Token
158 Quad_SVC::eval_AB(Value_P A, Value_P B)
159 {
160  if (A->get_rank() > 2) RANK_ERROR;
161  if (B->get_rank() > 2) RANK_ERROR;
162 
163 const ShapeItem var_count = B->get_rows();
164  if (A->get_rows() != var_count) LENGTH_ERROR;
165  if (A->get_rank() > 0 && A->get_cols() != 4) LENGTH_ERROR;
166 const UCS_string_vector vars(B.getref(), false);
167 
168 const Cell * cA = &A->get_ravel(0);
169 
170 Shape sh_Z;
171  if (var_count > 1) sh_Z.add_shape_item(var_count);
172  sh_Z.add_shape_item(4);
173 Value_P Z(sh_Z, LOC);
174 
175  loop(z, var_count)
176  {
177  // construct control value
178  //
179  int ctl = 0;
180  if (A->is_scalar())
181  {
182  const bool val = A->get_ravel(0).get_near_bool();
183  ctl = val ? ALL_SVAR_CONTROLS : NO_SVAR_CONTROL;
184  }
185  else if (A->is_vector())
186  {
187  if (A->get_ravel(0).get_near_bool()) ctl |= SET_BY_1;
188  if (A->get_ravel(1).get_near_bool()) ctl |= SET_BY_2;
189  if (A->get_ravel(2).get_near_bool()) ctl |= USE_BY_1;
190  if (A->get_ravel(3).get_near_bool()) ctl |= USE_BY_2;
191  }
192  else // matrix
193  {
194  if (cA++->get_near_bool()) ctl |= SET_BY_1;
195  if (cA++->get_near_bool()) ctl |= SET_BY_2;
196  if (cA++->get_near_bool()) ctl |= USE_BY_1;
197  if (cA++->get_near_bool()) ctl |= USE_BY_2;
198  }
199 
200  // set control.
201  //
202  Symbol * sym = Workspace::lookup_existing_symbol(vars[z]);
203  if (sym == 0) Error::throw_symbol_error(vars[z], LOC);
204 
205  const SV_key key = sym->get_SV_key();
206  Svar_DB::set_control(key, Svar_Control(ctl));
207  ctl = Svar_DB::get_control(key);
208 
209  new (Z->next_ravel()) IntCell(ctl & SET_BY_1 ? 1 : 0);
210  new (Z->next_ravel()) IntCell(ctl & SET_BY_2 ? 1 : 0);
211  new (Z->next_ravel()) IntCell(ctl & USE_BY_1 ? 1 : 0);
212  new (Z->next_ravel()) IntCell(ctl & USE_BY_2 ? 1 : 0);
213  }
214 
215  Z->check_value(LOC);
216  return Token(TOK_APL_VALUE1, Z);
217 }

256 void
257 Quad_SVE::assign(Value_P value, bool clone, const char * loc)
258 {
259  // âSVEâX merely remembers the timer expiration time. The timer is only
260  // started if âSVE is referenced (in Quad_SVE::get_apl_value())
261  //
262  if (!value->is_scalar()) RANK_ERROR;
263 
264 const APL_time_us duration = 1000000 * value->get_ravel(0).get_real_value();
265  if (duration < 0) DOMAIN_ERROR;
266 
267  if (duration == 0.0)
268  {
269  // âSVEâ0 causes get_apl_value() to return immediately. We achieve
270  // that by setting the end time to the past.
271  //
272  timer_end = now() - 1; // end time in the past
273  return;
274  }
275 
276  // set timer end
277  //
278  timer_end = now() + duration;
279 }
280 //-----------------------------------------------------------------------------
281 Value_P
282 Quad_SVE::get_apl_value() const
283 {
284  for (;;)
285  {
286  Svar_event events = SVE_NO_EVENTS;
287  SV_key key = Svar_DB::get_events(events, ProcessorID::get_id());
288 
289  if (key || (events != SVE_NO_EVENTS)) break; // something happend
290 
291  if (timer_end < now()) // timer expired: return 0
292  {
293  Svar_DB::clear_all_events(ProcessorID::get_id());
294  return IntScalar(0, LOC);
295  }
296 
297  usleep(50000); // wait 50 ms
298  }
299 
300  // referencing âSVE always clears all events
301  //
302  Svar_DB::clear_all_events(ProcessorID::get_id());
303 
304  // we have got an event; return remaining time.
305  //
306 const APL_Float remaining = timer_end - now();
307 
308  if (remaining < 0.0) return IntScalar(0, LOC);
309 
310 Value_P Z(LOC);
311  new (Z->next_ravel()) FloatCell(0.000001 * remaining);
312  return Z;
313 }

318 Token
319 Quad_SVO::eval_AB(Value_P A, Value_P B)
320 {
321  if (A->get_rank() > 1) RANK_ERROR;
322  if (B->get_rank() > 2) RANK_ERROR;
323 
324 const ShapeItem var_count = B->get_rows();
325  if (A->get_rank() == 1 && A->element_count() != var_count) LENGTH_ERROR;
326 
327 const UCS_string_vector apl_vars(B.getref(), false);
328 const UCS_string_vector surrogates(B.getref(), true);
329 
330 
331 Shape sh_Z;
332  if (var_count > 1) sh_Z.add_shape_item(var_count);
333 Value_P Z(sh_Z, LOC);
334 
335  loop(z, var_count)
336  {
337  ShapeItem a = A->is_scalar() ? 0 : z;
338  uint32_t vname[MAX_SVAR_NAMELEN + 1];
339  const int vlen = surrogates[z].size();
340  if (vlen > MAX_SVAR_NAMELEN) DOMAIN_ERROR;
341  bool bad_name = false;
342 
343  loop(v, vlen)
344  {
345  const Unicode uni = surrogates[z][v];
346  if (!Avec::is_symbol_char(uni)) bad_name = true;
347  vname[v] = uni;
348  }
349  vname[vlen] = 0;
350 
351  if (bad_name || vlen == 0)
352  {
353  new (Z->next_ravel()) IntCell(NO_COUPLING);
354  continue;
355  }
356 
357  // make sure current name class allows âSVO
358  //
359  Symbol * sym = Workspace::lookup_symbol(apl_vars[z]);
360  assert(sym);
361 
362  const AP_num proc = AP_num(A->get_ravel(a).get_int_value());
363 
364  ValueStackItem * vsp = sym->top_of_stack();
365  Assert(vsp);
366 
367  switch(sym->get_nc())
368  {
369  case NC_UNUSED_USER_NAME:
370  case NC_VARIABLE:
371  break;
372 
373  case NC_SHARED_VAR:
374  {
375  // CERR << "Shared variable " << apl_vars[z]
376  // << " is already shared" << endl;
377 
378  const SV_key key = sym->get_SV_key();
379  new (Z->next_ravel()) IntCell(Svar_DB::get_coupling(key));
380  }
381  continue; // next z
382 
383  default: // function etc.
384  Q1(sym->get_nc()) DOMAIN_ERROR;
385  }
386 
387  SV_Coupling coupling = NO_COUPLING;
388  const SV_key key = share_one_variable(proc, vname, coupling);
389  if (coupling >= SV_OFFERED) // offered or coupled
390  {
391  sym->share_var(key);
392  }
393 
394  new (Z->next_ravel()) IntCell(coupling);
395  }
396 
397  Z->check_value(LOC);
398  return Token(TOK_APL_VALUE1, Z);
399 }
400 //-----------------------------------------------------------------------------
401 SV_key
402 Quad_SVO::share_one_variable(AP_num to_ap, const uint32_t * vname,
403  SV_Coupling & coupling)
404 {
405  // if to_ap is a (dependent) AP then start it if necessary...
406  //
407 AP_num3 to_proc(to_ap, ProcessorID::get_own_ID(), ProcessorID::get_parent_ID());
408 bool auto_started = false;
409 
410  if (to_ap && to_ap < AP_FIRST_USER) // to_proc is an AP
411  {
412  if (!Svar_DB::is_registered_id(to_proc))
413  {
414  start_AP(to_ap);
415  auto_started = true;
416  }
417  }
418  else // to_proc is an independent processor
419  {
420  new (&to_proc) AP_num3(to_ap);
421  }
422 
423 Svar_partner from(ProcessorID::get_id(), NO_TCP_SOCKET);
424 
425 const SV_key key = Svar_DB::match_or_make(vname, to_proc, from);
426  coupling = Svar_DB::get_coupling(key);
427 
428  if (coupling == SV_COUPLED)
429  {
430  // if an AP was started automatically, then coupling could be
431  // SV_COUPLED or SV_OFFERED, depending on how fast the AP did start
432  // up. We pretend here that it started slowly so that auto-started
433  // APs always return SV_OFFERED
434  //
435  if (auto_started) coupling = SV_OFFERED;
436 
437  return key;
438  }
439 
440  // if something went wrong, complain.
441  //
442  if (coupling == NO_COUPLING)
443  {
444  Log(LOG_shared_variables)
445  CERR << "Svar_DB::match_or_store() returned unexpected 0" << endl;
446  return 0;
447  }
448 
449  Assert(coupling == SV_OFFERED);
450 
451  return key;
452 }

518 //-----------------------------------------------------------------------------
519 Value_P
520 Quad_SVQ::get_processors()
521 {
522  // shared variable processors are identified by one of the following:
523  //
524  // 1. auxiliary processors that can be started on demand
525  // (e.g. AP100, AP210). These APs are collocated with the apl binary,
526  // either in the same directory or in the 'APs' subdirectory,
527  //
528  // 2. running processors that have offered variables in Svar_DB.
529  //
530 std::vector<AP_num> processors;
531 
532  // case 1...
533  //
534 const char * dirs[] = { "", "/APs" };
535  enum { dircount = sizeof(dirs) / sizeof(*dirs) };
536 
537  for (int d = 0; d < dircount; ++d)
538  {
539  char dirname[APL_PATH_MAX + 1];
540  int slen = snprintf(dirname, APL_PATH_MAX, "%s%s", LibPaths::get_APL_bin_path(),
541  dirs[d]);
542  if (slen >= APL_PATH_MAX) dirname[APL_PATH_MAX] = 0;
543 
544  dirname[APL_PATH_MAX] = 0;
545  DIR * dir = opendir(dirname);
546 
547  // the APs directory only exists below the src directory (i.e when
548  // apl is started locally without installing it. We do not complain
549  // if the APs directory does not exist.
550  //
551  if (dir == 0)
552  {
553  if (*dirs[d] == 0)
554  {
555  CERR << "Could not open directory " << dirname << " : "
556  << strerror(errno) << " at " << LOC << endl;
557  }
558  continue;
559  }
560 
561  for (;;) // scan files in directory
562  {
563  dirent * entry = readdir(dir);
564  if (entry == 0) break; // directory done
565 
566 #ifdef _DIRENT_HAVE_D_TYPE
567  if (entry->d_type != DT_REG) continue; // not a regular file
568 #endif
569 
570  char filename[APL_PATH_MAX + 1];
571  int slen = snprintf(filename, APL_PATH_MAX, "%s/%s",
572  dirname, entry->d_name);
573  if (slen >= APL_PATH_MAX) filename[APL_PATH_MAX] = 0;
574 
575  if (!is_executable(filename)) continue;
576 
577  int apnum;
578  if (sscanf(entry->d_name, "AP%u", &apnum) != 1) continue;
579 
580  char expected[APL_PATH_MAX + 1];
581  slen = snprintf(expected, sizeof(expected), "AP%u", apnum);
582  if (slen >= APL_PATH_MAX) expected[APL_PATH_MAX] = 0;
583  if (strcmp(entry->d_name, expected)) continue;
584 
585  processors.push_back(AP_num(apnum));
586  }
587 
588  closedir(dir);
589  }
590 
591  // case 2...
592  //
593  Svar_DB::get_offering_processors(ProcessorID::get_own_ID(), processors);
594 
595  // sort and remove duplicates
596  //
597 std::vector<int32_t> sorted;
598  while (processors.size())
599  {
600  // find smallest
601  //
602  int smallest = processors[0];
603  for (size_t s = 1; s < processors.size(); ++s)
604  if (smallest > processors[s]) smallest = processors[s];
605 
606  // add smallest to sorted
607  //
608  sorted.push_back(smallest);
609 
610  // remove smallest from processors
611  //
612  for (size_t s = 0; s < processors.size();)
613  {
614  if (processors[s] != smallest)
615  {
616  ++s;
617  }
618  else
619  {
620  processors[s] = processors.back();
621  processors.pop_back();
622  }
623  }
624  }
625 
626 Value_P Z(sorted.size(), LOC);
627  loop(z, sorted.size()) new (Z->next_ravel()) IntCell(sorted[z]);
628  return Z;
629 }
630 //-----------------------------------------------------------------------------
631 Value_P
632 Quad_SVQ::get_variables(AP_num proc)
633 {
634 std::vector<uint32_t> varnames;
635  Svar_DB::get_offered_variables(ProcessorID::get_own_ID(), proc, varnames);
636 
637  // varnames is a sequence of 0-terminated Unicodes
638  //
639 std::vector<int> var_lengths; // including terminating 0
640 int last_zero = -1;
641  loop(v, varnames.size())
642  {
643  if (varnames[v] == 0)
644  {
645  var_lengths.push_back(v + 1 - last_zero);
646  last_zero = v;
647  }
648  }
649 
650 const int count = var_lengths.size();
651 ShapeItem max_len = 0;
652  loop(z, count)
653  if (max_len < (var_lengths[z] - 1)) max_len = var_lengths[z] - 1;
654 
655 const Shape shZ(count, max_len);
656 Value_P Z(shZ, LOC);
657 int v = 0;
658 
659  loop(z, count)
660  loop(c, max_len)
661  {
662  if (c < var_lengths[z])
663  {
664  if (varnames[v] == 0) ++v;
665  new (Z->next_ravel()) CharCell(Unicode(varnames[v++]));
666  }
667  else
668  {
669  new (Z->next_ravel()) CharCell(UNI_ASCII_SPACE);
670  }
671  }
672 
673  Z->set_default_Spc(); // prototype: character
674  return Z;
675 }

679 {

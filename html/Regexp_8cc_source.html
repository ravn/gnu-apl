25 static const PCRE2_UCHAR32 *
26 ucs_to_codepoints(const UCS_string &string)
27 {
28  int size = string.size();
29  PCRE2_UCHAR32 *buf = new PCRE2_UCHAR32[size];
30  PCRE2_UCHAR32 *p = buf;
31  UCS_string::iterator i = string.begin();
32  while(i.more()) {
33  *p++ = i.next();
34  }
35  return buf;
36 }

38 RegexpMatch::RegexpMatch(pcre2_code * code, const UCS_string & B,
39  PCRE2_SIZE start)
40  : matched_B(B)
41 {
42  match_data = pcre2_match_data_create_from_pattern_32(code, NULL);
43  match_result = pcre2_match_32(code, B.raw<PCRE2_UCHAR32>(), B.size(),
44  start, 0, match_data, NULL);
45  if (match_result == 0)
46  {
47  MORE_ERROR() << "Match buffer too small";
48  ovector_count = 0;
49  FIXME;
50  }
51  else if (match_result > 0)
52  {
53  ovector = pcre2_get_ovector_pointer_32(match_data);
54  ovector_count = pcre2_get_ovector_count_32(match_data);
55  }
56  else // match_result < 0
57  {
58  MORE_ERROR() << "libpcre error: " << Regexp::pcre_error(match_result);
59  ovector_count = 0;
60  ovector = 0;
61  }
62 }

95 Regexp::Regexp(const UCS_string & pattern, int flags)
96 {
97 const PCRE2_UCHAR32 * pattern_ucs = ucs_to_codepoints(pattern);
98 
99 int error_code = 0;
100 PCRE2_SIZE error_offset = -1;
101 
102  code = pcre2_compile_32(pattern_ucs, pattern.size(),
103  PCRE2_UTF | PCRE2_UCP | flags, &error_code,
104  &error_offset, 0);
105  delete[] pattern_ucs;
106 
107  if (code) return; // OK
108 
109  MORE_ERROR() <<
110 "in (say) Z←A ⎕RE B: error when compiling the regular expression A.\n"
111 " The cause of the error was: " << pcre_error(error_code) << ".\n"
112 " The regular expression A was: '" << pattern << "'\n"
113 " The error was detected at " << error_offset << "↓A.\n"
114 " the offending rest (" << error_offset << "↓A) of A was: '"
115  << UCS_string(pattern, error_offset, pattern.size() - error_offset) << "'";
116 
117  DOMAIN_ERROR;
118 }

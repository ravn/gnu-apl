38 XML_node::XML_node(XML_node * anchor, const UCS_string & string_B,
39  ShapeItem spos, ShapeItem slen)
40  : src(string_B),
41  src_pos(spos),
42  src_len(slen),
43  next(this),
44  prev(this),
45  err_loc(0),
46  level(-1), // set in translate()
47  position(-1) // set in collect()
48 {
49  if (!anchor) return;
50 
51  // insert this node at the end of the list that starts at anchor
52  //
53  prev = anchor->prev; // the (old) end of the list
54  anchor->prev = this; // make this node the new end of the list
55 
56  next = anchor; // this is the end, so this->next is the anchor
57  prev->next = this; // and this is the next of the old end
58 
59  // figure the node type
60  //
61  node_type = NT_error;
62  if (src[src_pos] != '<')
63  {
64  node_type = NT_text;
65  return;
66  }
67 
68  if (src_len < 3)
69  {
70  err_loc = LOC;
71  MORE_ERROR() << "âXML: Tag too short: "
72  << UCS_string(src, src_pos, src_len);
73  return;
74  }
75 
76  // <...
77  //
78  if (src[src_pos + src_len - 1] != '>')
79  {
80  err_loc = LOC;
81  MORE_ERROR() << "âXML: No tag end: "
82  << UCS_string(src, src_pos, src_len);
83  return;
84  }
85 
86  // <...>
87  //
88 const Unicode ucs1 = src[src_pos + 1]; // the character after <
89  if (ucs1 == '!') // XML declaration or comment <! ... >
90  {
91  if (src_len < 7) // at least <!---->
92  {
93  err_loc = LOC;
94  MORE_ERROR() << "âXML: comment not properly terminated: "
95  << UCS_string(src, src_pos, src_len);
96  return;
97  }
98 
99  if (src[src_pos + 2] =='D' &&
100  src[src_pos + 3] =='O' &&
101  src[src_pos + 4] =='C' &&
102  src[src_pos + 5] =='T' &&
103  src[src_pos + 6] =='Y' &&
104  src[src_pos + 7] =='P' &&
105  src[src_pos + 8] =='E') // <!DOCTYPE ...
106  {
107  node_type = NT_doctype;
108  return;
109  }
110 
111  if (src[src_pos + 2] !='-' ||
112  src[src_pos + 3] !='-') // not <!--
113  {
114  err_loc = LOC;
115  MORE_ERROR() << "âXML: bad comment start (expecting <!--) : "
116  << UCS_string(src, src_pos, src_len);
117  return;
118  }
119 
120  else if (src[src_pos + src_len - 2] != '-' ||
121  src[src_pos + src_len - 3] != '-')
122  {
123  err_loc = LOC;
124  MORE_ERROR() << "âXML: bad comment end (expecting --> ): "
125  << UCS_string(src, src_pos, src_len);
126  return;
127  }
128  node_type = NT_comment;
129  }
130  else if (ucs1 == '?') // <? ... >
131  {
132  if (src_len < 7) // at least <?xml?>
133  {
134  err_loc = LOC;
135  MORE_ERROR() << "âXML: processing instruction not terminated: "
136  << UCS_string(src, src_pos, src_len);
137  return;
138  }
139  else if (src[src_pos + 2] !='x' ||
140  src[src_pos + 3] !='m' ||
141  src[src_pos + 4] !='l') // not <?xml
142  {
143  err_loc = LOC;
144  MORE_ERROR() << "âXML: bad declaration start (expecting <?xml): "
145  << UCS_string(src, src_pos, src_len);
146  return;
147  }
148 
149  else if (src[src_pos + src_len - 2] != '?')
150  {
151  err_loc = LOC;
152  MORE_ERROR() << "âXML: bad declaration end (expecting ?>): "
153  << UCS_string(src, src_pos, src_len);
154  return;
155  }
156  node_type = NT_declaration;
157  }
158  else if (ucs1 == '/') // </ ... >
159  {
160  if (!is_first_name_char(src[src_pos + 2]))
161  {
162  err_loc = LOC;
163  MORE_ERROR() << "âXML: bad tag name in end tag : "
164  << UCS_string(src, src_pos, src_len);
165  return;
166  }
167  node_type = NT_end_tag;
168  }
169  else if (src[src_pos + src_len - 2] == '/') // leaf tag
170  {
171  if (!is_first_name_char(ucs1))
172  {
173  err_loc = LOC;
174  MORE_ERROR() << "âXML: bad tag name in empty (-leaf) tag: "
175  << UCS_string(src, src_pos, src_len);
176  return;
177  }
178  node_type = NT_leaf_tag;
179  }
180  else // start tag
181  {
182  if (!is_first_name_char(ucs1))
183  {
184  err_loc = LOC;
185  MORE_ERROR() << "âXML: bad tag name in start tag: "
186  << UCS_string(src, src_pos, src_len);
187  return;
188  }
189  node_type = NT_start_tag;
190  }
191 
192  Assert(node_type != NT_error);
193 }

273 //-----------------------------------------------------------------------------
274 void
275 Quad_XML::add_sorted_entities(vector<const UCS_string *> & entities,
276  const Value & B)
277 {
278  Assert(B.is_structured());
279 
280 std::vector<ShapeItem> member_indices;
281  B.sorted_members(member_indices, /* filter */ 0);
282 
283 UCS_string start_tag_name; // for constructing the corresponding end tag
284 UCS_string start_tag; // the start tag
285 bool start_tag_open = false;
286 bool tag_open = false;
287 
288  loop(m, member_indices.size())
289  {
290  ShapeItem member_pos; // the position in the XML file
291  Unicode category;
292  UCS_string name;
293  const Value & member_name = B.get_cravel(2*member_indices[m])
294  .get_pointer_value().getref();
295 
296  split_name(&category, &member_pos, &name, member_name);
297 
298  const Value & member_data = B.get_cravel(2*member_indices[m] + 1)
299  .get_pointer_value().getref();
300 
301  // one tag; the member_indices items occur in the following
302  // order (also below):
303  //
304  // â1 start tag name
305  // âNNNattr attribute N
306  // âNNN and _NNN declarations, unstructured text and subnodes
307  //
308  if (category == UNI_DELTA_UNDERBAR)
309  {
310  if (name.size() == 0) // start of next start tag
311  {
312  if (start_tag_open) // previous tag still open
313  {
314  start_tag.append_UTF8("/>");
315  entities.push_back(new UCS_string(start_tag));
316  start_tag_open = false;
317  }
318  else if (tag_open)
319  {
320  DOMAIN_ERROR;
321  }
322 
323  start_tag_name = UCS_string(member_data);
324  start_tag.clear();
325  start_tag += UNI_LESS; // <
326  start_tag.append(start_tag_name);
327  start_tag_open = true;
328  tag_open = true;
329  }
330  else // tag attribute
331  {
332  if (!tag_open)
333  {
334  DOMAIN_ERROR;
335  }
336 
337  start_tag += UNI_SPACE;
338  start_tag.append(UCS_string(name));
339  start_tag += UNI_EQUAL;
340  start_tag.append(XML_node::denormalize_attribute_value(
341  UCS_string(member_data), true));
342  }
343  }
344  else // â or _
345  {
346  if (start_tag_open) // start tag still open
347  {
348  start_tag.append_UTF8(">");
349  entities.push_back(new UCS_string(start_tag));
350  start_tag_open = false;
351  }
352 
353  if (category == UNI_DELTA)
354  {
355  entities.push_back(new UCS_string(member_data));
356  }
357  else if (category == UNI_UNDERSCORE)
358  {
359  add_sorted_entities(entities, member_data);
360  }
361  else
362  {
363  DOMAIN_ERROR;
364  }
365  }
366  }
367 
368  // all items processed.
369  //
370  if (tag_open)
371  {
372  if (start_tag_open) // <TAG att=val att=val... : close tag
373  {
374  start_tag.append_UTF8("/>");
375  }
376  else // <TAG> ... : add end tag
377  {
378  start_tag.clear();
379  start_tag.append_UTF8("</");
380  start_tag.append(start_tag_name);
381  start_tag += UNI_GREATER;
382  }
383  entities.push_back(new UCS_string(start_tag));
384  }
385 }
386 //-----------------------------------------------------------------------------
387 Value_P
388 Quad_XML::XML_to_APL(const Value & B)
389 {
390 const ShapeItem len_B = B.element_count();
391  if (len_B == 0) LENGTH_ERROR;
392 
393 UCS_string string_B(len_B, UNI_NUL);
394 
395 ShapeItem dest_B = 0;
396  loop(src_B, len_B)
397  {
398  const Unicode uni = B.get_cravel(src_B).get_char_value();
399  if (uni == UNI_CR) // CR
400  {
401  // see XML standard "2.11 End-of-Line Handling"
402  //
403  if (src_B == (len_B - 1)) // CR at the end of the document
404  {
405  string_B[dest_B++] = UNI_LF;
406  continue;
407  }
408 
409  const Unicode uni_1 = B.get_cravel(src_B + 1).get_char_value();
410  if (string_B[uni_1] == UNI_LF) // CR/LF
411  {
412  continue; // discard CR (don't ++dest_B)
413  }
414 
415  else // CR not followed by LF
416  {
417  string_B[dest_B++] = UNI_LF; // replace CR with LF
418  continue;
419  }
420 
421  // not reached
422  //
423  FIXME;
424  }
425 
426  if (XML_node::is_XML_char(uni)) // valid XML character
427  {
428  string_B[dest_B++] = uni;
429  }
430  else // invalid XML character
431  {
432  char uni_cc[20];
433  snprintf(uni_cc, sizeof(uni_cc), "U+%4.4X", uni);
434  MORE_ERROR() << "âXML B: Invalid XML character " << uni_cc
435  << " at B[" << (src_B + Workspace::get_IO()) << "]";
436  DOMAIN_ERROR;
437  }
438  }
439 
440  Assert(dest_B <= len_B);
441  string_B.resize(dest_B);
442 
443 XML_node anchor(0, string_B, 0, 0);
444 XML_node garbage(0, string_B, 0, 0);
445 ShapeItem text_start = 0;
446 Value_P Z = EmptyStruct(LOC);
447 bool error = true;
448 
449  // create a (flat) doubly-linked list, starting at anchor, and containing
450  // all XML_nodes of B.
451  //
452  loop(b, len_B)
453  {
454  const Unicode uni_b = B.get_cravel(b).get_char_value();
455  if (uni_b == '<')
456  {
457  if (text_start != b) // some text before the '<'
458  {
459  const ShapeItem text_len = b - text_start;
460  new XML_node(&anchor, string_B, text_start, text_len);
461  }
462 
463  const ShapeItem taglen = XML_node::get_taglen(string_B, b);
464  if (taglen == -1) goto cleanup;
465 
466  new XML_node(&anchor, string_B, b, taglen);
467  b += taglen;
468  text_start = b;
469  }
470  }
471 
472  if (text_start != len_B) // trailing unstructured text
473  new XML_node(&anchor, string_B, text_start, len_B - text_start);
474 
475  error = XML_node::translate(anchor, garbage);
476  if (!error) error = XML_node::collect(anchor, garbage, Z.get());
477 
478 cleanup:
479  while (anchor.get_next() != &anchor) delete anchor.get_next()->unlink();
480  while (garbage.get_next() != &garbage) delete garbage.get_next()->unlink();
481 
482  if (error) DOMAIN_ERROR;
483  return Z;
484 }
485 //-----------------------------------------------------------------------------
486 ShapeItem
487 XML_node::get_taglen(const UCS_string & string_B, ShapeItem offset)
488 {
489  Assert(string_B[offset] == '<');
490 
491 bool is_doctype = true;
492 const char * cdoc = "!DOCTYPE";
493 bool inside_dq = false; // inside "..."
494 bool inside_sq = false; // inside '...'
495 
496  for (ShapeItem pos = offset + 1; pos < string_B.size();)
497  {
498  const Unicode uni = string_B[pos++];
499 
500  if (*cdoc && uni != *cdoc++) is_doctype = false;
501 
502  if (uni == '\'' && !inside_dq) inside_sq = !inside_sq;
503  else if (uni == '"' && !inside_sq) inside_dq = !inside_dq;
504  else if (uni == '<') // invalid tag start inside tag attribute
505  {
506  if (inside_sq || inside_dq) MORE_ERROR() <<
507  "âXML B: Unescaped '<' in attribute value: '"
508  << UCS_string(string_B, offset, 20) << "'...";
509  else MORE_ERROR() <<
510  "âXML B: Unescaped '<' in attribute name: '"
511  << UCS_string(string_B, offset, 20) << "'...";
512  return -1;
513  }
514  else if (uni == '>' && !(inside_sq || inside_dq)) // end of tag
515  {
516  return pos - offset;
517  }
518 
519  if (is_doctype && !*cdoc) // rarely: <!DOCTYPE...
520  {
521  size_t level = 1;
522  for (; pos < string_B.size();)
523  {
524  const Unicode uni = string_B[pos++];
525  if (uni == '<') ++level;
526  else if (uni == '>')
527  {
528  --level;
529  if (level == 0)
530  {
531  return pos - offset;
532  }
533  }
534  }
535  }
536  }
537 
538 const int len = string_B.size() - offset;
539 const int len1 = len > 50 ? 50 : len;
540 UCS_string where(string_B, offset, len1);
541  if (len < 50) MORE_ERROR() << "No tag end found in: " << where;
542  else MORE_ERROR() << "No tag end found in: " << where;
543  return -1;
544 }
545 //-----------------------------------------------------------------------------
546 bool
547 XML_node::translate(XML_node & anchor, XML_node & garbage)
548 {
549 int level = 0;
550 ShapeItem count = 0;
551 
552  // reduce text and leaf nodes...
553  //
554  for (XML_node * node = anchor.get_next(); node != &anchor;
555  node = node->get_next())
556  {
557  node->level = level;
558  switch(node->get_node_type())
559  {
560  case NT_text:
561  case NT_comment:
562  case NT_declaration:
563  case NT_doctype:
564  if (node->get_src_len()) // not empty
565  {
566  const UCS_string item = node->get_item();
567  node->APL_value = Value_P(item, LOC);
568  }
569  else // empty string
570  {
571  node = node->get_prev(); // move back
572  garbage.append(node->get_next()->unlink());
573  }
574  break;
575 
576  case NT_start_tag:
577  ++level;
578  /* fall through */
579  case NT_leaf_tag:
580  if (node->parse_tag()) return true;
581  break;
582 
583  case NT_end_tag:
584  node->level = --level;
585  break;
586 
587  case NT_error:
588  return true;
589  }
590 
591  ++count;
592  }
593 
594  return false; // OK
595 }
596 //-----------------------------------------------------------------------------
597 bool
598 XML_node::collect(XML_node & anchor, XML_node & garbage, Value * Z)
599 {
600 XML_node * root = 0;
601 
602 vector<XML_node *> stack; // a stack of start tags
603 vector<size_t> pos_stack; // a stack of node positions
604 
605  // set the positions (before removing any nodes)
606  //
607  {
608  size_t position = Workspace::get_IO();
609  for (XML_node * n = anchor.get_next(); n != &anchor; n = n->get_next())
610  {
611  n->position = position++;
612  }
613  }
614 
615  for (XML_node * node = anchor.get_next(); node != &anchor;
616  node = node->get_next())
617  {
618  size_t position = node->position;
619  switch(node->get_node_type())
620  {
621  case NT_declaration:
622  if (stack.size()) // only allowed at top-level
623  {
624  MORE_ERROR() <<
625  "âXML: XML declaration below top-level: "
626  << node->get_item();
627  return true;
628  }
629 
630  add_member(Z, UNI_DELTA, "declaration", position,
631  node->APL_value.get());
632  node = node->get_prev(); // move back
633  garbage.append(node->get_next()->unlink());
634  break;
635 
636  case NT_doctype:
637  if (stack.size()) // only allowed at top-level
638  {
639  MORE_ERROR() <<
640  "âXML: !DOCTYPE below top-level: "
641  << node->get_item();
642  return true;
643  }
644 
645  add_member(Z, UNI_DELTA, "doctype", position,
646  node->APL_value.get());
647  node = node->get_prev(); // move back
648  garbage.append(node->get_next()->unlink());
649  break;
650 
651  case NT_comment:
652  if (stack.size()) continue; // only top-level comments
653 
654  add_member(Z, UNI_DELTA, "comment", position,
655  node->APL_value.get());
656  node = node->get_prev();
657  garbage.append(node->get_next()->unlink());
658  break;
659 
660  case NT_text:
661  if (stack.size()) continue; // only top-level texts
662  add_member(Z, UNI_DELTA, "text", position,
663  node->APL_value.get());
664  node = node->get_prev();
665  garbage.append(node->get_next()->unlink());
666  break;
667 
668  case NT_start_tag:
669  {
670  if (root == 0) root = node; // remember the root
671  else if (stack.size() == 0) // subsequent root
672  {
673  MORE_ERROR() << "âXML: more than one root";
674  return true;
675  }
676  Assert(stack.size() == size_t(node->level));
677  stack.push_back(node);
678  pos_stack.push_back(position);
679  position = Workspace::get_IO(); // âIO is â
680  }
681  break;
682 
683  case NT_end_tag:
684  {
685  if (stack.size() == 0) // no start tag
686  {
687  MORE_ERROR() << "âXML: end tag " << node->get_item()
688  << " without start tag";
689  return true;
690  }
691  XML_node * start = stack.back();
692  stack.pop_back();
693 
694  if (!start->matches(node))
695  {
696  MORE_ERROR() << "âXML: end tag " << node->get_item()
697  << " does not match start tag "
698  << start->get_item();
699  return true;
700  }
701 
702  position = pos_stack.back() + 1;
703  pos_stack.pop_back();
704  merge_range(*start, *node, garbage);
705  node = start;
706  }
707  break;
708 
709  case NT_leaf_tag:
710  if (root == 0) root = node; // remember the root
711  break;
712 
713  case NT_error: FIXME;
714  }
715  }
716 
717  if (root)
718  {
719  UTF8_string root_utf(root->get_tagname());
720  add_member(Z, UNI_UNDERSCORE, root_utf.c_str(), -1,
721  anchor.get_next()->APL_value.get());
722  }
723  else
724  {
725  MORE_ERROR() << "âXML B: no XML root element in B";
726  return true;
727  }
728 
729  return false; // OK
730 }

732 void

881 //-----------------------------------------------------------------------------
882 bool
883 XML_node::normalize_attribute_value(UCS_string & attval)
884 {
885  // XML standard "4.6 Predefined Entities"
886 
887 ShapeItem dest = 0;
888 
889 struct _mapping
890 {
891  const char * entity;
892  int string_len;
893  Unicode replacement;
894 };
895 
896 const _mapping predefined_entities[] =
897 { { "&lt;", 4, UNI_LESS },
898  { "&gt;", 4, UNI_GREATER },
899  { "&amp;", 5, UNI_AMPERSAND },
900  { "&apos;", 6, UNI_SINGLE_QUOTE },
901  { "&quot;", 6, UNI_DOUBLE_QUOTE }
902 };
903 
904 enum { PREDEFINED_COUNT = sizeof(predefined_entities)
905  / sizeof(*predefined_entities) };
906 
907  loop(src, attval.size())
908  {
909  const Unicode uni = attval[src];
910  if (uni < UNI_SPACE)
911  {
912  attval[dest++] = UNI_SPACE;
913  continue;
914  }
915 
916  if (uni != UNI_AMPERSAND) // normal char
917  {
918  attval[dest++] = uni;
919  continue;
920  }
921 
922  // &#xxx
923  //
924  if ((src + 2) < attval.size() && attval[src + 1] == '#')
925  {
926  src += 2; // skip &#
927  int number = 0;
928  Unicode bad_char = UNI_NUL;
929  if (attval[src] == 'x') // hex value &#xHH
930  {
931  ++src; // skip 'x'
932  while (src < attval.size() && !bad_char)
933  {
934  const Unicode digit = attval[src++];
935  if (digit == ';') break; // end of hex digits
936  const int digval = Avec::digit_value(digit, true);
937  if (digval == -1) bad_char = digit;
938  else number = 16*number + digval;
939  }
940  }
941  else // decimal value &#DDD
942  {
943  while (src < attval.size())
944  {
945  const Unicode digit = attval[src++];
946  if (digit == ';') break; // end of decimal digits
947  const int digval = Avec::digit_value(digit, false);
948  if (digval == -1) bad_char = digit;
949  else number = 10*number + digval;
950  }
951  }
952 
953  if (bad_char)
954  {
955  MORE_ERROR() << "âXML: bad character '"
956  << UCS_string(1, bad_char)
957  << "' in attribute value '" << attval << "'";
958  return true;
959  }
960  attval[dest++] = Unicode(number);
961  continue;
962  }
963 
964  // &... but not &#... Try predefined entities
965  //
966  loop(p, PREDEFINED_COUNT)
967  {
968  const _mapping & pred = predefined_entities[p];
969  const int len = pred.string_len;
970  if ((src + len) > attval.size()) continue; // too short
971 
972  bool match = true;
973  loop(ll, len)
974  {
975  if (pred.entity[ll] != attval[src + ll]) // mismatch
976  {
977  match = false;
978  break; // loop(ll)
979  }
980  }
981 
982  if (match)
983  {
984  attval[dest++] = pred.replacement;
985  src += len - 1; // except uni
986  break; // loop(p)
987  }
988  }
989  }
990 
991  attval.resize(dest);
992  return false; // OK
993 }

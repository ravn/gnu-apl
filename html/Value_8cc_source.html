443 Value_P
444 Value::get_cellrefs(const char * loc)
445 {
446  /* Create a non-nested (!) (left-) value Z from this value.
447  Z has the same shape and consists entirely of LvalCells that point
448  to the corresponding cells of this value.
449 
450  NOTE that if a cell of this value is PointerCell, then get_cellrefs()
451  will simply point to that cell rather than recursing into its sub-value
452  (which may happen at a later point in time, though).
453 
454  The reson for delaying the recursion into sub-values is that the
455  subsequent processing of Z may erase those sub-values and therefore
456  doing it now may render useless later.
457  */
458 
459 Value_P Z(get_shape(), loc);
460 
461  if (const ShapeItem ec = element_count())
462  {
463  loop(e, ec) Z->next_ravel_Cell(LvalCell(&get_wravel(e), this));
464  }
465  else // prototype
466  {
467  Z->set_ravel_Cell(0, LvalCell(&get_wproto(), this));
468 
469  }
470 
471  Z->check_value(LOC);
472  return Z;
473 }
474 //-----------------------------------------------------------------------------
475 void
476 Value::assign_cellrefs(Value_P new_value)
477 {
478 const ShapeItem dest_count = element_count();
479  if (dest_count == 0) return; // nothing to assign
480 
481 const int dest_incr = (dest_count == 1) ? 0 : 1;
482 
483 const ShapeItem value_count = new_value->nz_element_count();
484 const int src_incr = (new_value->nz_element_count() == 1) ? 0 : 1;
485 
486  if (src_incr && dest_incr && (dest_count != value_count)) LENGTH_ERROR;
487 
488  /* this: a value containing LvalCells and possibly PointerCells.
489 
490  test with:
491 
492  Câ"001011010010010110010100100001000010001101" â ((C='0')/C)â' ' â C
493  Kâ'RED' 'WHITE' 'BLUE' â (âK)â'YELLOW' â K
494  Vâ'v' 'vw' 'vwx' 'vwxyz' â (2âÂ¨V)â5 â 4 âCR V
495 
496  */
497 
498  // consistency check: all items are LvalCells
499  //
500  check_lval_consistency();
501 
502  if (is_scalar() && !new_value->is_scalar())
503  {
504  Cell * const C0 = &get_wscalar();
505  if (!C0->is_lval_cell()) LEFT_SYNTAX_ERROR;
506  const LvalCell * LVC0 = reinterpret_cast<const LvalCell *>(C0);
507  Cell * target = LVC0->get_lval_value(); // can be 0!
508  Value * owner = LVC0->get_cell_owner();
509  if (target) target->release(LOC); // free sub-values etc (if any)
510 
511  new (target) PointerCell(new_value.get(), *owner);
512  return;
513  }
514 
515 const Cell * src = &new_value->get_cfirst();
516  loop(d, dest_count)
517  {
518  Cell & C = get_wravel(d);
519  if (C.is_pointer_cell())
520  {
521  /*
522  NOTE: At this point C is one of the left cell in a selective
523  assignment, say:
524 
525  (f1 f2 ... fn VAR) â new_value.
526 
527  The initial cellrefs (i.e. of VAR) contain no PointerCells
528  (though possibly Lval cells pointing to PointerCells).
529 
530  Therefore C must have been created by one of the fi and then
531  all sub-values C must also be Lval Cells.
532  */
533  Value * sub = C.get_pointer_value().get(); // right-value
534  loop(s, sub->nz_element_count())
535  {
536  Cell * Csub = &sub->get_wravel(s);
537  if (!Csub->is_lval_cell()) LEFT_SYNTAX_ERROR;
538 
539  const LvalCell * LVC =
540  reinterpret_cast<const LvalCell *>(Csub);
541  if (Cell * target = LVC->get_lval_value()) // can be 0!
542  {
543  target->release(LOC); // free sub-values etc (if any)
544 
545  // erase the pointee when overriding a pointer-cell.
546  //
547  Value * owner = LVC->get_cell_owner();
548  target->init(*src, *owner, LOC);
549  }
550  }
551  }
552  else if (C.is_lval_cell())
553  {
554  const LvalCell & LVC = reinterpret_cast<const LvalCell &>(C);
555  if (Cell * target = C.get_lval_value()) // target can be 0!
556  {
557  target->release(LOC); // free sub-values etc (if any)
558 
559  // erase the pointee when overriding a pointer-cell.
560  //
561  Value * owner = LVC.get_cell_owner();
562  target->init(*src, *owner, LOC);
563  }
564  }
565  else LEFT_SYNTAX_ERROR;
566  src += src_incr;
567  }

935 {

937  //

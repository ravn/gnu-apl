102 {
103  if (B->get_rank() > 1) RANK_ERROR;
104  if (B->is_char_array())
105  {
106  const UCS_string gui_filename = B->get_UCS_ravel();
107  const int fd = open_window(gui_filename, /* no CSS */ 0);
108  return Token(TOK_APL_VALUE1, IntScalar(fd, LOC));
109  }
110 
111  if (!B->is_int_scalar())
112  {
113  MORE_ERROR() << "âGTK B expects B to be an integer scalar"
114  " or a text vector";
115  DOMAIN_ERROR;
116  }
117 
118 const int function = B->get_ravel(0).get_int_value();
119  switch(function)
120  {
121  case 0: // list of open fds
122  return Token(TOK_APL_VALUE1, window_list());
123 
124  case 1: // blocking poll for next event
125  case 2: // non-blocking wait for next event
126  poll_all();
127  if (function == 2 && event_queue.size() == 0) // non-blocking
128  return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
129 
130  // blocking
131  //
132  while (event_queue.size() == 0 && !interrupt_is_raised())
133  poll_all();
134 
135  // at this point either event_queue.size() > 0 or an interrupt
136  // was raised
137  //
138  if (event_queue.size() == 0) // hence interrupt was raised
139  {
140  clear_interrupt_raised(LOC);
141  return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
142  }
143 
144  {
145  UCS_string HWF = event_queue[0]; // fd, widget : fun
146  event_queue.erase(event_queue.begin());
147 
148  // split (Unicode)Handle,"widget:function"
149  // into a 3-element APL vector Handle (â"widget") (â"function")
150  //
151  UCS_string_vector args;
152  UCS_string arg;
153  for (ShapeItem j = 1; j < HWF.size(); ++j)
154  {
155  if (HWF[j] == UNI_ASCII_COLON)
156  {
157  args.push_back(arg);
158  arg.clear();
159  }
160  else
161  {
162  arg.append(HWF[j]);
163  }
164  }
165  args.push_back(arg);
166 
167  Value_P Z(1 + args.size(), LOC);
168  new (Z->next_ravel()) IntCell(HWF[0]);
169  loop(a, args.size())
170  {
171  Value_P Za(args[a], LOC);
172  new (Z->next_ravel()) PointerCell(Za.get(), Z.getref());
173  }
174  Z->check_value(LOC);
175  return Token(TOK_APL_VALUE1, Z);
176  }
177 
178  default: MORE_ERROR() << "Invalid function number Bi=" << function
179  << " in âGTK Bi";
180  DOMAIN_ERROR;
181  }
182 
183  MORE_ERROR() << "Unexpected B in âGTK B";
184  DOMAIN_ERROR;

306 Value_P
307 Quad_GTK::read_fd(int fd, int tag)
308 {
309  // tag == -1 indicates a poll for ANY tag (= an event as opposed to a
310  // command response). In that case the poll is non-blocking.
311 
312 char TLV[1000];
313 
314  errno = 0;
315 const ssize_t rx_len = read(fd, TLV, sizeof(TLV));
316 
317  if (rx_len < 8)
318  {
319  MORE_ERROR() << "read() failed in Quad_GTK::read_fd(): "
320  << strerror(errno);
321  DOMAIN_ERROR;
322  }
323 
324 const int TLV_tag = (TLV[0] & 0xFF) << 24 | (TLV[1] & 0xFF) << 16
325  | (TLV[2] & 0xFF) << 8 | (TLV[3] & 0xFF);
326 
327 
328 const int V_len = (TLV[4] & 0xFF) << 24 | (TLV[5] & 0xFF) << 16
329  | (TLV[6] & 0xFF) << 8 | (TLV[7] & 0xFF);
330 
331  Assert(rx_len == (V_len + 8));
332 
333 char * V = TLV + 8;
334  V[V_len] = 0;
335 
336  // 1. check for events from generic_callback()
337  //
338  if (TLV_tag == Event_widget_fun || // "H:button1:clicked"
339  TLV_tag == Event_widget_fun_id_class || // dito + :id:class
340  TLV_tag == Event_toplevel_window_done)
341  {
342  // V is a string of the form H%s:%s:%s:% where H is a placeholder
343  // for the fd over which we have received V.
344  //
345  // replace H by the actual fd and store the result in event_queue.
346  //
347  UTF8_string data_utf; // H:widget:callback
348  loop(v, V_len) data_utf += V[v];
349  UCS_string data_ucs(data_utf);
350  data_ucs[0] = (Unicode(fd));
351 
352  event_queue.push_back(data_ucs);
353 
354  return Value_P(); // i.e. NULL
355  }
356 
357  if (tag == -1) // not waiting for a specific tag
358  {
359  CERR << "*** âGTK: ignoring event: " << V << endl;
360  return Value_P(); // i.e. NULL
361  }
362 
363  if (TLV_tag != tag)
364  {
365  CERR << "Got unexpected tag " << TLV_tag
366  << " when waiting for response " << tag << endl;
367  return Value_P();
368  }
369 
370  // TLV_tag is the expected one. Strip off the response offset
371  // Response_0 (= 2000) from the TLV_tag to get the function number
372  // and return APL vector function,"result-value"
373  //
374 UTF8_string utf;
375  loop(u, V_len) utf += V[u];
376  UCS_string ucs(utf);
377 
378 Value_P Z(1 + ucs.size(), LOC);
379  new (Z->next_ravel()) IntCell(TLV_tag - Response_0);
380  loop(u, ucs.size()) new (Z->next_ravel()) CharCell(ucs[u]);
381  Z->check_value(LOC);
382  return Z;
383 }
384 //-----------------------------------------------------------------------------
385 void
386 Quad_GTK::poll_all()
387 {
388 const int count = open_windows.size();
389 pollfd fds[count];
390  loop(w, count)
391  {
392  fds[w].fd = open_windows[w].fd;
393 #ifdef POLLRDHUP
394  fds[w].events = POLLIN | POLLRDHUP;
395 #else
396  fds[w].events = POLLIN;
397 #endif
398  fds[w].revents = 0;
399  }
400 
401  errno = 0;
402 const int ready = poll(fds, count, 0);
403  if (ready == 0) return;
404  if (ready > 0)
405  {
406  loop(w, count)
407  {
408  if (fds[w].revents) read_fd(fds[w].fd, -1);
409  }
410  }
411  else
412  {
413  if (errno == EINTR) return; // user hits ^C
414  CERR << "*** poll() failed: " << strerror(errno) << endl;
415  }
416 }
417 //-----------------------------------------------------------------------------
418 Value_P
419 Quad_GTK::poll_response(int fd, int tag)
420 {
421 again:
422 
423 pollfd pfd;
424  pfd.fd = fd;
425 #ifdef POLLRDHUP
426  pfd.events = POLLIN | POLLRDHUP;
427 #else
428  pfd.events = POLLIN;
429 #endif
430  pfd.revents = 0;
431 
432 const int ready = poll(&pfd, 1, 0);
433  if (ready == 0)
434  {
435  usleep(100000);
436  goto again;
437  }
438 
439  if (ready > 0 && pfd.revents)
440  {
441  Value_P Z = read_fd(fd, tag);
442  if (!Z) goto again;
443  return Z;
444  }
445 
446  CERR << "*** poll() failed" << endl;
447  DOMAIN_ERROR;
448 }

556 int
557 Quad_GTK::write_TL0(int fd, int tag)
558 {
559 unsigned char TLV[8];
560  memset(TLV, 0, 8);
561  TLV[0] = tag >> 24;
562  TLV[1] = tag >> 16;
563  TLV[2] = tag >> 8;
564  TLV[3] = tag;
565 
566  errno = 0;
567 const size_t tx_len = write(fd, TLV, 8);
568  if (tx_len != 8)
569  {
570  CERR << "write(Tag = " << tag << ") failed in âGTK::write_TL0()"
571  << endl << " tx_len = " << tx_len << endl
572  << " errno says: " << strerror(errno) << endl;
573  return -1;
574  }
575 
576  return 0;
577 }
578 //-----------------------------------------------------------------------------
579 int
580 Quad_GTK::write_TLV(int fd, int tag, const UTF8_string & value)
581 {
582 const int TLV_len = 8 + value.size();
583 unsigned char TLV[TLV_len];
584  TLV[0] = tag >> 24 & 0xFF;
585  TLV[1] = tag >> 16 & 0xFF;
586  TLV[2] = tag >> 8 & 0xFF;
587  TLV[3] = tag & 0xFF;
588  TLV[4] = value.size() >> 24;
589  TLV[5] = value.size() >> 16;
590  TLV[6] = value.size() >> 8;
591  TLV[7] = value.size();
592  loop(s, value.size()) TLV[8 + s] = value[s];
593 
594  if (write(fd, TLV, TLV_len) != TLV_len)
595  {
596  CERR << "write(Tag = " << tag << ") failed in âGTK::write_TLV()";
597  return -1;
598  }
599 
600  return 0;
601 }
602 //-----------------------------------------------------------------------------
603 int
604 Quad_GTK::open_window(const UCS_string & gui_filename,
605  const UCS_string * css_filename)
606 {
607  // locate the Gtk_server. It should live in one of two places:
608  //
609  // 1, for an installed apl: in the same directory as apl, or
610  // 2. during development in subdir Gtk of the src directory
611  //
612 char path1[APL_PATH_MAX + 1 + 8];
613 char path2[APL_PATH_MAX + 1 + 8];
614 const char * bin_dir = LibPaths::get_APL_bin_path();
615 int slen = snprintf(path1, APL_PATH_MAX, "%s/Gtk/Gtk_server", bin_dir);
616  if (slen >= APL_PATH_MAX) path1[APL_PATH_MAX] = 0;
617  slen = snprintf(path2, APL_PATH_MAX, "%s/Gtk_server", bin_dir);
618  if (slen >= APL_PATH_MAX) path2[APL_PATH_MAX] = 0;
619 
620 const char * path = 0;
621  if (!access(path1, X_OK)) path = path1;
622  else if (!access(path2, X_OK)) path = path2;
623  else
624  {
625  MORE_ERROR() << "No Gtk_server found in " << path1
626  << "\nor " << path2;
627  DOMAIN_ERROR;
628  }
629 
630 const char * evars[] = { "DISPLAY", "XAUTHORITY", "XAUTHLOCALHOST" };
631 enum { evar_count = sizeof(evars) / sizeof(*evars) };
632 
633 char * envp[evar_count + 1] = { 0 };
634 int envp_idx = 0;
635  loop(c, evar_count)
636  {
637  const char * var = evars[c];
638  if (char * val = getenv(var))
639  {
640  char * env = new char[strlen(var) + 2 + strlen(val)];
641  sprintf(env, "%s=%s", var, val);
642  envp[envp_idx++] = env;
643  envp[envp_idx] = 0;
644  }
645  }
646 const int fd = Quad_FIO::fun->do_FIO_57(path, envp);
647 
648  // write TLVs 1 and 3 or 1, 2, and 3 to Gtk_server...
649  //
650 UTF8_string gui_utf8(gui_filename);
651  slen = snprintf(path1 + 8, APL_PATH_MAX, "%s", gui_utf8.c_str());
652  if (slen >= APL_PATH_MAX) path1[APL_PATH_MAX] = 0;
653  memset(path1, 0, 8);
654  path1[3] = 1; // tag = 1: gui filename
655  path1[6] = gui_utf8.size() >> 8;
656  path1[7] = gui_utf8.size() & 0xFF;

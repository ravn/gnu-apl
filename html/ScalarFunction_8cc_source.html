115 //-----------------------------------------------------------------------------
116 Token
117 ScalarFunction::eval_scalar_B(Value_P B, prim_f1 fun)
118 {
119 const ShapeItem len_Z = B->element_count();
120  if (len_Z == 0) return eval_fill_B(B);
121 
122 PERFORMANCE_START(start)
123 
124 ErrorCode ec = E_NO_ERROR;
125 Value_P Z = do_scalar_B(ec, B, fun);
126  if (ec != E_NO_ERROR)
127  {
128  loop(a, Thread_context::get_active_core_count())
129  Thread_context::get_context(CoreNumber(a))->joblist_B.cancel_jobs();
130  throw_apl_error(ec, LOC);
131  }
132 
133 PERFORMANCE_END(fs_SCALAR_B, start, Z->nz_element_count());
134 
135  loop(a, Thread_context::get_active_core_count())
136  Assert(Thread_context::get_context(CoreNumber(a))
137  ->joblist_B.get_size() == 0);
138 
139  return Token(TOK_APL_VALUE1, Z);
140 }
141 //-----------------------------------------------------------------------------
142 Value_P
143 ScalarFunction::do_scalar_B(ErrorCode & ec, Value_P B, prim_f1 fun)
144 {
145 Value_P Z(B->get_shape(), LOC);
146 
147  // create a worklist with one item that computes Z. If nested values are
148  // detected while computing Z then jobs for them are added to the worklist.
149  //
150  {
151  PJob_scalar_B job_B(Z.get(), B.getref());
152  Thread_context::get_master().joblist_B.start(job_B, LOC);
153  }
154 
155 #if PARALLEL_ENABLED
156 const bool maybe_parallel = Parallel::run_parallel &&
157  Thread_context::get_active_core_count() > 1;
158 #endif
159 
160  for (;;)
161  {
162  loop(a, Thread_context::get_active_core_count())
163  {
164  job_B = Thread_context::get_context(CoreNumber(a))
165  ->joblist_B.next_job();
166  if (job_B) break;
167  }
168  if (job_B == 0) break; // all jobs done
169 
170 #if PARALLEL_ENABLED
171  if (maybe_parallel && job_B->len_Z > get_monadic_threshold())
172  {
173  // parallel execution...
174  //
175  job_B->fun = this;
176  job_B->fun1 = fun;
177  Thread_context::do_work = PF_scalar_B;
178  Thread_context::M_fork("eval_scalar_B"); // start pool
179  PF_scalar_B(Thread_context::get_master());
180 PERFORMANCE_START(start_M_join)
181  Thread_context::M_join();
182  if (job_B->error != E_NO_ERROR)
183  {
184  ec =job_B->error;
185  return Value_P();
186  }
187 PERFORMANCE_END(fs_M_join_B, start_M_join, 1);
188  }
189  else
190 #endif // PARALLEL_ENABLED
191  {
192  // sequential execution...
193  //
194  loop(z, job_B->len_Z)
195  {
196  const Cell & cell_B = job_B->B_at(z);
197  Cell & cell_Z = job_B->Z_at(z);
198 
199  if (cell_B.is_pointer_cell())
200  {
201  Value_P B1 = cell_B.get_pointer_value();
202  Value_P Z1(B1->get_shape(), LOC);
203  new (&cell_Z) PointerCell(Z1.get(), *job_B->value_Z);
204 
205  PJob_scalar_B j1(Z1.get(), B1.getref());
206  Thread_context::get_master().joblist_B.add_job(j1);
207  }
208  else
209  {
210 PERFORMANCE_START(start_2)
211  ec = (cell_B.*fun)(&cell_Z);
212  if (ec != E_NO_ERROR) return Value_P();
213 CELL_PERFORMANCE_END(get_statistics_B(), start_2, z)
214  }
215  }
216  }
217  job_B->value_Z->check_value(LOC);
218  }
219 
220  Z->check_value(LOC);
221 
222  return Z;
223 }
224 //-----------------------------------------------------------------------------
225 void
226 ScalarFunction::PF_scalar_B(Thread_context & tctx)
227 {
228 const int cores = Thread_context::get_active_core_count();
229 
230 const ShapeItem slice_len = (job_B->len_Z + cores - 1) / cores;
231 ShapeItem z = tctx.get_N() * slice_len;
232 ShapeItem end_z = z + slice_len;
233  if (end_z > job_B->len_Z) end_z = job_B->len_Z;
234 
235  for (; z < end_z; ++z)
236  {
237  const Cell & cell_B = job_B->B_at(z);
238  Cell & cell_Z = job_B->Z_at(z);
239 
240  if (cell_B.is_pointer_cell())
241  {
242  // B is nested
243  //
244  Value_P B1 = cell_B.get_pointer_value();
245 
246  const ShapeItem len_Z1 = B1->element_count();
247  if (len_Z1 == 0)
248  {
249  POOL_LOCK(parallel_jobs_lock,
250  Value_P Z1 = B1->clone(LOC))
251  Z1->to_proto();
252  new (&cell_Z) PointerCell(Z1.get(), *job_B->value_Z);
253  }
254  else
255  {
256  POOL_LOCK(parallel_jobs_lock,
257  Value_P Z1(B1->get_shape(), LOC))
258  new (&cell_Z) PointerCell(Z1.get(), *job_B->value_Z);
259 
260  PJob_scalar_B j1(Z1.get(), B1.getref());
261  tctx.joblist_B.add_job(j1);
262  }
263  }
264  else
265  {
266  // B not nested: execute fun
267  //
268 PERFORMANCE_START(start_2)
269  job_B->error = (cell_B.*job_B->fun1)(&cell_Z);
270  if (job_B->error != E_NO_ERROR) return;
271 
272 CELL_PERFORMANCE_END(job_B->fun->get_statistics_B(), start_2, z)
273  }
274  }

276 //-----------------------------------------------------------------------------
277 void
278 ScalarFunction::expand_pointers(Cell * cell_Z, Value & Z_owner,
279  const Cell * cell_A, const Cell * cell_B,
280  prim_f2 fun)
281 {
282  if (cell_A->is_pointer_cell())
283  {
284  if (cell_B->is_pointer_cell()) // A and B are both pointers
285  {
286  Value_P value_A = cell_A->get_pointer_value();
287  Value_P value_B = cell_B->get_pointer_value();
288  POOL_LOCK(parallel_jobs_lock,
289  Token token = eval_scalar_AB(value_A, value_B, fun))
290  new (cell_Z) PointerCell(token.get_apl_val().get(), Z_owner);
291  }
292  else // A is pointer, B is simple
293  {
294  Value_P value_A = cell_A->get_pointer_value();
295  Value_P scalar_B(*cell_B, LOC);
296  POOL_LOCK(parallel_jobs_lock,
297  Token token = eval_scalar_AB(value_A, scalar_B, fun))
298  new (cell_Z) PointerCell(token.get_apl_val().get(), Z_owner);
299  }
300  }
301  else // A is simple
302  {
303  if (cell_B->is_pointer_cell()) // A is simple, B is pointer
304  {
305  Value_P scalar_A(*cell_A, LOC);
306  Value_P value_B = cell_B->get_pointer_value();
307  POOL_LOCK(parallel_jobs_lock,
308  Token token = eval_scalar_AB(scalar_A, value_B, fun))
309  new (cell_Z) PointerCell(token.get_apl_val().get(), Z_owner);
310  }
311  else // A and B are both plain
312  {
313  (cell_B->*fun)(cell_Z, cell_A);
314  }
315  }
316 }

340 //-----------------------------------------------------------------------------
341 Value_P
342 ScalarFunction::do_scalar_AB(ErrorCode & ec, Value_P A, Value_P B, prim_f2 fun)
343 {
344 const int inc_A = A->get_increment();
345 const int inc_B = B->get_increment();
346 
347 const Shape * shape_Z = 0;
348  if (A->is_scalar()) shape_Z = &B->get_shape();
349  else if (B->is_scalar()) shape_Z = &A->get_shape();
350  else if (inc_A == 0) shape_Z = &B->get_shape();
351  else if (inc_B == 0) shape_Z = &A->get_shape();
352  else if (A->same_shape(*B)) shape_Z = &B->get_shape();
353  else
354  {
355  if (!A->same_rank(*B)) ec = E_RANK_ERROR;
356  else ec = E_LENGTH_ERROR;
357  return Value_P();
358  }
359 
360 const ShapeItem len_Z = shape_Z->get_volume();
361  if (len_Z == 0) return eval_fill_AB(A, B).get_apl_val();
362 
363 Value_P Z(*shape_Z, LOC);
364 
365  // create a worklist with one item that computes Z. If nested values are
366  // detected while computing Z then jobs for them are added to the worklist.
367  //
368  {
369  PJob_scalar_AB job_AB(Z.get(), &A->get_ravel(0), inc_A,
370  &B->get_ravel(0), inc_B);
371  Thread_context::get_master().joblist_AB.start(job_AB, LOC);
372  }
373 
374 #if PARALLEL_ENABLED
375 const bool maybe_parallel = Parallel::run_parallel &&
376  Thread_context::get_active_core_count() > 1;
377 #endif
378 
379  for (;;)
380  {
381  loop(a, Thread_context::get_active_core_count())
382  {
383  job_AB = Thread_context::get_context(CoreNumber(a))
384  ->joblist_AB.next_job();
385  if (job_AB) break;
386  }
387  if (job_AB == 0) break; // all jobs done
388 
389 #if PARALLEL_ENABLED
390  if (maybe_parallel && job_AB->len_Z > get_dyadic_threshold())
391  {
392  // parallel execution...
393  //
394  job_AB->fun = this;
395  job_AB->fun2 = fun;
396  Thread_context::do_work = PF_scalar_AB;
397  Thread_context::M_fork("eval_scalar_AB"); // start pool
398  PF_scalar_AB(Thread_context::get_master());
399 PERFORMANCE_START(start_M_join)
400  Thread_context::M_join();
401  ec = job_AB->error;
402  if (ec != E_NO_ERROR) return Value_P();
403 PERFORMANCE_END(fs_M_join_AB, start_M_join, 1);
404  }
405  else
406 #endif // PARALLEL_ENABLED
407  {
408  // sequential execution...
409  //
410  loop(z, job_AB->len_Z)
411  {
412  const Cell & cell_A = job_AB->A_at(z);
413  const Cell & cell_B = job_AB->B_at(z);
414  Cell & cell_Z = job_AB->Z_at(z);
415 
416  if (cell_A.is_pointer_cell())
417  if (cell_B.is_pointer_cell())
418  {
419  // both A and B are nested
420  //
421  Value_P A1 = cell_A.get_pointer_value();
422  Value_P B1 = cell_B.get_pointer_value();
423  const int inc_A1 = A1->get_increment();
424  const int inc_B1 = B1->get_increment();
425  const Shape * sh_Z1 = &B1->get_shape();
426  if (A1->is_scalar()) sh_Z1 = &B1->get_shape();
427  else if (B1->is_scalar()) sh_Z1 = &A1->get_shape();
428  else if (inc_B1 == 0) sh_Z1 = &A1->get_shape();
429 
430  if (inc_A1 && inc_B1 && !A1->same_shape(*B1))
431  {
432  if (!A1->same_rank(*B1)) ec = E_RANK_ERROR;
433  else ec = E_LENGTH_ERROR;
434  return Value_P();
435  }
436 
437  const ShapeItem len_Z1 = sh_Z1->get_volume();
438  if (len_Z1 == 0)
439  {
440  Value_P Z1 =
441  eval_fill_AB(A1, B1).get_apl_val();
442  new (&cell_Z) PointerCell(Z1.get(),
443  *job_AB->value_Z);
444  continue;
445  }
446 
447  Value_P Z1(*sh_Z1, LOC);
448  new (&cell_Z)
449  PointerCell(Z1.get(), *job_AB->value_Z,
450  0x6B616769);
451 
452  PJob_scalar_AB j1(Z1.get(),
453  &A1->get_ravel(0), inc_A1,
454  &B1->get_ravel(0), inc_B1);
455  Thread_context::get_master()
456  .joblist_AB.add_job(j1);
457  }
458  else
459  {
460  // A is nested, B is not
461  //
462  Value_P A1 = cell_A.get_pointer_value();
463  const int inc_A1 = A1->get_increment();
464 
465  const ShapeItem len_Z1 = A1->element_count();
466  if (len_Z1 == 0)
467  {
468  Value_P Z1 = eval_fill_AB(A1, B).get_apl_val();
469  new (&cell_Z) PointerCell(Z1.get(),
470  *job_AB->value_Z);
471  }
472  else
473  {
474  Value_P Z1(A1->get_shape(), LOC);
475  if (!Z1) WS_FULL;
476  new (&cell_Z)
477  PointerCell(Z1.get(),*job_AB->value_Z,
478  0x6B616769);
479 
480  PJob_scalar_AB j1(Z1.get(),
481  &A1->get_ravel(0), inc_A1,
482  &cell_B, 0);
483  Thread_context::get_master().joblist_AB
484  .add_job(j1);
485  }
486  }
487  else
488  if (cell_B.is_pointer_cell())
489  {
490  // B is nested, A is not
491  //
492  Value_P B1 = cell_B.get_pointer_value();
493  const int inc_B1 = B1->get_increment();
494 
495  const ShapeItem len_Z1 = B1->element_count();
496  if (len_Z1 == 0)
497  {
498  Value_P Z1 = eval_fill_AB(A, B1).get_apl_val();
499  new (&cell_Z) PointerCell(Z1.get(),
500  *job_AB->value_Z);
501  }
502  else
503  {
504  Value_P Z1(B1->get_shape(), LOC);
505 
506  new (&cell_Z)
507  PointerCell(Z1.get(), *job_AB->value_Z,
508  0x6B616769);
509 
510  PJob_scalar_AB j1(Z1.get(), &cell_A, 0,
511  &B1->get_ravel(0), inc_B1);
512  Thread_context::get_master()
513  .joblist_AB.add_job(j1);
514  }
515  }
516  else
517  {
518  // neither A nor B are nested: execute fun
519  //
520 PERFORMANCE_START(start_2)
521 
522  ec = (cell_B.*fun)(&cell_Z, &cell_A);
523  if (ec != E_NO_ERROR) return Value_P();
524 CELL_PERFORMANCE_END(get_statistics_AB(), start_2, z)
525  }
526  }
527  }
528  job_AB->value_Z->check_value(LOC);
529  }
530 
531  Z->set_default(*B.get(), LOC);
532  Z->check_value(LOC);
533 
534  return Z;
535 }
536 //-----------------------------------------------------------------------------
537 void
538 ScalarFunction::PF_scalar_AB(Thread_context & tctx)
539 {
540 const CoreCount cores = Thread_context::get_active_core_count();

1123 //-----------------------------------------------------------------------------
1124 Value_P
1125 Bif_F12_WITHOUT::large_eval_AB(const Value * A, const Value * B)
1126 {
1127 const ShapeItem len_A = A->element_count();
1128 const ShapeItem len_B = B->element_count();
1129 
1130  /* pack pointers to the cells of the arguments A and B and of the
1131  result Z into one big array:
1132 
1133  len_A len_A len_B
1134  ┌─────────┬─────────┬─────────┐
1135  │ cells_A │ cells_Z │ cells_B │
1136  └─────────┴─────────┴─────────┘
1137  */
1138 const Cell ** cells_A = new const Cell *[2*len_A + len_B];
1139 const Cell ** cells_Z = cells_A + len_A;
1140 const Cell ** cells_B = cells_A + 2*len_A;
1141 
1142  loop(a, len_A) cells_A[a] = &A->get_ravel(a);
1143  loop(b, len_B) cells_B[b] = &B->get_ravel(b);
1144 
1145  // sort the A-cells and the B-cells ascendingly
1146  //
1147  Heapsort<const Cell *>::sort(cells_A, len_A, 0, Cell::compare_stable);
1148  Heapsort<const Cell *>::sort(cells_B, len_B, 0, Cell::compare_stable);
1149 
1150  // store those cells_A pointers that are not in cells_B into cells_Z. Use
1151  // the fact that cells_A and cells_B are sorted.
1152 ShapeItem len_Z = 0;
1153  {
1154  ShapeItem idx_A = 0;
1155  ShapeItem idx_B = 0;
1156  const double qct = Workspace::get_CT();
1157 
1158  while (idx_A < len_A && idx_B < len_B)
1159  {
1160  const Cell & ref_A = *cells_A[idx_A];
1161  const Cell & ref_B = *cells_B[idx_B];
1162  if (ref_A.equal(ref_B, qct)) ++idx_A; // A is in B → not in Z
1163  else if (ref_A.greater(ref_B)) ++idx_B;
1164  else cells_Z[len_Z++] = cells_A[idx_A++]; // A is in Z
1165  }
1166

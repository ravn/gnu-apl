33 Token
34 Bif_OPER1_EACH::eval_ALB(Value_P A, Token & _LO, Value_P B)
35 {
36  // dyadic EACH: call _LO for corresponding items of A and B
37 
38  if (!A->same_shape(*B))
39  {
40  // if the shapes differ then either A or B must be a scalar or
41  // 1-element vector.
42  //
43  if (A->get_rank() != B->get_rank())
44  {
45  if (A->is_scalar_or_len1_vector()) ; // OK
46  else if (B->is_scalar_or_len1_vector()) ; // OK
47  else if (A->get_rank() != B->get_rank()) RANK_ERROR;
48  else LENGTH_ERROR;
49  }
50  }
51 
52 Function * LO = _LO.get_function();
53  Assert1(LO);
54 
55  // for the ambiguous operators /. â¿, \, and â is_operator() returns true,
56  // which is incorrect in this context. We use get_signature() instead.
57  //
58  if ((LO->get_signature() & SIG_DYA) != SIG_DYA) VALENCE_ERROR;
59 
60  if (A->is_empty() || B->is_empty())
61  {
62  if (!LO->has_result()) return Token(TOK_VOID);
63 
64  Value_P First_A = Bif_F12_TAKE::first(A);
65  Value_P First_B = Bif_F12_TAKE::first(B);
66  Shape shape_Z;
67 
68  if (A->is_empty()) shape_Z = A->get_shape();
69  else if (!A->is_scalar()) DOMAIN_ERROR;
70 
71  if (B->is_empty()) shape_Z = B->get_shape();
72  else if (!B->is_scalar()) DOMAIN_ERROR;
73 
74  Value_P Z1 = LO->eval_fill_AB(First_A, First_B).get_apl_val();
75  Value_P Z(shape_Z, LOC);
76  if (Z1->is_simple_scalar())
77  Z->get_ravel(0).init(Z1->get_ravel(0), Z.getref(), LOC);
78  else
79  new (&Z->get_ravel(0)) PointerCell(Z1.get(), Z.getref());
80  Z->check_value(LOC);
81  return Token(TOK_APL_VALUE1, Z);
82  }
83 
84  if (LO->may_push_SI()) // user defined LO
85  {
86  const bool scalar_A = A->is_scalar();
87  const bool scalar_B = B->is_scalar();
88 
89  Macro * macro = 0;
90  if (LO->has_result())
91  {
92  if (scalar_A)
93  {
94  macro = Macro::get_macro(scalar_B
95  ? Macro::MAC_Z__sA_LO_EACH_sB
96  : Macro::MAC_Z__sA_LO_EACH_vB);
97  }
98  else
99  {
100  macro = Macro::get_macro(scalar_B
101  ? Macro::MAC_Z__vA_LO_EACH_sB
102  : Macro::MAC_Z__vA_LO_EACH_vB);
103  }
104  }
105  else // LO has no result, so we can ignore the shape of the result
106  {
107  if (scalar_A && scalar_B)
108  {
109  macro = Macro::get_macro(Macro::MAC_sA_LO_EACH_sB);
110  }
111 
112  if (scalar_B)
113  {
114  macro = Macro::get_macro(Macro::MAC_vA_LO_EACH_sB);
115  }
116  else if (scalar_A)
117  {
118  macro = Macro::get_macro(Macro::MAC_sA_LO_EACH_vB);
119  }
120  else
121  {
122  macro = Macro::get_macro(Macro::MAC_vA_LO_EACH_vB);
123  }
124  }
125 
126  return macro->eval_ALB(A, _LO, B);
127  }
128 
129 Value_P Z;
130 int dA = 1; // assume A is non-scalar
131 int dB = 1; // assume B is non-scalar
132 ShapeItem len_Z = 0;
133 
134  if (A->is_scalar()) // scalar-extend A
135  {
136  len_Z = B->element_count();
137  dA = 0;
138  if (LO->has_result()) Z = Value_P(B->get_shape(), LOC);
139  }
140  else if (B->is_scalar()) // scalar-extend B
141  {
142  dB = 0;
143  len_Z = A->element_count();
144  if (LO->has_result()) Z = Value_P(A->get_shape(), LOC);
145  }
146  else if (A->same_shape(*B)) // A and B are non-scalars with identical shapes
147  {
148  len_Z = B->element_count();
149  if (LO->has_result()) Z = Value_P(B->get_shape(), LOC);
150  }
151  else // the shapes of A and B differ
152  {
153  if (A->get_rank() != B->get_rank()) RANK_ERROR;
154  else LENGTH_ERROR;
155  }
156 
157 
158  loop(z, len_Z)
159  {
160  const Cell * cA = &A->get_ravel(dA * z);
161  const Cell * cB = &B->get_ravel(dB * z);
162  const bool left_val = cB->is_lval_cell();
163  Value_P LO_A = cA->to_value(LOC); // left argument of LO
164  Value_P LO_B = cB->to_value(LOC); // right argument of LO;
165  if (left_val)
166  {
167  Cell * dest = cB->get_lval_value();
168  if (dest->is_pointer_cell())
169  {
170  Value_P sub = dest->get_pointer_value();
171  LO_B = sub->get_cellrefs(LOC);
172  }
173  }
174 
175  Token result = LO->eval_AB(LO_A, LO_B);
176 
177  // if LO was a primitive function, then result may be a value.
178  // if LO was a user defined function then result may be TOK_SI_PUSHED.
179  // in both cases result could be TOK_ERROR.
180  //
181  if (result.get_Class() == TC_VALUE)
182  {
183  Value_P vZ = result.get_apl_val();
184 
185  Cell * cZ = Z->next_ravel();
186  if (vZ->is_simple_scalar() || (left_val && vZ->is_scalar()))
187  cZ->init(vZ->get_ravel(0), Z.getref(), LOC);
188  else
189  new (cZ) PointerCell(vZ.get(), Z.getref());
190 
191  continue; // next z
192  }
193 
194  if (result.get_tag() == TOK_ERROR) return result;
195 
196  Q1(result); FIXME;
197  }
198 
199  if (!Z) return Token(TOK_VOID); // LO without result
200 
201  Z->set_default(*B.get(), LOC);
202  Z->check_value(LOC);
203  return Token(TOK_APL_VALUE1, Z);
204 }

206 Token
207 Bif_OPER1_EACH::eval_LB(Token & _LO, Value_P B)
208 {
209  // monadic EACH: call _LO for every item of B
210 
211 Function * LO = _LO.get_function();
212  Assert1(LO);
213  if (LO->is_operator()) SYNTAX_ERROR;
214  if (!(LO->get_signature() & SIG_B)) VALENCE_ERROR;
215 
216  if (B->is_empty())
217  {
218  if (!LO->has_result()) return Token(TOK_VOID);
219 
220  Value_P First_B = Bif_F12_TAKE::first(B);
221  Token tZ = LO->eval_fill_B(First_B);
222  Value_P Z1 = tZ.get_apl_val();
223 
224  if (Z1->is_simple_scalar())
225  {
226  Z1->set_shape(B->get_shape());
227  return Token(TOK_APL_VALUE1, Z1);
228  }
229 
230  Value_P Z(B->get_shape(), LOC);
231  new (&Z->get_ravel(0)) PointerCell(Z1.get(), Z.getref());
232  Z->check_value(LOC);
233  return Token(TOK_APL_VALUE1, Z);
234  }
235 
236  if (LO->may_push_SI()) // user defined LO
237  {
238  if (!LO->has_result())
239  return Macro::get_macro(Macro::MAC_LO_EACH_B)->eval_LB(_LO, B);
240 
241  if (LO == Bif_F1_EXECUTE::fun)
242  return Macro::get_macro(Macro::MAC_Z__EXEC_EACH_B)->eval_B(B);
243 
244  return Macro::get_macro(Macro::MAC_Z__LO_EACH_B)->eval_LB(_LO, B);
245  }
246 
247 const ShapeItem len_Z = B->element_count();
248 Value_P Z;
249  if (LO->has_result()) Z = Value_P(B->get_shape(), LOC);
250 
251  loop (z, len_Z)
252  {
253  if (LO->get_fun_valence() == 0)
254  {
255  // we allow niladic functions N so that one can loop over them with
256  // N Â¨ 1 2 3 4
257  //
258  Token result = LO->eval_();
259 
260  if (result.get_Class() == TC_VALUE)
261  {
262  Value_P vZ = result.get_apl_val();
263 
264  Cell * cZ = Z->next_ravel();
265  if (vZ->is_simple_scalar())
266  cZ->init(vZ->get_ravel(0), Z.getref(), LOC);
267  else
268  new (cZ) PointerCell(vZ.get(), Z.getref());
269 
270  continue; // next z
271  }
272 
273  if (result.get_tag() == TOK_VOID) continue; // next z
274 
275  if (result.get_tag() == TOK_ERROR) return result;
276 
277  Q1(result); FIXME;
278  }
279  else
280  {
281  const Cell * cB = &B->get_ravel(z);
282  const bool left_val = cB->is_lval_cell();
283  Value_P LO_B = cB->to_value(LOC); // right argument of LO
284 
285  if (left_val)
286  {
287  Cell * dest = cB->get_lval_value();
288  if (dest->is_pointer_cell())
289  {
290  Value_P sub = dest->get_pointer_value();
291  LO_B = sub->get_cellrefs(LOC);
292  }
293  }
294 
295  Token result = LO->eval_B(LO_B);
296  if (result.get_Class() == TC_VALUE)
297  {
298  Value * vZ = result.get_apl_val().get();
299 
300  Cell * cZ = Z->next_ravel();
301  if (0)
302  cZ->init_from_value(vZ, Z.getref(), LOC);
303  else if (vZ->is_simple_scalar() ||
304  (left_val && vZ->is_scalar()))
305  cZ->init(vZ->get_ravel(0), Z.getref(), LOC);
306  else
307  new (cZ) PointerCell(vZ, Z.getref());
308 
309  continue; // next z
310  }
311 
312  if (result.get_tag() == TOK_VOID) continue; // next z
313 
314  if (result.get_tag() == TOK_ERROR) return result;
315 
316  Q1(result); Q1(*LO) FIXME;
317  }
318  }
319 
320  if (!Z) return Token(TOK_VOID); // LO without result
321 
322  Z->set_default(*B.get(), LOC);
323  Z->check_value(LOC);
324  return Token(TOK_APL_VALUE1, Z);
325 }

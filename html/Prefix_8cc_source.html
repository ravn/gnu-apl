71 Prefix::syntax_error(const char * loc)
72 {
73  // move the PC back to the beginning of the failed statement
74  //
75  while (PC > 0)
76  {
77  --PC;
78  if (body[PC].get_Class() == TC_END)
79  {
80  ++PC;
81  break;

85  // clear values in FIFO
86  //
87  loop (s, size())
88  {
89  Token & tok = at(s).tok;
90  if (tok.get_Class() == TC_VALUE)
91  {
92  Value_P val = tok.get_apl_val();
93  }
94  }
95 
96  // see if error was caused by a function not returning a value.
97  // in that case we throw a value error instead of a syntax error.
98  //
99  loop (s, size())
100  {
101  if (at(s).tok.get_Class() == TC_VOID)
102  throw_apl_error(E_VALUE_ERROR, loc);

104 
105  throw_apl_error(get_assign_state() == ASS_none ? E_SYNTAX_ERROR
106  : E_LEFT_SYNTAX_ERROR, loc);

124 //-----------------------------------------------------------------------------
125 bool
126 Prefix::is_value_parenthesis(int pc) const
127 {
128  // we have ) XXX with XXX on the stack and need to know if the evaluation
129  // of (... ) will be a value as in e.g. (1 + 1) or a function as in (+/).
130  //
131  Assert1(body[pc].get_Class() == TC_R_PARENT);
132 
133  ++pc;
134  if (pc >= int(body.size())) return true; // syntax error
135 
136 TokenClass next = body[pc].get_Class();
137 
138  if (next == TC_R_BRACK) // skip [ ... ]
139  {
140  const int offset = body[pc].get_int_val2();
141  pc += offset;
142  Assert1(body[pc].get_Class() == TC_L_BRACK); // opening [
143  if (pc >= Function_PC(body.size())) return true; // syntax error
144  next = body[pc].get_Class();
145  }
146 
147  if (next == TC_SYMBOL) // resolve symbol if necessary
148  {
149  const Symbol * sym = body[pc].get_sym_ptr();
150  const NameClass nc = sym->get_nc();
151 
152  if (nc == NC_FUNCTION) return false;
153  if (nc == NC_OPERATOR) return false;
154  return true;
155  }
156 
157  if (next == TC_OPER1) return false;
158  if (next == TC_OPER2) return false;
159  if (next == TC_FUN12) return false;
160 
161  if (next == TC_L_PARENT) // )) XXX
162  {
163  ++pc;
164  if (!is_value_parenthesis(pc)) return false; // (fun)) XXX
165  const int offset = body[pc].get_int_val2();
166  pc += offset;
167  if (pc >= Function_PC(body.size())) return true; // syntax error
168  next = body[pc].get_Class();
169  Assert1(next == TC_L_PARENT); // opening (
170  ++pc;
171  if (pc >= Function_PC(body.size())) return true; // syntax error
172 
173  // (val)) XXX
174  // ^
175  // pc
176  //
177  // result is a value unless (val) is the right function operand
178  // of a dyadic operator
179  //
180  next = body[pc].get_Class();
181  if (next == TC_OPER2) return false;
182  if (next == TC_SYMBOL) // resolve symbol if necessary
183  {
184  const Symbol * sym = body[pc].get_sym_ptr();
185  const Function * fun = sym->get_function();
186  return ! (fun && fun->is_operator() &&
187  fun->get_oper_valence() == 2);
188  }
189  return true;
190  }
191 
192  // dyadic operator with numeric function argument, for example: â¤ 0
193  //
194  if (next == TC_VALUE &&
195  pc < Function_PC(body.size() - 1) &&
196  body[pc+1].get_Class() == TC_OPER2) return false;
197 
198  return true;
199 }
200 //-----------------------------------------------------------------------------
201 bool
202 Prefix::is_fun_or_oper(int pc) const
203 {
204  // this function is called when / â¿ \ or â shall be resolved. pc points
205  // to the token left of / â¿ \ or â.
206  //
207 TokenTag tag_LO = body[pc].get_tag();
208 
209  if (tag_LO == TOK_R_BRACK)
210  {
211  // e.g. fun[...]/ or value[...]/ Skip over [...]
212  //
213  pc += body[pc].get_int_val2();
214  Assert1(body[pc++].get_Class() == TC_L_BRACK); // opening [
215  tag_LO = body[pc].get_tag();
216  }
217 
218  if (tag_LO == TOK_R_PARENT) return !is_value_parenthesis(pc);
219  if (body[pc].get_Class() == TC_OPER2) return false; // make / a function
220 
221  if ((tag_LO & TV_MASK) == TV_FUN) return true;
222 
223  if (tag_LO == TOK_SYMBOL)
224  {
225  Symbol * sym = body[pc].get_sym_ptr();
226  if (sym == 0) return false;
227  return sym->get_function() != 0;
228  }
229 
230  return false; // not a function or operator
231 }

337 //-----------------------------------------------------------------------------
338 bool
339 Prefix::value_expected()
340 {
341  // return true iff the current saved_lookahead token (which has been tested
342  // to be a TC_INDEX token) is the index of a value and false it it is
343  // a function axis
344 
345  // if it contains semicolons then get_ValueType() is TV_INDEX and
346  // it MUST be a value.
347  //
348  if (saved_lookahead.tok.get_ValueType() == TV_INDEX) return true;
349 
350  for (int pc = PC; pc < int(body.size());)
351  {
352  const Token & tok = body[pc++];
353  switch(tok.get_Class())
354  {
355  case TC_R_BRACK: // skip over [...] (func axis or value index)
356  //
357  pc += tok.get_int_val2();
358  continue;
359 
360  case TC_END: return false; // syntax error
361 
362  case TC_FUN0: return true; // niladic function is a value
363  case TC_FUN12: return false; // function
364 
365  case TC_SYMBOL:
366  {
367  const Symbol * sym = tok.get_sym_ptr();
368  const NameClass nc = sym->get_nc();
369 
370  if (nc == NC_FUNCTION) return false;
371  if (nc == NC_OPERATOR) return false;
372  return true; // value
373  }
374 
375  case TC_RETURN: return false; // syntax error
376  case TC_VALUE: return true;
377 
378  default: continue;
379  }
380  }
381 
382  // this is a syntax error.
383  //
384  return false;
385 }

400 
401 // a hash table with all prefixes that can be reduced...
402 #define PH(name, idx, prio, misc, len, fun) \
403  { #name, idx, prio, misc, len, &Prefix::reduce_ ## fun, #fun },

407 Token
408 Prefix::reduce_statements()
409 {
410  Log(LOG_prefix_parser)
411  {
412  CERR << endl << "changed to Prefix[si=" << si.get_level()
413  << "]) ============================================" << endl;
414  }
415 
416  if (size() > 0) goto again;
417 
418 grow:
419  // the current stack does not contain a valid phrase.
420  // Push one more token onto the stack and continue
421  //
422  {
423  if (saved_lookahead.tok.get_tag() != TOK_VOID)
424  {
425  // there is a MISC token from a MISC phrase. Use it.
426  //
427  push(saved_lookahead);
428  saved_lookahead.tok.clear(LOC);
429  goto again; // success
430  }
431 
432  // if END was reached, then there are no more token in current-statement
433  //
434  if (size() > 0 && at0().get_Class() == TC_END)
435  {
436  Log(LOG_prefix_parser) print_stack(CERR, LOC);
437 
438  // provide help on some common cases...
439  //
440  for (int j = 1; j < (size() - 1); ++j)
441  {
442  if ( (at(j).tok.get_Class() == TC_ASSIGN) &&
443  (at(j + 1).tok.get_Class() == TC_VALUE))
444  {
445  if (at(j - 1).tok.get_Class() == TC_FUN0 ||
446  at(j - 1).tok.get_Class() == TC_FUN12)
447  {
448  MORE_ERROR() <<
449  "Cannot assign a value to a function";
450  }
451  else if (at(j - 1).tok.get_Class() == TC_OPER1 ||
452  at(j - 1).tok.get_Class() == TC_OPER1)
453  {
454  MORE_ERROR() <<
455  "Cannot assign a value to an operator";
456  }
457  }
458  }
459  Log(LOG_prefix_parser) print_stack(CERR, LOC);
460  syntax_error(LOC); // no more token
461  }
462 
463  Token_loc tl = lookahead();
464  Log(LOG_prefix_parser)
465  {
466  CERR << " [si=" << si.get_level() << " PC=" << (PC - 1)
467  << "] Read token[" << size()
468  << "] (â" << get_assign_state() << "â) " << tl.tok << " "
469  << Token::class_name(tl.tok.get_Class()) << endl;
470  }
471 
472  lookahead_high = tl.pc;
473  TokenClass tcl = tl.tok.get_Class();
474 
475  if (tcl == TC_SYMBOL) // resolve symbol if necessary
476  {
477  // reset the PC back to the previous token, so that a failed
478  // resolve() (aka VALUE ERROR) will re-fetch the token
479  //
480  // But not if symbol is âLC because that would make the first
481  // element of âLC too low!
482  //
483  if (tl.tok.get_tag() != TOK_Quad_LC) --PC;
484 
485  Symbol * sym = tl.tok.get_sym_ptr();
486  if (tl.tok.get_tag() == TOK_LSYMB2)
487  {
488  // this is the last token C of a vector assignment
489  // (A B ... C)â. We return C and let the caller do the rest
490  //
491  sym->resolve(tl.tok, true);
492  Log(LOG_prefix_parser)
493  CERR << "TOK_LSYMB2 " << sym->get_name()
494  << "resolved to " << tl.tok << endl;
495  }
496  else
497  {
498  const bool left_sym = get_assign_state() == ASS_arrow_seen;
499  bool resolved = false;
500  if (left_sym)
501  {
502  // assignment to only variable or undefined names
503  const NameClass nc = sym->get_nc();
504 
505  if (NC_VARIABLE != nc &&
506  NC_UNUSED_USER_NAME != nc &&
507  NC_SHARED_VAR != nc &&
508  NC_INVALID != nc) // â, â, âxx
509  syntax_error(LOC);
510  }
511 
512  if (size() > 0 && at(0).tok.get_Class() == TC_INDEX &&
513  tl.tok.get_tag() == TOK_SYMBOL) // user defined variable
514  {
515  // indexed reference, e.g. A[N]. Calling sym->resolve()
516  // would copy the entire variable and then index it, which
517  // is inefficient if the variable is big. We rather call
518  // Symbol::get_value() directly in order to avoid that
519  //
520  Value_P val = sym->get_value();
521  if (!!val && !left_sym)
522  {
523  Token tok(TOK_APL_VALUE1, val);
524  tl.tok.move_1(tok, LOC);
525  resolved = true;
526  }
527  }
528  if (!resolved) sym->resolve(tl.tok, left_sym);
529 
530  if (left_sym) set_assign_state(ASS_var_seen);
531  Log(LOG_prefix_parser)
532  {
533  if (left_sym) CERR << "TOK_LSYMB ";
534  else CERR << "TOK_SYMBOL ";
535  CERR << "resolved to " << tl.tok << endl;
536  }
537  }
538  PC = lookahead_high + 1; // resolve() succeeded: restore PC
539 
540  Log(LOG_prefix_parser)
541  {
542  CERR << " resolved symbol " << sym->get_name()
543  << " to " << tl.tok.get_Class() << endl;
544  }
545 
546  if (tl.tok.get_tag() == TOK_SI_PUSHED)
547  {
548  // Quad_Quad::resolve() calls â which returns TOK_SI_PUSHED.
549  //
550  push(tl);
551  return Token(TOK_SI_PUSHED);
552  }
553  }
554  else if (tcl == TC_ASSIGN) // resolve symbol if necessary
555  {
556  if (get_assign_state() != ASS_none) syntax_error(LOC);
557  set_assign_state(ASS_arrow_seen);
558  }
559 
560  push(tl);
561  }
562 
563 again:
564  Log(LOG_prefix_parser) print_stack(CERR, LOC);
565 
566  // search prefixes in phrase table...
567  //
568  {
569  const int hash_0 = at0().get_Class();
570 
571  if (size() >= 3)
572  {
573  const int hash_01 = hash_0 | at1().get_Class() << 5;
574  const int hash_012 = hash_01 | at2().get_Class() << 10;
575 
576  if (size() >= 4)
577  {
578  const int hash_0123 = hash_012 | at3().get_Class() << 15;
579  best = hash_table + hash_0123 % PHRASE_MODU;
580  if (best->phrase_hash == hash_0123) goto found_prefix;
581  }
582 
583  best = hash_table + hash_012 % PHRASE_MODU;
584  if (best->phrase_hash == hash_012) goto found_prefix;
585 
586  best = hash_table + hash_01 % PHRASE_MODU;
587  if (best->phrase_hash == hash_01) goto found_prefix;
588 
589  best = hash_table + hash_0 % PHRASE_MODU;
590  if (best->phrase_hash != hash_0) goto grow; // no matching phrase
591  }
592  else // 0 < size() < 3
593  {
594  if (size() >= 2)
595  {
596  const int hash_01 = hash_0 | at1().get_Class() << 5;
597  best = hash_table + hash_01 % PHRASE_MODU;
598  if (best->phrase_hash == hash_01) goto found_prefix;
599  }
600 
601  best = hash_table + hash_0 % PHRASE_MODU;
602  if (best->phrase_hash != hash_0) goto grow;
603  }
604  }
605 
606 found_prefix:
607 
608  // found a reducible prefix. See if the next token class binds stronger
609  // than best->prio
610  //
611  {
612  TokenClass next = TC_INVALID;
613  if (PC < Function_PC(body.size()))
614  {
615  const Token & tok = body[PC];
616 
617  next = tok.get_Class();
618  if (next == TC_SYMBOL)
619  {
620  Symbol * sym = tok.get_sym_ptr();
621  const bool left_sym = get_assign_state() == ASS_arrow_seen;
622  next = sym->resolve_class(left_sym);
623  }
624  }
625 
626  if (best->misc && (at0().get_Class() == TC_R_BRACK))
627  {
628  // the next symbol is a ] and the matching phrase is a MISC
629  // phrase (monadic call of a possibly dyadic function).
630  // The ] could belong to:
631  //
632  // 1. an indexed value, e.g. A[X] or
633  // 2. a function with an axis, e.g. +[2]
634  //
635  // These cases lead to different reductions:
636  //
637  // 1. A[X] Ã B should evalate Ã dyadically, while
638  // 2. +[1] Ã B should evalate Ã monadically,
639  //
640  // We solve this by computing the indexed value first
641  //
642  if (is_value_bracket()) // case 1.
643  {
644  // we call reduce_RBRA____, which pushes a partial index list
645  // onto the stack. The following token are processed until the
646  // entire indexed value A[ ... ] is computed
647  prefix_len = 1;
648  reduce_RBRA___();
649  goto grow;
650  }
651  }
652 
653 // Q1(next) Q1(at0())
654 
655  // we could reduce, but we could also shift. Compute more, which is true
656  // if we should shift.
657  //
658  const bool shift = dont_reduce(next);
659  if (shift)
660  {
661  Log(LOG_prefix_parser) CERR
662  << " phrase #" << (best - hash_table)
663  << ": " << best->phrase_name
664  << " matches, but prio " << best->prio
665  << " is too small to call " << best->reduce_name
666  << "()" << endl;
667  goto grow;
668  }
669  }
670 
671  Log(LOG_prefix_parser) CERR
672  << " phrase #" << (best - hash_table)
673  << ": " << best->phrase_name
674  << " matches, prio " << best->prio
675  << ", calling reduce_" << best->reduce_name
676  << "()" << endl;
677 
678  action = RA_FIXME;
679  prefix_len = best->phrase_len;
680  if (best->misc) // MISC phrase: save X and remove it
681  {
682  Assert(saved_lookahead.tok.get_tag() == TOK_VOID);
683  saved_lookahead.copy(pop(), LOC);
684  --prefix_len;
685  }
686 
687 const uint64_t inst = instance;
688  (this->*best->reduce_fun)();
689 
690  if (inst != Workspace::SI_top()->get_prefix().instance)
691  {
692  // the reduce_fun() above has changed the )SI stack. As a consequence
693  // the 'this' pointer is no longer valid and we must not access members
694  // of this Prefix instance.
695  //
696  return Token(TOK_SI_PUSHED);
697  }
698 
699  Log(LOG_prefix_parser)
700  CERR << " reduce_" << best->reduce_name << "() returned: ";
701 
702  // handle action (with decreasing likelihood)
703  //
704  if (action == RA_CONTINUE)
705  {
706  Log(LOG_prefix_parser) CERR << "RA_CONTINUE" << endl;
707  goto again;
708  }
709 
710  if (action == RA_PUSH_NEXT)
711  {
712  Log(LOG_prefix_parser) CERR << "RA_PUSH_NEXT" << endl;
713  goto grow;
714  }
715 
716  if (action == RA_SI_PUSHED)
717  {
718  Log(LOG_prefix_parser) CERR << "RA_SI_PUSHED" << endl;
719  return Token(TOK_SI_PUSHED);
720  }
721 
722  if (action == RA_RETURN)
723  {
724  Log(LOG_prefix_parser) CERR << "RA_RETURN" << endl;
725  return pop().tok;

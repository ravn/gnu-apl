37 Value_P
38 Bif_F12_TAKE::first(Value_P B)
39 {
40 const Cell & first_B = B->get_cfirst();
41  if (B->element_count() == 0) // empty value: return prototype
42  {
43  if (first_B.is_lval_cell()) // (↑...)←V
44  {
45  Value_P Z(LOC);
46  Z->next_ravel_Cell(first_B);
47  return Z;
48  }
49 
50  // normal (right-) value
51  //
52  Value_P Z = B->prototype(LOC);
53  Z->check_value(LOC);
54  return Z;
55  }
56 
57  if (!first_B.is_pointer_cell()) // simple cell
58  {
59  Value_P Z(LOC);
60  Z->get_wscalar().init(first_B, Z.getref(), LOC);
61  Z->check_value(LOC);
62  return Z;
63  }
64 
65 Value_P v1 = first_B.get_pointer_value();
66 Value * v1_owner = v1->get_lval_cellowner();
67  if (v1_owner) // B is a left value
68  {
69  Value_P B1(LOC);
70  B1->next_ravel_Pointer(v1.get());
71  B1->check_value(LOC);
72 
73  Value_P Z(LOC);
74  Z->next_ravel_Lval(&B1->get_wscalar(), v1_owner);
75 
76  Z->check_value(LOC);
77  return Z;
78  }
79  else
80  {
81  const ShapeItem ec = v1->element_count();
82  Value_P Z(v1->get_shape(), LOC);
83  if (ec == 0) Z->get_wproto().init(v1->get_cproto(), Z.getref(), LOC);
84 
85  loop(e, ec) Z->next_ravel_Cell(v1->get_cravel(e));
86 
87  Z->check_value(LOC);
88  return Z;
89  }
90 }

92 Token
93 Bif_F12_TAKE::eval_AXB(Value_P A, Value_P X, Value_P B) const
94 {
95  if (X->element_count() == 0) // no axes
96  {
97  Token result(TOK_APL_VALUE1, B->clone(LOC));
98  return result;
99  }
100 
101  // A↑[X]B ←→ ⊃[X](⊂A)↑¨⊂[X]B
102  //
103 const Shape shape_X = Value::to_shape(X.get());
104 Value_P cA = Bif_F12_PARTITION::do_eval_B(A); // ⊂A
105 Value_P cB = Bif_F12_PARTITION::enclose_with_axes(shape_X, B); // ⊂[X]B
106 Token take(TOK_FUN2, Bif_F12_TAKE::fun);
107 Token cT = Bif_OPER1_EACH::fun->eval_ALB(cA, take, cB); // cA↑¨cB
108 
109 Token result = Bif_F12_PICK::fun->eval_XB(X, cT.get_apl_val());
110  return result;
111 }

113 Token
114 Bif_F12_TAKE::eval_AB(Value_P A, Value_P B) const
115 {
116 Shape ravel_A1(A.get(), /* ⎕IO */ 0); // checks that 1 ≤ ⍴⍴A and ⍴A ≤ MAX_RANK
117 
118  if (B->is_scalar())
119  {
120  Shape shape_B1;
121  loop(a, ravel_A1.get_rank()) shape_B1.add_shape_item(1);
122  Value_P B1 = B->clone(LOC);
123  B1->set_shape(shape_B1);
124  return Token(TOK_APL_VALUE1, do_take(ravel_A1, B1));
125  }
126  else
127  {
128  if (ravel_A1.get_rank() != B->get_rank()) LENGTH_ERROR;
129  return Token(TOK_APL_VALUE1, do_take(ravel_A1, B));
130  }
131 }

133 Value_P
134 Bif_F12_TAKE::do_take(const Shape & ravel_A1, Value_P B)
135 {
136  // ravel_A1 can have negative items (for take from the end).
137  //
138 Value_P Z(ravel_A1.abs(), LOC);
139 
140  if (ravel_A1.is_empty())
141  Z->get_wproto().init_type(B->get_cfirst(), Z.getref(), LOC);
142  else
143  fill(ravel_A1, Z.getref(), B.getref());
144  Z->check_value(LOC);
145  return Z;
146 }

149 Bif_F12_TAKE::fill(const Shape & shape_Zi, Value & Z, const Value & B)
150 {
151  for (TakeDropIterator i(true, shape_Zi, B.get_shape()); i.more(); ++i)
152  {
153  const ShapeItem offset = i();
154  if (offset == -1) // invalid cell
155  {
156  Z.next_ravel_Proto(B.get_cproto());
157  }
158  else // valid Cell
159  {
160  Z.next_ravel_Cell(B.get_cravel(offset));
161  }
162  }
163 }

165 Token
166 Bif_F12_DROP::eval_AB(Value_P A, Value_P B) const
167 {
168 const Shape ravel_A(A.get(), /* ⎕IO */ 0);
169  if (A->get_rank() > 1) RANK_ERROR;
170 
171  if (B->is_scalar())
172  {
173  /* Z←A↓B with scalar B has an element count() of 1 (if all items of A
174  are are 0 thus nothing is dropped) or else 1.
175 
176  ⍴⍴Z is ⍴⍴A and ⍴Z[j] ←→ A[j] = 0
177  */
178  Shape shape_Z;
179  loop(r, ravel_A.get_rank())
180  {
181  // the r'th shape item is either 0 (if anything is dropped)
182  // or 1 if not.
183  if (ravel_A.get_shape_item(r)) shape_Z.add_shape_item(0);
184  else shape_Z.add_shape_item(1);
185  }
186 
187  Value_P Z(shape_Z, LOC);
188 
189  Z->set_ravel_Cell(0, B->get_cfirst());
190  if (shape_Z.get_volume() == 0) Z->to_proto();
191  Z->check_value(LOC);
192  return Token(TOK_APL_VALUE1, Z);
193  }
194 
195  if (ravel_A.get_rank() != B->get_rank()) LENGTH_ERROR;
196 
197 Shape sh_Z;
198  loop(r, ravel_A.get_rank())
199  {
200  const ShapeItem sA = ravel_A.get_shape_item(r);
201  const ShapeItem sB = B->get_shape_item(r);
202  const ShapeItem pA = sA < 0 ? -sA : sA;
203  if (pA >= sB) sh_Z.add_shape_item(0); // over-drop
204  else sh_Z.add_shape_item(sB - pA);
205  }
206 
207 Value_P Z(sh_Z, LOC);
208  if (sh_Z.is_empty()) // empty Z, e.g. from overdrop
209  {
210  Value_P Z(sh_Z, LOC);
211  Z->set_default(*B.get(), LOC);
212  Z->check_value(LOC);
213  return Token(TOK_APL_VALUE1, Z);
214  }
215 
216  for (TakeDropIterator i(false, ravel_A, B->get_shape()); i.more(); ++i)
217  {
218  const ShapeItem offset = i();
219  Z->next_ravel_Cell(B->get_cravel(offset));
220  }
221 
222  Z->check_value(LOC);
223  return Token(TOK_APL_VALUE1, Z);
224 }

226 Token
227 Bif_F12_DROP::eval_AXB(Value_P A, Value_P X, Value_P B) const
228 {
229  if (X->element_count() == 0) // no axes
230  {
231  Token result(TOK_APL_VALUE1, B->clone(LOC));
232  return result;
233  }
234 
235  if (X->get_rank() > 1) INDEX_ERROR;
236 
237 const uint64_t len_X = X->element_count();
238  if (len_X > MAX_RANK) INDEX_ERROR;
239  if (len_X == 0) INDEX_ERROR;
240 
241  if (A->get_rank() > 1) RANK_ERROR;
242 
243 uint64_t len_A = A->element_count();
244  if (len_A != len_X) LENGTH_ERROR;
245 
246 const APL_Integer qio = Workspace::get_IO();
247 
248  // init ravel_A = shape_B and seen.
249  //
250 Shape ravel_A(B->get_shape());
251 bool seen[MAX_RANK];
252  loop(r, B->get_rank()) seen[r] = false;
253 
254  loop(r, len_X)
255  {
256  const APL_Integer a = A->get_cravel(r).get_near_int();
257  const APL_Integer x = X->get_cravel(r).get_near_int() - qio;
258 
259  if (x >= B->get_rank()) INDEX_ERROR;
260  if (seen[x]) INDEX_ERROR;
261  seen[x] = true;
262 
263  const ShapeItem amax = B->get_shape_item(x);
264  if (a >= amax) ravel_A.set_shape_item(x, 0);
265  else if (a >= 0) ravel_A.set_shape_item(x, a - amax);
266  else if (a > -amax) ravel_A.set_shape_item(x, amax + a);
267  else ravel_A.set_shape_item(x, 0);
268  }
269 
270  return Token(TOK_APL_VALUE1, Bif_F12_TAKE::do_take(ravel_A, B));
271 }

37 Value_P
38 Bif_F12_TAKE::first(Value_P B)
39 {
40 const Cell & first_B = B->get_ravel(0);
41  if (B->element_count() == 0) // empty value: return prototype
42  {
43  if (first_B.is_lval_cell()) // (
44  {
45  Value_P Z(LOC);
46  Z->next_ravel()->init(first_B, B.getref(), LOC);
47  return Z;
48  }
49  Value_P Z = B->prototype(LOC);
50  Z->check_value(LOC);
51  return Z;
52  }
53 
54  if (!first_B.is_pointer_cell()) // simple cell
55  {
56  Value_P Z(LOC);
57  Z->get_ravel(0).init(first_B, Z.getref(), LOC);
58  Z->check_value(LOC);
59  return Z;
60  }
61 
62 Value_P v1 = first_B.get_pointer_value();
63 Value * v1_owner = v1->get_lval_cellowner();
64  if (v1_owner) // B is a left value
65  {
66  Value_P B1(LOC);
67  new (&B1->get_ravel(0)) PointerCell(v1.get(), B1.getref());
68 
69  Value_P Z(LOC);
70  new (&Z->get_ravel(0)) LvalCell(&B1->get_ravel(0), v1_owner);
71 
72  Z->check_value(LOC);
73  return Z;
74  }
75  else
76  {
77  const ShapeItem ec = v1->element_count();
78  Value_P Z(v1->get_shape(), LOC);
79  if (ec == 0) Z->get_ravel(0).init(v1->get_ravel(0), Z.getref(), LOC);
80 
81  loop(e, ec) Z->next_ravel()->init(v1->get_ravel(e), Z.getref(), LOC);
82 
83  Z->check_value(LOC);
84  return Z;
85  }
86 }

90 {
91  if (X->element_count() == 0) // no axes
92  {
93  Token result(TOK_APL_VALUE1, B->clone(LOC));
94  return result;
95  }
96 
97  // A↑[X]B ←→ ⊃[X](⊂A)↑¨⊂[X]B
98  //
99 Value_P cA = Bif_F12_PARTITION::fun->do_eval_B(A); // ⊂A
100 Value_P cB = Bif_F12_PARTITION::fun->do_eval_XB(X, B); // ⊂[X]B
101 Token take(TOK_FUN2, Bif_F12_TAKE::fun);
102 Token cT = Bif_OPER1_EACH::fun->eval_ALB(cA, take, cB); // cA↑¨cB
103 
104 Token result = Bif_F12_PICK::fun->eval_XB(X, cT.get_apl_val());
105  return result;
106 }

108 Token
109 Bif_F12_TAKE::eval_AB(Value_P A, Value_P B)
110 {
111 Shape ravel_A1(A.get(), /* ⎕IO */ 0); // checks that 1 ≤ ⍴⍴A and ⍴A ≤ MAX_RANK
112 
113  if (B->is_scalar())
114  {
115  Shape shape_B1;
116  loop(a, ravel_A1.get_rank()) shape_B1.add_shape_item(1);
117  Value_P B1 = B->clone(LOC);
118  B1->set_shape(shape_B1);
119  return Token(TOK_APL_VALUE1, do_take(ravel_A1, B1));
120  }
121  else
122  {
123  if (ravel_A1.get_rank() != B->get_rank()) LENGTH_ERROR;
124  return Token(TOK_APL_VALUE1, do_take(ravel_A1, B));
125  }
126 }

129 Bif_F12_TAKE::do_take(const Shape & ravel_A1, Value_P B)
130 {
131  // ravel_A1 can have negative items (for take from the end).
132  //
133 Value_P Z(ravel_A1.abs(), LOC);
134 
135  if (ravel_A1.is_empty()) Z->set_default(*B.get(), LOC); // empty Z
136  else fill(ravel_A1, &Z->get_ravel(0), Z.getref(), B);
137  Z->check_value(LOC);
138  return Z;

153 Token
154 Bif_F12_DROP::eval_AB(Value_P A, Value_P B)
155 {
156 Shape ravel_A(A.get(), /* ⎕IO */ 0);
157  if (A->get_rank() > 1) RANK_ERROR;
158 
159  if (B->is_scalar())
160  {
161  Shape shape_Z;
162  loop(r, ravel_A.get_rank())
163  {
164  // the r'th shape item is either 0 (if anything is dropped)
165  // or 1 if not.
166  if (ravel_A.get_shape_item(r)) shape_Z.add_shape_item(0);
167  else shape_Z.add_shape_item(1);
168  }
169 
170  Value_P Z(shape_Z, LOC);
171 
172  Z->get_ravel(0).init(B->get_ravel(0), Z.getref(), LOC);
173  if (shape_Z.get_volume() == 0) Z->to_proto();
174  Z->check_value(LOC);
175  return Token(TOK_APL_VALUE1, Z);
176  }
177 
178  if (ravel_A.get_rank() == 0) // ''↓B
179  {
180  if (!B->is_scalar()) LENGTH_ERROR;
181  ravel_A.add_shape_item(1); // A = ,A
182  }
183 
184  if (ravel_A.get_rank() != B->get_rank()) LENGTH_ERROR;
185 
186 Shape sh_Z;
187  loop(r, ravel_A.get_rank())
188  {
189  const ShapeItem sA = ravel_A.get_shape_item(r);
190  const ShapeItem sB = B->get_shape_item(r);
191  const ShapeItem pA = sA < 0 ? -sA : sA;
192  if (pA >= sB) sh_Z.add_shape_item(0); // over-drop
193  else sh_Z.add_shape_item(sB - pA);
194  }
195 
196 Value_P Z(sh_Z, LOC);
197  if (sh_Z.is_empty()) // empty Z, e.g. from overdrop
198  {
199  Value_P Z(sh_Z, LOC);
200  Z->set_default(*B.get(), LOC);
201  Z->check_value(LOC);
202  return Token(TOK_APL_VALUE1, Z);
203  }
204 
205  for (TakeDropIterator i(false, ravel_A, B->get_shape()); i.more(); ++i)
206  {
207  const ShapeItem offset = i();
208  Z->next_ravel()->init(B->get_ravel(offset), Z.getref(), LOC);
209  }
210 
211  Z->check_value(LOC);
212  return Token(TOK_APL_VALUE1, Z);
213 }

215 Token
216 Bif_F12_DROP::eval_AXB(Value_P A, Value_P X, Value_P B)
217 {
218  if (X->element_count() == 0) // no axes
219  {
220  Token result(TOK_APL_VALUE1, B->clone(LOC));
221  return result;
222  }
223 
224  if (X->get_rank() > 1) INDEX_ERROR;
225 
226 const uint64_t len_X = X->element_count();
227  if (len_X > MAX_RANK) INDEX_ERROR;
228  if (len_X == 0) INDEX_ERROR;
229 
230  if (A->get_rank() > 1) RANK_ERROR;
231 
232 uint64_t len_A = A->element_count();
233  if (len_A != len_X) LENGTH_ERROR;
234 
235 const APL_Integer qio = Workspace::get_IO();
236 
237  // init ravel_A = shape_B and seen.
238  //
239 Shape ravel_A(B->get_shape());
240 bool seen[MAX_RANK];
241  loop(r, B->get_rank()) seen[r] = false;
242 
243  loop(r, len_X)
244  {
245  const APL_Integer a = A->get_ravel(r).get_near_int();
246  const APL_Integer x = X->get_ravel(r).get_near_int() - qio;
247 
248  if (x >= B->get_rank()) INDEX_ERROR;
249  if (seen[x]) INDEX_ERROR;
250  seen[x] = true;
251 
252  const ShapeItem amax = B->get_shape_item(x);
253  if (a >= amax) ravel_A.set_shape_item(x, 0);
254  else if (a >= 0) ravel_A.set_shape_item(x, a - amax);
255  else if (a > -amax) ravel_A.set_shape_item(x, amax + a);
256  else ravel_A.set_shape_item(x, 0);
257  }
258 
259  return Token(TOK_APL_VALUE1, Bif_F12_TAKE::do_take(ravel_A, B));
260 }

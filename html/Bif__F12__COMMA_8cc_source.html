46 Value_P
47 Bif_COMMA::prepend_scalar(const Cell & cell_A, uAxis axis, Value_P B)
48 {
49  if (B->is_empty())
50  {
51  Shape shape_Z = B->get_shape();
52  shape_Z.set_shape_item(axis, shape_Z.get_shape_item(axis) + 1);
53  Value_P Z(shape_Z, LOC);
54  if (Z->is_empty())
55  {
56  Z->get_ravel(0).init(cell_A, Z.getref(), LOC); // prototype
57  }
58  else
59  {
60  loop(z, Z->element_count())
61  Z->next_ravel()->init(cell_A, Z.getref(), LOC);
62  }
63  return Z;
64  }
65 
66  if (B->is_scalar())
67  {
68  Value_P Z(2, LOC);
69  Z->next_ravel()->init(cell_A, Z.getref(), LOC);
70  Z->next_ravel()->init(B->get_ravel(0), Z.getref(), LOC);
71  Z->check_value(LOC);
72  return Z;
73  }
74 
75  if (axis >= B->get_rank()) INDEX_ERROR;
76 
77 Shape shape_Z(B->get_shape());
78  shape_Z.set_shape_item(axis, shape_Z.get_shape_item(axis) + 1);
79 
80 Value_P Z(shape_Z, LOC);
81 
82 const Shape3 shape_B3(B->get_shape(), axis);
83  const ShapeItem slice_a = shape_B3.l();
84  const ShapeItem slice_b = shape_B3.l() * B->get_shape_item(axis);
85 
86 const Cell * cB = &B->get_ravel(0);
87 
88  loop(hz, shape_B3.h())
89  {
90  loop(lz, slice_a) Z->next_ravel()->init(cell_A, Z.getref(), LOC);
91 
92  Cell::copy(*Z.get(), cB, slice_b);
93  }
94 
95  Z->check_value(LOC);
96  return Z;
97 }

143 Token
144 Bif_COMMA::catenate(Value_P A, Axis axis, Value_P B)
145 {
146  // NOTE: the case A->is_scalar() && B->is_scalar() was supposedly ruled out
147  // before calling catenate()
148 
149  if (A->is_scalar())
150  {
151  const Cell & cell_A = A->get_ravel(0);
152  Value_P Z = prepend_scalar(cell_A, axis, B);
153  Z->check_value(LOC);
154  return Token(TOK_APL_VALUE1, Z);
155  }
156 
157  if (B->is_scalar())
158  {
159  const Cell & cell_B = B->get_ravel(0);
160  Value_P Z = append_scalar(A, axis, cell_B);
161  Z->check_value(LOC);
162  return Token(TOK_APL_VALUE1, Z);
163  }
164 
165  if ((A->get_rank() + 1) == B->get_rank())
166  {
167  Shape shape_Z;
168 
169  // check shape conformance.
170  {
171  ShapeItem ra = 0;
172  loop(rb, B->get_rank())
173  {
174  if (rb != axis)
175  {
176  shape_Z.add_shape_item(B->get_shape_item(rb));
177  if (A->get_shape_item(ra) != B->get_shape_item(rb))
178  {
179  LENGTH_ERROR;
180  }
181  ++ra;
182  }
183  else
184  {
185  shape_Z.add_shape_item(B->get_shape_item(rb) + 1);
186  }
187  }
188  }
189 
190  Value_P Z(shape_Z, LOC);
191 
192  Z->set_default(*B.get(), LOC);
193 
194  const Shape3 shape_B3(B->get_shape(), axis);
195  const ShapeItem slice_a = shape_B3.l();
196  const ShapeItem slice_b = shape_B3.l() * B->get_shape_item(axis);
197 
198  const Cell * cA = &A->get_ravel(0);
199  const Cell * cB = &B->get_ravel(0);
200 
201  loop(hz, shape_B3.h())
202  {
203  Cell::copy(*Z.get(), cA, slice_a);
204  Cell::copy(*Z.get(), cB, slice_b);
205  }
206 
207  Z->check_value(LOC);
208  return Token(TOK_APL_VALUE1, Z);
209  }
210 
211  if (A->get_rank() == (B->get_rank() + 1))
212  {
213  // e.g.» ∆∆∆ , 3
214  // ∆∆∆ 4
215  //
216  Shape shape_Z;
217 
218  // check shape conformance. We step ranks ra and rb, except for the
219  // axis where only ra is incremented.
220  {
221  uint32_t rb = 0;
222  loop(ra, A->get_rank())
223  {
224  if (ra != axis)
225  {
226  if (A->get_shape_item(ra) != B->get_shape_item(rb))
227  LENGTH_ERROR;
228 
229  shape_Z.add_shape_item(A->get_shape_item(ra));
230  ++rb;
231  }
232  else
233  {
234  shape_Z.add_shape_item(A->get_shape_item(ra) + 1);
235  }
236  }
237  }
238 
239  Value_P Z(shape_Z, LOC);
240 
241  Z->set_default(*B.get(), LOC);
242 
243  const Shape3 shape_A3(A->get_shape(), axis);
244  const ShapeItem slice_a = shape_A3.l() * A->get_shape_item(axis);
245  const ShapeItem slice_b = shape_A3.l();
246 
247  const Cell * cA = &A->get_ravel(0);
248  const Cell * cB = &B->get_ravel(0);
249 
250  loop (hz, shape_A3.h())
251  {
252  Cell::copy(*Z.get(), cA, slice_a);
253  Cell::copy(*Z.get(), cB, slice_b);
254  }
255 
256  Z->set_default(*B.get(), LOC);
257 
258  Z->check_value(LOC);
259  return Token(TOK_APL_VALUE1, Z);
260  }
261  if (A->get_rank() != B->get_rank()) RANK_ERROR;
262 
263  // A and B have the same rank. The shapes need to agree, except for the
264  // dimension corresponding to the axis.
265  //
266 Shape shape_Z;
267 
268  loop(r, A->get_rank())
269  {
270  if (r != axis)
271  {
272  if (A->get_shape_item(r) != B->get_shape_item(r))
273  LENGTH_ERROR;
274 
275  shape_Z.add_shape_item(A->get_shape_item(r));
276  }
277  else
278  {
279  shape_Z.add_shape_item(A->get_shape_item(r) +
280  + B->get_shape_item(r));
281  }
282  }
283 
284 const Shape3 shape_A3(A->get_shape(), axis);
285 
286 const Cell * cA = &A->get_ravel(0);
287 const Cell * cB = &B->get_ravel(0);
288 const ShapeItem slice_a = shape_A3.l() * A->get_shape_item(axis);
289 const ShapeItem slice_b = shape_A3.l() * B->get_shape_item(axis);
290 
291 Value_P Z(shape_Z, LOC);
292  loop(hz, shape_A3.h())
293  {
294  Cell::copy(*Z.get(), cA, slice_a);
295  Cell::copy(*Z.get(), cB, slice_b);
296  }
297 
298  Z->set_default(*B.get(), LOC);
299  Z->check_value(LOC);
300  return Token(TOK_APL_VALUE1, Z);
301 }

303 Token
304 Bif_COMMA::laminate(Value_P A, Axis axis, Value_P B)
305 {
306  // shapes of A and B must be the same, unless one of them is a scalar.
307  //
308  if (!A->is_scalar() && !B->is_scalar())
309  A->get_shape().check_same(B->get_shape(),
310  E_INDEX_ERROR, E_LENGTH_ERROR, LOC);
311 
312 const Shape shape_Z = A->is_scalar() ? B->get_shape().insert_axis(axis, 2)
313  : A->get_shape().insert_axis(axis, 2);
314 
315 Value_P Z(shape_Z, LOC);
316 
317 const Shape3 shape_Z3(shape_Z, axis);
318  if (shape_Z3.m() != 2) AXIS_ERROR;
319 
320 const Cell * cA = &A->get_ravel(0);
321 const Cell * cB = &B->get_ravel(0);
322  if (A->is_scalar())
323  {
324  if (B->is_scalar())
325  {
326  Z->next_ravel()->init(*cA, Z.getref(), LOC);
327  Z->next_ravel()->init(*cB, Z.getref(), LOC);
328  }
329  else
330  {
331  loop(h, shape_Z3.h())
332  {
333  loop(l, shape_Z3.l())
334  Z->next_ravel()->init(*cA, Z.getref(), LOC);
335  Cell::copy(*Z.get(), cB, shape_Z3.l());
336  }
337  }
338  }
339  else
340  {
341  if (B->is_scalar())
342  {
343  loop(h, shape_Z3.h())
344  {
345  Cell::copy(*Z.get(), cA, shape_Z3.l());
346  loop(l, shape_Z3.l())
347  Z->next_ravel()->init(*cB, Z.getref(), LOC);
348  }
349  }
350  else
351  {
352  loop(h, shape_Z3.h())
353  {
354  Cell::copy(*Z.get(), cA, shape_Z3.l());
355  Cell::copy(*Z.get(), cB, shape_Z3.l());
356  }
357  }
358  }
359 
360  Z->set_default(*B.get(), LOC);
361 
362  Z->check_value(LOC);
363  return Token(TOK_APL_VALUE1, Z);
364 }

366 Token
367 Bif_COMMA::ravel_axis(Value_P X, Value_P B, uAxis axis)
368 {
369 const APL_Integer qio = Workspace::get_IO();
370 
371  if (!X) SYNTAX_ERROR;
372 
373  // I must be a (integer or real) scalar or simple integer vector
374  //
375  if (X->get_rank() > 1) INDEX_ERROR;
376 
377  // There are 3 variants, determined by I:
378  //
379  // 1) I is empty: append a new last axis of length 1
380  // 2) I is a fraction: append a new axis before I
381  // 3a) I is integer scalar: return B
382  // 3b) I is integer vector: combine axes
383 
384  // case 1: ,['']B or ,[⍳0]B : append new first or last axis of length 1.
385  //
386  if (X->element_count() == 0)
387  {
388  if (B->get_rank() == MAX_RANK) AXIS_ERROR;
389 
390  const Shape shape_Z = B->get_shape().insert_axis(axis, 1);
391  return ravel(shape_Z, B);
392  }
393 
394  // case 2: ,[x.y]B : insert axis before axis x+1
395  //
396  if (!X->get_ravel(0).is_near_int()) // fraction: insert an axis
397  {
398  if (B->get_rank() == MAX_RANK) INDEX_ERROR;
399 
400  const APL_Float new_axis = X->get_ravel(0).get_real_value() - qio;
401  Axis axis = new_axis; if (new_axis < 0.0) axis = -1;
402  const Shape shape_Z = B->get_shape().insert_axis(axis + 1, 1);
403  return ravel(shape_Z, B);
404  }
405  // case 3a: ,[n]B : return B (combine single axis doesn't change anything)
406  //
407  if (X->is_scalar_or_len1_vector()) // single int: return B->
408  {
409  Token result(TOK_APL_VALUE1, B->clone(LOC));
410  return result;
411  }
412 
413  // case 3b: ,[n1 ... nk]B : combine axes.
414  //
415 const Shape axes(X.get(), qio);
416 
417 const ShapeItem from = axes.get_shape_item(0);
418  if (from < 0) AXIS_ERROR;
419 
420 const ShapeItem to = axes.get_last_shape_item();
421  if (to >= B->get_rank()) AXIS_ERROR;
422 
423  // check that the axes are contiguous and compute the number of elements
424  // in the combined axes
425  //
426 ShapeItem count = 1;
427  loop(a, axes.get_rank())
428  {
429  if (axes.get_shape_item(a) != (from + a)) AXIS_ERROR;
430  count *= B->get_shape_item(from + a);
431  }
432 
433 Shape shape_Z;
434  loop (r, from) shape_Z.add_shape_item(B->get_shape_item(r));
435  shape_Z.add_shape_item(count);
436  for (uRank r = to + 1; r < B->get_rank(); ++r)
437  shape_Z.add_shape_item(B->get_shape_item(r));
438 
439  return ravel(shape_Z, B);
440 }

470 Token
471 Bif_F12_COMMA::eval_AB(Value_P A, Value_P B)
472 {
473  if (A->is_scalar() && B->is_scalar())
474  {
475  Value_P Z(2, LOC);
476  Z->next_ravel()->init(A->get_ravel(0), Z.getref(), LOC);
477  Z->next_ravel()->init(B->get_ravel(0), Z.getref(), LOC);
478  Z->check_value(LOC);
479  return Token(TOK_APL_VALUE1, Z);
480  }
481 
482 uRank max_rank = A->get_rank();
483  if (max_rank < B->get_rank()) max_rank = B->get_rank();
484  return catenate(A, max_rank - 1, B);
485 }

487 Token
488 Bif_F12_COMMA::eval_AXB(Value_P A, Value_P X, Value_P B)
489 {
490  if (A->is_scalar() && B->is_scalar()) RANK_ERROR;
491 
492  // catenate or laminate
493  //
494  if (!X->is_scalar_or_len1_vector()) AXIS_ERROR;
495 
496 const Cell & cX = X->get_ravel(0);
497 const APL_Integer qio = Workspace::get_IO();
498 
499  if (cX.is_near_int()) // catenate along existing axis
500  {
501  const Axis axis = cX.get_checked_near_int() - qio;
502  if (axis < 0) AXIS_ERROR;
503  if (uAxis(axis) >= A->get_rank() && uAxis(axis) >= B->get_rank())
504  AXIS_ERROR;
505  return catenate(A, axis, B);
506  }
507 
508 const APL_Float axis = cX.get_real_value() - qio;
509  if (axis <= -1.0) AXIS_ERROR;
510  if (axis >= (A->get_rank() + 1.0) &&
511  axis >= (B->get_rank() + 1.0)) AXIS_ERROR;
512  return laminate(A, Axis(axis + 1.0), B);
513 }

516 Bif_F12_COMMA1::eval_B(Value_P B)
517 {
518  // turn B into a matrix
519  //

521 ShapeItem c2 = 1; // assume B is scalar;
522 
523  if (B->get_rank() >= 1)
524  {
525  c1 = B->get_shape_item(0);
526  for (uRank r = 1; r < B->get_rank(); ++r) c2 *= B->get_shape_item(r);
527  }
528 
529 Shape shape_Z(c1, c2);
530  if (B->get_owner_count() == 2 &&
531  this == Workspace::SI_top()->get_prefix().get_monadic_fun())
532  {
533  Log(LOG_optimization) CERR << "optimizing ,B" << endl;
534 
535  B->set_shape(shape_Z);
536  return Token(TOK_APL_VALUE1, B);
537  }
538  return ravel(shape_Z, B);
539 }

240 char *
241 format_tick(double val)
242 {
243 static char cc[40];
244 char * cp = cc;
245  if (val == 0)
246  {
247  *cp++ = '0';
248  *cp++ = 0;
249  return cc;
250  }
251 
252  if (val < 0)
253  {
254  *cp++ = '-';
255  val = -val;
256  }
257 
258 const char * unit = 0;
259  if (val >= 1e3)
260  {
261  if (val >= 1e15) ;
262  else if (val >= 1e12) { unit = "T"; val *= 1e-12; }
263  else if (val >= 1e9) { unit = "G"; val *= 1e-9; }
264  else if (val >= 1e6) { unit = "M"; val *= 1e-6; }
265  else { unit = "k"; val *= 1e-3; }
266  }
267  else
268  {
269 
270  if (val >= 1e0) { unit = ""; }
271  else if (val >= 1e-3) { unit = "m"; val *= 1e3; }
272  else if (val >= 1e-6) { unit = "Î¼"; val *= 1e6; }
273  else if (val >= 1e-9) { unit = "n"; val *= 1e9; }
274  else if (val >= 1e-12) { unit = "p"; val *= 1e12; }
275  }
276 
277  if (unit == 0) // very large or very small
278  {
279  snprintf(cp, sizeof(cc) - 1, "%.2E", val);
280  return cc;
281  }
282 
283  // at this point: 1 â¤ val < 1000
284  //
285  if (val < 10) snprintf(cp, sizeof(cc) - 1, "%.2f%s", val, unit);
286  else if (val < 100) snprintf(cp, sizeof(cc) - 1, "%.1f%s", val, unit);
287  else snprintf(cp, sizeof(cc) - 1, "%.0f%s", val, unit);
288  return cc;
289 }

306 inline double
307 intersection(double Ax, double Ay, double Bx, double By,
308  double Cx, double Cy, double Dx, double Dy)
309 {
328  /* first, translate A, B, and D by -C, so that C=(0, 0):
329 
330  A
331  \ B
332  \ /
333  M
334  / \
335  / \
336  0 D
337  */
338  Ax -= Cx; Bx -= Cx; Dx -= Cx;
339  Ay -= Cy; By -= Cy; Dy -= Cy;
340 
341  /* Thus:
342 
343  (Ax-Bx) + alpha*(Dx-Ax) == beta*(-Bx)
344  (Ay-By) + alpha*(Dy-Ay) == beta*(-By)
345 
346  (Ax-Bx)*(Dy-Ay) + alpha*(Dx-Ax)*(Dy-Ay) == beta*(-Bx)*(Dy-Ay)
347  (Ay-By)*(Dx-Ax) + alpha*(Dy-Ay)*(Dx-Ax) == beta*(-By)*(Dx-Ax)
348  ======================================= ==================
349  (Ax-Bx)*(Dy-Ay) - (Ay-By)*(Dx-Ax) == beta*((-Bx)*(Dy-Ay) - (-By)*(Dx-Ax))
350  **/
351 
352 const double numer = (Ax-Bx)*(Dy-Ay) - (Ay-By)*(Dx-Ax); // left side
353 const double denom = ((-Bx)*(Dy-Ay) - (-By)*(Dx-Ax)); // right factor
354  if (denom == 0.0)
355  {
356  Ax += Cx; Bx += Cx; Dx += Cx;
357  Ay += Cy; By += Cy; Dy += Cy;
358  MORE_ERROR() << "âPLOT: one of the X-Z squares has bad coordinates:"
359  << "P1=(" << Ax << ":" << Ay << "), "
360  "P2=(" << Bx << ":" << By << "), "
361  "P3=(" << Cx << ":" << Cy << ") "
362  "P4=(" << Dx << ":" << Dy << ")";
363  DOMAIN_ERROR;
364  }
365 
366  return /* beta == */ numer/denom;
367 }

369 void
370 draw_triangle(const Plot_context & pctx, int verbosity,
371  Pixel_XY P0, double H0, Pixel_XY P1, Pixel_XY P2, double H12)
372 {
373  // draw a triangle with P1 and P2 at the same level H12
374  //
375  Assert(H0 >= 0.0); Assert(H0 <= 1.0);
376  Assert(H12 >= 0.0); Assert(H12 <= 1.0);
377 
378  if (verbosity & SHOW_DRAW)
379  CERR << " â2: P0(" << P0.x << ":" << P0.y << ") @H0=" << H0
380  << " P1(" << P1.x << ":" << P1.y << ") @H12=" << H12
381  << " P2(" << P2.x << ":" << P2.y << ") @H12=" << H12 << endl;
382 
383 const double d1 = P0.distance2(P1);
384 const double d2 = P0.distance2(P2);
385 const int steps = sqrt(d1 > d2 ? d1 : d2);
386 const double dH = (H12 - H0) / steps;
387  loop(s, steps)
388  {
389  const double alpha = H0 + s*dH;
390  const double beta = 1.0 * s/ steps;
391  const uint32_t color = pctx.w_props.get_color(alpha);
392  xcb_change_gc(pctx.conn, pctx.fill, XCB_GC_FOREGROUND, &color);
393 
394  const Pixel_XY P0_P1(P0.x + beta * (P1.x - P0.x),
395  P0.y + beta * (P1.y - P0.y));
396  const Pixel_XY P0_P2(P0.x + beta * (P2.x - P0.x),
397  P0.y + beta * (P2.y - P0.y));
398  draw_line(pctx, pctx.fill, P0_P1, P0_P2);
399  }
400 }
401 //-----------------------------------------------------------------------------
402 void
403 draw_triangle(const Plot_context & pctx, int verbosity, Pixel_XY P0, double H0,
404  Pixel_XY P1, double H1, Pixel_XY P2, double H2)
405 {
406  // draw a triangle with P0, P1 and P2 at levels H0, H1, and H2
407 
408  Assert(H0 >= 0); Assert(H0 <= 1.0);
409  Assert(H1 >= 0); Assert(H1 <= 1.0);
410  Assert(H2 >= 0); Assert(H2 <= 1.0);
411 
412  if (verbosity & SHOW_DRAW)
413  CERR << "\nâ1: P0(" << P0.x << ":" << P0.y << ")@H=" << H0
414  << " P1(" << P1.x << ":" << P1.y << ")@H=" << H1
415  << " P2(" << P2.x << ":" << P2.y << ")@H=" << H2 << endl;
416 
417  if (H0 < H1) pv_swap(P0, H0, P1, H1); // then H0 >= H1
418  if (H0 < H2) pv_swap(P0, H0, P2, H2); // then H0 >= H2
419  if (H1 < H2) pv_swap(P1, H1, P2, H2); // then H1 >= H2
420 
421  // here H0 >= H1 >= H2
422  //
423  Assert(H0 >= H1);
424  Assert(H1 >= H2);
425 
426  if (H0 == H1) // P0 and P1 have the same height
427  {
428  draw_triangle(pctx, verbosity, P2, H2, P0, P1, H0);
429  }
430  else if (H1 == H2) // P1 and P2 have the same height
431  {
432  draw_triangle(pctx, verbosity, P0, H0, P1, P2, H1);
433  }
434  else // P2 lies below P1
435  {
436  const double alpha = (H0 - H1) / (H0 - H2); // alpha â 1 as P1 â P2
437  Assert(alpha >= 0.0);
438  Assert(alpha <= 1.0);
439 
440  // compute the point P on P0-P2 that has the heigth H1 (of P1)
441  const Pixel_XY P(P0.x + alpha*(P2.x - P0.x),
442  P0.y + alpha*(P2.y - P0.y));
443  draw_triangle(pctx, verbosity, P0, H0, P1, P, H1);
444  draw_triangle(pctx, verbosity, P2, H2, P1, P, H1);
445  }
446 }

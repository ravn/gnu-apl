117  xcb_str[s].byte2 = string[s];

119 
120 const xcb_query_text_extents_cookie_t cookie =
121  xcb_query_text_extents(pctx.conn, pctx.font, string_length, xcb_str);
122 
123  if ( xcb_query_text_extents_reply_t * reply =
124  xcb_query_text_extents_reply(pctx.conn, cookie, 0))
125  {
126  width = reply->overall_width;
127  height = reply->font_ascent + reply->font_descent;
128  // origin = reply->font_ascent;

130  }

132  {
133  puts("XCB ERROR: xcb_query_text_extents_reply() failed.");
134  width = 24;

240 char *
241 format_tick(double val)
242 {
243 static char cc[40];
244 char * cp = cc;
245  if (val == 0)
246  {
247  *cp++ = '0';
248  *cp++ = 0;
249  return cc;
250  }
251 
252  if (val < 0)
253  {
254  *cp++ = '-';
255  val = -val;
256  }
257 
258 const char * unit = 0;
259  if (val >= 1e3)
260  {
261  if (val >= 1e15) ;
262  else if (val >= 1e12) { unit = "T"; val *= 1e-12; }
263  else if (val >= 1e9) { unit = "G"; val *= 1e-9; }
264  else if (val >= 1e6) { unit = "M"; val *= 1e-6; }
265  else { unit = "k"; val *= 1e-3; }
266  }
267  else
268  {
269 
270  if (val >= 1e0) { unit = ""; }
271  else if (val >= 1e-3) { unit = "m"; val *= 1e3; }
272  else if (val >= 1e-6) { unit = "μ"; val *= 1e6; }
273  else if (val >= 1e-9) { unit = "n"; val *= 1e9; }
274  else if (val >= 1e-12) { unit = "p"; val *= 1e12; }
275  }
276 
277  if (unit == 0) // very large or very small
278  {
279  snprintf(cp, sizeof(cc) - 1, "%.2E", val);
280  return cc;
281  }
282 
283  // at this point: 1 ≤ val < 1000
284  //
285  if (val < 10) snprintf(cp, sizeof(cc) - 1, "%.2f%s", val, unit);
286  else if (val < 100) snprintf(cp, sizeof(cc) - 1, "%.1f%s", val, unit);
287  else snprintf(cp, sizeof(cc) - 1, "%.0f%s", val, unit);
288  return cc;
289 }

306 inline double
307 intersection(double Ax, double Ay, double Bx, double By,
308  double Cx, double Cy, double Dx, double Dy)
309 {
328  /* first, translate A, B, and D by -C, so that C=(0, 0):
329 
330  A
331  \ B
332  \ /
333  M
334  / \
335  / \
336  0 D
337  */
338  Ax -= Cx; Bx -= Cx; Dx -= Cx;
339  Ay -= Cy; By -= Cy; Dy -= Cy;
340 
341  /* Thus:
342 
343  (Ax-Bx) + alpha*(Dx-Ax) == beta*(-Bx)
344  (Ay-By) + alpha*(Dy-Ay) == beta*(-By)
345 
346  (Ax-Bx)*(Dy-Ay) + alpha*(Dx-Ax)*(Dy-Ay) == beta*(-Bx)*(Dy-Ay)
347  (Ay-By)*(Dx-Ax) + alpha*(Dy-Ay)*(Dx-Ax) == beta*(-By)*(Dx-Ax)
348  ======================================= ==================
349  (Ax-Bx)*(Dy-Ay) - (Ay-By)*(Dx-Ax) == beta*((-Bx)*(Dy-Ay) - (-By)*(Dx-Ax))
350  **/
351 
352 const double numer = (Ax-Bx)*(Dy-Ay) - (Ay-By)*(Dx-Ax); // left side
353 const double denom = ((-Bx)*(Dy-Ay) - (-By)*(Dx-Ax)); // right factor
354  if (denom == 0.0)
355  {
356  Ax += Cx; Bx += Cx; Dx += Cx;
357  Ay += Cy; By += Cy; Dy += Cy;
358  MORE_ERROR() << "⎕PLOT: one of the X-Z squares has bad coordinates:"
359  << "P1=(" << Ax << ":" << Ay << "), "
360  "P2=(" << Bx << ":" << By << "), "
361  "P3=(" << Cx << ":" << Cy << ") "
362  "P4=(" << Dx << ":" << Dy << ")";
363  DOMAIN_ERROR;
364  }
365 
366  return /* beta == */ numer/denom;
367 }

388 void
389 draw_triangle(const Plot_context & pctx, int verbosity,
390  Pixel_XY P0, double H0, Pixel_XY P1, Pixel_XY P2, double H12)
391 {
392  // draw a triangle with P1 and P2 at the same level H12
393  //
394  Assert(H0 >= 0.0); Assert(H0 <= 1.0);
395  Assert(H12 >= 0.0); Assert(H12 <= 1.0);
396 
397  if (verbosity & SHOW_DRAW)
398  CERR << " ∆2: P0(" << P0.x << ":" << P0.y << ") @H0=" << H0
399  << " P1(" << P1.x << ":" << P1.y << ") @H12=" << H12
400  << " P2(" << P2.x << ":" << P2.y << ") @H12=" << H12 << endl;
401 
402 const double d1 = P0.distance2(P1);
403 const double d2 = P0.distance2(P2);
404 const int steps = sqrt(d1 > d2 ? d1 : d2);
405 const double dH = (H12 - H0) / steps;
406  loop(s, steps)
407  {
408  const double alpha = H0 + s*dH;
409  const double beta = 1.0 * s/ steps;
410  const uint32_t color = pctx.w_props.get_color(alpha);
411  set_GC_foreground(pctx, pctx.fill, color);
412 
413  const Pixel_XY P0_P1(P0.x + beta * (P1.x - P0.x),
414  P0.y + beta * (P1.y - P0.y));
415  const Pixel_XY P0_P2(P0.x + beta * (P2.x - P0.x),
416  P0.y + beta * (P2.y - P0.y));
417  draw_line(pctx, pctx.fill, P0_P1, P0_P2);
418  }
419 }
420 //-----------------------------------------------------------------------------
421 void
422 draw_triangle(const Plot_context & pctx, int verbosity, Pixel_XY P0, double H0,
423  Pixel_XY P1, double H1, Pixel_XY P2, double H2)
424 {
425  // draw a triangle with P0, P1 and P2 at levels H0, H1, and H2
426 
427  Assert(H0 >= 0); Assert(H0 <= 1.0);
428  Assert(H1 >= 0); Assert(H1 <= 1.0);
429  Assert(H2 >= 0); Assert(H2 <= 1.0);
430 
431  if (verbosity & SHOW_DRAW)
432  CERR << "\n∆1: P0(" << P0.x << ":" << P0.y << ")@H=" << H0
433  << " P1(" << P1.x << ":" << P1.y << ")@H=" << H1
434  << " P2(" << P2.x << ":" << P2.y << ")@H=" << H2 << endl;
435 
436  if (H0 < H1) pv_swap(P0, H0, P1, H1); // then H0 >= H1
437  if (H0 < H2) pv_swap(P0, H0, P2, H2); // then H0 >= H2
438  if (H1 < H2) pv_swap(P1, H1, P2, H2); // then H1 >= H2
439 
440  // here H0 >= H1 >= H2
441  //
442  Assert(H0 >= H1);
443  Assert(H1 >= H2);
444 
445  if (H0 == H1) // P0 and P1 have the same height
446  {
447  draw_triangle(pctx, verbosity, P2, H2, P0, P1, H0);
448  }
449  else if (H1 == H2) // P1 and P2 have the same height
450  {
451  draw_triangle(pctx, verbosity, P0, H0, P1, P2, H1);
452  }
453  else // P2 lies below P1
454  {
455  const double alpha = (H0 - H1) / (H0 - H2); // alpha → 1 as P1 → P2
456  Assert(alpha >= 0.0);
457  Assert(alpha <= 1.0);
458 
459  // compute the point P on P0-P2 that has the heigth H1 (of P1)
460  const Pixel_XY P(P0.x + alpha*(P2.x - P0.x),
461  P0.y + alpha*(P2.y - P0.y));
462  draw_triangle(pctx, verbosity, P0, H0, P1, P, H1);
463  draw_triangle(pctx, verbosity, P2, H2, P1, P, H1);
464  }
465 }

467 void
468 draw_point(const Plot_context & pctx, const Pixel_XY & xy, Color canvas_color,
469  const Plot_line_properties & l_props)
470 {
471 const Pixel_X point_size = l_props.get_point_size(); // outer point diameter
472 const Pixel_X point_size2 = l_props.get_point_size2(); // inner point diameter
473 const int16_t half = point_size >> 1; // outer point radius
474 const int point_style = l_props.get_point_style();
475 const Color point_color = l_props.get_point_color();
476 
477  // set outer point color for all point styles (and maybe inner point colors
478  // later
479  //
480  set_GC_foreground(pctx, pctx.point, point_color);
481  if (point_style == 1) // circle
482  {
483  xcb_arc_t arc = { I1616(xy.x - half, xy.y - half), // positions
484  point_size, point_size, // diameters
485  0, 360 << 6 }; // angles
486  xcb_poly_fill_arc(pctx.conn, pctx.window, pctx.point, 1, &arc);
487 
488  if (point_size2) // hollow
489  {
490  const int16_t half2 = point_size2 >> 1;
491  xcb_arc_t arc2 = { I1616(xy.x - half2, xy.y - half2), // pos
492  point_size2, point_size2, // dia
493  0, int16_t(360 << 6) }; // ang
494  set_GC_foreground(pctx, pctx.point, canvas_color);
495  xcb_poly_fill_arc(pctx.conn, pctx.window, pctx.point, 1, &arc2);
496  set_GC_foreground(pctx, pctx.point, point_color);
497  }
498  }
499  else if (point_style == 2) // triangle ∆
500  {
501  const int16_t center_y = xy.y - 2*point_size/3;
502  const int16_t hypo_y = half*1.732;
503  const xcb_point_t points[3] = { { int16_t(xy.x), center_y },
504  { half, hypo_y },
505  { int16_t(-2*half), 0 } };
506  xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
507  XCB_POLY_SHAPE_CONVEX,
508  XCB_COORD_MODE_PREVIOUS, ITEMS(points));
509  if (point_size2) // hollow
510  {
511  const int16_t half2 = point_size2 >> 1;
512  const int16_t center2_y = xy.y - 2*point_size2/3;
513  const int16_t hypo2_y = half2*1.732;
514  const xcb_point_t points2[3] = {
515  { int16_t(xy.x), center2_y },
516  { int16_t(half2), hypo2_y },
517  { int16_t(-2*half2), 0 } };
518  set_GC_foreground(pctx, pctx.point, canvas_color);
519  xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
520  XCB_POLY_SHAPE_CONVEX,
521  XCB_COORD_MODE_PREVIOUS, ITEMS(points2));
522  set_GC_foreground(pctx, pctx.point, point_color);
523  }
524  }
525  else if (point_style == 3) // triangle ∇
526  {
527  const int16_t center_y = xy.y + 2*point_size/3;
528  const int16_t hypo_y = - half*1.732;
529  const xcb_point_t points[3] = {
530  { int16_t(xy.x), center_y },
531  { int16_t(half), hypo_y },
532  { int16_t(-2*half), 0 } };
533  xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
534  XCB_POLY_SHAPE_CONVEX,
535  XCB_COORD_MODE_PREVIOUS, ITEMS(points));
536  if (point_size2) // hollow
537  {
538  const int16_t half2 = point_size2 >> 1;
539  const int16_t center2_y = xy.y + 2*point_size2/3;
540  const int16_t hypo2_y = - half2*1.732;
541  const xcb_point_t points2[3] = {
542  { int16_t(xy.x), center2_y },
543  { int16_t(half2), hypo2_y },
544  { int16_t(-2*half2), 0 } };
545  set_GC_foreground(pctx, pctx.point, canvas_color);
546  xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
547  XCB_POLY_SHAPE_CONVEX,
548  XCB_COORD_MODE_PREVIOUS, ITEMS(points2));
549  set_GC_foreground(pctx, pctx.point, point_color);
550  }
551  }
552  else if (point_style == 4) // caro
553  {
554  const xcb_point_t points[4] = {
555  { int16_t(xy.x), int16_t(xy.y - half) },
556  { half, half },
557  { int16_t(-half), half },
558  { int16_t(-half), int16_t(-half) } };
559  xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
560  XCB_POLY_SHAPE_CONVEX,
561  XCB_COORD_MODE_PREVIOUS, ITEMS(points));
562  if (point_size2) // hollow
563  {
564  const int16_t half2 = point_size2 >> 1;
565  const xcb_point_t points2[4] = {
566  { I1616(xy.x, xy.y - half2) },
567  { I1616( half2, half2) },
568  { I1616(-half2, half2) },
569  { I1616(-half2, -half2) } };
570  set_GC_foreground(pctx, pctx.point, canvas_color);
571  xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
572  XCB_POLY_SHAPE_CONVEX,
573  XCB_COORD_MODE_PREVIOUS,
574  ITEMS(points2));
575  set_GC_foreground(pctx, pctx.point, point_color);
576  }
577  }
578  else if (point_style == 5) // square
579  {
580  const xcb_point_t points[4] = {
581  { I1616(xy.x - half, xy.y - half) },
582  { I1616(point_size, 0) },
583  { I1616(0, point_size) },
584  { I1616(-point_size, 0) } };
585  xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
586  XCB_POLY_SHAPE_CONVEX,
587  XCB_COORD_MODE_PREVIOUS, ITEMS(points));
588  if (point_size2) // hollow
589  {
590  const int16_t half2 = point_size2 >> 1;
591  const xcb_point_t points2[4] =

850 Plot_line_properties const * const * l_props = w_props.get_line_properties();
851 const Plot_line_properties & lp0 = *l_props[0];
852 
853  // 1. setup graphic contexts
854  //
855  set_GC_foreground(pctx, pctx.line, lp0.get_line_color());

861  // 2. draw areas between plot lines...
862  //
863 const double Hmin = w_props.get_min_Y();
864 const double dH = w_props.get_max_Y() - Hmin; // 100%
865 const int verbosity = w_props.get_verbosity();
866 
867  loop(row, data.get_row_count() - 1)
868  loop(col, data[row].get_N() - 1)
869  {
870  if (verbosity & SHOW_DATA)
871  CERR << "B[" << row << ";" << col << "]"
872  << " X=" << data.get_X(row, col)
873  << " Y=" << data.get_Y(row, col)

875

877 // if (row != 3) continue; // show only given row

907

909  plane Y=0, interpolate the respective heights, and draw
910  4 triangles from PM to each of the four edges of the square
911 
912  B. fold the surface at its shorter edge into 2 triangles that
913  have the edge in common. (triangles are always coplanar).
914 
915  Option B seems to look better.
916  */
917

920

946  }
947  else // P1-P2 is shorter
948  {
949  draw_triangle(pctx, verbosity, P0, H0, P1, H1, P2, H2);
950  draw_triangle(pctx, verbosity, P3, H3, P1, H1, P2, H2);
951  }
952 #endif
953  }
954 
955  // 3. draw lines between plot points
956  //

977  const double Y2 = data.get_Y(row + 1, col);

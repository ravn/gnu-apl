46 //-----------------------------------------------------------------------------
47 bool
48 FloatCell::greater(const Cell & other) const
49 {
50 const APL_Float this_val = get_real_value();
51 
52  switch(other.get_cell_type())
53  {
54  case CT_INT:
55  {
56  const APL_Float other_val(other.get_int_value());
57  if (this_val == other_val) return this > &other;
58  return this_val > other_val;
59  }
60 
61  case CT_FLOAT:
62  {
63  const APL_Float other_val = other.get_real_value();
64  if (this_val == other_val) return this > &other;
65  return this_val > other_val;
66  }
67 
68  case CT_CHAR: return true;
69  case CT_COMPLEX: break;
70  case CT_POINTER: return false;
71  case CT_CELLREF: DOMAIN_ERROR;
72  default: Assert(0 && "Bad celltype");
73  }
74 
75 const Comp_result comp = compare(other);
76  if (comp == COMP_EQ) return this > &other;
77  return (comp == COMP_GT);
78 }

95 //-----------------------------------------------------------------------------
96 Comp_result
97 FloatCell::compare(const Cell & other) const
98 {
99  if (other.is_integer_cell()) // integer
100  {
101  const double qct = Workspace::get_CT();
102  if (equal(other, qct)) return COMP_EQ;
103  return (dfval() <= other.get_int_value()) ? COMP_LT : COMP_GT;
104  }
105 
106  if (other.is_float_cell())
107  {
108  const double qct = Workspace::get_CT();
109  if (equal(other, qct)) return COMP_EQ;
110  return (dfval() <= other.get_real_value()) ? COMP_LT : COMP_GT;
111  }
112 
113  if (other.is_complex_cell()) return Comp_result(-other.compare(*this));
114 
115  if (other.is_character_cell()) return COMP_GT; // numeric > char
116  if (other.is_pointer_cell()) return COMP_LT; // numeric < nested
117  DOMAIN_ERROR;
118 }

263 ErrorCode
264 FloatCell::bif_ceiling(Cell * Z) const
265 {
266 #ifdef RATIONAL_NUMBERS_WANTED
267  if (const APL_Integer denom = get_denominator())
268  {
269  // since the quotient is exact, we ignore âCT
270  //
271  const APL_Integer numer = get_numerator();
272  APL_Integer quotient = numer / denom;
273  if (numer > (quotient * denom)) ++quotient;
274  return IntCell::zv(Z, quotient);
275  }
276 #endif
277 
278  // see comments for bif_floor below.
279 
280 const APL_Float b = dfval();
281  // if b is large then return it as is.
282  //
283  if (b >= LARGE_INT) return zv(Z, b);
284  if (b <= SMALL_INT) return zv(Z, b);
285 
286 APL_Integer bi = b;
287  while (bi < b) ++bi;
288  while ((bi - 1) > b) --bi;
289  if (bi == b) return IntCell::zv(Z, bi); // b already equal to its floor
290 
291 const APL_Float D = bi - b;
292 
293  if (D >= (1.0 - Workspace::get_CT())) --bi;
294  return IntCell::zv(Z, bi);
295 }
296 //-----------------------------------------------------------------------------
297 ErrorCode
298 FloatCell::bif_floor(Cell * Z) const
299 {
300 #ifdef RATIONAL_NUMBERS_WANTED
301  if (const APL_Integer denom = get_denominator())
302  {
303  // since the quotient is exact, we ignore âCT
304  //
305  const APL_Integer numer = get_numerator();
306  APL_Integer quotient = numer / denom;
307  if (numer < (quotient * denom)) --quotient;
308  return IntCell::zv(Z, quotient);
309  }
310 #endif
311 
312 /* Informal description (iso p. 78):
313  For real-numbers, Z is the greatest integer tolerantly less than
314  or equal to B. Uses comparison-tolerance.
315 
316  Formal description:
317  Return the tolerant-floor of B within comparison-tolerance.
318 
319  tolerant-floor (p.19) is defined for complex A:
320  Let A be a member of the set of numbers in the unit-square at the
321  complex-integer C, and let D be A minus C.
322  If the sum of the real and imaginary parts of D is tolerantly-less-than
323  one within B, then Z is C.
324  Otherwise, if the imaginary-part of D is greater-than the real-part of D,
325  then Z is C plus imaginary-one.
326  Otherwise, Z is C plus one.
327 
328  Unfortunately tolerantly-less-than is not defined in the standard. We
329  interpret it as meaning 'less than and not tolerrantly-equal'.
330 
331  Replacing B with âCT, and A with B, and considering that the imaginary
332  part of B is always 0 if B is real this becomes:
333 
334  tolerant-floor of (real) B within âCT:
335  Let B be a member of the set of numbers in the hals-open interval [C, C+1),
336  and let D be B minus C.
337  If D is tolerantly-less-than one within âCT, then Z is C.
338  Otherwise, Z is C plus one.
339 
340  In other word, Let RB be B rounded down. Then Z is RB if B < RB + 1 - âCT
341  and RB+1 otherwise.
342 
343  Note: if B cannot fit into int64_t then, due to the smaller precision
344  of double, it is already equal to its floor and we return it unchanged.
345 */
346 
347 const APL_Float b = dfval();
348  // if b is large then return it as is.
349  //
350  if (b >= LARGE_INT) return zv(Z, b);
351  if (b <= SMALL_INT) return zv(Z, b);
352 
353 APL_Integer bi = b;
354  while (bi > b) --bi;
355  while ((bi + 1) < b) ++bi;
356  if (bi == b) return IntCell::zv(Z, bi); // b already equal to its floor
357 
358 const APL_Float D = b - bi;
359 
360  if (D >= (1.0 - Workspace::get_CT())) ++bi;
361  return IntCell::zv(Z, bi);
362 }

386 //-----------------------------------------------------------------------------
387 // dyadic built-in functions...
388 //
389 // where possible a function with non-real A is delegated to the corresponding
390 // member function of A. For numeric cells that is the ComplexCell function
391 // and otherwise the default function (that returns E_DOMAIN_ERROR.
392 //
393 //-----------------------------------------------------------------------------
394 ErrorCode
395 FloatCell::bif_add(Cell * Z, const Cell * A) const
396 {
397  if (A->is_real_cell())
398  {
399 #ifdef RATIONAL_NUMBERS_WANTED
400  if (const APL_Integer denom_B = get_denominator())
401  if (const APL_Integer denom_A = A->get_denominator())
402  {
403  // both A and B are rational
404  //
405  if (Cell::prod_overflow(denom_A, denom_B)) goto big;
406 
407  // compute common denominator...
408  const APL_Integer gcd_AB = gcd(denom_A, denom_B);
409  const APL_Integer mult_A = denom_A / gcd_AB;
410  const APL_Integer mult_B = denom_B / gcd_AB;
411  const APL_Integer denom_AB = denom_A * mult_B;
412  if (Cell::prod_overflow(denom_AB, mult_B)) goto big;
413 
414  // compute numerators...
415  const APL_Integer numer_A = A->get_numerator();
416  if (Cell::prod_overflow(numer_A, mult_B)) goto big;
417  const APL_Integer numer_A1 = numer_A * mult_B;
418  const APL_Integer numer_B = get_numerator();
419  if (Cell::prod_overflow(numer_B, mult_A)) goto big;
420  const APL_Integer numer_B1 = numer_B * mult_A;
421 
422  const APL_Integer sum_AB = numer_A1 + numer_B1;
423  if (Cell::sum_overflow(sum_AB, numer_A1, numer_B1)) goto big;
424  const APL_Integer sum_gcd = gcd(sum_AB, denom_AB);
425  if (sum_gcd == denom_AB) return IntCell::zv(Z, sum_AB / denom_AB);
426  if (sum_gcd == 1) return FloatCell::zv(Z, sum_AB, denom_AB);
427  return FloatCell::zv(Z, sum_AB/sum_gcd, denom_AB/sum_gcd);
428  }
429  big:
430 
431 #endif
432 
433  return FloatCell::zv(Z, A->get_real_value() + get_real_value());
434  }
435 
436  // delegate to A
437  //
438  return A->bif_add(Z, this);
439 }

448 FloatCell::bif_subtract(Cell * Z, const Cell * A) const
449 {
450  if (A->is_real_cell()) // real result
451  {
452 #ifdef RATIONAL_NUMBERS_WANTED
453  if (const APL_Integer denom_B = get_denominator())
454  if (const APL_Integer denom_A = A->get_denominator())
455  {
456  // both A and B are rational
457  //
458  if (Cell::prod_overflow(denom_A, denom_B)) goto big;
459 
460  // compute common denominator...
461  const APL_Integer gcd_AB = gcd(denom_A, denom_B);
462  const APL_Integer mult_A = denom_A / gcd_AB;
463  const APL_Integer mult_B = denom_B / gcd_AB;
464  const APL_Integer denom_AB = denom_A * mult_B;
465 
466  // compute numerators...
467  const APL_Integer numer_A = A->get_numerator();
468  if (Cell::prod_overflow(numer_A, mult_B)) goto big;
469  const APL_Integer numer_A1 = numer_A * mult_B;
470  const APL_Integer numer_B = get_numerator();
471  if (Cell::prod_overflow(numer_B, mult_A)) goto big;
472  const APL_Integer numer_B1 = numer_B * mult_A;
473 
474  const APL_Integer diff_AB = numer_A1 - numer_B1;
475  if (Cell::diff_overflow(diff_AB, numer_A1, numer_B1)) goto big;
476  const APL_Integer diff_gcd = gcd(diff_AB, denom_AB);
477  if (diff_gcd == denom_AB) return IntCell::zv(Z, diff_AB / denom_AB);
478  if (diff_gcd == 1) return FloatCell::zv(Z, diff_AB, denom_AB);
479  return FloatCell::zv(Z, diff_AB/diff_gcd, denom_AB/diff_gcd);
480  }
481  big:
482 
483 #endif
484 
485  return zv(Z, A->get_real_value() - get_real_value());
486  }
487 
488  if (A->is_complex_cell()) // complex result
489  {
490  return ComplexCell::zv(Z, A->get_real_value() - get_real_value(),
491  A->get_imag_value());
492  }
493

495 }

497 ErrorCode
498 FloatCell::bif_multiply(Cell * Z, const Cell * A) const
499 {
500 #ifdef RATIONAL_NUMBERS_WANTED
501  if (APL_Integer denom_B = get_denominator())
502  if (APL_Integer denom_A = A->get_denominator())
503  {
504  // both A and B are rational
505  //
506  APL_Integer numer_A = A->get_numerator();
507  APL_Integer numer_B = get_numerator();
508  const APL_Integer gcd_A_B = gcd(numer_A, denom_B);
509  if (gcd_A_B > 1) { numer_A /= gcd_A_B; denom_B /= gcd_A_B; }
510 
511  const APL_Integer gcd_B_A = gcd(numer_B, denom_A);
512  if (gcd_B_A > 1) { numer_B /= gcd_B_A; denom_A /= gcd_B_A; }
513 
514  if (Cell::prod_overflow(numer_A, numer_B)) goto big;
515  if (Cell::prod_overflow(denom_A, denom_B)) goto big;
516 
517  const APL_Integer numer = numer_A * numer_B;
518  const APL_Integer denom = denom_A * denom_B;
519  const APL_Integer prod_gcd = gcd(numer, denom);
520  if (prod_gcd == denom) return IntCell::zv(Z, numer / denom);
521  if (prod_gcd == 1) return FloatCell::zv(Z, numer, denom);
522  return FloatCell::zv(Z, numer/prod_gcd, denom/prod_gcd);
523  }
524  big:
525 
526 #endif
527 
528  if (!A->is_numeric()) return E_DOMAIN_ERROR;
529 
530 const APL_Float ar = A->get_real_value();
531 const APL_Float ai = A->get_imag_value();
532 
533  if (ai == 0.0) // real result
534  {
535  const APL_Float z = ar * dfval();
536  if (!isfinite(z)) return E_DOMAIN_ERROR;
537  return FloatCell::zv(Z, z);
538  }
539 
540  // complex result
541  //
542 const double zr = ar * dfval();
543 const double zi = ai * dfval();
544  if (!isfinite(zr)) return E_DOMAIN_ERROR;
545  if (!isfinite(zi)) return E_DOMAIN_ERROR;
546  return ComplexCell::zv(Z, zr, zi);
547 }

556 FloatCell::bif_divide(Cell * Z, const Cell * A) const
557 {
558 #ifdef RATIONAL_NUMBERS_WANTED
559  if (const APL_Integer B_denom = get_denominator()) // B is rational
560  {
561  const APL_Integer B_numer = get_numerator();
562  if (B_numer == 0) // A Ã· 0
563  {
564  if (A->is_near_zero()) return IntCell::z1(Z); // 0Ã·0 is 1
565  return E_DOMAIN_ERROR;
566  }
567  const FloatCell inv_B(B_denom, B_numer);
568  return inv_B.bif_multiply(Z, A);
569  }
570 #endif
571 
572  if (!A->is_numeric()) return E_DOMAIN_ERROR;
573 
574 const APL_Float ar = A->get_real_value();
575 const APL_Float ai = A->get_imag_value();
576 
577  if (dfval() == 0.0) // A Ã· 0
578  {
579  if (ar != 0.0) return E_DOMAIN_ERROR;
580  if (ai != 0.0) return E_DOMAIN_ERROR;
581 
582  return IntCell::z1(Z); // 0Ã·0 is 1 in APL
583  }
584 
585 
586  if (ai == 0.0) // real result
587  {
588  const APL_Float real = ar / dfval() ;
589  if (isfinite(real)) return FloatCell::zv(Z, real);
590  return E_DOMAIN_ERROR;
591  }
592 
593  // complex result
594  //
595 const double zar = ar / dfval();
596 const double zai = ai / dfval();
597  if (!isfinite(zar)) return E_DOMAIN_ERROR;
598  if (!isfinite(zai)) return E_DOMAIN_ERROR;

600 }

603 FloatCell::bif_power(Cell * Z, const Cell * A) const
604 {
605  // some A to the real B-th power
606  //
607  if (!A->is_numeric()) return E_DOMAIN_ERROR;
608 
609 const APL_Float ar = A->get_real_value();
610 const APL_Float ai = A->get_imag_value();
611 
612  // 1. A == 0
613  //
614  if (ar == 0.0 && ai == 0.0)
615  {
616  if (dfval() == 0.0) return IntCell::z1(Z); // 0â0 is 1
617  if (dfval() > 0.0) return IntCell::z0(Z); // 0âN is 0
618  return E_DOMAIN_ERROR; // 0âÂ¯N = 1Ã·0
619  }
620 
621  // 2. real A > 0 (real result)
622  //
623  if (ai == 0.0)
624  {
625  if (ar == 1.0) return IntCell::z1(Z); // 1âb = 1

630  }

636  if (!isfinite(z.real())) return E_DOMAIN_ERROR;
637  if (!isfinite(z.imag())) return E_DOMAIN_ERROR;
638

38  static void sort(T * a, int64_t heapsize, const void * comp_arg,
39  greater_fun gf)
40  {
41  // turn a[] into a heap, i.e. a[i] > a[2i] and a[i] > a[2i+1]
42  // for all i. At heapsize/2 + 1 ... the heap property is already
43  // fulfilled, because these items have no children. So we move
44  // downwards from heapsize/2 to 0.
45  //
46  for (int64_t p = heapsize/2 - 1; p >= 0; --p)
47  make_heap(a, heapsize, p, comp_arg, gf);
48 
49  // here a[] is a heap (and therefore a[0] is the largest element)
50 
51  for (--heapsize; heapsize > 0; heapsize--)
52  {
53  // The root a[0] is the largest element in a[0] ... a[k].
54  // Exchange a[0] and a[k], decrease the heap size,
55  // and re-establish the heap property of the new a[0].
56  //
57  Hswap(a[heapsize], a[0]);
58 
59  // re-establish the heap property of the new a[0]
60  //
61  make_heap(a, heapsize, 0, comp_arg, gf);
62  }
63  }

66  template<typename K>
67  static const T * search(const K & key, const T * array,
68  int64_t /* array size */ u,
69  int (*compare)(const K & key, const T & item))
70  {
71  for (int64_t l = 0; l < u;)
72  {
73  const int64_t half = (l + u) / 2;
74  const T * middle = &array[half];
75  const int comp = (*compare)(key, *middle);
76  if (comp < 0) u = half;
77  else if (comp > 0) l = half + 1;
78  else return middle;
79  }
80 
81  return 0;
82  }

86  static void make_heap(T * a, int64_t heapsize, int64_t parent,
87  const void * comp_arg, greater_fun gf)
88  {
89  for (;;)
90  {
91  const int64_t left = 2*parent + 1; // left child of parent.
92  const int64_t right = left + 1; // right child of parent.
93  int64_t max = parent; // assume parent is the max.
94 
95  // set max to the position of the largest of a[i], a[l], and a[r]
96  //
97  if ((left < heapsize) && (*gf)(a[left], a[max], comp_arg))
98  max = left; // left child is larger
99 
100  if ((right < heapsize) && (*gf)(a[right], a[max], comp_arg))
101  max = right; // right child is larger
102 
103  if (max == parent) return; // parent was the max: done
104 
105  // left or right was the max. exchange and continue
106  Hswap(a[max], a[parent]);
107  parent = max;
108  }
109  }
110 };

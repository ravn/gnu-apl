40  static void sort(T * a, int64_t heapsize, const void * comp_arg,
41  greater_fun gf)
42  {
43  // turn a[] into a heap, i.e. a[i] > a[2i] and a[i] > a[2i+1]
44  // for all i. At heapsize/2 + 1 ... the heap property is already
45  // fulfilled, because these items have no children. So we move
46  // downwards from heapsize/2 to 0.
47  //
48  for (int64_t p = heapsize/2 - 1; p >= 0; --p)
49  make_heap(a, heapsize, p, comp_arg, gf);
50 
51  // here a[] is a heap (and therefore a[0] is the largest element)
52 
53  for (--heapsize; heapsize > 0; heapsize--)
54  {
55  // The root a[0] is the largest element in a[0] ... a[k].
56  // Exchange a[0] and a[k], decrease the heap size,
57  // and re-establish the heap property of the new a[0].
58  //
59  Hswap(a[heapsize], a[0]);
60 
61  // re-establish the heap property of the new a[0]
62  //
63  make_heap(a, heapsize, 0, comp_arg, gf);
64  }
65  }

75  template<typename K>
76  static const T * search(const K & key,
77  const T * array,
78  int64_t /* array size */ u,
79  int (*compare)(const K & key, const T & item,
80  const void * comp_ctx),
81  const void * ctx)
82  {
83  for (int64_t l = 0; l < u;)
84  {
85  const int64_t half = (l + u) / 2;
86  const T * middle = &array[half];
87  const int comp = (*compare)(key, *middle, ctx);
88  if (comp > 0) l = half + 1; // search in key upper half
89  else if (comp < 0) u = half; // search in key lower half
90  else return middle; // key found
91  }
92 
93  return 0;
94  }

106 protected:
108  static void make_heap(T * a, int64_t heapsize, int64_t parent,
109  const void * comp_arg, greater_fun gf)
110  {
111  for (;;)
112  {
113  const int64_t left = 2*parent + 1; // left child of parent.
114  const int64_t right = left + 1; // right child of parent.
115  int64_t max = parent; // assume parent is the max.
116 
117  // set max to the position of the largest of a[i], a[l], and a[r]
118  //
119  if ((left < heapsize) && (*gf)(a[left], a[max], comp_arg))
120  max = left; // left child is larger
121 
122  if ((right < heapsize) && (*gf)(a[right], a[max], comp_arg))
123  max = right; // right child is larger
124 
125  if (max == parent) return; // parent was the max: done
126 
127  // left or right was the max. exchange and continue
128  Hswap(a[max], a[parent]);
129  parent = max;
130  }
131  }

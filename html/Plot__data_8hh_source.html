39 struct Pixel_XY
40 {
42  Pixel_XY(Pixel_X px, Pixel_Y py)
43  : x(px),
44  y(py)
45  {}
46 
48  const double distance2(const Pixel_XY other) const
49  { const double dx = x - other.x;
50  const double dy = y - other.y;
51  return dx*dx + dy*dy;
52  }
53 
54 
56  const double distance(const Pixel_XY other) const
57  { return sqrt(distance2(other)); }
58 
59  Pixel_X x;
60  Pixel_Y y;
61 };

64 class Plot_data_row
65 {
66 public:
68  Plot_data_row(const double * pX, const double * pY, const double * pZ,
69  uint32_t idx, uint32_t len)
70  : X(pX), Y(pY), Z(pZ), row_num(idx), N(len)
71  {
72  min_X = max_X = get_X(0);
73  min_Y = max_Y = get_Y(0);
74  min_Z = max_Z = get_Z(0);
75 
76  for (unsigned int n = 1; n < N; ++n)
77  {
78  const double Xn = get_X(n);
79  if (min_X > Xn) min_X = Xn;
80  if (max_X < Xn) max_X = Xn;
81 
82  const double Yn = get_Y(n);
83  if (min_Y > Yn) min_Y = Yn;
84  if (max_Y < Yn) max_Y = Yn;
85 
86  const double Zn = get_Z(n);
87  if (min_Z > Zn) min_Z = Zn;
88  if (max_Z < Zn) max_Z = Zn;
89  }
90  }
91 
93  ~Plot_data_row()
94  {
95  // all rows belong to the same double *, so we only delete row 0
96  //
97  if (row_num == 0) delete[] X;
98  }
99 
101  uint32_t get_N() const
102  { return N; }
103 
105  double get_X(uint32_t n) const
106  { return X ? X[n] : n + 1; }
107 
109  double get_Y(uint32_t n) const
110  { return Y[n]; }
111 
113  double get_Z(uint32_t n) const
114  { return Z ? Z[n] : row_num + 1; }
115 
117  double get_min_X() const

184 class Plot_data
185 {
186 public:
188  Plot_data(uint32_t rows)
189  : surface(false),
190  row_count(rows),
191  idx(0)
192  {
193  data_rows = new const Plot_data_row *[row_count];
194  }
195 
197  ~Plot_data()
198  {
199  loop(r, idx) delete data_rows[r];
200  delete data_rows;
201  }

209  {

249  { return data_rows[row]->get_Z(col); }

252  bool is_surface_plot() const { return surface; }
253 
255  void add_row(const Plot_data_row * row)
256  {
257  Assert(idx < row_count);
258  data_rows[idx++] = row;
259  if (idx == 1) // first row
260  {
261  min_X = row->get_min_X();
262  max_X = row->get_max_X();
263  min_Y = row->get_min_Y();
264  max_Y = row->get_max_Y();
265  min_Z = row->get_min_Y();
266  max_Z = row->get_max_Y();
267  }
268  else
269  {
270  if (min_X > row->get_min_X()) min_X = row->get_min_X();
271  if (max_X < row->get_max_X()) max_X = row->get_max_X();
272  if (min_Y > row->get_min_Y()) min_Y = row->get_min_Y();
273  if (max_Y < row->get_max_Y()) max_Y = row->get_max_Y();
274  if (min_Z > row->get_min_Z()) min_Z = row->get_min_Z();
275  if (max_Z < row->get_max_Z()) max_Z = row->get_max_Z();
276  }
277  }

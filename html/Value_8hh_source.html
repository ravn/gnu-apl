644  static void * operator new(size_t sz)
645  {
646  if (deleted_values) // we have deleted values: recycle one
647  {
648  --deleted_values_count;
649  void * ret = deleted_values;
650  deleted_values = deleted_values->next;
651  ++fast_new;
652  return ret;
653  }
654 
655  ++slow_new;
656  return ::operator new(sz);
657  }
658 
660  static void operator delete(void * ptr)
661  {
662  if (deleted_values_count < deleted_values_MAX) // we have space
663  {
664  ++deleted_values_count;
665  reinterpret_cast<_deleted_value *>(ptr)->next = deleted_values;
666  deleted_values = reinterpret_cast<_deleted_value *>(ptr);
667  }
668  else // no more space
669  {
670  free(ptr);
671  }
672  }
673 
674 #endif

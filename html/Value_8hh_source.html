881  static void operator delete(void * ptr)
882  {
883  if (deleted_values_count < deleted_values_MAX) // we have space
884  {
885  ++deleted_values_count;
886  reinterpret_cast<_deleted_value *>(ptr)->next = deleted_values;
887  deleted_values = reinterpret_cast<_deleted_value *>(ptr);
888  }
889  else // no more space
890  {
891  free(ptr);
892  }
893  }

61 Token
62 Bif_OPER2_RANK::do_LyXB(Token & _LO, Value_P X, Value_P B, Rank rank_chunk_B)
63 {
64 Function_P LO = _LO.get_function();
65  Assert(LO);
66  if (!LO->has_result()) DOMAIN_ERROR;
67 
68  // split shape of B into high (=frame) and low (= chunk) shapes.
69  //
70 const Shape shape_Z = B->get_shape().high_shape(B->get_rank() - rank_chunk_B);
71  if (shape_Z.is_empty())
72  {
73  Value_P Fill_B = Bif_F12_TAKE::first(B);
74  Token tZ = LO->eval_fill_B(Fill_B);
75  Value_P Z = tZ.get_apl_val();
76  Z->set_shape(B->get_shape());
77  Z->check_value(LOC);
78  return Token(TOK_APL_VALUE1, Z);
79  }
80 
81 const Shape shape_B = B->get_shape().low_shape(rank_chunk_B);
82 
83 Value_P vsh_B(shape_B.get_rank(), LOC);
84  vsh_B->set_proto_Int(); // prototype
85  loop(sh, shape_B.get_rank())
86  vsh_B->next_ravel_Int(shape_B.get_shape_item(sh));
87  vsh_B->check_value(LOC);
88 
89 Value_P vsh_Z(shape_Z.get_rank(), LOC);
90  loop(sh, shape_Z.get_rank())
91  vsh_Z->next_ravel_Int(shape_Z.get_shape_item(sh));
92  vsh_Z->check_value(LOC);
93 
94 Value_P X5(5, LOC);
95  if (!X) X5->next_ravel_Int(-1); // ¯1 means no X
96  else X5->next_ravel_Pointer(X.get());
97 
98  X5->next_ravel_Int(shape_B.get_volume()); // LB
99  X5->next_ravel_Pointer(vsh_B.get()); // rho_B
100  X5->next_ravel_Int(shape_Z.get_volume()); // N_max
101  X5->next_ravel_Pointer(vsh_Z.get()); // rho_Z
102  X5->check_value(LOC);
103 
104  return Macro::get_macro(Macro::MAC_Z__LO_RANK_X5_B)->eval_LXB(_LO, X5, B);
105 }

134 Token
135 Bif_OPER2_RANK::do_ALyXB(Value_P A, Rank rank_chunk_A, Token & _LO,
136  Value_P X, Value_P B, Rank rank_chunk_B)
137 {
138 Function_P LO = _LO.get_function();
139  Assert(LO);
140  if (!LO->has_result()) DOMAIN_ERROR;
141 
142 Rank rk_A_frame = A->get_rank() - rank_chunk_A; // rk_A_frame is y8
143 Rank rk_B_frame = B->get_rank() - rank_chunk_B; // rk_B_frame is y9
144 
145  // if both high-ranks are 0, then return A LO B.
146  //
147  if (rk_A_frame == 0 && rk_B_frame == 0) return LO->eval_AB(A, B);
148 
149  // split shapes of A1 and B1 into high (frame) and low (chunk) shapes.
150  // Even though A and B have the same shape, rk_A_frame and rk_B_frame
151  // could be different, leading to different split shapes for A and B
152  //
153 const Shape shape_Z = rk_B_frame ? B->get_shape().high_shape(rk_B_frame)
154  : A->get_shape().high_shape(rk_A_frame);
155 
156  if (rk_A_frame && rk_B_frame) // A and B frames non-scalar
157  {
158  if (rk_A_frame != rk_B_frame) RANK_ERROR;
159  if (shape_Z != A->get_shape().high_shape(rk_A_frame)) LENGTH_ERROR;
160  }
161 
162  if (shape_Z.is_empty())
163  {
164  Value_P Fill_A = Bif_F12_TAKE::first(A);
165  Value_P Fill_B = Bif_F12_TAKE::first(B);
166  Shape shape_Z;
167 
168  if (A->is_empty()) shape_Z = A->get_shape();
169  else if (!A->is_scalar()) DOMAIN_ERROR;
170 
171  if (B->is_empty()) shape_Z = B->get_shape();
172  else if (!B->is_scalar()) DOMAIN_ERROR;
173 
174  Value_P Z1 = LO->eval_fill_AB(Fill_A, Fill_B).get_apl_val();
175 
176  Value_P Z(shape_Z, LOC);
177  Z->set_ravel_Value(0, Z1.get());
178  Z->check_value(LOC);
179  return Token(TOK_APL_VALUE1, Z);
180  }
181 
182 const Shape low_A = A->get_shape().low_shape(rank_chunk_A);
183 const Shape low_B = B->get_shape().low_shape(rank_chunk_B);
184 
185 Value_P vsh_A(LOC, &low_A);
186 Value_P vsh_B(LOC, &low_B);
187 Value_P vsh_Z(LOC, &shape_Z);
188 
189 Value_P X7(7, LOC);
190  if (!X) X7->next_ravel_Int(-1); // no X
191  else X7->next_ravel_Value(X.get());
192 
193  X7->next_ravel_Int(low_A.get_volume()); // LA
194  X7->next_ravel_Value(vsh_A.get());
195  X7->next_ravel_Int(low_B.get_volume()); // LB
196  X7->next_ravel_Value(vsh_B.get());
197  X7->next_ravel_Int(shape_Z.get_volume()); // N_max
198  X7->next_ravel_Value(vsh_Z.get());
199  X7->check_value(LOC);
200  return Macro::get_macro(Macro::MAC_Z__A_LO_RANK_X7_B)
201  ->eval_ALXB(A, _LO, X7, B);
202 }

204 void
205 Bif_OPER2_RANK::y123_to_B(Value_P y123, Rank & rank_B)
206 {
207  // y123_to_AB() splits the ranks of A and B into a (higher-dimensions)
208  // "frame" and a (lower-dimensions) "chunk" as specified by y123.
209 
210  // 1. on entry rank_B is the rank of B.
211  //
212  // Remember the rank of B to limit rank_B
213  // if values in y123 should exceed them.
214  //
215 const Rank rk_B = rank_B;
216 
217  if (!y123) VALUE_ERROR;
218  if ( y123->get_rank() > 1) DOMAIN_ERROR;
219 
220  // 2. the number of elements in y determine how rank_B shall be computed:
221  //
222  // -- monadic f⍤ -- -- dyadic f⍤ --
223  // rank_A rank_B rank_A rank_B
224  // ---------------------------------------------------------
225  // y : N/A y y y
226  // yA yB : N/A yB yA yB
227  // yM yA yB : N/A yM yA yB
228  // ---------------------------------------------------------
229 
230  switch(y123->element_count())
231  {
232  case 1: rank_B = y123->get_cfirst().get_near_int(); break;
233 
234  case 2: y123->get_cfirst().get_near_int();
235  rank_B = y123->get_cravel(1).get_near_int(); break;
236 
237  case 3: rank_B = y123->get_cfirst().get_near_int();
238  y123->get_cravel(1).get_near_int();
239  y123->get_cravel(2).get_near_int(); break;
240 
241  default: LENGTH_ERROR;
242  }
243 
244  // 3. adjust rank_B if they exceed its initial value or
245  // if it is negative
246  //
247  if (rank_B > rk_B) rank_B = rk_B;
248  if (rank_B < 0) rank_B += rk_B;
249  if (rank_B < 0) rank_B = 0;
250 }
251 //-----------------------------------------------------------------------------
252 void
253 Bif_OPER2_RANK::y123_to_AB(Value_P y123, Rank & rank_A, Rank & rank_B)
254 {
255  // y123_to_AB() splits the ranks of A and B into a (higher-dimensions)
256  // "frame" and a (lower-dimensions) "chunk" as specified by y123.
257 
258  // 1. on entry rank_A and rank_B are the ranks of A and B.
259  //
260  // Remember the ranks of A and B to limit rank_A and rank_B
261  // if values in y123 should exceed them.
262  //
263 const Rank rk_A = rank_A;
264 const Rank rk_B = rank_B;
265 
266  if (!y123) VALUE_ERROR;
267  if ( y123->get_rank() > 1) DOMAIN_ERROR;
268 
269  // 2. the number of elements in y determine how rank_A and rank_B
270  // shall be computed:
271  //
272  // -- monadic f⍤ -- -- dyadic f⍤ --
273  // rank_A rank_B rank_A rank_B
274  // ---------------------------------------------------------
275  // y : N/A y y y
276  // yA yB : N/A yB yA yB
277  // yM yA yB : N/A yM yA yB
278  // ---------------------------------------------------------
279 
280  switch(y123->element_count())
281  {
282  case 1: rank_A = y123->get_cfirst().get_near_int();
283  rank_B = rank_A; break;
284 
285  case 2: rank_A = y123->get_cfirst().get_near_int();
286  rank_B = y123->get_cravel(1).get_near_int(); break;
287 
288  case 3: y123->get_cfirst().get_near_int();
289  rank_A = y123->get_cravel(1).get_near_int();
290  rank_B = y123->get_cravel(2).get_near_int(); break;
291 
292  default: LENGTH_ERROR;
293  }
294 
295  // 3. adjust rank_A and rank_B if they exceed their initial value or
296  // if they are negative
297  //
298  if (rank_A > rk_A) rank_A = rk_A;
299  if (rank_A < 0) rank_A += rk_A;
300  if (rank_A < 0) rank_A = 0;
301 
302  if (rank_B > rk_B) rank_B = rk_B;
303  if (rank_B < 0) rank_B += rk_B;
304  if (rank_B < 0) rank_B = 0;
305 }
306 //-----------------------------------------------------------------------------
307 void
308 Bif_OPER2_RANK::split_y123_B(Value_P y123_B, Value_P & y123, Value_P & B)
309 {
310  // The ISO standard and NARS define the reduction pattern for the RANK
311  // operator ⍤ as:
312  //
313  // Z ← A f ⍤ y B (ISO)
314  // Z ← f ⍤ y B (ISO)
315  // Z ← A f ⍤ [X] y B (NARS)
316  // Z ← f ⍤ [X] y B (NARS)
317  //
318  // GNU APL may bind y to B at tokenization time if y and B are constants
319  // This function tries to "unbind" its argument y123_B into the original
320  // components y123 (= y in the standard) and B. The tokenization time
321  // binding is shown as y123:B
322  //
323  // Usage y123 : B Result: j123 B
324  //-------------------------------------------------------------------------
325  // 1. f ⍤ (y123):B... nested any y123 B
326  // 2. (f ⍤ y123:⍬) simple empty y123 -
327  // 3. f ⍤ y123:(B) simple nested skalar y123 B
328  // 4a. f ⍤ y123:B... simple any y123 B...
329  //
330 
331  // y123_B shall be a scalar or vector
332  //
333  if (y123_B->get_rank() > 1) RANK_ERROR;
334 
335 const ShapeItem length = y123_B->element_count();
336  if (length == 0) LENGTH_ERROR;
337 
338  // check for case 1 (the only one with nested first element)
339  //
340  if (y123_B->get_cfirst().is_pointer_cell()) // (y123)
341  {
342  y123 = y123_B->get_cfirst().get_pointer_value();
343  if (length == 1) // empty B
344  {
345  }
346  else if (length == 2) // skalar B
347  {
348  const Cell & B0 = y123_B->get_cravel(1);
349  if (B0.is_pointer_cell()) // (B)
350  {
351  B = B0.get_pointer_value();
352  }
353  else
354  {
355  B = Value_P(LOC);
356  B->next_ravel_Cell(B0);
357  }
358  }
359  else // vector B
360  {
361  B = Value_P(length - 1, LOC);
362  loop(l, length - 1)
363  B->next_ravel_Cell(y123_B->get_cravel(l + 1));
364  }
365  y123->check_value(LOC);
366  B->check_value(LOC);
367  return;
368  }
369 
370  // case 1. ruled out, so the first 1, 2, or 3 cells are j123.
371  // see how many (at most)
372  //
373 int y123_len = 0;
374  loop(yy, 3)
375  {
376  if (yy >= length) break;
377  const Cell & cy = y123_B->get_cravel(yy);
378  if (cy.is_near_int()) ++y123_len;
379  else break;
380  }
381  if (y123_len == 0) LENGTH_ERROR; // at least y1 is needed
382 
383  // cases 2.-4. start with integers of length 1, 2, or 3
384  //
385  if (length == y123_len) // case 2: y123:⍬
386  {
387  y123 = y123_B;
388  //
389  // NOTE: B is NOT assigned so that Prefix::reduce_F_D_B_() can detect
390  // that y123_ was only y123 !
391  //
392  return;
393  }
394 
395  if (length == (y123_len + 1) &&
396  y123_B->get_cravel(y123_len).is_pointer_cell()) // case 3. y123:⊂B
397  {
398  y123 = Value_P(y123_len, LOC);
399  loop(yy, y123_len) y123->next_ravel_Cell(y123_B->get_cravel(yy));
400  B = y123_B->get_cravel(y123_len).get_pointer_value();
401  y123->check_value(LOC);
402  B->check_value(LOC);
403  return;
404  }
405 
406  // case 4: y123:B...
407  //
408  y123 = Value_P(y123_len, LOC);
409  loop(yy, y123_len) y123->next_ravel_Cell(y123_B->get_cravel(yy));
410 
411 const ShapeItem B_len = length - y123_len;
412  B = Value_P(B_len, LOC);
413  loop(bb, B_len) B->next_ravel_Cell(y123_B->get_cravel(bb + y123_len));
414  B->check_value(LOC);
415 }

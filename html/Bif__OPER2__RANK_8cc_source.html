61 Token
62 Bif_OPER2_RANK::do_LyXB(Token & _LO, Value_P X, Value_P B, Rank rank_chunk_B)
63 {
64 Function * LO = _LO.get_function();
65  Assert(LO);
66  if (!LO->has_result()) DOMAIN_ERROR;
67 
68  // split shape of B into high (=frame) and low (= chunk) shapes.
69  //
70 const Shape shape_Z = B->get_shape().high_shape(B->get_rank() - rank_chunk_B);
71  if (shape_Z.is_empty())
72  {
73  Value_P Fill_B = Bif_F12_TAKE::first(B);
74  Token tZ = LO->eval_fill_B(Fill_B);
75  Value_P Z = tZ.get_apl_val();
76  Z->set_shape(B->get_shape());
77  Z->check_value(LOC);
78  return Token(TOK_APL_VALUE1, Z);
79  }
80 
81 const Shape shape_B = B->get_shape().low_shape(rank_chunk_B);
82 
83 Value_P vsh_B(shape_B.get_rank(), LOC);
84  new (&vsh_B->get_ravel(0)) IntCell(0); // prototype
85  loop(sh, shape_B.get_rank())
86  new (vsh_B->next_ravel()) IntCell(shape_B.get_shape_item(sh));
87  vsh_B->check_value(LOC);
88 
89 Value_P vsh_Z(shape_Z.get_rank(), LOC);
90  loop(sh, shape_Z.get_rank())
91  new (vsh_Z->next_ravel()) IntCell(shape_Z.get_shape_item(sh));
92  vsh_Z->check_value(LOC);
93 
94 Value_P X5(5, LOC);
95  if (!X) new (X5->next_ravel()) IntCell(-1); // no X
96  else new (X5->next_ravel()) PointerCell(X.get(), X5.getref()); // X
97 
98  new (X5->next_ravel()) IntCell(shape_B.get_volume()); // LB
99  new (X5->next_ravel()) PointerCell(vsh_B.get(), X5.getref()); // rho_B
100  new (X5->next_ravel()) IntCell(shape_Z.get_volume()); // N_max
101  new (X5->next_ravel()) PointerCell(vsh_Z.get(), X5.getref()); // rho_Z
102  X5->check_value(LOC);
103 
104  return Macro::get_macro(Macro::MAC_Z__LO_RANK_X5_B)->eval_LXB(_LO, X5, B);
105 }

135 Token
136 Bif_OPER2_RANK::do_ALyXB(Value_P A, Rank rank_chunk_A, Token & _LO,
137  Value_P X, Value_P B, Rank rank_chunk_B)
138 {
139 Function * LO = _LO.get_function();
140  Assert(LO);
141  if (!LO->has_result()) DOMAIN_ERROR;
142 
143 Rank rk_A_frame = A->get_rank() - rank_chunk_A; // rk_A_frame is y8
144 Rank rk_B_frame = B->get_rank() - rank_chunk_B; // rk_B_frame is y9
145 
146  // if both high-ranks are 0, then return A LO B.
147  //
148  if (rk_A_frame == 0 && rk_B_frame == 0) return LO->eval_AB(A, B);
149 
150  // split shapes of A1 and B1 into high (frame) and low (chunk) shapes.
151  // Even though A and B have the same shape, rk_A_frame and rk_B_frame
152  // could be different, leading to different split shapes for A and B
153  //
154 const Shape shape_Z = rk_B_frame ? B->get_shape().high_shape(rk_B_frame)
155  : A->get_shape().high_shape(rk_A_frame);
156 
157  if (rk_A_frame && rk_B_frame) // A and B frames non-scalar
158  {
159  if (rk_A_frame != rk_B_frame) RANK_ERROR;
160  if (shape_Z != A->get_shape().high_shape(rk_A_frame)) LENGTH_ERROR;
161  }
162 
163  if (shape_Z.is_empty())
164  {
165  Value_P Fill_A = Bif_F12_TAKE::first(A);
166  Value_P Fill_B = Bif_F12_TAKE::first(B);
167  Shape shape_Z;
168 
169  if (A->is_empty()) shape_Z = A->get_shape();
170  else if (!A->is_scalar()) DOMAIN_ERROR;
171 
172  if (B->is_empty()) shape_Z = B->get_shape();
173  else if (!B->is_scalar()) DOMAIN_ERROR;
174 
175  Value_P Z1 = LO->eval_fill_AB(Fill_A, Fill_B).get_apl_val();
176 
177  Value_P Z(shape_Z, LOC);
178  Z->get_ravel(0).init_from_value(Z1.get(), Z.getref(), LOC);
179  Z->check_value(LOC);
180  return Token(TOK_APL_VALUE1, Z);
181  }
182 
183 const Shape low_A = A->get_shape().low_shape(rank_chunk_A);
184 const Shape low_B = B->get_shape().low_shape(rank_chunk_B);
185 
186 Value_P vsh_A(LOC, &low_A);
187 Value_P vsh_B(LOC, &low_B);
188 Value_P vsh_Z(LOC, &shape_Z);
189 
190 Value_P X7(7, LOC);
191  if (!X) new (X7->next_ravel()) IntCell(-1); // no X
192  else X7->next_ravel()->init_from_value(X.get(), X7.getref(), LOC);
193 
194  new (X7->next_ravel()) IntCell(low_A.get_volume()); // LA
195 
196  X7->next_ravel()->init_from_value(vsh_A.get(), X7.getref(), LOC);
197 
198  new (X7->next_ravel()) IntCell(low_B.get_volume()); // LB
199 
200  X7->next_ravel()->init_from_value(vsh_B.get(), X7.getref(), LOC);
201 
202  new (X7->next_ravel()) IntCell(shape_Z.get_volume()); // N_max
203 
204  X7->next_ravel()->init_from_value(vsh_Z.get(), X7.getref(), LOC);
205 
206  X7->check_value(LOC);
207  return Macro::get_macro(Macro::MAC_Z__A_LO_RANK_X7_B)
208  ->eval_ALXB(A, _LO, X7, B);
209 }

211 void
212 Bif_OPER2_RANK::y123_to_B(Value_P y123, Rank & rank_B)
213 {
214  // y123_to_AB() splits the ranks of A and B into a (higher-dimensions)
215  // "frame" and a (lower-dimensions) "chunk" as specified by y123.
216 
217  // 1. on entry rank_B is the rank of B.
218  //
219  // Remember the rank of B to limit rank_B
220  // if values in y123 should exceed them.
221  //
222 const Rank rk_B = rank_B;
223 
224  if (!y123) VALUE_ERROR;
225  if ( y123->get_rank() > 1) DOMAIN_ERROR;
226 
227  // 2. the number of elements in y determine how rank_B shall be computed:
228  //
229  // -- monadic fâ¤ -- -- dyadic fâ¤ --
230  // rank_A rank_B rank_A rank_B
231  // ---------------------------------------------------------
232  // y : N/A y y y
233  // yA yB : N/A yB yA yB
234  // yM yA yB : N/A yM yA yB
235  // ---------------------------------------------------------
236 
237  switch(y123->element_count())
238  {
239  case 1: rank_B = y123->get_ravel(0).get_near_int(); break;
240 
241  case 2: y123->get_ravel(0).get_near_int();
242  rank_B = y123->get_ravel(1).get_near_int(); break;
243 
244  case 3: rank_B = y123->get_ravel(0).get_near_int();
245  y123->get_ravel(1).get_near_int();
246  y123->get_ravel(2).get_near_int(); break;
247 
248  default: LENGTH_ERROR;
249  }
250 
251  // 3. adjust rank_B if they exceed its initial value or
252  // if it is negative
253  //
254  if (rank_B > rk_B) rank_B = rk_B;
255  if (rank_B < 0) rank_B += rk_B;
256  if (rank_B < 0) rank_B = 0;
257 }

259 void
260 Bif_OPER2_RANK::y123_to_AB(Value_P y123, Rank & rank_A, Rank & rank_B)
261 {
262  // y123_to_AB() splits the ranks of A and B into a (higher-dimensions)
263  // "frame" and a (lower-dimensions) "chunk" as specified by y123.
264 
265  // 1. on entry rank_A and rank_B are the ranks of A and B.
266  //
267  // Remember the ranks of A and B to limit rank_A and rank_B
268  // if values in y123 should exceed them.
269  //
270 const Rank rk_A = rank_A;
271 const Rank rk_B = rank_B;
272 
273  if (!y123) VALUE_ERROR;
274  if ( y123->get_rank() > 1) DOMAIN_ERROR;
275 
276  // 2. the number of elements in y determine how rank_A and rank_B
277  // shall be computed:
278  //
279  // -- monadic fâ¤ -- -- dyadic fâ¤ --
280  // rank_A rank_B rank_A rank_B
281  // ---------------------------------------------------------
282  // y : N/A y y y
283  // yA yB : N/A yB yA yB
284  // yM yA yB : N/A yM yA yB
285  // ---------------------------------------------------------
286 
287  switch(y123->element_count())
288  {
289  case 1: rank_A = y123->get_ravel(0).get_near_int();
290  rank_B = rank_A; break;
291 
292  case 2: rank_A = y123->get_ravel(0).get_near_int();
293  rank_B = y123->get_ravel(1).get_near_int(); break;
294 
295  case 3: y123->get_ravel(0).get_near_int();
296  rank_A = y123->get_ravel(1).get_near_int();
297  rank_B = y123->get_ravel(2).get_near_int(); break;
298 
299  default: LENGTH_ERROR;
300  }
301 
302  // 3. adjust rank_A and rank_B if they exceed their initial value or
303  // if they are negative
304  //
305  if (rank_A > rk_A) rank_A = rk_A;
306  if (rank_A < 0) rank_A += rk_A;
307  if (rank_A < 0) rank_A = 0;
308 
309  if (rank_B > rk_B) rank_B = rk_B;
310  if (rank_B < 0) rank_B += rk_B;
311  if (rank_B < 0) rank_B = 0;
312 }

314 void
315 Bif_OPER2_RANK::split_y123_B(Value_P y123_B, Value_P & y123, Value_P & B)
316 {
317  // The ISO standard and NARS define the reduction pattern for the RANK
318  // operator â¤ as:
319  //
320  // Z â A f â¤ y B (ISO)
321  // Z â f â¤ y B (ISO)
322  // Z â A f â¤ [X] y B (NARS)
323  // Z â f â¤ [X] y B (NARS)
324  //
325  // GNU APL may bind y to B at tokenization time if y and B are constants
326  // This function tries to "unbind" its argument y123_B into the original
327  // components y123 (= y in the standard) and B. The tokenization time
328  // binding is shown as y123:B
329  //
330  // Usage y123 : B Result: j123 B
331  //-------------------------------------------------------------------------
332  // 1. f â¤ (y123):B... nested any y123 B
333  // 2. (f â¤ y123:â¬) simple empty y123 -
334  // 3. f â¤ y123:(B) simple nested skalar y123 B
335  // 4a. f â¤ y123:B... simple any y123 B...
336  //
337 
338  // y123_B shall be a scalar or vector
339  //
340  if (y123_B->get_rank() > 1) RANK_ERROR;
341 
342 const ShapeItem length = y123_B->element_count();
343  if (length == 0) LENGTH_ERROR;
344 
345  // check for case 1 (the only one with nested first element)
346  //
347  if (y123_B->get_ravel(0).is_pointer_cell()) // (y123)
348  {
349  y123 = y123_B->get_ravel(0).get_pointer_value();
350  if (length == 1) // empty B
351  {
352  }
353  else if (length == 2) // skalar B
354  {
355  const Cell & B0 = y123_B->get_ravel(1);
356  if (B0.is_pointer_cell()) // (B)
357  {
358  B = B0.get_pointer_value();
359  }
360  else
361  {
362  B = Value_P(LOC);
363  B->next_ravel()->init(B0, B.getref(), LOC);
364  }
365  }
366  else // vector B
367  {
368  B = Value_P(length - 1, LOC);
369  loop(l, length - 1)
370  B->next_ravel()->init(y123_B->get_ravel(l + 1),
371  B.getref(), LOC);
372  }
373  y123->check_value(LOC);
374  B->check_value(LOC);
375  return;
376  }
377 
378  // case 1. ruled out, so the first 1, 2, or 3 cells are j123.
379  // see how many (at most)
380  //
381 int y123_len = 0;
382  loop(yy, 3)
383  {
384  if (yy >= length) break;
385  const Cell & cy = y123_B->get_ravel(yy);
386  if (cy.is_near_int()) ++y123_len;
387  else break;
388  }
389  if (y123_len == 0) LENGTH_ERROR; // at least y1 is needed
390 
391  // cases 2.-4. start with integers of length 1, 2, or 3
392  //
393  if (length == y123_len) // case 2: y123:â¬
394  {
395  y123 = y123_B;
396  //
397  // NOTE: B is NOT assigned so that Prefix::reduce_F_D_B_() can detect
398  // that y123_ was only y123 !
399  //
400  return;
401  }
402 
403  if (length == (y123_len + 1) &&
404  y123_B->get_ravel(y123_len).is_pointer_cell()) // case 3. y123:âB
405  {
406  y123 = Value_P(y123_len, LOC);
407  loop(yy, y123_len)
408  y123->next_ravel()->init(y123_B->get_ravel(yy), y123.getref(), LOC);
409  B = y123_B->get_ravel(y123_len).get_pointer_value();
410  y123->check_value(LOC);
411  B->check_value(LOC);
412  return;
413  }
414 
415  // case 4: y123:B...
416  //
417  y123 = Value_P(y123_len, LOC);
418  loop(yy, y123_len)
419  y123->next_ravel()->init(y123_B->get_ravel(yy), y123.getref(), LOC);
420 
421 const ShapeItem B_len = length - y123_len;
422  B = Value_P(B_len, LOC);
423  loop(bb, B_len)
424  B->next_ravel()->init(y123_B->get_ravel(bb + y123_len), B.getref(), LOC);
425  B->check_value(LOC);
426 }

55  << "now printed at " << loc << endl;
56 
57  return;
58  }
59 
60  Log(LOG_verbose_error)
61  {
62  out << endl
63  << "--------------------------" << endl;
64 
65  if (error_code == E_NO_ERROR)
66  {
67  out << error_name(E_NO_ERROR) << endl
68  << "--------------------------" << endl;
69 
70  return;
71  }
72 
73  if (*error_message_1)
74  {
75  out << error_message_1 << endl;
76  }
77  else
78  {
79  out << error_name(error_code);
80  if (Workspace::more_error().size()) out << UNI_ASCII_PLUS;
81  out << endl;
82  }
83 
84  if (parser_loc) out << " Parser LOC: " << parser_loc << endl;
85  if (print_loc) out << " Print LOC: " << print_loc << endl;
86  out << " loc: " << loc << endl;
87  loc = print_loc;
88 
89  if (*symbol_name)
90  out << " Symbol: " << symbol_name << endl;
91 
92  out << " Thrown at: " << throw_loc << endl
93  << "--------------------------" << endl
94  << endl;
95  }

144 UCS_string
145 Error::get_error_line_3() const
146 {
147  if (error_code == E_NO_ERROR) return UCS_string();
148 
149  if (left_caret < 0) return UCS_string();
150 
151 UCS_string ret;
152  if (left_caret > 0) ret.append(UCS_string(left_caret, UNI_ASCII_SPACE));
153  ret.append(UNI_ASCII_CIRCUMFLEX);
154 
155 const int diff = right_caret - left_caret;
156  if (diff <= 0) return ret;
157  if (diff > 1) ret.append(UCS_string(diff - 1, UNI_ASCII_SPACE));
158  ret.append(UNI_ASCII_CIRCUMFLEX);
159 
160  return ret;
161 }

285 //-----------------------------------------------------------------------------
286 void
287 Error::update_error_info(StateIndicator * si)
288 {
289 bool locked = false;
290 const UserFunction * ufun = si->get_executable()->get_ufun();
291 
292  // prepare second error line (failed statement)
293  //
294  if (ufun)
295  {
296  if (get_show_locked() || ufun->get_exec_properties()[1])
297  {
298  locked = true;
299  set_error_line_2(" ");
300  set_left_caret(6);
301  }
302  else
303  {
304  UCS_string ucs(ufun->get_name_and_line(si->get_PC()));
305  ucs.append(UNI_ASCII_SPACE);
306  ucs.append(UNI_ASCII_SPACE);
307  UTF8_string utf(ucs);
308  set_error_line_2(utf.c_str());
309  set_left_caret(ucs.size());
310  }
311  }
312  else
313  {
314  set_error_line_2(" ");
315  set_left_caret(6);
316  }
317 
318  // prepare third line (carets)
319  //
320  set_right_caret(-1);
321 
322  if (locked)
323  {
324  si->get_executable()->get_ufun()->set_locked_error_info(*this);
325  }
326  else
327  {
328  const Function_PC from = si->get_prefix().get_range_low();
329  Function_PC to = si->get_prefix().get_range_high();
330  const Function_PC2 error_range(from, to);
331  si->get_executable()->set_error_info(*this, error_range);
332  }
333 
334  // print error, unless we are in safe execution mode.
335  //
336  {
337  bool print_error = true;
338  for (const StateIndicator * si1 = si; si1; si1 = si1->get_parent())
339  {
340  if (si1->get_safe_execution())
341  {
342  print_error = false;
343  break;
344  }
345  }
346  if (print_error) print_em(UERR, LOC);
347  }
348 
349  StateIndicator::get_error(si) = *this;
350 }

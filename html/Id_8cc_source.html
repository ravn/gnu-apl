104  bsearch(&id, id2ucs, sizeof(id2ucs) / sizeof(Id_name),
105  sizeof(Id_name), Id_name::compare);
106 
107  Assert(result);
108 Id_name * idn = static_cast<Id_name *>(result);
109  if (const UTF8 * utf = idn->utf_name) return utf;
110 
111  // the name was not yet constructed. Do it now
112  //
113 const char * name = "unknown ID";
114  switch(id)
115  {
116 #define pp(i, _u, _v) case ID_ ## i: name = #i; break;
117 #define qf(i, u, _v) case ID_Quad_ ## i: name = u; break;
118 #define qv(i, u, _v) case ID_Quad_ ## i: name = u; break;
119 #define sf(i, u, _v) case ID_ ## i: name = u; break;
120 #define st(i, u, _v) case ID_ ## i: name = u; break;

122 #include "Id.def"

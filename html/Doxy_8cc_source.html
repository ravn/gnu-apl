40 static bool
41 symcomp(const Symbol * const & s1, const Symbol * const & s2, const void *)
42 {
43  return s2->get_name().compare(s1->get_name()) == COMP_LT;

45 //-----------------------------------------------------------------------------
46 Doxy::Doxy(ostream & cout, const UCS_string & dest_dir)
47  : out(cout),
48  root_dir(dest_dir),
49  errors(0)
50 {
51  ws_name = Workspace::get_WS_name();
52  if (ws_name.compare(UCS_string("CLEAR WS")) == 0)
53  ws_name = UCS_string("CLEAR-WS");
54  else if (Avec::is_digit(ws_name[0]))
55  {
56  // workspace name starts with a library reference number. Skip the
57  // library reference number and whitespace directly after it
58  //
59  int pos = 1; // first blank after the library reference number
60  while (pos < ws_name.size() && Avec::is_white(ws_name[pos])) ++pos;
61  ws_name = UCS_string(ws_name, pos, ws_name.size() - pos);
62  }
63 
64  root_dir.append_ASCII("/");
65  root_dir.append_UTF8(UTF8_string(ws_name));
66 
67  Log(LOG_command_DOXY)
68  out << "Creating output directory " << root_dir << endl;
69 
70  errno = 0;
71  if (mkdir(root_dir.c_str(), 0777))
72  {
73  const char * why = strerror(errno);
74 
75  // EEXIST says "File exists" which is a little misleading
76  if (errno == EEXIST) why = "Directory already exists";
77 
78  CERR << "Cannot create fresh destination directory "
79  << root_dir << ": " << why << endl;
80  ++errors;
81  DOMAIN_ERROR;
82  }
83  write_css();
84 }

121 //-----------------------------------------------------------------------------
122 void
123 Doxy::gen()
124 {
125 const SymbolTable & symtab = Workspace::get_symbol_table();
126 
127 std::vector<const Symbol *> all_symbols = symtab.get_all_symbols();
128 std::vector<const Symbol *> variables;
129 
130  loop(a, all_symbols.size())
131  {
132  const Symbol * sym = all_symbols[a];
133  if (sym->is_erased()) continue;
134  if (sym->get_name()[0] == UNI_MUE) continue; // macro
135 
136  bool is_function = false;
137  bool is_variable = false;
138  loop(si, sym->value_stack_size())
139  {
140  switch((*sym)[si].name_class)
141  {
142  case NC_VARIABLE: is_variable = true; break;
143  case NC_FUNCTION: is_function = true; break;
144  case NC_OPERATOR: is_function = true; break;
145  default: ;
146  }
147  }
148  if (is_function)
149  all_functions.push_back(sym);
150 
151 
152  if (is_variable)
153  variables.push_back(sym);
154  }
155 
156  if (all_functions.size() > 1)
157  Heapsort<const Symbol *>::
158  sort(&all_functions[0], all_functions.size(), 0, symcomp);
159 
160  if (variables.size() > 1)
161  Heapsort<const Symbol *>::
162  sort(&variables[0], variables.size(), 0, symcomp);
163 
164 UTF8_string index_filename(root_dir);
165  index_filename.append_ASCII("/index.html");
166 
167  Log(LOG_command_DOXY)
168  out << "Writing top-level HTML file " << index_filename << endl;
169 
170 ofstream page(index_filename.c_str());
171  page <<
172 "<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.01//EN\"" CRLF
173 " \"http://www.w3.org/TR/html4/strict.dtd\">" CRLF
174 "<HTML>" CRLF
175 " <HEAD>" CRLF
176 " <TITLE>Documentation of " << ws_name << "</title> " CRLF
177 " <META http-equiv=\"content-type\" " CRLF
178 " content=\"text/html; charset=UTF-8\">" CRLF
179 " <LINK rel='stylesheet' type='text/css' href='apl_doxy.css'>" CRLF
180 " </HEAD>" CRLF
181 " <BODY>" CRLF
182 " <H1>Documentation of workspace " << ws_name << "</H1><HR>" CRLF;
183 
184  make_call_graph(all_functions);
185  functions_table(all_functions, page);
186  variables_table(variables, page);
187  SI_table(page);
188 
189  set_call_graph_root(0);
190 const UCS_string alias = "all_functions";
191  if (write_call_graph(0, alias, false) == 0)
192  {
193  page <<
194 " <H3>Global Function Call Graph</H3>" CRLF
195 " <IMG class=cg1 usemap=#CG src=cg_" << alias << ".png>" CRLF;
196 
197  // copy the cmap file for the IMG (generated by dot) into the page
198  {
199  UTF8_string cmapx_filename(root_dir);
200  cmapx_filename.append_ASCII("/cg_");
201  cmapx_filename.append_UTF8(UTF8_string(alias));
202  cmapx_filename.append_ASCII(".cmapx");
203  if (FILE * cmap = fopen(cmapx_filename.c_str(), "r"))
204  {
205  char buffer[400];
206  for (;;)
207  {
208  const char * s = fgets(buffer, sizeof(buffer), cmap);
209  if (s == 0) break;
210  buffer[sizeof(buffer) - 1] = 0;
211  page << " " << buffer;
212  }
213  fclose(cmap);
214  }
215  else
216  {
217  CERR << "*** Cannot open " << cmapx_filename
218  << ": " << strerror(errno) << endl;
219  }
220 
221  Log(LOG_command_DOXY) {} else unlink(cmapx_filename.c_str());
222  }
223  }
224 
225  page <<
226 " </BODY>" CRLF
227 "</HTML>" CRLF;
228  page.close();
229 }
230 //-----------------------------------------------------------------------------
231 void
232 Doxy::functions_table(const std::vector<const Symbol *> & functions,
233  ofstream & page)
234 {
235  if (functions.size() == 0) return;
236 
237  page <<
238 " <H3>Defined Functions</H3>" CRLF
239 " <TABLE class=funtab>" CRLF
240 " <TR>" CRLF
241 " <TH>Function" CRLF
242 " <TH>L" CRLF
243 " <TH>SI" CRLF
244 " <TH>Lines" CRLF
245 " <TH>Header" CRLF
246 " <TH class=doxy_comment>Doxy Comments" CRLF;
247 
248 int total_lines = 0;
249  loop(f, functions.size())
250  {
251  const Symbol * fun_sym = functions[f];
252  loop(si, fun_sym->value_stack_size())
253  {
254  if ((*fun_sym)[si].name_class != NC_FUNCTION &&
255  (*fun_sym)[si].name_class != NC_OPERATOR) continue;
256 
257  const Function * fp = (*fun_sym)[si].sym_val.function;
258  Assert(fp);
259  const UserFunction * ufun = fp->get_ufun1();
260  const int si_level = fun_sym->get_SI_level(fp);
261 
262  // colunm 1: function name/link
263  //
264  page <<
265 " <TR>" CRLF
266 " <TD class=code>";
267 
268  int line_count = 0;
269  if (fp->is_native())
270  {
271  page << fun_sym->get_name() << " (native)" CRLF;
272  }
273  else if (fp->is_lambda())
274  {
275  line_count = 1;
276  function_page(ufun, fun_sym->get_name());
277  page << fun_anchor(fun_sym->get_name()) << CRLF;
278  }
279  else if (ufun)
280  {
281  line_count = ufun->get_text_size();
282  function_page(ufun, ufun->get_name());
283  page << fun_anchor(fun_sym->get_name()) << CRLF;
284  }
285  else
286  {
287  page << fun_sym->get_name() << CRLF;
288  }
289 
290  // column 2: local stack depth
291  //
292  page <<
293 " <TD class='code center'>" << si << CRLF;
294 
295  // column 3: SI level
296  //
297  if (si_level) page <<
298 " <TD class='code center'>" << si_level << CRLF;
299  else page <<
300 " <TD class='code center'>G" CRLF;
301 
302  // column 4: number of lines
303  //
304  page <<
305 " <TD class='code center'>" << line_count << CRLF;
306 
307  total_lines += line_count;
308  // column 5: function header (if any)
309  //
310  page <<
311 " <TD class=code>";
312  if (fp->is_native())
313  page << reinterpret_cast<const NativeFunction *>
314  (fp)->get_so_path();
315  else if (fp->is_lambda())
316  {
317  Assert(ufun);
318  const UCS_string & text = ufun->get_text(1);
319  page << "{" << UCS_string(text, 2, text.size() - 2)
320  << "}" CRLF;
321  }
322  else if (ufun)
323  bold_name(page, ufun);
324  else
325  page << "-";
326  page << CRLF;
327 
328  // column 6: Doxygen comments
329  //
330  page <<
331 " <TD class=doxy_comment>";
332  if (fp->is_lambda())
333  page << "(named Î»)" CRLF;
334  else if (ufun && !fp->is_native())
335  {
336  loop(l, ufun->get_text_size())
337  {
338  UCS_string line = ufun->get_text(l);
339  line.remove_leading_and_trailing_whitespaces();
340  if (line.size() >= 2 &&
341  line[0] == UNI_COMMENT &&
342  line[1] == UNI_COMMENT)
343  page << line.to_HTML(2, false) << CRLF;
344  }
345  }
346  else
347  page << "-" CRLF;
348  }
349  }
350 
351  // summary line
352  //
353  page <<
354 " <TR><TD><TD colspan=2 class='code center'>Total<TD class='code center'>"
355  << total_lines << CRLF
356 " </TABLE>" CRLF;
357 }

429 //-----------------------------------------------------------------------------
430 void
431 Doxy::SI_table(ofstream & page)
432 {
433  // collect SI entries in reverse order...
434  //
435 std::vector<const StateIndicator *> stack;
436 
437  for (const StateIndicator * si = Workspace::SI_top();
438  si; si = si->get_parent())
439  {
440  stack.push_back(si);
441  }
442 
443  if (stack.size() == 0) return;
444 
445  page <<
446 " <H3>SI Stack</H3>" CRLF
447 " <TABLE class=sitab>" CRLF
448 " <TR>" CRLF
449 " <TH>SI" CRLF
450 " <TH>T" CRLF
451 " <TH>Fun[line]" CRLF
452 " <TH>Statement" CRLF
453 " <TH>Local" CRLF
454 " <TH>Error" CRLF;
455 
456  loop(d, stack.size())
457  {
458  const StateIndicator * si = stack[stack.size() - d - 1];
459  const Executable * exec = si->get_executable();
460  Assert(exec);
461  const UserFunction * ufun = exec->get_ufun(); // possibly 0!
462 
463  page <<
464 " <TR>" CRLF
465 " <TD class='code center'>" << si->get_level() << CRLF
466 " <TD class='code center'>" << si->get_parse_mode_name() << CRLF;
467  if (si->get_parse_mode() == PM_FUNCTION)
468  {
469  Assert(ufun);
470  const Function_PC PC = si->get_prefix().get_error_PC();
471  if (const ErrorCode ec =
472  StateIndicator::get_error(si).get_error_code())
473  {
474  // get_error_line_2() is something like fun[line] statement.
475  // find the space after fun[line].
476  //
477  int spc2 = 0;
478  const UCS_string el2(UTF8_string(
479  StateIndicator::get_error(si).get_error_line_2()));
480  loop(e, el2.size())
481  {
482  if (el2[e] == ']')
483  {
484  spc2 = e + 1;
485  break;
486  }
487  }
488 
489  int spe2 = spc2;
490  while (spe2 < el2.size() &&
491  el2[spe2] == UNI_ASCII_SPACE) ++spe2;
492  const int spl2 = el2.size() - spe2;
493  const UCS_string & el3 =
494  StateIndicator::get_error(si).get_error_line_3();
495 
496  // put fun[line] into one <TD> and the failed statement into
497  // the next <TD>
498  //
499  page <<
500 " <TD class=code>" << UCS_string(el2, 0, spc2) << CRLF
501 " <TD class=code>" << UCS_string(el2, spe2, spl2) << BRLF
502 " " << el3.to_HTML(spe2, true) << CRLF
503 " <TD class=code>>"; if (ufun) ufun->print_local_vars(page);
504  page << CRLF
505 " <TD class=code>" << Error::error_name(ec) << CRLF;
506  }
507  else
508  {
509  page <<
510 " <TD class=code>" << ufun->get_name_and_line(PC) << CRLF
511 " <TD class=code>" << exec->statement_text(PC) << CRLF
512 " <TD class=code>"; if (ufun) ufun->print_local_vars(page);
513  page << CRLF;
514  }
515  }
516  else if (si->get_parse_mode() == PM_STATEMENT_LIST)
517  {
518  page <<
519 " <TD 'code center'>-" CRLF
520 " <TD class=code>" << exec->get_text(0) << CRLF;
521  }
522  else // PM_EXECUTE
523  {
524  page <<
525 " <TD'code center'>-" CRLF
526 " <TD class=code>" << exec->get_text(0) << CRLF;
527  }
528  }
529  page <<
530 " </TABLE>" CRLF;
531 }

650 //-----------------------------------------------------------------------------

656

659  if (header.LO()) // operator

693 void
694 Doxy::add_fun_to_call_graph(const Symbol * caller_sym,
695  const UserFunction * ufun)
696 {
697  Log(LOG_command_DOXY)
698  out << " add (caller-) Symbol " << caller_sym->get_name() << endl;
699 
700 const Token_string & body = ufun->get_body();
701 
702  // scan the body of ufun for symbol Token and filter those that are
703  // functions. This is rather heuriostic and therefore not reliable.
704  //
705  // It can fail if the same name is used for both functions and variables.
706  //
707  loop(b, body.size())
708  {
709  const Token & tok = body[b];
710  if (tok.get_Class() != TC_SYMBOL) continue;
711 
712  const Symbol * callee_ptr = tok.get_sym_ptr();
713  Assert(callee_ptr);
714 
715  // assume (maybe incorrectly) that local symbol of ufun are variables.
716  //
717  if (ufun->localizes(callee_ptr)) continue;
718 
719  const Symbol & callee_sym = *callee_ptr;
720  loop(si, callee_sym.value_stack_size())
721  {
722  if (callee_sym[si].name_class == NC_FUNCTION ||
723  callee_sym[si].name_class == NC_OPERATOR)
724  {
725  const Function * fp = callee_sym[si].sym_val.function;
726  Assert(fp);
727  const UserFunction * callee = fp->get_ufun1();
728  if (!callee) continue;
729 
730  // ignore multiple calls of the same callee from the
731  // same caller...
732  //
733  bool have_call = false;
734  loop(cg, call_graph.size())
735  {
736  if (call_graph[cg].caller == ufun &&
737  call_graph[cg].callee == callee)
738  {
739  have_call = true;
740  break; // loop(cg, ...
741  }
742  }
743 
744  if (!have_call)
745  {
746  const fcall_edge edge(ufun, caller_sym->get_name(),
747  callee, callee_sym.get_name());
748 
749  call_graph.push_back(edge);
750  Log(LOG_command_DOXY)
751  out << " " << caller_sym->get_name()
752  << " calls " << callee->get_name() << endl;
753  }
754  }
755  }
756  }

758 //-----------------------------------------------------------------------------
759 void
760 Doxy::set_call_graph_root(const UserFunction * ufun)
761 {
762 const int MAX = 2 * call_graph.size(); // MAX means unreachable
763 UCS_string root_name("all");
764  if (ufun) root_name = ufun->get_name();
765 
766  if (call_graph.size() == 0)
767  {
768  // there are no calls (all functions are leafs). If the call graph
769  // root is the top-level (i.e. ufun == 0), then add all nodes
770  // to nodes.
771  //
772  if (ufun == 0)
773  {
774  nodes.clear();
775  aliases.clear();
776  loop(f, all_functions.size())
777  {
778  const Symbol * sym = all_functions[f];
779  const Function * fun = sym->get_function();
780  const UserFunction * ufun = fun->get_ufun1();
781  nodes.push_back(ufun);
782  aliases.push_back(ufun->get_name());
783  }
784  return;
785  }
786  }
787 
788  // 1. set all edges starting at ufun to 0 and all others unreachable
789  //
790  loop(cg, call_graph.size())
791  {
792  fcall_edge & edge = call_graph[cg];
793  if (edge.caller == ufun) edge.value = 0; // root
794  else if (ufun == 0) edge.value = 1; // no root, all edges
795  else edge.value = MAX; // unreachable
796  }
797 
798 bool progress = true;
799  for (int depth = 0; progress; ++depth)
800  {
801  progress = false;
802  loop(f, call_graph.size())
803  {
804  const fcall_edge & from = call_graph[f];
805  if (from.value != depth) continue;
806 
807  // 'from' starts at level depth.
808  // Mark all edges 'to'at level depth + 1
809  //
810  loop(t, call_graph.size())
811  {
812  fcall_edge & to = call_graph[t];
813  if (from.callee != to.caller) continue;
814  if (to.value < (depth + 1)) continue;
815  to.value = depth + 1;
816  progress = true;
817  }
818  }
819  }
820 
821  Log(LOG_command_DOXY)
822  loop(e, call_graph.size())
823  {
824  const fcall_edge & edge = call_graph[e];
825  if (edge.value == MAX) continue; // not reachable
826 
827  out << "edge " << edge.caller->get_name() << " â "
828  << edge.callee->get_name()
829  << " has distance " << edge.value
830  << " from " << root_name << endl;
831  }
832 
833  // create a list of nodes that are reachable
834  //
835  nodes.clear();
836  aliases.clear();
837  loop(e, call_graph.size())
838  {
839  const fcall_edge & edge = call_graph[e];
840  if (edge.value == MAX) continue; // not reachable
841 
842  bool caller_is_new = true;
843  bool callee_is_new = true;
844  loop(n, nodes.size())
845  {
846  if (nodes[n] == edge.caller) caller_is_new = false;
847  if (nodes[n] == edge.callee) callee_is_new = false;
848  }
849 
850  if (caller_is_new)
851  {
852  nodes.push_back(edge.caller);
853  aliases.push_back(edge.caller_name);
854  }
855  if (callee_is_new)
856  {
857  nodes.push_back(edge.callee);
858  aliases.push_back(edge.callee_name);
859  }
860  }
861 
862  Log(LOG_command_DOXY)
863  {
864  loop(n, nodes.size())
865  out << "Node: " << nodes[n]->get_name()
866  << " alias: " << aliases[n] << endl;
867  }
868 }

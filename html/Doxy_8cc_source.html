107 //-----------------------------------------------------------------------------
108 void
109 Doxy::gen()
110 {
111 const SymbolTable & symtab = Workspace::get_symbol_table();
112 
113 std::vector<const Symbol *> all_symbols = symtab.get_all_symbols();
114 std::vector<const Symbol *> functions;
115 std::vector<const Symbol *> variables;
116 
117  loop(a, all_symbols.size())
118  {
119  const Symbol * sym = all_symbols[a];
120  if (sym->is_erased()) continue;
121  if (sym->get_name()[0] == UNI_MUE) continue; // macro
122 
123  bool is_function = false;
124  bool is_variable = false;
125  loop(si, sym->value_stack_size())
126  {
127  switch((*sym)[si].name_class)
128  {
129  case NC_VARIABLE: is_variable = true; break;
130  case NC_FUNCTION: is_function = true; break;
131  case NC_OPERATOR: is_function = true; break;
132  default: ;
133  }
134  }
135  if (is_function) functions.push_back(sym);
136  if (is_variable) variables.push_back(sym);
137  }
138 
139  if (functions.size() > 1)
140  Heapsort<const Symbol *>::
141  sort(&functions[0], functions.size(), 0, symcomp);
142 
143  if (variables.size() > 1)
144  Heapsort<const Symbol *>::
145  sort(&variables[0], variables.size(), 0, symcomp);
146 
147 UTF8_string index_filename(root_dir);
148  index_filename.append_ASCII("/index.html");
149 
150  Log(LOG_command_DOXY)
151  out << "Writing top-level HTML file " << index_filename << endl;
152 
153 ofstream page(index_filename.c_str());
154  page <<
155 "<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.01//EN\"" CRLF
156 " \"http://www.w3.org/TR/html4/strict.dtd\">" CRLF
157 "<HTML>" CRLF
158 " <HEAD>" CRLF
159 " <TITLE>Documentation of " << ws_name << "</title> " CRLF
160 " <META http-equiv=\"content-type\" " CRLF
161 " content=\"text/html; charset=UTF-8\">" CRLF
162 " <LINK rel='stylesheet' type='text/css' href='apl_doxy.css'>" CRLF
163 " </HEAD>" CRLF
164 " <BODY>" CRLF
165 " <H1>Documentation of workspace " << ws_name << "</H1><HR>" CRLF;
166 
167  make_call_graph(functions);
168  functions_table(functions, page);
169  variables_table(variables, page);
170  SI_table(page);
171 
172  set_call_graph_root(0);
173 const UCS_string alias = "all-functions";
174  if (write_call_graph(0, alias, false) == 0)
175  {
176  page <<
177 " <H3>Global Function Call Graph</H3>" CRLF
178 " <IMG class=cg1 usemap=#CG src=cg_" << alias << ".png>" CRLF;
179 
180  // copy the cmap file for the IMG (generated by dot) into the page
181  {
182  UTF8_string cmapx_filename(root_dir);
183  cmapx_filename.append_ASCII("/cg_");
184  cmapx_filename.append_UTF8(UTF8_string(alias));
185  cmapx_filename.append_ASCII(".cmapx");
186  FILE * cmap = fopen(cmapx_filename.c_str(), "r");
187  Assert(cmap);
188  char buffer[400];
189  for (;;)
190  {
191  const char * s = fgets(buffer, sizeof(buffer), cmap);
192  if (s == 0) break;
193  buffer[sizeof(buffer) - 1] = 0;
194  page << " " << buffer;
195  }
196  fclose(cmap);
197  Log(LOG_command_DOXY) {} else unlink(cmapx_filename.c_str());
198  }
199  }
200 
201  page <<
202 " </BODY>" CRLF
203 "</HTML>" CRLF;
204  page.close();
205 }

208 Doxy::functions_table(const std::vector<const Symbol *> & functions,
209  ofstream & page)
210 {
211  if (functions.size() == 0) return;
212 
213  page <<
214 " <H3>Defined Functions</H3>" CRLF
215 " <TABLE class=funtab>" CRLF
216 " <TR>" CRLF
217 " <TH>Function" CRLF

219 " <TH>SI" CRLF
220 " <TH>Lines" CRLF
221 " <TH>Header" CRLF
222 " <TH class=doxy_comment>Doxy Comments" CRLF;
223 
224 int total_lines = 0;
225  loop(f, functions.size())
226  {
227  const Symbol * fun_sym = functions[f];
228  loop(si, fun_sym->value_stack_size())
229  {
230  if ((*fun_sym)[si].name_class != NC_FUNCTION &&
231  (*fun_sym)[si].name_class != NC_OPERATOR) continue;
232 
233  const Function * fp = (*fun_sym)[si].sym_val.function;
234  Assert(fp);
235  const UserFunction * ufun = fp->get_ufun1();
236  const int si_level = fun_sym->get_SI_level(fp);
237 
238  // colunm 1: function name/link
239  //
240  page <<
241 " <TR>" CRLF
242 " <TD class=code>";
243 
244  int line_count = 0;
245  if (fp->is_native())
246  {
247  page << fun_sym->get_name() << " (native)" CRLF;
248  }
249  else if (fp->is_lambda())
250  {
251  line_count = 1;
252  function_page(ufun, fun_sym->get_name());
253  page << fun_anchor(fun_sym->get_name()) << CRLF;
254  }
255  else if (ufun)
256  {
257  line_count = ufun->get_text_size();
258  function_page(ufun, ufun->get_name());
259  page << fun_anchor(fun_sym->get_name()) << CRLF;
260  }
261  else
262  {
263  page << fun_sym->get_name() << CRLF;
264  }
265 
266  // column 2: local stack depth
267  //
268  page <<
269 " <TD class='code center'>" << si << CRLF;
270 
271  // column 3: SI level
272  //
273  if (si_level) page <<
274 " <TD class='code center'>" << si_level << CRLF;
275  else page <<
276 " <TD class='code center'>G" CRLF;
277 
278  // column 4: number of lines
279  //
280  page <<
281 " <TD class='code center'>" << line_count << CRLF;
282 
283  total_lines += line_count;
284  // column 5: function header (if any)
285  //
286  page <<
287 " <TD class=code>";
288  if (fp->is_native())
289  page << reinterpret_cast<const NativeFunction *>
290  (fp)->get_so_path();
291  else if (fp->is_lambda())
292  {
293  Assert(ufun);
294  const UCS_string & text = ufun->get_text(1);
295  page << "{" << UCS_string(text, 2, text.size() - 2)
296  << "}" CRLF;
297  }
298  else if (ufun)
299  bold_name(page, ufun);
300  else
301  page << "-";
302  page << CRLF;
303 
304  // column 6: Doxygen comments
305  //
306  page <<
307 " <TD class=doxy_comment>";
308  if (fp->is_lambda())
309  page << "(named Î»)" CRLF;
310  else if (ufun && !fp->is_native())
311  {
312  loop(l, ufun->get_text_size())
313  {
314  const UCS_string & line = ufun->get_text(l);
315  if (line.size() >= 2 &&
316  line[0] == UNI_COMMENT &&
317  line[1] == UNI_COMMENT)
318  page << line.to_HTML(2, false) << CRLF;
319  }
320  }
321  else
322  page << "-" CRLF;
323  }
324  }
325 
326  // summary line
327  //
328  page <<
329 " <TR><TD><TD colspan=2 class=code center>Total<TD class=code center>"
330  << total_lines << CRLF
331 " </TABLE>" CRLF;
332 }

404 //-----------------------------------------------------------------------------
405 void
406 Doxy::SI_table(ofstream & page)
407 {
408  // collect SI entries in reverse order...
409  //
410 std::vector<const StateIndicator *> stack;
411 
412  for (const StateIndicator * si = Workspace::SI_top();
413  si; si = si->get_parent())
414  {
415  stack.push_back(si);
416  }
417 
418  if (stack.size() == 0) return;
419 
420  page <<
421 " <H3>SI Stack</H3>" CRLF
422 " <TABLE class=sitab>" CRLF
423 " <TR>" CRLF
424 " <TH>SI" CRLF
425 " <TH>T" CRLF
426 " <TH>Fun[line]" CRLF
427 " <TH>Statement" CRLF
428 " <TH>Local" CRLF
429 " <TH>Error" CRLF;
430 
431  loop(d, stack.size())
432  {
433  const StateIndicator * si = stack[stack.size() - d - 1];
434  const Executable * exec = si->get_executable();
435  Assert(exec);
436  const UserFunction * ufun = exec->get_ufun(); // possibly 0!
437 
438  page <<
439 " <TR>" CRLF
440 " <TD class='code center'>" << si->get_level() << CRLF
441 " <TD class='code center'>" << si->get_parse_mode_name() << CRLF;
442  if (si->get_parse_mode() == PM_FUNCTION)
443  {
444  Assert(ufun);
445  const Function_PC PC = si->get_prefix().get_error_PC();
446  if (const ErrorCode ec =
447  StateIndicator::get_error(si).get_error_code())
448  {
449  // get_error_line_2() is something like fun[line] statement.
450  // find the space after fun[line].
451  //
452  int spc2 = 0;
453  const UCS_string el2(UTF8_string(
454  StateIndicator::get_error(si).get_error_line_2()));
455  loop(e, el2.size())
456  {
457  if (el2[e] == ']')
458  {
459  spc2 = e + 1;
460  break;
461  }
462  }
463 
464  int spe2 = spc2;
465  while (spe2 < el2.size() &&
466  el2[spe2] == UNI_ASCII_SPACE) ++spe2;
467  const int spl2 = el2.size() - spe2;
468  const UCS_string & el3 =
469  StateIndicator::get_error(si).get_error_line_3();
470 
471  // put fun[line] into one <TD> and the failed statement into
472  // the next <TD>
473  //
474  page <<
475 " <TD class=code>" << UCS_string(el2, 0, spc2) << CRLF
476 " <TD class=code>" << UCS_string(el2, spe2, spl2) << BRLF
477 " " << el3.to_HTML(spe2, true) << CRLF
478 " <TD class=code>>"; if (ufun) ufun->print_local_vars(page);
479  page << CRLF
480 " <TD class=code>" << Error::error_name(ec) << CRLF;
481  }
482  else
483  {
484  page <<
485 " <TD class=code>" << ufun->get_name_and_line(PC) << CRLF
486 " <TD class=code>" << exec->statement_text(PC) << CRLF
487 " <TD class=code>"; if (ufun) ufun->print_local_vars(page);
488  page << CRLF;
489  }
490  }
491  else if (si->get_parse_mode() == PM_STATEMENT_LIST)
492  {
493  page <<
494 " <TD 'code center'>-" CRLF
495 " <TD class=code>" << exec->get_text(0) << CRLF;
496  }
497  else // PM_EXECUTE
498  {
499  page <<
500 " <TD'code center'>-" CRLF
501 " <TD class=code>" << exec->get_text(0) << CRLF;
502  }
503  }
504  page <<
505 " </TABLE>" CRLF;
506 }

647 //-----------------------------------------------------------------------------

669 Doxy::add_fun_to_call_graph(const Symbol * caller_sym,
670  const UserFunction * ufun)
671 {
672  Log(LOG_command_DOXY)
673  out << "add (caller) Symbol " << caller_sym->get_name() << endl;
674 
675 const Token_string & body = ufun->get_body();
676  loop(b, body.size())
677  {
678  const Token & tok = body[b];
679  if (tok.get_Class() != TC_SYMBOL) continue;
680  const Symbol * callee_ptr = tok.get_sym_ptr();

683  loop(si, callee_sym.value_stack_size())
684  {
685  if (callee_sym[si].name_class == NC_FUNCTION ||
686  callee_sym[si].name_class == NC_OPERATOR)

689  Assert(fp);

691  if (!callee) continue;
692 
693  // ignore multiple calls of the same callee from the
694  // same caller...
695  //
696  bool have_call = false;
697  loop(cg, call_graph.size())
698  {
699  if (call_graph[cg].caller == ufun &&
700  call_graph[cg].callee == callee)
701  {
702  have_call = true;
703  break; // loop(cg, ...
704  }
705  }
706 
707  if (!have_call)
708  {
709  UCS_string caller_name = caller_sym->get_name();
710  UCS_string callee_name = callee_sym.get_name();
711  const fcall_edge edge(ufun, &caller_name,
712  callee, &callee_name);
713 
714  call_graph.push_back(edge);
715  Log(LOG_command_DOXY)
716  out << " " << caller_sym->get_name()
717  << " calls " << callee->get_name() << endl;
718  }
719  }
720  }
721  }

723 //-----------------------------------------------------------------------------
724 void
725 Doxy::set_call_graph_root(const UserFunction * ufun)
726 {
727 const int MAX = 2 * call_graph.size(); // MAX means unreachable
728 UCS_string root_name("all");
729  if (ufun) root_name = ufun->get_name();
730 
731 
732  // 1. set all edges starting at ufun to 0 and all others unreachable
733  //
734  loop(cg, call_graph.size())
735  {
736  fcall_edge & edge = call_graph[cg];
737  if (edge.caller == ufun) edge.value = 0; // root
738  else if (ufun == 0) edge.value = 1; // no root, all edges
739  else edge.value = MAX; // unreachable
740  }
741 
742 bool progress = true;
743  for (int depth = 0; progress; ++depth)
744  {
745  progress = false;
746  loop(f, call_graph.size())
747  {
748  const fcall_edge & from = call_graph[f];
749  if (from.value != depth) continue;
750 
751  // 'from' starts at level depth.
752  // Mark all edges 'to'at level depth + 1
753  //
754  loop(t, call_graph.size())
755  {
756  fcall_edge & to = call_graph[t];
757  if (from.callee != to.caller) continue;
758  if (to.value < (depth + 1)) continue;
759  to.value = depth + 1;
760  progress = true;
761  }
762  }
763  }
764 
765  Log(LOG_command_DOXY)
766  loop(e, call_graph.size())
767  {
768  const fcall_edge & edge = call_graph[e];
769  if (edge.value == MAX) continue; // not reachable
770 
771  out << "edge " << edge.caller->get_name() << " â "
772  << edge.callee->get_name()
773  << " has distance " << edge.value
774  << " from " << root_name << endl;
775  }
776 
777  // create a list of nodes that are reachable
778  //
779  nodes.clear();
780  aliases.clear();
781  loop(e, call_graph.size())
782  {
783  const fcall_edge & edge = call_graph[e];
784  if (edge.value == MAX) continue; // not reachable
785  bool caller_is_new = true;
786  bool callee_is_new = true;
787  loop(n, nodes.size())
788  {
789  if (nodes[n] == edge.caller) caller_is_new = false;
790  if (nodes[n] == edge.callee) callee_is_new = false;
791  }
792 
793  if (caller_is_new)
794  {
795  nodes.push_back(edge.caller);
796  aliases.push_back(*edge.caller_name);
797  }
798  if (callee_is_new)
799  {
800  nodes.push_back(edge.callee);
801  aliases.push_back(*edge.callee_name);
802  }
803  }
804 
805  Log(LOG_command_DOXY)
806  {
807  loop(n, nodes.size())
808  out << "Node: " << nodes[n]->get_name()
809  << " alias: " << aliases[n] <<endl;
810  }
811 }

841

846  {

848  }
849  else // normal graph, > 0 edges

852  //

865  // create edges
866  //

879  }

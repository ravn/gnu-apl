178 void
179 Output::set_color_mode(Output::ColorMode mode)
180 {
181  if (!colors_enabled) return; // colors disabled
182  if (color_mode == mode) return; // no change in color mode
183 
184  // mode changed
185  //
186  color_mode = mode;
187  colors_changed = true; // to reset them in reset_colors()
188 
189  switch(color_mode)
190  {
191  case COLM_INPUT: cerr << color_CIN << clear_EOL; break;
192 
193  case COLM_OUTPUT: cout << color_COUT << clear_EOL; break;
194 
195  case COLM_ERROR: cerr << color_CERR << clear_EOL; break;
196 
197  case COLM_UERROR: cout << color_UERR << clear_EOL; break;
198 
199  default: break;
200  }
201 }

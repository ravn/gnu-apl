189 void
190 Output::init(bool logit)
191 {
192  if (!isatty(fileno(stdout)))
193  {
194  cout.flush();
195  cout.setf(ios::unitbuf);
196  }
197 
198 #if CURSES_USABLE
199 
200  if (logit)
201  {
202  if (use_curses)
203  CERR << "initializing output ESC sequences from libcurses" << endl;
204  else
205  CERR << "using ANSI terminal output ESC sequences (or those "
206  "configured in your preferences file(s))" << endl;
207 
208 
209  if (keys_curses)
210  CERR << "initializing keyboard ESC sequences from libcurses" <<endl;
211  else
212  CERR << "using ANSI terminal input ESC sequences(or those "
213  "configured in your preferences file(s))" << endl;
214  }
215 
216 int errors = 0;
217 
218  if (use_curses)
219  {
220  const int ret = setupterm(0, STDOUT_FILENO, 0);
221  if (ret != 0) ++errors;
222 
223  // read some ESC sequences
224  //
225 #define READ_Seq(dest, app, cap, p1) \
226  errors += read_ESC_sequence(dest, MAX_ESC_LEN, app, #cap, cap, p1);
227 
228  READ_Seq(clear_EOL, 0, clr_eol, 0);
229  READ_Seq(clear_EOS, 0, clr_eos, 0);
230  READ_Seq(exit_attr_mode, 0, exit_attribute_mode, 0);
231 
232  READ_Seq(color_CIN, 0, set_foreground, color_CIN_foreground);
233  READ_Seq(color_CIN, 1, set_background, color_CIN_background);
234 
235  READ_Seq(color_COUT, 0, set_foreground, color_COUT_foreground);
236  READ_Seq(color_COUT, 1, set_background, color_COUT_background);
237 
238  READ_Seq(color_CERR, 0, set_foreground, color_CERR_foreground);
239  READ_Seq(color_CERR, 1, set_background, color_CERR_background);
240 
241  READ_Seq(color_UERR, 0, set_foreground, color_UERR_foreground);
242  READ_Seq(color_UERR, 1, set_background, color_UERR_background);
243  }
244 
245  // cursor keys. This does not work currently because the keys reported
246  // by: key_up, key_down, etc are different from the hardwired VT100 keys
247  //
248  // The other group: cursor_up, cursor_down, etc is closer but cursor_down
249  // is linefeed
250  //
251  if (keys_curses)
252  {
253  READ_Seq(ESC_CursorUp, 0, key_up, 0);
254  READ_Seq(ESC_CursorDown, 0, key_down, 0);
255  READ_Seq(ESC_CursorLeft, 0, key_left, 0);
256  READ_Seq(ESC_CursorRight, 0, key_right, 0);
257  READ_Seq(ESC_CursorEnd, 0, key_end, 0);
258  READ_Seq(ESC_CursorHome, 0, key_home, 0);
259  READ_Seq(ESC_InsertMode, 0, key_ic, 0);
260  READ_Seq(ESC_Delete, 0, key_dc, 0);
261 
262  ESCmap::refresh_lengths();
263  }
264 
265  if (errors)
266  {
267  CERR <<
268 "\n*** use of libcurses was requested, but something went wrong during its\n"
269 "initialization. Expect garbled output and non-functional keys." << endl;
270  use_curses = false;
271  }
272 #endif

275 int
276 Output::read_ESC_sequence(char * dest, int destlen, int append,
277  const char * capname, char * str, int p1)
278 {
279 #if CURSES_USABLE
280  if (str == 0)
281  {
282  const char * term = getenv("TERM");
283  CERR << "capability '" << capname
284  << "' is not contained in the description";
285  if (term) CERR << " of terminal " << term;
286  CERR << endl;
287  return 1;
288  }
289 
290  if (str == reinterpret_cast<const char *>(-1))
291  {
292  const char * term = getenv("TERM");
293  CERR << "capability '" << capname
294  << "' is not a string capability";
295  CERR << endl;
296  if (term) CERR << " of terminal " << term;
297  return 1;
298  }
299 
300 // CERR << "BEFORE: ";
301 // for (int i = 0; i < strlen(dest); ++i) CERR << " " << HEX2(dest[i]);
302 // CERR << endl;
303 
304  if (!append) *dest = 0;
305 
306 const int offset = strlen(dest);
307 const char * seq = tparm(str, p1, 0, 0, 0, 0, 0, 0, 0, 0);
308 const int seq_len = strlen(seq);
309 
310  if (seq_len + offset >= (destlen - 1))
311  {
312  CERR << "ESC sequence too long" << endl;
313  return 1;
314  }
315 
316  strncpy(dest + offset, seq, destlen - offset - 1);
317 
318 // CERR << "AFTER: ";
319 // for (int i = 0; i < strlen(dest); ++i) CERR << " " << HEX2(dest[i]);
320 // CERR << endl;
321 
322  return 0;
323 
324 #else
325  return 1; // should not happen
326 #endif
327 }

352 void
353 Output::set_color_mode(Output::ColorMode mode)
354 {
355  if (!colors_enabled) return; // colors disabled
356  if (color_mode == mode) return; // no change in color mode
357 
358  // mode changed
359  //
360  color_mode = mode;
361  colors_changed = true; // to reset them in reset_colors()
362 
363  switch(color_mode)
364  {
365  case COLM_INPUT: cerr << color_CIN << clear_EOL; break;
366 
367  case COLM_OUTPUT:
368  if (use_curses) cout << exit_attr_mode << clear_EOL;
369  else cout << color_COUT << clear_EOL;
370  break;
371 
372  case COLM_ERROR: cerr << color_CERR << clear_EOL; break;
373 
374  case COLM_UERROR: cout << color_UERR << clear_EOL; break;
375 
376  default: break;
377  }
378 }

395 CIN_ostream::set_cursor(int y, int x)
396 {
397  if (uprefs.raw_cin) return;
398 
399 #if CURSES_USABLE

401  {
402  if (y < 0)
403  {
404  // y < 0 means from bottom upwards
405  //
406  *this << CSI << "30;" << (1 + x) << "H" << CSI << "99B";
407  if (y < -1) *this <<tparm(parm_down_cursor,
408  -(y + 1), 0, 0, 0, 0, 0, 0, 0, 0);
409  *this << std::flush;
410  }
411  else
412  {
413  // y â¥ 0 is from top downwards. This is currently not used.
414  //
415  *this << tparm(cursor_address, y, x, 0, 0, 0, 0, 0, 0, 0)
416  << std::flush;
417  }
418  }
419  else
420 #endif
421  {
422  if (y < 0)
423  {
424  // y < 0 means from bottom upwards
425  //
426  *this << CSI << "30;" << (1 + x) << 'H'
427  << CSI << "99B" << std::flush;
428  if (y < -1) *this << CSI << (-(y + 1)) << "A";
429  }
430  else
431  {
432  // y â¥ 0 is from top downwards. This is currently not used.
433  *this << CSI << (1 + y) << ";" << (1 + x) << 'H' << std::flush;
434  }

35 //-----------------------------------------------------------------------------
36 Token
37 Bif_F12_FORMAT::eval_B(Value_P B) const
38 {
39  // ISO and lrm: If B is a character array, then Z is B
40  //
41  if (!B->NOTCHAR()) return Token(TOK_APL_VALUE1, B);
42 
43  if (B->is_empty())
44  {
45  Value_P Z(B->get_shape(), LOC);
46  Z->set_proto_Spc();
47  Z->check_value(LOC);
48  return Token(TOK_APL_VALUE1, Z);
49  }
50 
51  if (!B->is_simple())
52  {
53  PrintBuffer pb(*B, Workspace::get_PrintContext(PR_APL), 0);
54  Assert(pb.is_rectangular());
55  const ShapeItem cols = pb.get_width(0);
56  const ShapeItem rows = pb.get_height();
57  Shape shape_Z(rows, cols);
58  Value_P Z(shape_Z, LOC);
59  loop(y, rows)
60  {
61  UCS_string row = pb.get_line(y);
62  row.map_pad();
63  loop(x, cols) Z->next_ravel_Char(row[x]);
64  }
65 
66  Z->check_value(LOC);
67 
68  // turn 1-line matrices into vectors
69  //
70  if (Z->get_rank() == 2 && Z->get_shape().get_shape_item(0) == 1)
71  {
72  Shape sh(Z->get_shape().get_shape_item(1));
73  Z->set_shape(sh);
74  }
75 
76  return Token(TOK_APL_VALUE1, Z);
77  }
78 
79 Value_P Z;
80  if (B->get_rank() > 2)
81  {
82  // reduce N>2 dimensions to 2 dimensions
83  //
84  const Shape shape_B1(B->get_rows(), B->get_cols());
85  Value_P B1 = B->clone(LOC);
86  B1->set_shape(shape_B1);
87 
88  Z = monadic_format(B1);
89 
90  Shape shape_Z(B->get_shape());
91  shape_Z.set_last_shape_item(Z->get_last_shape_item());
92  Assert(shape_Z.get_volume() == Z->element_count());
93  Z->set_shape(shape_Z);
94  }
95  else // B->get_rank() is 0, 1, or 2
96  {
97  Z = monadic_format(B);
98  }
99 
100  // adjust rank as needed:
101  // ρρZ ←→ ,1⌈⍴ρB if B is simple
102  // ρρZ ←→ ,1 or ,2 if B is nested
103  //
104 const APL_types::Depth depth = B->compute_depth();
105 Shape sZ;
106  if (depth > 1) // B is nested ⊢ ⍴⍴R is 1 or 2
107  {
108  // the examples in lrm contradict the text in lrm.
109  //
110  // Page 136 shows: R←2 3ρ'ONE' 1 1 'TWO' 2 22 which contains only
111  // scalar and vector items (R itself being a matrix) and therefore
112  // ⍕R shoule be a vector according to page 137:
113  //
114  // "Nested Arrays: When R is a nested array, Z is a vector if all"
115  // items of R at any depth are scalars or vectors."
116  //
117  // lrm also contradicts the ISO standard regarding the rank of Z.
118  //
119  // We try our best...
120  //
121  if (B->is_one_dimensional())
122  {
123  sZ = Shape(Z->element_count());
124  }
125  else
126  {
127  sZ = Z->get_shape(); // leave Z as is (2-dimensional)
128  }
129  }
130  else // B is simple: result rank: ,1⌈⍴ρB
131  {
132  if (B->get_rank() < 2)
133  {
134  sZ = Shape(Z->get_cols());
135  }
136  else
137  {
138  sZ = B->get_shape();
139  sZ.set_last_shape_item(Z->get_last_shape_item());
140  }
141  }
142 
143  if (sZ != Z->get_shape())
144  {
145  Assert(sZ.get_volume() == Z->element_count());
146  Z->set_shape(sZ);
147  }
148 
149  Z->check_value(LOC);
150  return Token(TOK_APL_VALUE1, Z);
151 }

153 Token
154 Bif_F12_FORMAT::eval_AB(Value_P A, Value_P B) const
155 {
156 Value_P Z;
157 
158  // any A should be a scalar or a vcector
159  //
160  if (A->get_rank() > 1) RANK_ERROR;
161 
162  if (A->is_char_array()) Z = format_by_example(A, B);
163  else if (A->is_int_array()) Z = format_by_specification(A, B);
164  else
165  {
166  MORE_ERROR() = "Bad left argument of ⍕";
167  DOMAIN_ERROR;
168  }
169 
170  Z->set_proto_Spc();
171  Z->check_value(LOC);
172  return Token(TOK_APL_VALUE1, Z);
173 }
174 //-----------------------------------------------------------------------------
175 Value_P
176 Bif_F12_FORMAT::monadic_format(Value_P B)
177 {
178  Assert(B->get_rank() <= 2);
179 
180 const PrintStyle style(PrintStyle(PR_APL | PST_NO_FRACT_0));
181 const PrintContext pctx = Workspace::get_PrintContext(style);
182 
183 const PrintBuffer pb(*B, pctx, 0);
184 
185 const ShapeItem width = pb.get_width(0);
186 const ShapeItem height = pb.get_height();
187 
188  // monadic_format() returns the Value with rank 1 or 2 which may be changed
189  // in Bif_F12_FORMAT::eval_B() later on to match the rather arbitrary rules
190  // in lrm.
191  //
192 Value_P Z;
193  if (height == 1) Z = Value_P(width, LOC);
194  else Z = Value_P(Shape(height, width), LOC);
195 
196  loop(h, height)
197  loop(w, width)
198  {
199  const Unicode uni = pb.get_char(w, h);
200  if (is_iPAD_char(uni)) Z->next_ravel_Char(UNI_SPACE);
201  else Z->next_ravel_Char(uni);
202  }
203 
204  Z->check_value(LOC);
205  return Z;
206 }

208 Value_P
209 Bif_F12_FORMAT::format_by_example(Value_P A, Value_P B)
210 {
211  // convert the ravel of char vector A into UCS_string 'format'.
212  //
213 UCS_string all_formats = A->get_UCS_ravel();
214  if (all_formats.size() == 0) LENGTH_ERROR;
215 
216 const ShapeItem cols = B->get_cols();
217 const ShapeItem rows = B->get_rows();
218 
219  // split string all_formats into individual format fields, one per column.
220  // If there is only one format field, then repeat it cols times.
221  //
222 vector<UCS_string> col_formats;
223  split_example_into_columns(all_formats, col_formats);
224  if (col_formats. size() == 1)
225  {
226  const UCS_string f = all_formats;
227  const UCS_string col0 = col_formats[0]; // keep it out of loop!
228  loop(c, cols - 1)
229  {
230  all_formats.append(f);
231  col_formats.push_back(col0);
232  }
233  }
234 
235  if (cols != ShapeItem(col_formats.size())) LENGTH_ERROR;
236 
237  // convert each column format string into a Format_LIFER
238  //
239 vector<Format_LIFER> col_items;
240  loop(c, col_formats.size())
241  col_items.push_back(Format_LIFER(col_formats[c]));
242 
243  Log(LOG_Bif_F12_FORMAT)
244  {
245  Q1(all_formats)
246  Q1(all_formats.size())
247  Q1(col_formats.size())
248 
249  loop(c, col_items.size())
250  {
251  CERR << "At " LOC " in format_by_example()" << endl
252  << " col_items[c].left_deco: '"
253  << col_items[c].left_deco << "'" << endl
254  << " col_items[c].int_part: '"
255  << col_items[c].int_part << "'" << endl
256  << " col_items[c].fract_part: '"
257  << col_items[c].fract_part << "'" << endl
258  << " col_items[c].expo_deco: '"
259  << col_items[c].expo_deco << "'" << endl
260  << " col_items[c].expo_negative: "
261  << col_items[c].expo_negative << endl
262  << " col_items[c].right_deco: '"
263  << col_items[c].right_deco << "'" << endl;
264  }
265  }
266 
267 Shape shape_Z(B->get_shape());
268  if (B->is_scalar()) shape_Z.add_shape_item(1);
269  shape_Z.set_last_shape_item(all_formats.size());
270 
271 Value_P Z(shape_Z, LOC);
272 
273  try
274  {
275  loop(r, rows)
276  {
277  UCS_string row;
278  loop(c, cols)
279  {
280  const Cell & cB = B->get_cravel(c + r*cols);
281  if (!cB.is_real_cell()) DOMAIN_ERROR;
282 
283  const APL_Float value = cB.get_real_value();
284 
285  const UCS_string item = col_items[c].format_example(value);
286  Log(LOG_Bif_F12_FORMAT) Q1(item)
287  row.append(item);
288  }
289 
290  Log(LOG_Bif_F12_FORMAT) { Q1(row) Q1(all_formats) }
291 
292  Assert(row.size() == all_formats.size());
293  loop(c, row.size())
294  Z->set_ravel_Char(r*all_formats.size() + c, row[c]);
295  }
296  }
297  catch (Error err)
298  {
299  throw err; // rethrow
300  }
301  catch (...)
302  {
303  }
304 
305  return Z;
306 }
307 //-----------------------------------------------------------------------------
308 void
309 Bif_F12_FORMAT::split_example_into_columns(const UCS_string & all_formats,
310  vector<UCS_string> & col_formats)
311 {
312  // split string 'all_formats' into fields, where:
313  //
314  // 1. a field contains at least one digit and is delimited by either
315  // 1a. a space (which belongs to the next field), or
316  // 1b. digit '6' (which belongs to the field).
317  //
318 bool digit_seen = false;
319 UCS_string current_format;
320  loop(f, all_formats.size())
321  {
322  const Unicode cc = all_formats[f];
323  current_format.append(cc);
324 
325  if (Avec::is_digit(cc)) digit_seen = true;
326 
327  if ((cc == UNI_SPACE) && digit_seen) // end of field, case 1a.
328  {
329  col_formats.push_back(current_format);
330  current_format.clear(); // start a new field;
331  digit_seen = false;
332  }
333  else if (cc == UNI_6) // end of field, case 1b.
334  {
335  ++f; // next char is right decorator (and end of field)
336  if (f < all_formats.size())
337  current_format.append(all_formats[f]);
338 
339  col_formats.push_back(current_format);
340  current_format.clear(); // start a new field;
341  digit_seen = false;
342  continue; // next char
343  }
344  }
345 
346  if (col_formats.size() && !digit_seen)
347  {
348  col_formats.back().append(current_format);
349  }
350  else
351  {
352  col_formats.push_back(current_format);
353  }
354 }
355 //-----------------------------------------------------------------------------
356 UCS_string
357 Bif_F12_FORMAT::Format_LIFER::format_example(APL_Float value)
358 {
359 UCS_string data_int;
360 UCS_string data_fract;
361 UCS_string data_expo;
362 const APL_Float val_1 = (value < 0.0) ? APL_Float(-value) : value;
363 
364 bool overflow = false;
365  fill_data_fields(val_1, data_int, data_fract, data_expo, overflow);
366  Log(LOG_Bif_F12_FORMAT)
367  {
368  Q1(overflow)
369  Q1(data_int)
370  Q1(data_fract)
371  Q1(data_expo)
372  }
373  if (overflow) return UCS_string(out_size(), Workspace::get_FC(3));
374 
375 const UCS_string left = format_left_side(data_int, value < 0.0, overflow);
376  if (overflow) return UCS_string(out_size(), Workspace::get_FC(3));
377  Assert(left.size() == (left_deco.out_len + int_part.out_len));
378 
379 const UCS_string right = format_right_side(data_fract, value < 0.0, data_expo);
380  Assert(right.size() == (fract_part.out_len + expo_deco.out_len +
381  exponent.out_len + right_deco.out_len));
382 
383  return left + right;
384 }
385 //-----------------------------------------------------------------------------
386 UCS_string
387 Bif_F12_FORMAT::Format_LIFER::format_left_side(const UCS_string data_int,
388  bool negative, bool & overflow)
389 {
390 const UCS_string data = int_part.insert_int_commas(data_int, overflow);
391 
392  if (overflow) return UCS_string();
393 
394  Assert(int_part.size() >= data.size());
395 const Unicode pad_char = int_part.pad_char(Workspace::get_FC(2));
396 const UCS_string pad(int_part.size() - data.size(), pad_char);
397 
398 UCS_string ucs;
399  if (int_part.no_float()) // floating disabled.
400  {
401  ucs.append(left_deco.format);
402  ucs.append(pad);
403  }
404  else if (int_part.do_float(negative)) // floating enabled, deco shown
405  {
406  ucs.append(pad);
407  ucs.append(left_deco.format);
408  }
409  else // floating enabled, deco hidden
410  {
411  ucs.append(pad);
412  ucs.append(UCS_string(left_deco.size(), UNI_SPACE));
413  }
414 
415  ucs.append(data);
416  return ucs;
417 }
418 //-----------------------------------------------------------------------------
419 UCS_string
420 Bif_F12_FORMAT::Format_LIFER::format_right_side(const UCS_string data_fract,
421  bool negative,
422  const UCS_string data_expo)
423 {
424 int pad_count = 0;
425 UCS_string ucs;
426 
427  if (fract_part.out_len)
428  {
429  const UCS_string data = fract_part.insert_fract_commas(data_fract);
430  Assert(fract_part.size() >= data.size());
431  pad_count += fract_part.size() - data.size();
432 
433  // print nothing instead of .
434  if (data.size() == 0) ++pad_count;
435  else ucs.append(Workspace::get_FC(0));
436  ucs.append(data);
437  }
438 
439  if (exponent.size())
440  {
441  UCS_string data(1, exponent_char); // the 'E'
442 
443  const Unicode pad_char = exponent.pad_char(Workspace::get_FC(2));
444  if (exponent.no_float()) // floating disabled.
445  {
446  data.append(expo_deco.format);
447  data.append(data_expo);
448  }
449  else if (exponent.do_float(expo_negative)) // floating, deco shown
450  {
451  data.append(expo_deco.format);
452  data.append(data_expo);
453  }
454  else // floating enabled and deco hidden
455  {
456  data.append(UCS_string(expo_deco.format.size(), pad_char));
457  data.append(data_expo);
458  }
459 
460  pad_count += expo_deco.out_len + exponent.out_len - data.size();
461  ucs.append(data);
462  }
463 
464  // now do the right side padding.
465  {
466  const Unicode pad_char = fract_part.pad_char(Workspace::get_FC(2));
467  const UCS_string pad(pad_count, pad_char);
468 
469  if (fract_part.no_float()) // floating disabled.
470  {
471  ucs.append(right_deco.format);
472  ucs.append(pad);
473  }
474  else if (fract_part.do_float(negative)) // floating, deco shown
475  {
476  ucs.append(pad);
477  ucs.append(right_deco.format);
478  }
479  else // floating, deco hidden
480  {
481  ucs.append(pad);
482  ucs.append(UCS_string(right_deco.size(), pad_char));
483  }
484  }
485 
486  return ucs;
487 }
488 //-----------------------------------------------------------------------------
489 Bif_F12_FORMAT::Format_LIFER::Format_LIFER(const UCS_string format)
490  : exponent_char(UNI_E),
491  expo_negative(false)
492 {
493  // split one column format into sub-format chunks...
494  //
495 int f = 0;
496 bool exponent_pending = false;
497 bool have_decimal_point = false;
498 
499 // left_decorator:
500  while (f < format.size())
501  {
502  if (is_control_char(format[f])) goto integral_part;
503  else left_deco.format.append(format[f++]);
504  }
505  goto fields_done;
506 
507 integral_part:
508  while (f < format.size())
509  {
510  const Unicode cc = format[f++];
511 
512  if (cc == UNI_FULLSTOP)
513  {
514  have_decimal_point = true;
515  goto fractional_part;
516  }
517  if (cc == UNI_E) goto exponent_part;
518  if (is_control_char(cc))
519  {
520  int_part.format.append(cc);
521  if (cc == UNI_6) goto right_decorator;
522  }
523  else
524  {
525  --f;
526  goto right_decorator;
527  }
528  }
529  goto fields_done;
530 
531 fractional_part:
532  while (f < format.size())
533  {
534  const Unicode cc = format[f++];
535  if (cc == UNI_7) exponent_pending = true;
536 
537  if (is_control_char(cc))
538  {
539  fract_part.format.append(cc);
540  if (cc == UNI_6) goto right_decorator;
541  }
542  else
543  {
544  exponent_char = cc;
545  goto exponent_decorator;
546  }
547  }
548  goto right_decorator;
549 
550 exponent_decorator:
551  if (!exponent_pending) { --f; goto right_decorator; }
552 
553  while (f < format.size())
554  {
555  const Unicode cc = format[f++];
556 
557  if (!is_control_char(cc))
558  {
559  expo_deco.format.append(cc);
560  }
561  else
562  {
563  --f;
564  goto exponent_part;
565  }
566  }
567 
568 exponent_part:
569  while (f < format.size())
570  {
571  const Unicode cc = format[f++];
572 
573  if (is_control_char(cc))
574  {
575  exponent.format.append(cc);
576  if (cc == UNI_6) goto right_decorator;
577  }
578  else
579  {
580  --f;
581  goto right_decorator;
582  }
583  }
584 
585 right_decorator:
586 
587  if (have_decimal_point &&
588  exponent.format.size() == 0 &&
589  fract_part.format.size() == 0) // quirk!
590  {
591  // this is ambiguous as to whether the decimal point belongs to
592  // the number or to the right decorator. lrm says all non-digits
593  // are decorators but that seems to be wrong anyhow. (e.g. ',').
594  //
595  // we put a trailing decimat point into the right decorator.
596  //
597  right_deco.format.append(UNI_FULLSTOP);
598  }
599 
600  while (f < format.size())
601  right_deco.format.append(format[f++]);
602 
603 fields_done:
604  left_deco.out_len = left_deco .format.size();
605 
606  int_part .out_len = int_part .format.size();
607 
608  if (fract_part.format.size())
609  fract_part.out_len = fract_part.format.size() + 1;
610 
611  if (exponent.size())
612  {
613  expo_deco.out_len = expo_deco.size();
614  exponent.out_len = exponent.format.size() + 1;
615  }
616 
617  right_deco.out_len = right_deco.format.size();
618 
619  // compute floating flags.
620  //
621  {
622  const int count = int_part .map_field(0)
623  + fract_part.map_field(1);
624 
625  if (count == 1) // flags apply to both sides.
626  {
627  int_part. flt_mask |= fract_part.flt_mask & (BIT_1 | BIT_2 | BIT_3);
628  fract_part.flt_mask |= int_part .flt_mask & (BIT_1 | BIT_2 | BIT_3);
629  }
630 
631  exponent.map_field(2);
632  }
633 }
634 //-----------------------------------------------------------------------------
635 int
636 Format_sub::map_field(int type)
637 {
638 int flt_cnt = 0;
639 
640  loop(f, format.size())
641  {
642  switch(format[f])
643  {
644  case UNI_0: flt_mask |= BIT_0; break;
645  case UNI_1: ++flt_cnt; flt_mask |= BIT_1; break;
646  case UNI_2: ++flt_cnt; flt_mask |= BIT_2; break;
647  case UNI_3: ++flt_cnt; flt_mask |= BIT_3; break;
648  case UNI_4: ++flt_cnt; flt_mask |= BIT_4; break;
649  case UNI_5: flt_mask |= BIT_5; break;
650  case UNI_6: flt_mask |= BIT_6; break;
651  case UNI_7: flt_mask |= BIT_7; break;
652  case UNI_8: flt_mask |= BIT_8; break;
653  case UNI_9: flt_mask |= BIT_9; break;
654  default: break;
655  }
656  }
657 
658  if (flt_mask & (BIT_0 | BIT_9))
659  {
660  min_len = format.size();
661  loop(f, format.size())
662  {
663  const size_t pos = (type == 1) ? format.size() - f - 1 : f;
664  const Unicode uni = format[pos];
665  if (uni == UNI_0) break;
666  if (uni == UNI_9) break;
667  --min_len;
668  }
669  }
670 
671  return flt_cnt;
672 }
673 //-----------------------------------------------------------------------------
674 ostream &
675 Format_sub::print(ostream & out) const
676 {
677  out << "format: '" << format << "',"
678  " min: " << min_len << ", out_len " << out_len << ", flags: ";
679  loop(d, 32) if (flt_mask & (1 << d)) out << char('0' + d);
680  return out;
681 }
682 //-----------------------------------------------------------------------------
683 void
684 Bif_F12_FORMAT::Format_LIFER::fill_data_fields(APL_Float value,
685  UCS_string & data_int, UCS_string & data_fract,
686  UCS_string & data_expo, bool & overflow)
687 {
688 char format[40];
689 const int data_buf_len = int_part.out_len + 1 // 123.
690  + fract_part.out_len // 456
691  + 1 + exponent.out_len + 1; // E-22
692 
693  if (data_buf_len > 100) DOMAIN_ERROR;
694 
695 char data_buf[101];
696 char * fract_end = 0;
697 
698  if (exponent.size()) // E in format string
699  {
700  // create a format like %.5E (if fract_part has 5 digits)
701  //
702  const int flen = snprintf(format, sizeof(format), "%%.%luE",
703  unsigned_long(fract_part.size()));
704  Assert(flen < int(sizeof(format))); // format was big enough.
705 
706  const int dlen = snprintf(&data_buf[0], data_buf_len, format, value);
707  Assert(dlen < data_buf_len);
708 
709  char * ep = strchr(&data_buf[0], 'E');
710  Assert(ep);
711  fract_end = ep++;
712  if (*ep == '+') ++ep; // skip +
713  else if (*ep == '-') { expo_negative = true; ++ep; } // skip -
714  if (*ep == '0' && ep[1]) ++ep; // skip leading 0 in exponent
715 
716  int elen = strlen(ep);
717 
718  // insert leading zeros until we have at least min_len digits.
719  //
720  for (; elen < exponent.min_len; ++elen) data_expo.append(UNI_0);
721 
722  data_expo.append(UCS_string(UTF8_string(ep)));
723  }
724  else // no exponent in format string.
725  {
726  const int flen = snprintf(format, sizeof(format), "%%.%luf",
727  unsigned_long(fract_part.size()));
728  Assert(flen < int(sizeof(format))); // assume no snprintf() overflow
729 
730  const int dlen = snprintf(&data_buf[0], data_buf_len, format, value);
731  data_buf[data_buf_len - 1] = 0;
732 
733  // the int part could be longer than allowed by the example string.
734  //
735  const char * dot = strchr(data_buf, '.');
736  const int ilen = dot ? (dot - data_buf)
737  : strlen(data_buf);
738  if (ilen > int_part.out_len)
739  {
740  if (Workspace::get_FC(3) == UNI_0)
741  {
742  MORE_ERROR() << "Overflow in integer part";
743  DOMAIN_ERROR;
744  }
745  overflow = true;
746  return ;
747  }
748 
749  Assert(dlen < data_buf_len);
750  fract_end = &data_buf[dlen];
751  }
752 
753 char * int_end = strchr(&data_buf[0], '.');
754  if (fract_part.size() == 0)
755  {
756  Assert(int_end == 0);
757  int_end = fract_end;
758  }
759  else
760  {
761  Assert(int_end);
762  char * fract_digits = int_end + 1;
763 
764  int flen = fract_end - fract_digits;
765 
766  // remove trailing zeros, but leave at least min_len chars.
767  while (fract_digits[flen - 1] == '0' && flen > fract_part.min_len)
768  fract_digits[--flen] = 0;
769 
770  loop(f, flen) data_fract.append(Unicode(fract_digits[f]));
771  }
772 
773 const int ilen = int_end - &data_buf[0];
774 
775  // insert leading zeros so that we will have at least min_len digits
776  // after appending the integer data.
777  //
778  loop(d, int_part.min_len - ilen) data_int.append(UNI_0);
779 
780  loop(i, ilen) data_int.append(Unicode(data_buf[i]));
781 
782  // convert 0.xxx to .xxx
783  //
784  if ( data_int.size() == 1
785  && data_int[0] == UNI_0
786  && int_part.min_len == 0
787  && fract_part.size()) data_int.clear();
788 
789  Log(LOG_Bif_F12_FORMAT)
790  {
791  CERR << "At " LOC " in Format_LIFER::fill_data_fields()" << endl
792  << " format: '" << format << "'" << endl
793  << " value: " << value << endl
794  << " data_int: '" << data_int << "'" << endl
795  << " data_fract: '" << data_fract << "'" << endl
796  << " data_expo: '" << data_expo << "'" << endl;
797  }

799 //-----------------------------------------------------------------------------
800 UCS_string
801 Format_sub::insert_int_commas(const UCS_string & data, bool & overflow) const
802 {
803 size_t fill_pos = -1;
804 Unicode fill_char = UNI_SPACE;
805 
806  // BIT_0: pad with 0
807  // BIT_8: fille with ⎕FC[3] (* by default)
808  // BIT_9: pad with 0
809  if (flt_mask & (BIT_0 | BIT_8 | BIT_9)) // format has a '0', '8', or '9'
810  {
811  loop(f, format.size())
812  {
813  const Unicode format_char = format[f];
814  if (format_char == UNI_0 || format_char == UNI_9)
815  {
816  fill_pos = f;
817  fill_char = UNI_0;
818  break;
819  }
820 
821  if (format_char == UNI_8)
822  {
823  fill_pos = f;
824  fill_char = Workspace::get_FC(2);
825  break;
826  }
827  }
828  }
829 
830  // we move backwards from the end of the format field, filling it
831  // with chars from data of from format.
832  //
833 UCS_string ucs;
834 size_t d = data.size();
835 
836  // d runs downwards from data.size() to 0
837  // f1 runs upwards from 0 to format.size()
838  // f runs downwards from format.size() to 0
839  //
840  loop(f1, format.size())
841  {
842  const size_t f = format.size() - f1 - 1;
843  const Unicode format_char = format[f];
844  if (format_char == UNI_COMMA)
845  {
846  // Workspace::get_FC(1) is ⎕FC[2] when ⎕IO is 1
847  //
848  if (d) ucs.append(Workspace::get_FC(1));
849  else if (f >= fill_pos) ucs.append(fill_char);
850  else break;
851  }
852  else if (Avec::is_digit(format_char))
853  {
854  if (d) ucs.append(data[--d]);
855  else if (f >= fill_pos) ucs.append(fill_char);
856  else break;
857  }
858  else
859  {
860  CERR << "Offending format char [" << f << "] : '"
861  << format_char << "' at " << LOC << endl;
862  break;
863  }
864  }
865 
866  if (d) // format too short
867  {
868  if (Workspace::get_FC(3) == UNI_0) DOMAIN_ERROR;
869  else overflow = true;
870  }
871 
872  if ((flt_mask & BIT_9) && ucs.all_zeroes()) ucs.clear();
873 
874  return ucs.reverse();
875 }

878 Format_sub::insert_fract_commas(const UCS_string & data) const
879 {

881 int d = 0;
882 
883  loop(f, format.size())
884  {

893  else break;
894  }
895  else
896  {
897  CERR << "Offending format char [" << f << "] : '"
898  << format_char << "' at " << LOC << endl;
899  }
900  }
901 
902  if ((flt_mask & BIT_9) && ucs.all_zeroes()) ucs.clear();
903 
904  return ucs;

994 //-----------------------------------------------------------------------------
995 PrintBuffer
996 Bif_F12_FORMAT::format_one_col_by_spec(int width, int precision,
997  const Cell * cB, ShapeItem cols,
998  ShapeItem rows)
999 {
1000 PrintBuffer ret;
1001 
1002 bool has_char = false;
1003 bool has_num = false;
1004 bool has_complex = false;
1005 
1006  // determine the data types (character/numbers/complex) in B
1007  loop(r, rows)
1008  {
1009  const Cell & cell = cB[r*cols];
1010  if (cell.is_numeric())
1011  {
1012  has_num = true;
1013  if (cell.is_complex_cell()) has_complex = true;
1014  }
1015  else has_char = true;
1016  }
1017 
1018  if (has_complex)
1019  {
1020  // split cB into vectors real and imag...
1021  Value_P real(rows, LOC);
1022  Value_P imag(rows, LOC);
1023  loop(r, rows)
1024  {
1025  const Cell & cell = cB[r*cols];
1026  if (cell.is_complex_cell())
1027  {
1028  real->next_ravel_Float(cell.get_real_value());
1029  imag->next_ravel_Float(cell.get_imag_value());
1030  }
1031  else
1032  {
1033  real->next_ravel_Cell(cell);
1034  imag->next_ravel_Char(UNI_SPACE);
1035  }
1036  }
1037 
1038  PrintBuffer pb_real = format_one_col_by_spec(width, precision,
1039  &real->get_cfirst(), 1,
1040  rows);
1041  PrintBuffer pb_imag = format_one_col_by_spec(width, precision,
1042  &imag->get_cfirst(), 1,
1043  rows);
1044 
1045  PrintBuffer pb_real_imag;
1046  loop(r,rows)
1047  {
1048  UCS_string real_imag = pb_real.get_line(r);
1049  UCS_string imag = pb_imag.get_line(r);
1050  imag.remove_leading_whitespaces();
1051  if (imag.size())
1052  {
1053  real_imag += UNI_J;
1054  real_imag.append(imag);
1055  }
1056  pb_real_imag.append_ucs(real_imag);
1057  }
1058  return pb_real_imag;
1059  }
1060 
1061  // real B. (if B was initially complex then we arrive here twice: once
1062  // for the real parts and once for the imag parts).
1063  //
1064  loop(r, rows)
1065  {
1066  const Cell & cell = cB[r*cols];
1067  if (cell.is_character_cell())
1068  {
1069  UCS_string data = UCS_string(cell.get_char_value());
1070 
1071  add_row(ret, r, has_char, has_num, UNI_E, data);
1072  continue;
1073  }
1074 
1075  if (cell.is_pointer_cell())
1076  {
1077  Value_P value = cell.get_pointer_value();
1078  UCS_string data = value->get_UCS_ravel();
1079 
1080  if (width && data.size() > width) // overflow
1081  {
1082  if (Workspace::get_FC(3) == UNI_0) DOMAIN_ERROR;
1083 
1084  data = UCS_string(width, Workspace::get_FC(3));
1085  }
1086 
1087  add_row(ret, r, has_char, has_num, UNI_E, data);
1088  continue;
1089  }
1090 
1091  if (!cell.is_real_cell()) DOMAIN_ERROR;
1092 
1093  APL_Float value = cB[r*cols].get_real_value();
1094 
1095  if (precision >= 0) // floating format
1096  {
1097  // fix values close to 0 as 0 so that, for example, we never
1098  // see ¯.0000 in the formatted output.
1099  //
1100  double minval = 0.01;
1101  loop(p, precision) minval /= 10;
1102  if (value < minval && value > -minval) value = 0.0;
1103 
1104  UCS_string data = format_spec_float(value, precision);
1105  if (width && data.size() > width) // overflow
1106  {
1107  if (Workspace::get_FC(3) == UNI_0) DOMAIN_ERROR;
1108  data = UCS_string(width, Workspace::get_FC(3));
1109  }
1110 
1111  add_row(ret, r, has_char, has_num, UNI_E, data);
1112  }
1113  else // exponential format
1114  {
1115  UCS_string data = UCS_string::from_double_expo_prec(value,
1116  -precision - 1);
1117  add_row(ret, r, has_char, has_num, UNI_E, data);
1118  }
1119  }
1120 
1121  if (width && ret.get_width(0) < width)
1122  ret.pad_l(UNI_SPACE, width - ret.get_width(0));
1123 
1124  return ret;
1125 }

1128 Bif_F12_FORMAT::add_row(PrintBuffer & ret, int row, bool has_char,
1129  bool has_num, Unicode align_char, UCS_string & data)
1130 {

1132  {

1154 //-----------------------------------------------------------------------------
1155 UCS_string
1156 Bif_F12_FORMAT::format_spec_float(APL_Float value, int precision)
1157 {
1158 UCS_string ret = UCS_string::from_double_fixed_prec(value, precision);
1159 
1160  // Note: the examples in the apl standard use a leading 0 (like 0.00)
1161  // while lrm shows .00 instead. We follow lrm and remove a leading 0.
1162  // Interestingly, the exponential format leaves a leading 0 in both
1163  // the standard and lrm
1164  //
1165  if (precision) // we have a '.'
1166  {
1167  if (ret[0] == UNI_0 &&
1168  ret[1] == UNI_FULLSTOP) // 0.xxx → .xxx
1169  {
1170  ret.erase(0);
1171  }
1172  else if (ret[0] == UNI_OVERBAR &&
1173  ret[1] == UNI_0 &&
1174  ret[2] == UNI_FULLSTOP) // ¯0.x → ¯.x
1175  {
1176  ret[1] = UNI_OVERBAR;
1177  ret.erase(0);
1178  }
1179  }
1180  else if (ret.size() == 2 &&
1181  ret[0] == UNI_OVERBAR &&
1182  ret[1] == UNI_0) // ¯0 → 0
1183  {
1184  ret.erase(0);
1185  }
1186 
1187  return ret;
1188 }

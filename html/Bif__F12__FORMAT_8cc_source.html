35 //-----------------------------------------------------------------------------
36 Token
37 Bif_F12_FORMAT::eval_B(Value_P B)
38 {
39  // ISO and lrm: If B is a character array, then Z is B
40  //
41  if (!B->NOTCHAR()) return Token(TOK_APL_VALUE1, B);
42 
43  if (B->is_empty())
44  {
45  Value_P Z(B->get_shape(), LOC);
46  Z->set_proto_Spc();
47  Z->check_value(LOC);
48  return Token(TOK_APL_VALUE1, Z);
49  }
50 
51  if (!B->is_simple())
52  {
53  PrintBuffer pb(*B, Workspace::get_PrintContext(PR_APL), 0);
54  Assert(pb.is_rectangular());
55  const ShapeItem cols = pb.get_width(0);
56  const ShapeItem rows = pb.get_height();
57  Shape shape_Z(rows, cols);
58  Value_P Z(shape_Z, LOC);
59  loop(y, rows)
60  {
61  UCS_string row = pb.get_line(y);
62  row.map_pad();
63  loop(x, cols) new (Z->next_ravel()) CharCell(row[x]);
64  }
65 
66  if (Z->get_rank() == 2 && Z->get_shape().get_shape_item(0) == 1)
67  {
68  Shape sh(Z->get_shape().get_shape_item(1));
69  Z->set_shape(sh);
70  }
71 
72  Z->check_value(LOC);
73  return Token(TOK_APL_VALUE1, Z);
74  }
75 
76 Value_P Z;
77  if (B->get_rank() > 2)
78  {
79  // reduce N>2 dimensions to 2 dimensions
80  //
81  const Shape shape_B1(B->get_rows(), B->get_cols());
82  Value_P B1 = B->clone(LOC);
83  B1->set_shape(shape_B1);
84 
85  Z = monadic_format(B1);
86 
87  Shape shape_Z(B->get_shape());
88  shape_Z.set_last_shape_item(Z->get_last_shape_item());
89  Assert(shape_Z.get_volume() == Z->element_count());
90  Z->set_shape(shape_Z);
91  }
92  else // B->get_rank() is 0, 1, or 2
93  {
94  Z = monadic_format(B);
95  }
96 
97  // adjust rank as needed:
98  // ρρZ ←→ ,1⌈⍴ρB if B is simple
99  // ρρZ ←→ ,1 or ,2 if B is nested
100  //
101 const APL_types::Depth depth = B->compute_depth();
102 Shape sZ;
103  if (depth > 1) // B is nested ⊢ ⍴⍴R is 1 or 2
104  {
105  // the examples in lrm contradict the text in lrm.
106  //
107  // Page 136 shows: R←2 3ρ'ONE' 1 1 'TWO' 2 22 which contains only
108  // scalar and vector items (R itself being a matrix) and therefore
109  // ⍕R shoule be a vector according to page 137:
110  //
111  // "Nested Arrays: When R is a nested array, Z is a vector if all"
112  // items of R at any depth are scalars or vectors."
113  //
114  // lrm also contradicts the ISO standard regarding the rank of Z.
115  //
116  // We try our best...
117  //
118  if (B->is_one_dimensional())
119  {
120  sZ = Shape(Z->element_count());
121  }
122  else
123  {
124  sZ = Z->get_shape(); // leave Z as is (2-dimensional)
125  }
126  }
127  else // B is simple: result rank: ,1⌈⍴ρB
128  {
129  if (B->get_rank() < 2)
130  {
131  sZ = Shape(Z->get_cols());
132  }
133  else
134  {
135  sZ = B->get_shape();
136  sZ.set_last_shape_item(Z->get_last_shape_item());
137  }
138  }
139 
140  if (sZ != Z->get_shape())
141  {
142  Assert(sZ.get_volume() == Z->element_count());
143  Z->set_shape(sZ);
144  }
145 
146  Z->check_value(LOC);
147  return Token(TOK_APL_VALUE1, Z);
148 }
149 //-----------------------------------------------------------------------------
150 Token
151 Bif_F12_FORMAT::eval_AB(Value_P A, Value_P B)
152 {
153 Value_P Z;
154 
155  // any A should be a scalar or a vcector
156  //
157  if (A->get_rank() > 1) RANK_ERROR;
158 
159  if (A->is_char_array()) Z = format_by_example(A, B);
160  else if (A->is_int_array()) Z = format_by_specification(A, B);
161  else
162  {
163  MORE_ERROR() = "Bad left argument of ⍕";
164  DOMAIN_ERROR;
165  }
166 
167  Z->set_default_Spc();
168  Z->check_value(LOC);
169  return Token(TOK_APL_VALUE1, Z);
170 }
171 //-----------------------------------------------------------------------------
172 Value_P
173 Bif_F12_FORMAT::monadic_format(Value_P B)
174 {
175  Assert(B->get_rank() <= 2);
176 
177 const PrintStyle style(PrintStyle(PR_APL | PST_NO_FRACT_0));
178 const PrintContext pctx = Workspace::get_PrintContext(style);
179 
180 const PrintBuffer pb(*B, pctx, 0);
181 
182 const ShapeItem width = pb.get_width(0);
183 const ShapeItem height = pb.get_height();
184 
185  // monadic_format() returns the Value with rank 1 or 2 which may be changed
186  // in Bif_F12_FORMAT::eval_B() later on to match the rather arbitrary rules
187  // in lrm.
188  //
189 Value_P Z;
190  if (height == 1) Z = Value_P(width, LOC);
191  else Z = Value_P(Shape(height, width), LOC);
192 
193  loop(h, height)
194  loop(w, width)
195  {
196  const Unicode uni = pb.get_char(w, h);
197  if (is_iPAD_char(uni)) new (Z->next_ravel()) CharCell(UNI_ASCII_SPACE);
198  else new (Z->next_ravel()) CharCell(uni);
199  }
200 
201  Z->check_value(LOC);
202  return Z;
203 }
204 //-----------------------------------------------------------------------------
205 Value_P
206 Bif_F12_FORMAT::format_by_example(Value_P A, Value_P B)
207 {
208  // convert the ravel of char vector A into UCS_string 'format'.
209  //
210 UCS_string all_formats = A->get_UCS_ravel();
211  if (all_formats.size() == 0) LENGTH_ERROR;
212 
213 const ShapeItem cols = B->get_cols();
214 const ShapeItem rows = B->get_rows();
215 
216  // split string all_formats into individual format fields, one per column.
217  // If there is only one format field, then repeat it cols times.
218  //
219 vector<UCS_string> col_formats;
220  split_example_into_columns(all_formats, col_formats);
221  if (col_formats. size() == 1)
222  {
223  const UCS_string f = all_formats;
224  const UCS_string col0 = col_formats[0]; // keep it out of loop!
225  loop(c, cols - 1)
226  {
227  all_formats.append(f);
228  col_formats.push_back(col0);
229  }
230  }
231 
232  if (cols != ShapeItem(col_formats.size())) LENGTH_ERROR;
233 
234  // convert each column format string into a Format_LIFER
235  //
236 vector<Format_LIFER> col_items;
237  loop(c, col_formats.size())
238  col_items.push_back(Format_LIFER(col_formats[c]));
239 
240  Log(LOG_Bif_F12_FORMAT)
241  {
242  Q1(all_formats)
243  Q1(all_formats.size())
244  Q1(col_formats.size())
245 
246  loop(c, col_items.size())
247  {
248  CERR << "At " LOC " in format_by_example()" << endl
249  << " col_items[c].left_deco: '"
250  << col_items[c].left_deco << "'" << endl
251  << " col_items[c].int_part: '"
252  << col_items[c].int_part << "'" << endl
253  << " col_items[c].fract_part: '"
254  << col_items[c].fract_part << "'" << endl
255  << " col_items[c].expo_deco: '"
256  << col_items[c].expo_deco << "'" << endl
257  << " col_items[c].expo_negative: "
258  << col_items[c].expo_negative << endl
259  << " col_items[c].right_deco: '"
260  << col_items[c].right_deco << "'" << endl;
261  }
262  }
263 
264 Shape shape_Z(B->get_shape());
265  if (B->is_scalar()) shape_Z.add_shape_item(1);
266  shape_Z.set_last_shape_item(all_formats.size());
267 
268 Value_P Z(shape_Z, LOC);
269 
270  try
271  {
272  loop(r, rows)
273  {
274  UCS_string row;
275  loop(c, cols)
276  {
277  const Cell & cB = B->get_ravel(c + r*cols);
278  if (!cB.is_real_cell()) DOMAIN_ERROR;
279 
280  const APL_Float value = cB.get_real_value();
281 
282  const UCS_string item = col_items[c].format_example(value);
283  Log(LOG_Bif_F12_FORMAT) Q1(item)
284  row.append(item);
285  }
286 
287  Log(LOG_Bif_F12_FORMAT) { Q1(row) Q1(all_formats) }
288 
289  Assert(row.size() == all_formats.size());
290  loop(c, row.size())
291  new (&Z->get_ravel(r*all_formats.size() + c)) CharCell(row[c]);
292  }
293  }
294  catch (Error err)
295  {
296  throw err; // rethrow
297  }
298  catch (...)
299  {
300  }
301 
302  return Z;
303 }
304 //-----------------------------------------------------------------------------
305 void
306 Bif_F12_FORMAT::split_example_into_columns(const UCS_string & all_formats,
307  vector<UCS_string> & col_formats)
308 {
309  // split string 'all_formats' into fields, where:
310  //
311  // 1. a field contains at least one digit and is delimited by either
312  // 1a. a space (which belongs to the next field), or
313  // 1b. digit '6' (which belongs to the field).
314  //
315 bool digit_seen = false;
316 UCS_string current_format;
317  loop(f, all_formats.size())
318  {
319  const Unicode cc = all_formats[f];
320  current_format.append(cc);
321 
322  if (Avec::is_digit(cc)) digit_seen = true;
323 
324  if ((cc == UNI_ASCII_SPACE) && digit_seen) // end of field, case 1a.
325  {
326  col_formats.push_back(current_format);
327  current_format.clear(); // start a new field;
328  digit_seen = false;
329  }
330  else if (cc == UNI_ASCII_6) // end of field, case 1b.
331  {
332  ++f; // next char is right decorator (and end of field)
333  if (f < all_formats.size())
334  current_format.append(all_formats[f]);
335 
336  col_formats.push_back(current_format);
337  current_format.clear(); // start a new field;
338  digit_seen = false;
339  continue; // next char
340  }
341  }
342 
343  if (col_formats.size() && !digit_seen)
344  {
345  col_formats.back().append(current_format);
346  }
347  else
348  {
349  col_formats.push_back(current_format);
350  }
351 }
352 //-----------------------------------------------------------------------------
353 UCS_string
354 Bif_F12_FORMAT::Format_LIFER::format_example(APL_Float value)
355 {
356 UCS_string data_int;
357 UCS_string data_fract;
358 UCS_string data_expo;
359 const APL_Float val_1 = (value < 0.0) ? APL_Float(-value) : value;
360 
361 bool overflow = false;
362  fill_data_fields(val_1, data_int, data_fract, data_expo, overflow);
363  Log(LOG_Bif_F12_FORMAT)
364  {
365  Q1(overflow)
366  Q1(data_int)
367  Q1(data_fract)
368  Q1(data_expo)
369  }
370  if (overflow) return UCS_string(out_size(), Workspace::get_FC(3));
371 
372 const UCS_string left = format_left_side(data_int, value < 0.0, overflow);
373  if (overflow) return UCS_string(out_size(), Workspace::get_FC(3));
374  Assert(left.size() == (left_deco.out_len + int_part.out_len));
375 
376 const UCS_string right = format_right_side(data_fract, value < 0.0, data_expo);
377  Assert(right.size() == (fract_part.out_len + expo_deco.out_len +
378  exponent.out_len + right_deco.out_len));
379 
380  return left + right;
381 }
382 //-----------------------------------------------------------------------------
383 UCS_string
384 Bif_F12_FORMAT::Format_LIFER::format_left_side(const UCS_string data_int,
385  bool negative, bool & overflow)
386 {
387 const UCS_string data = int_part.insert_int_commas(data_int, overflow);
388 
389  if (overflow) return UCS_string();
390 
391  Assert(int_part.size() >= data.size());
392 const Unicode pad_char = int_part.pad_char(Workspace::get_FC(2));
393 const UCS_string pad(int_part.size() - data.size(), pad_char);
394 
395 UCS_string ucs;
396  if (int_part.no_float()) // floating disabled.
397  {
398  ucs.append(left_deco.format);
399  ucs.append(pad);
400  }
401  else if (int_part.do_float(negative)) // floating enabled, deco shown
402  {
403  ucs.append(pad);
404  ucs.append(left_deco.format);
405  }
406  else // floating enabled, deco hidden
407  {
408  ucs.append(pad);
409  ucs.append(UCS_string(left_deco.size(), UNI_ASCII_SPACE));
410  }
411 
412  ucs.append(data);
413  return ucs;
414 }
415 //-----------------------------------------------------------------------------
416 UCS_string
417 Bif_F12_FORMAT::Format_LIFER::format_right_side(const UCS_string data_fract,
418  bool negative,
419  const UCS_string data_expo)
420 {
421 int pad_count = 0;
422 UCS_string ucs;
423 
424  if (fract_part.out_len)
425  {
426  const UCS_string data = fract_part.insert_fract_commas(data_fract);
427  Assert(fract_part.size() >= data.size());
428  pad_count += fract_part.size() - data.size();
429 
430  // print nothing instead of .
431  if (data.size() == 0) ++pad_count;
432  else ucs.append(Workspace::get_FC(0));
433  ucs.append(data);
434  }
435 
436  if (exponent.size())
437  {
438  UCS_string data(1, exponent_char); // the 'E'
439 
440  const Unicode pad_char = exponent.pad_char(Workspace::get_FC(2));
441  if (exponent.no_float()) // floating disabled.
442  {
443  data.append(expo_deco.format);
444  data.append(data_expo);
445  }
446  else if (exponent.do_float(expo_negative)) // floating, deco shown
447  {
448  data.append(expo_deco.format);
449  data.append(data_expo);
450  }
451  else // floating enabled and deco hidden
452  {
453  data.append(UCS_string(expo_deco.format.size(), pad_char));
454  data.append(data_expo);
455  }
456 
457  pad_count += expo_deco.out_len + exponent.out_len - data.size();
458  ucs.append(data);
459  }
460 
461  // now do the right side padding.
462  {
463  const Unicode pad_char = fract_part.pad_char(Workspace::get_FC(2));
464  const UCS_string pad(pad_count, pad_char);
465 
466  if (fract_part.no_float()) // floating disabled.
467  {
468  ucs.append(right_deco.format);
469  ucs.append(pad);
470  }
471  else if (fract_part.do_float(negative)) // floating, deco shown
472  {
473  ucs.append(pad);
474  ucs.append(right_deco.format);
475  }
476  else // floating, deco hidden
477  {
478  ucs.append(pad);
479  ucs.append(UCS_string(right_deco.size(), pad_char));
480  }
481  }
482 
483  return ucs;
484 }
485 //-----------------------------------------------------------------------------
486 Bif_F12_FORMAT::Format_LIFER::Format_LIFER(const UCS_string format)
487  : exponent_char(UNI_ASCII_E),
488  expo_negative(false)
489 {
490  // split one column format into sub-format chunks...
491  //
492 int f = 0;
493 bool exponent_pending = false;
494 bool have_decimal_point = false;
495 
496 // left_decorator:
497  while (f < format.size())
498  {
499  if (is_control_char(format[f])) goto integral_part;
500  else left_deco.format.append(format[f++]);
501  }
502  goto fields_done;
503 
504 integral_part:
505  while (f < format.size())
506  {
507  const Unicode cc = format[f++];
508 
509  if (cc == UNI_ASCII_FULLSTOP)
510  {
511  have_decimal_point = true;
512  goto fractional_part;
513  }
514  if (cc == UNI_ASCII_E) goto exponent_part;
515  if (is_control_char(cc))
516  {
517  int_part.format.append(cc);
518  if (cc == UNI_ASCII_6) goto right_decorator;
519  }
520  else
521  {
522  --f;
523  goto right_decorator;
524  }
525  }
526  goto fields_done;
527 
528 fractional_part:
529  while (f < format.size())
530  {
531  const Unicode cc = format[f++];
532  if (cc == UNI_ASCII_7) exponent_pending = true;
533 
534  if (is_control_char(cc))
535  {
536  fract_part.format.append(cc);
537  if (cc == UNI_ASCII_6) goto right_decorator;
538  }
539  else
540  {
541  exponent_char = cc;
542  goto exponent_decorator;
543  }
544  }
545  goto right_decorator;
546 
547 exponent_decorator:
548  if (!exponent_pending) { --f; goto right_decorator; }
549 
550  while (f < format.size())
551  {
552  const Unicode cc = format[f++];
553 
554  if (!is_control_char(cc))
555  {
556  expo_deco.format.append(cc);
557  }
558  else
559  {
560  --f;
561  goto exponent_part;
562  }
563  }
564 
565 exponent_part:
566  while (f < format.size())
567  {
568  const Unicode cc = format[f++];
569 
570  if (is_control_char(cc))
571  {
572  exponent.format.append(cc);
573  if (cc == UNI_ASCII_6) goto right_decorator;
574  }
575  else
576  {
577  --f;
578  goto right_decorator;
579  }
580  }
581 
582 right_decorator:
583 
584  if (have_decimal_point &&
585  exponent.format.size() == 0 &&
586  fract_part.format.size() == 0) // quirk!
587  {
588  // this is ambiguous as to whether the decimal point belongs to
589  // the number or to the right decorator. lrm says all non-digits
590  // are decorators but that seems to be wrong anyhow. (e.g. ',').
591  //
592  // we put a trailing decimat point into the right decorator.
593  //
594  right_deco.format.append(UNI_ASCII_FULLSTOP);
595  }
596 
597  while (f < format.size())
598  right_deco.format.append(format[f++]);
599 
600 fields_done:
601 int sum = 0;
602  sum += left_deco.out_len = left_deco .format.size();
603 
604  sum += int_part .out_len = int_part .format.size();
605 
606  if (fract_part.format.size())
607  sum += fract_part.out_len = fract_part.format.size() + 1;
608 
609  if (exponent.size())
610  {
611  sum += expo_deco.out_len = expo_deco.size();
612  sum += exponent.out_len = exponent.format.size() + 1;
613  }
614 
615  right_deco.out_len = right_deco.format.size();
616 
617  // compute floating flags.
618  //
619  {
620  const int count = int_part .map_field(0)
621  + fract_part.map_field(1);
622 
623  if (count == 1) // flags apply to both sides.
624  {
625  int_part. flt_mask |= fract_part.flt_mask & (BIT_1 | BIT_2 | BIT_3);
626  fract_part.flt_mask |= int_part .flt_mask & (BIT_1 | BIT_2 | BIT_3);
627  }
628 
629  exponent.map_field(2);
630  }
631 }
632 //-----------------------------------------------------------------------------
633 int
634 Format_sub::map_field(int type)
635 {
636 int flt_cnt = 0;
637 
638  loop(f, format.size())
639  {
640  switch(format[f])
641  {
642  case UNI_ASCII_0: flt_mask |= BIT_0; break;
643  case UNI_ASCII_1: ++flt_cnt; flt_mask |= BIT_1; break;
644  case UNI_ASCII_2: ++flt_cnt; flt_mask |= BIT_2; break;
645  case UNI_ASCII_3: ++flt_cnt; flt_mask |= BIT_3; break;
646  case UNI_ASCII_4: ++flt_cnt; flt_mask |= BIT_4; break;
647  case UNI_ASCII_5: flt_mask |= BIT_5; break;
648  case UNI_ASCII_6: flt_mask |= BIT_6; break;
649  case UNI_ASCII_7: flt_mask |= BIT_7; break;
650  case UNI_ASCII_8: flt_mask |= BIT_8; break;
651  case UNI_ASCII_9: flt_mask |= BIT_9; break;
652  default: break;
653  }
654  }
655 
656  if (flt_mask & (BIT_0 | BIT_9))
657  {
658  min_len = format.size();
659  loop(f, format.size())
660  {
661  const size_t pos = (type == 1) ? format.size() - f - 1 : f;
662  const Unicode uni = format[pos];
663  if (uni == UNI_ASCII_0) break;
664  if (uni == UNI_ASCII_9) break;
665  --min_len;
666  }
667  }
668 
669  return flt_cnt;
670 }
671 //-----------------------------------------------------------------------------
672 ostream &
673 Format_sub::print(ostream & out) const
674 {
675  out << "format: '" << format << "',"
676  " min: " << min_len << ", out_len " << out_len << ", flags: ";
677  loop(d, 32) if (flt_mask & (1 << d)) out << char('0' + d);
678  return out;
679 }
680 //-----------------------------------------------------------------------------
681 void
682 Bif_F12_FORMAT::Format_LIFER::fill_data_fields(APL_Float value,
683  UCS_string & data_int, UCS_string & data_fract,
684  UCS_string & data_expo, bool & overflow)
685 {
686 char format[40];
687 const int data_buf_len = int_part.out_len + 1 // 123.
688  + fract_part.out_len // 456

797 //-----------------------------------------------------------------------------
798 UCS_string
799 Format_sub::insert_int_commas(const UCS_string & data, bool & overflow) const
800 {
801 size_t fill_pos = -1;
802 Unicode fill_char = UNI_ASCII_SPACE;
803 
804  // BIT_0: pad with 0
805  // BIT_8: fille with ⎕FC[3] (* by default)
806  // BIT_9: pad with 0
807  if (flt_mask & (BIT_0 | BIT_8 | BIT_9)) // format has a '0', '8', or '9'
808  {
809  loop(f, format.size())
810  {
811  const Unicode format_char = format[f];
812  if (format_char == UNI_ASCII_0 || format_char == UNI_ASCII_9)
813  {
814  fill_pos = f;
815  fill_char = UNI_ASCII_0;
816  break;
817  }
818 
819  if (format_char == UNI_ASCII_8)
820  {
821  fill_pos = f;
822  fill_char = Workspace::get_FC(2);
823  break;
824  }
825  }
826  }
827 
828  // we move backwards from the end of the format field, filling it
829  // with chars from data of from format.
830  //
831 UCS_string ucs;
832 size_t d = data.size();
833 
834  // d runs downwards from data.size() to 0
835  // f1 runs upwards from 0 to format.size()
836  // f runs downwards from format.size() to 0
837  //
838  loop(f1, format.size())
839  {
840  const size_t f = format.size() - f1 - 1;
841  const Unicode format_char = format[f];
842  if (format_char == UNI_ASCII_COMMA)
843  {
844  // Workspace::get_FC(1) is ⎕FC[2] when ⎕IO is 1
845  //
846  if (d) ucs.append(Workspace::get_FC(1));
847  else if (f >= fill_pos) ucs.append(fill_char);
848  else break;
849  }
850  else if (Avec::is_digit(format_char))
851  {
852  if (d) ucs.append(data[--d]);
853  else if (f >= fill_pos) ucs.append(fill_char);
854  else break;
855  }
856  else
857  {
858  CERR << "Offending format char [" << f << "] : '"
859  << format_char << "' at " << LOC << endl;
860  break;
861  }
862  }
863 
864  if (d) // format too short
865  {
866  if (Workspace::get_FC(3) == UNI_ASCII_0) DOMAIN_ERROR;
867  else overflow = true;
868  }
869 
870  if ((flt_mask & BIT_9) && ucs.all_zeroes()) ucs.clear();
871 
872  return ucs.reverse();
873 }

875 UCS_string
876 Format_sub::insert_fract_commas(const UCS_string & data) const
877 {
878 UCS_string ucs;
879 int d = 0;
880 
881  loop(f, format.size())
882  {
883  const Unicode format_char = format[f];
884  if (format_char == UNI_ASCII_COMMA)
885  {
886  ucs.append(Workspace::get_FC(1));
887  }
888  else if (Avec::is_digit(format_char))
889  {
890  if (d < data.size()) ucs.append(data[d++]);
891  else break;
892  }
893  else
894  {
895  CERR << "Offending format char [" << f << "] : '"
896  << format_char << "' at " << LOC << endl;
897  }
898  }
899 
900  if ((flt_mask & BIT_9) && ucs.all_zeroes()) ucs.clear();
901

992 //-----------------------------------------------------------------------------
993 PrintBuffer
994 Bif_F12_FORMAT::format_one_col_by_spec(int width, int precision,
995  const Cell * cB, ShapeItem cols,
996  ShapeItem rows)
997 {
998 PrintBuffer ret;
999 
1000 bool has_char = false;
1001 bool has_num = false;
1002 bool has_complex = false;
1003 
1004  // determine the data types (character/numbers/complex) in B
1005  loop(r, rows)
1006  {
1007  const Cell & cell = cB[r*cols];
1008  if (cell.is_numeric())
1009  {
1010  has_num = true;
1011  if (cell.is_complex_cell()) has_complex = true;
1012  }
1013  else has_char = true;
1014  }
1015 
1016  if (has_complex)
1017  {
1018  // split cB into real and imag parts...
1019  Value_P real(rows, LOC);
1020  Value_P imag(rows, LOC);
1021  loop(r, rows)
1022  {
1023  const Cell & cell = cB[r*cols];
1024  if (cell.is_complex_cell())
1025  {
1026  new (real->next_ravel()) FloatCell(cell.get_real_value());
1027  new (imag->next_ravel()) FloatCell(cell.get_imag_value());
1028  }
1029  else
1030  {
1031  real->next_ravel()->init(cell, real.getref(), LOC);
1032  new (imag->next_ravel()) CharCell(UNI_ASCII_SPACE);
1033  }
1034  }
1035 
1036  PrintBuffer pb_real = format_one_col_by_spec(width, precision,
1037  &real->get_ravel(0), 1,
1038  rows);
1039  PrintBuffer pb_imag = format_one_col_by_spec(width, precision,
1040  &imag->get_ravel(0), 1,
1041  rows);
1042 
1043  PrintBuffer pb_real_imag;
1044  loop(r,rows)
1045  {
1046  UCS_string real_imag = pb_real.get_line(r);
1047  UCS_string imag = pb_imag.get_line(r);
1048  imag.remove_leading_whitespaces();
1049  if (imag.size())
1050  {
1051  real_imag += UNI_ASCII_J;
1052  real_imag.append(imag);
1053  }
1054  pb_real_imag.append_ucs(real_imag);
1055  }
1056  return pb_real_imag;
1057  }
1058 
1059  // real B. (if B was initially complex then we arrive here twice: once
1060  // for the real parts and once for the imag parts).
1061  //
1062  loop(r, rows)
1063  {
1064  const Cell & cell = cB[r*cols];
1065  if (cell.is_character_cell())
1066  {
1067  UCS_string data = UCS_string(cell.get_char_value());
1068 
1069  add_row(ret, r, has_char, has_num, UNI_ASCII_E, data);
1070  continue;
1071  }
1072 
1073  if (cell.is_pointer_cell())
1074  {
1075  Value_P value = cell.get_pointer_value();
1076  UCS_string data = value->get_UCS_ravel();
1077 
1078  if (width && data.size() > width) // overflow
1079  {
1080  if (Workspace::get_FC(3) == UNI_ASCII_0) DOMAIN_ERROR;
1081 
1082  data = UCS_string(width, Workspace::get_FC(3));
1083  }
1084 
1085  add_row(ret, r, has_char, has_num, UNI_ASCII_E, data);
1086  continue;
1087  }
1088 
1089  if (!cell.is_real_cell()) DOMAIN_ERROR;
1090 
1091  APL_Float value = cB[r*cols].get_real_value();
1092 
1093  if (precision >= 0) // floating format
1094  {
1095  // fix values close to 0 as 0 so that, for example, we never
1096  // see ¯.0000 in the formatted output.
1097  //
1098  double minval = 0.01;
1099  loop(p, precision) minval /= 10;
1100  if (value < minval && value > -minval) value = 0.0;
1101 
1102  UCS_string data = format_spec_float(value, precision);
1103  if (width && data.size() > width) // overflow
1104  {
1105  if (Workspace::get_FC(3) == UNI_ASCII_0) DOMAIN_ERROR;
1106  data = UCS_string(width, Workspace::get_FC(3));
1107  }
1108 
1109  add_row(ret, r, has_char, has_num, UNI_ASCII_E, data);
1110  }
1111  else // exponential format
1112  {
1113  UCS_string data = UCS_string::from_double_expo_prec(value,
1114  -precision - 1);
1115  add_row(ret, r, has_char, has_num, UNI_ASCII_E, data);
1116  }
1117  }
1118 
1119  if (width && ret.get_width(0) < width)
1120  ret.pad_l(UNI_ASCII_SPACE, width - ret.get_width(0));
1121 
1122  return ret;
1123 }

1153 UCS_string
1154 Bif_F12_FORMAT::format_spec_float(APL_Float value, int precision)
1155 {
1156 UCS_string ret = UCS_string::from_double_fixed_prec(value, precision);
1157

1159  // while lrm shows .00 instead. We follow lrm and remove a leading 0.
1160  // Interestingly, the exponential format leaves a leading 0 in both
1161  // the standard and lrm
1162  //
1163  if (precision) // we have a '.'

1167  {
1168  ret.erase(0);
1169  }
1170  else if (ret[0] == UNI_OVERBAR &&
1171  ret[1] == UNI_ASCII_0 &&
1172  ret[2] == UNI_ASCII_FULLSTOP) // ¯0.x → ¯.x
1173  {
1174  ret[1] = UNI_OVERBAR;
1175  ret.erase(0);
1176  }
1177  }
1178  else if (ret.size() == 2 &&
1179  ret[0] == UNI_OVERBAR &&
1180  ret[1] == UNI_ASCII_0) // ¯0 → 0
1181  {
1182  ret.erase(0);
1183  }
1184 
1185  return ret;
1186 }

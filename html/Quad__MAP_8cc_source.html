29  8 ⎕CR A←5 2⍴'eEwWaAzZ92' ◊ 8 ⎕CR B←'Halloween' ◊ A ⎕MAP B
30  */
31 Token
32 Quad_MAP::eval_AB(Value_P A, Value_P B) const
33 {
34 bool recursive = false;
35 
36  // a nested scalar A indicates recursive ⎕MAP
37  //
38  if (A->get_rank() == 0 && A->get_cfirst().is_pointer_cell())
39  {
40  recursive = true;
41  A = A->get_cfirst().get_pointer_value();
42  }
43 
44  if (A->get_rank() != 2) RANK_ERROR;
45  if (A->get_cols() != 2) LENGTH_ERROR;
46 const ShapeItem map_len = A->get_rows();
47  if (map_len == 0) LENGTH_ERROR;
48 
49 ShapeItem * indices = new ShapeItem[map_len];
50  if (indices == 0) WS_FULL;
51 
52  loop(m, map_len) indices[m] = m;
53 
54 const ravel_comp_len ctx = { &A->get_cfirst(), 1};
55  Heapsort<ShapeItem>::sort(indices, map_len, &ctx, &Quad_MAP::greater_map);
56 
57  // complain about duplicated keys
58  //
59 const double qct = Workspace::get_CT();
60  for (ShapeItem m = 1; m < map_len; ++m)
61  {
62  const Cell & cm1 = A->get_cravel(2*indices[m - 1]);
63  const Cell & cm = A->get_cravel(2*indices[m ]);
64  if (cm1.equal(cm, qct))
65  {
66  const int qio = Workspace::get_IO();
67  MORE_ERROR() << "Duplicate keys (e.g. A["
68  << (qio + indices[m - 1]) << "] and A["
69  << (qio + indices[m]) << "]) in 'A ⎕MAP B'";
70  delete[] indices;
71  DOMAIN_ERROR;
72  }
73  }
74 
75 Value_P Z = do_map(&A->get_cfirst(), map_len, indices, B.get(), recursive);
76  delete[] indices;
77  return Token(TOK_APL_VALUE1, Z);
78 }

80 bool
81 Quad_MAP::greater_map(const ShapeItem & a, const ShapeItem & b,
82  const void * ctx)
83 {
84 const ravel_comp_len * rcl = reinterpret_cast<const ravel_comp_len *>(ctx);
85 const Cell * cells = rcl->ravel;
86 
87  if (const Comp_result cr = cells[2*a].compare(cells[2*b])) // A[1] ≠ B[1]
88  return cr == COMP_GT;
89 
90  return a > b;
91 }

103 Value_P
104 Quad_MAP::do_map(const Cell * ravel_A, ShapeItem len_A,
105  const ShapeItem * sorted_indices_A, const Value * B,
106  bool recursive)
107 {
108 Value_P Z(B->get_shape(), LOC);
109 const ravel_comp_len ctx = { ravel_A, // start of the ravel
110  1 // number of characters to compare
111  };
112 
113 const ShapeItem len_B = B->element_count();
114  if (len_B == 0) // empty value
115  {
116  const Cell & cell_B = B->get_cfirst();
117  if (const ShapeItem * map =
118  Heapsort<ShapeItem>::search<const Cell &>
119  (cell_B,
120  sorted_indices_A,
121  len_A,
122  compare_MAP,
123  &ctx))
124  {
125  const Cell & cell_A = ravel_A[*map*2 + 1];
126  if (cell_A.is_pointer_cell())
127  {
128  Cell & cell_Z0 = Z->get_wproto();
129  cell_Z0.init(cell_A, Z.getref(), LOC);
130  cell_Z0.get_pointer_value()->to_proto();
131  }
132  else if (cell_A.is_character_cell())
133  Z->set_proto_Spc();
134  else if (cell_A.is_numeric())
135  Z->set_proto_Int();
136  else
137  DOMAIN_ERROR;
138  }
139  else // not mapped, simple
140  {
141  Z->get_wproto().init(cell_B, Z.getref(), LOC);
142  }
143  }
144 
145  loop(b, len_B)
146  {
147  const Cell & cell_B = B->get_cravel(b);
148  if (const ShapeItem * map = Heapsort<ShapeItem>::search<const Cell &>
149  (cell_B, sorted_indices_A, len_A, compare_MAP, &ctx))
150  {
151  Z->next_ravel_Cell(ravel_A[*map*2 + 1]);
152  }
153  else // cell_B shall not be mapped
154  {
155  if (recursive && cell_B.is_pointer_cell()) // nested: recursive
156  {
157  Value_P sub_B = cell_B.get_pointer_value();
158  Value_P sub_Z = do_map(ravel_A, len_A, sorted_indices_A,
159  sub_B.get(), true);
160  Z->next_ravel_Pointer(sub_Z.get());
161  }
162  else // not mapped, simple
163  {
164  Z->next_ravel_Cell(cell_B);
165  }
166  }
167  }
168 
169  Z->check_value(LOC);
170  return Z;
171 }

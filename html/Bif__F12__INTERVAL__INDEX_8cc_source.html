38 Token
39 Bif_F12_INTERVAL_INDEX::eval_B(Value_P B) const
40 {
41  // B must be boolean. Che that and count the number of 1s in B.
42  //
43 const APL_Integer qio = Workspace::get_IO();
44 const ShapeItem ec_B = B->element_count();
45 ShapeItem count = 0;
46  loop(b, ec_B)
47  {
48  const Cell & cell = B->get_cravel(b);
49  if (!cell.is_near_int())
50  {
51  MORE_ERROR() << "non-integer item in the argument of monadic ⍸";
52  DOMAIN_ERROR;
53  }
54 
55  const APL_Integer Bi = B->get_cravel(b).get_near_int();
56  if (Bi < 0)
57  {
58  MORE_ERROR() << "negative item in the argument of monadic ⍸";
59  DOMAIN_ERROR;
60  }
61 
62  count += Bi;
63  }
64 
65 Value_P Z(count, LOC);
66 
67 const uRank rank = B->get_rank();
68  loop(b, ec_B)
69  {
70  const Cell & cell = B->get_cravel(b);
71  const APL_Integer Bi = cell.get_near_int(); // number of repetitions
72  if (!Bi) continue; // nothing to do
73 
74  const Shape sh_b = B->get_shape().offset_to_index(b, qio);
75  Assert(sh_b.get_rank() == rank);
76  loop(rep, Bi)
77  {
78  if (rank <= 1) // simple shape
79  {
80  Z->next_ravel_Int(b + qio);
81  }
82  else // nested shape
83  {
84  Value_P ZZ(rank, LOC);
85  loop(r, rank)
86  {
87  const ShapeItem sh_r = sh_b.get_shape_item(r);
88  ZZ->next_ravel_Int(sh_r);
89  }
90  ZZ->set_proto_Int();
91  ZZ->check_value(LOC);
92  Z->next_ravel_Pointer(ZZ.get());
93  }
94  }
95  }
96 
97  Z->set_proto_Int();
98  Z->check_value(LOC);
99  return Token(TOK_APL_VALUE1, Z);
100 }

104 Token
105 Bif_F12_INTERVAL_INDEX::eval_AB(Value_P A, Value_P B) const
106 {
107  // A must be a non-empty sorted vector
108  //
109  if (A->get_rank() > 1) RANK_ERROR;
110 const ShapeItem ec_A = A->element_count();
111  if (ec_A < 1)
112  {
113  MORE_ERROR() << "the left argument of A ⍸ B is empty";
114  LENGTH_ERROR;
115  }
116 
117  for(ShapeItem a = 1; a < ec_A; ++a)
118  {
119  const Cell & c1 = A->get_cravel(a-1);
120  const Cell & c2 = A->get_cravel(a);
121  const Comp_result c1_c2 = c1.compare(c2);
122  if (c1_c2 != COMP_LT)
123  {
124  MORE_ERROR() << "the left argument of A ⍸ B "
125  "is not sorted ascendingly";
126  DOMAIN_ERROR;
127  }
128  }
129 
130  // from here on nothing can fail.
131  //
132 Value_P Z(B->get_shape(), LOC);
133 const ShapeItem ec_B = B->element_count();
134 const Cell * cells_A = &A->get_cfirst();
135 const APL_Integer qio = Workspace::get_IO();
136 
137  loop(b, ec_B)
138  {
139  const ShapeItem z = find_range(B->get_cravel(b), cells_A, ec_A);
140  Z->next_ravel_Int(z + qio);
141  }
142 
143  Z->set_proto_Int();
144  Z->check_value(LOC);
145  return Token(TOK_APL_VALUE1, Z);
146 }

148 ShapeItem
149 Bif_F12_INTERVAL_INDEX::find_range(const Cell & cell, const Cell * ranges,
150  ShapeItem range_count)
151 {
152  // first check if cell is below or above it
153  //
154  {
155  const Comp_result c0 = cell.compare(ranges[0]);
156  if (c0 == COMP_LT) return -1; // == 0 with ⎕IO = 1
157 
158  const Comp_result cN = cell.compare(ranges[range_count - 1]);
159  if (cN != COMP_LT) return range_count-1;
160  }
161 
162  // divide and conquer the ranges
163  //
164 ShapeItem ret = 0;
165  while (range_count > 1)
166  {
167  const ShapeItem middle = range_count >> 1;
168  const Comp_result cm = cell.compare(ranges[ret + middle]);
169  if (cm == COMP_LT) // cell is below the middle
170  {
171  range_count = middle;
172  }
173  else // cell is above the middle
174  {
175  ret += middle;
176  range_count -= middle;
177  }
178  }
179 
180 
181  return ret;
182 }

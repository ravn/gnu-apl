43 //-----------------------------------------------------------------------------
44 Token
45 Quad_FX::eval_AB(Value_P A, Value_P B)
46 {
47  CHECK_SECURITY(disable_native_functions);
48 
49  if (A->get_rank() > 1) RANK_ERROR;
50 
51  // dyadic ⎕FX supports the following formats:
52  //
53  // 1. "libname.so" ⎕FX "APL-name" (native function)
54  //
55  // exec properties creator
56  // ------------------------
57  // 2a. N ⎕FX "APL-text" N N N N "⎕FX"
58  // 2b. N "creator" ⎕FX "APL-text" N N N N "creator"
59  // 2c. N1 N2 N3 N4 ⎕FX "APL-text" N1 N2 N3 N4 "⎕FX"
60  // 2d. N1 N2 N3 N4 "creator" ⎕FX "APL-text" N1 N2 N3 N4 "creator"
61  //
62 
63  if (A->is_char_string()) return do_native_FX(A, -1, B);
64 
65 int eprops[4];
66 UTF8_string creator("⎕FX");
67 
68  switch(A->element_count())
69  {
70  case 2: // format 2b.
71  {
72  const Value & C = *A->get_ravel(1).get_pointer_value().get();
73  UCS_string creator_ucs(C);
74  creator = UTF8_string(creator_ucs);
75  }
76  /* no break */
77  case 1: // format 2a.
78  eprops[0] = A->get_ravel(0).get_int_value();
79  if (eprops[0] < 0) DOMAIN_ERROR;
80  if (eprops[0] > 1) DOMAIN_ERROR;
81  eprops[3] = eprops[2] = eprops[1] = eprops[0];
82  break;
83 
84  case 5: // format 2d.
85  {
86  const Value & C = *A->get_ravel(4).get_pointer_value().get();
87  UCS_string creator_ucs(C);
88  creator = UTF8_string(creator_ucs);
89  }
90  /* no break */
91  case 4: // format 2c.
92  loop(e, 4)
93  {
94  eprops[e] = A->get_ravel(e).get_int_value();
95  if (eprops[e] < 0) DOMAIN_ERROR;
96  if (eprops[e] > 1) DOMAIN_ERROR;
97  }
98  break;
99 
100  default: LENGTH_ERROR;
101  }
102 
103  return do_quad_FX(eprops, B, creator, false);
104 }

119 Token
120 Quad_FX::do_quad_FX(const int * exec_props, Value_P B,
121  const UTF8_string & creator, bool tolerant)
122 {
123  if (B->get_rank() > 2) RANK_ERROR;
124  if (B->get_rank() < 1) RANK_ERROR;
125 
126 UCS_string text;
127 
128  // ⎕FX accepts two kinds of B argments:
129  //
130  // 1. A vector whose elements are the (nested) lines of the function, or
131  // 2. A character matrix whose rows are the lines of the function.
132  //
133  // we convert each format into text, which is a UCS string with
134  // lines separated by ASCII_LF.
135  //
136  if (B->compute_depth() >= 2) // case 1: vector of simple character vectors
137  {
138  const ShapeItem rows = B->element_count();
139  loop(row, rows)
140  {
141  const Cell & cell = B->get_ravel(row);
142  if (cell.is_character_cell())
143  {
144  const Unicode uni = cell.get_char_value();
145  if (uni > UNI_ASCII_SPACE) text.append(uni);
146  text.append(UNI_ASCII_LF);
147  continue;
148  }
149 
150  // row has more than 1 chatacter, so it must be nested
151  if (!cell.is_pointer_cell())
152  {
153  MORE_ERROR() << "⎕FX: Function line " << row
154  << " is not a string";
155  DOMAIN_ERROR;
156  }
157 
158  Value_P line = cell.get_pointer_value();
159  Assert(!!line);
160 
161  Log(LOG_quad_FX)
162  {
163  CERR << "[" << setw(2) << row << "] " << *line << endl;
164  }
165 
166  if (line->is_char_vector())
167  {
168  const ShapeItem line_len = line->element_count();
169  bool skipping = false;
170  loop(l, line_len)
171  {
172  const Cell & c = line->get_ravel(l);
173  if (!c.is_character_cell())
174  {
175  MORE_ERROR() << "non-char in line at " LOC;
176  DOMAIN_ERROR;
177  }
178 
179  const Unicode uni = c.get_char_value();
180  if (l == 0 || skipping)
181  skipping = (uni <= UNI_ASCII_SPACE);
182  if (!skipping) text.append(uni);
183  }
184  }
185  else if (line->is_scalar())
186  {
187  const Cell & c1 = line->get_ravel(0);
188  if (!c1.is_character_cell())
189  {
190  MORE_ERROR() << "non-char in line at " LOC;
191  DOMAIN_ERROR;
192  }
193  const Unicode uni = c1.get_char_value();
194  if (uni > UNI_ASCII_SPACE) text.append(uni);
195  }
196  else
197  {
198 line->print_boxed(CERR);
199  MORE_ERROR() << "bad line at " LOC;
200  DOMAIN_ERROR;
201  }
202 
203  text.append(UNI_ASCII_LF);
204  }
205  }
206  else // case 2: simple character matrix
207  {
208  const ShapeItem rows = B->get_rows();
209  const ShapeItem cols = B->get_cols();
210  const Cell * cB = &B->get_ravel(0);
211 
212  loop(row, rows)
213  {
214  bool skipping = false;
215  loop(col, cols)
216  {
217  const Unicode uni = cB++->get_char_value();
218  if (col == 0 || skipping)
219  skipping = (uni <= UNI_ASCII_SPACE);
220  if (!skipping) text.append(uni);
221  }
222  text.append(UNI_ASCII_LF);
223  }
224  }
225 
226  return do_quad_FX(exec_props, text, creator, tolerant);
227 }

229 Token
230 Quad_FX::do_quad_FX(const int * exec_props, const UCS_string & text,
231  const UTF8_string & creator, bool tolerant)
232 {
233 int error_line = 0;
234 UserFunction * fun = UserFunction::fix(text, error_line, false, LOC, creator,
235  tolerant);
236  if (fun == 0)
237  {
238  Value_P Z(LOC);
239  new (Z->next_ravel()) IntCell(error_line + Workspace::get_IO());
240  Z->check_value(LOC);
241  return Token(TOK_APL_VALUE1, Z);
242  }
243 
244  fun->set_exec_properties(exec_props);
245 
246 const UCS_string fun_name = fun->get_name();
247 Value_P Z(fun_name, LOC);
248 
249  Z->check_value(LOC);
250  return Token(TOK_APL_VALUE1, Z);
251 }

253 Token
254 Quad_FX::do_native_FX(Value_P A, Axis axis, Value_P B)
255 {
256  if (uprefs.safe_mode) DOMAIN_ERROR;
257 
258 UCS_string so_name = A->get_UCS_ravel();
259 UCS_string function_name = B->get_UCS_ravel();
260 
261  if (so_name.size() == 0) LENGTH_ERROR;
262  if (function_name.size() == 0) LENGTH_ERROR;
263 
264 NativeFunction * fun = NativeFunction::fix(so_name, function_name);
265  if (fun == 0) return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
266 
267 Value_P Z = B;
268  Z->check_value(LOC);
269  return Token(TOK_APL_VALUE1, Z);
270 }

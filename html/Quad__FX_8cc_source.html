44 Token
45 Quad_FX::eval_AB(Value_P A, Value_P B)
46 {
47  CHECK_SECURITY(disable_native_functions);
48 
49  if (A->get_rank() > 1) RANK_ERROR;
50 
51  // dyadic ⎕FX supports the following formats:
52  //
53  // 1. "libname.so" ⎕FX "APL-name" (native function)
54  //
55  // exec properties creator
56  // ------------------------
57  // 2a. N ⎕FX "APL-text" N N N N "⎕FX"
58  // 2b. N "creator" ⎕FX "APL-text" N N N N "creator"
59  // 2c. N1 N2 N3 N4 ⎕FX "APL-text" N1 N2 N3 N4 "⎕FX"
60  // 2d. N1 N2 N3 N4 "creator" ⎕FX "APL-text" N1 N2 N3 N4 "creator"
61  //
62 
63  if (A->is_char_string()) return do_native_FX(A, -1, B);
64 
65 int eprops[4];
66 UTF8_string creator("⎕FX");
67 
68  switch(A->element_count())
69  {
70  case 2: // format 2b.
71  {
72  const Value & C = *A->get_ravel(1).get_pointer_value().get();
73  UCS_string creator_ucs(C);
74  creator = UTF8_string(creator_ucs);
75  }
76  /* no break */
77  case 1: // format 2a.
78  eprops[0] = A->get_ravel(0).get_int_value();
79  if (eprops[0] < 0) DOMAIN_ERROR;
80  if (eprops[0] > 1) DOMAIN_ERROR;
81  eprops[3] = eprops[2] = eprops[1] = eprops[0];
82  break;
83 
84  case 5: // format 2d.
85  {
86  const Value & C = *A->get_ravel(4).get_pointer_value().get();
87  UCS_string creator_ucs(C);
88  creator = UTF8_string(creator_ucs);
89  }
90  /* no break */
91  case 4: // format 2c.
92  loop(e, 4)
93  {
94  eprops[e] = A->get_ravel(e).get_int_value();
95  if (eprops[e] < 0) DOMAIN_ERROR;
96  if (eprops[e] > 1) DOMAIN_ERROR;
97  }
98  break;
99 
100  default: LENGTH_ERROR;
101  }
102 
103  return do_quad_FX(eprops, B, creator, false);
104 }

119 Token
120 Quad_FX::do_quad_FX(const int * exec_props, Value_P B,
121  const UTF8_string & creator, bool tolerant)
122 {
123  if (B->get_rank() > 2) RANK_ERROR;
124  if (B->get_rank() < 1) RANK_ERROR;
125 
126 UCS_string text;
127 
128  // ⎕FX accepts two kinds of B argments:
129  //
130  // 1. A vector whose elements are the lines of the function, or
131  // 2. A character matrix whose rows are the lines of the function.
132  //
133  // we convert each format into text, which is a UCS string with
134  // lines separated by ASCII_LF.
135  //
136  if (B->compute_depth() >= 2) // case 1: vector of simple character vectors
137  {
138  const ShapeItem ec = B->element_count();
139  loop(e, ec)
140  {
141  const Cell & cell = B->get_ravel(e);
142  if (cell.is_character_cell())
143  {
144  text.append(cell.get_char_value());
145  text.append(UNI_ASCII_LF);
146  continue;
147  }
148 
149  if (!cell.is_pointer_cell()) DOMAIN_ERROR;
150  Value_P line = cell.get_pointer_value();
151  Assert(!!line);
152 
153  Log(LOG_quad_FX)
154  {
155  CERR << "[" << setw(2) << e << "] " << *line << endl;
156  }
157 
158  if (line->is_char_vector())
159  {
160  const uint64_t line_len = line->element_count();
161  loop(l, line_len)
162  {
163  const Cell & c = line->get_ravel(l);
164  if (!c.is_character_cell()) DOMAIN_ERROR;
165  text.append(c.get_char_value());
166  }
167  }
168  else if (line->is_scalar())
169  {
170  const Cell & c1 = line->get_ravel(0);
171  if (!c1.is_character_cell()) DOMAIN_ERROR;
172  text.append(c1.get_char_value());
173  }
174  else
175  {
176  DOMAIN_ERROR;
177  }
178 
179  text.append(UNI_ASCII_LF);
180  }
181  }
182  else // case 2: simple character matrix
183  {
184  const ShapeItem rows = B->get_rows();
185  const ShapeItem cols = B->get_cols();
186  const Cell * cB = &B->get_ravel(0);
187 
188  loop(row, rows)
189  {
190  loop(col, cols) text.append(cB++->get_char_value());
191  text.append(UNI_ASCII_LF);
192  }
193  }
194 
195  return do_quad_FX(exec_props, text, creator, tolerant);
196 }

198 Token
199 Quad_FX::do_quad_FX(const int * exec_props, const UCS_string & text,
200  const UTF8_string & creator, bool tolerant)
201 {
202 int error_line = 0;
203 UserFunction * fun = UserFunction::fix(text, error_line, false, LOC, creator,
204  tolerant);
205  if (fun == 0)
206  {
207  Value_P Z(LOC);
208  new (Z->next_ravel()) IntCell(error_line + Workspace::get_IO());
209  Z->check_value(LOC);
210  return Token(TOK_APL_VALUE1, Z);
211  }
212 
213  fun->set_exec_properties(exec_props);
214 
215 const UCS_string fun_name = fun->get_name();
216 Value_P Z(fun_name, LOC);
217 
218  Z->check_value(LOC);
219  return Token(TOK_APL_VALUE1, Z);
220 }

222 Token
223 Quad_FX::do_native_FX(Value_P A, Axis axis, Value_P B)
224 {
225  if (uprefs.safe_mode) DOMAIN_ERROR;
226 
227 UCS_string so_name = A->get_UCS_ravel();
228 UCS_string function_name = B->get_UCS_ravel();
229 
230  if (so_name.size() == 0) LENGTH_ERROR;
231  if (function_name.size() == 0) LENGTH_ERROR;
232 
233 NativeFunction * fun = NativeFunction::fix(so_name, function_name);
234  if (fun == 0) return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
235 
236 Value_P Z = B;
237  Z->check_value(LOC);
238  return Token(TOK_APL_VALUE1, Z);
239 }

140 #if HAVE_RDTSC
141 inline uint64_t cycle_counter()
142 {
143 uint32_t lo, hi;
144  __asm__ __volatile__ ("rdtsc" : "=a" (lo), "=d" (hi));
145  return uint64_t(hi) << 32 | lo;
146 }
147 
148 #elif HAVE_GETTIMEOFDAY
149 
150 inline uint64_t cycle_counter()
151 {
152 timeval tv;
153  gettimeofday(&tv, 0);
154  return tv.tv_sec * 1000000ULL + tv.tv_usec;
155 }
156 #else // neither HAVE_RDTSC nor HAVE_GETTIMEOFDAY
157 #define cycle_counter() 0
158 #endif // HAVE_RDTSC

187 class Probe
189 {
190 public:
192  enum { PROBE_COUNT = 100,
193  PROBE_LEN = 20
194  };
195 
197  Probe()
198  { init(); }

201  void init()
202  {
203  idx = 0;
204  start_p = &dummy;
205  stop_p = &dummy;
206  }
207 
209  static int init(int p)
210  { if (p >= PROBE_COUNT) return -3;
211  probes[p].init();
212  return 0;
213  }
214 
216  static void init_all()
217  { loop(p, PROBE_COUNT) init(p); }
218 
220  void start()
221  {
222  if (idx < PROBE_LEN)
223  {
224  measurement & m = measurements[idx];
225  start_p = &m.cycles_from;
226  stop_p = &m.cycles_to;
227  }
228  else
229  {
230  start_p = &dummy;
231  stop_p = &dummy;
232  }
233 
234  // write to *start_p and *stop_p so that they are loaded into the cache
235  //
236  *stop_p = cycle_counter();
237  *start_p = cycle_counter();
238 
239  // now the real start of the measurment
240  //
241  *start_p = cycle_counter();
242  }
243 
245  void stop()
246  {
247  // the real end of the measurment
248  //
249  *stop_p = cycle_counter();
250  ++idx;
251  }
252 
254  int64_t get_time(int m) const
255  { if (m >= idx) return -1;
256  const int64_t diff = measurements[m].cycles_to
257  - measurements[m].cycles_from;
258  if (diff < 0) return -2;
259  return diff;
260  }
261 
263  static int get_time(int p, int m)
264  { if (p >= PROBE_COUNT) return -3;
265  return probes[p].get_time(m);
266  }

269  int64_t get_start(int m) const

272  }

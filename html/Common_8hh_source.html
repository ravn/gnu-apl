131 #if HAVE_RDTSC
132 inline uint64_t cycle_counter()
133 {
134 uint32_t lo, hi;
135  __asm__ __volatile__ ("rdtsc" : "=a" (lo), "=d" (hi));
136  return uint64_t(hi) << 32 | lo;
137 }
138 
139 #else // not HAVE_RDTSC
140 
141 inline uint64_t cycle_counter()
142 {
143 timeval tv;
144  gettimeofday(&tv, 0);
145  return tv.tv_sec * 1000000ULL + tv.tv_usec;
146 }
147 #endif // HAVE_RDTSC

176 class Probe
178 {
179 public:
181  enum { PROBE_COUNT = 100,
182  PROBE_LEN = 20
183  };
184 
186  Probe()
187  { init(); }
188 
190  void init()
191  {
192  idx = 0;
193  start_p = &dummy;
194  stop_p = &dummy;
195  }
196 
198  static int init(int p)
199  { if (p >= PROBE_COUNT) return -3;
200  probes[p].init();
201  return 0;
202  }
203 
205  static void init_all()
206  { loop(p, PROBE_COUNT) init(p); }
207 
209  void start()
210  {
211  if (idx < PROBE_LEN)
212  {
213  measurement & m = measurements[idx];
214  start_p = &m.cycles_from;
215  stop_p = &m.cycles_to;
216  }
217  else
218  {
219  start_p = &dummy;
220  stop_p = &dummy;
221  }
222 
223  // write to *start_p and *stop_p so that they are loaded into the cache
224  //
225  *stop_p = cycle_counter();
226  *start_p = cycle_counter();
227 
228  // now the real start of the measurment
229  //
230  *start_p = cycle_counter();
231  }
232 
234  void stop()
235  {
236  // the real end of the measurment
237  //
238  *stop_p = cycle_counter();
239  ++idx;
240  }
241 
243  int64_t get_time(int m) const
244  { if (m >= idx) return -1;
245  const int64_t diff = measurements[m].cycles_to
246  - measurements[m].cycles_from;
247  if (diff < 0) return -2;
248  return diff;
249  }
250 
252  static int get_time(int p, int m)
253  { if (p >= PROBE_COUNT) return -3;
254  return probes[p].get_time(m);
255  }
256 
258  int64_t get_start(int m) const
259  { if (m >= idx) return -1;
260  return measurements[m].cycles_from;
261  }
262 
264  static int get_start(int p, int m)
265  { if (p >= PROBE_COUNT) return -3;
266  return probes[p].get_start(m);
267  }
268 
270  int64_t get_stop(int m) const
271  { if (m >= idx) return -1;
272  return measurements[m].cycles_to;
273  }
274 
276  static int get_stop(int p, int m)
277  { if (p >= PROBE_COUNT) return -3;
278  return probes[p].get_stop(m);
279  }
280 
282  static int get_length(int p)
283  { if (p >= PROBE_COUNT) return -3;
284  return probes[p].idx;
285  }

42 //-----------------------------------------------------------------------------
43 Token
44 Bif_F12_DOMINO::eval_B(Value_P B) const
45 {
46  if (B->is_scalar())
47  {
48  Value_P Z(LOC);
49 
50  Cell cZ;
51  B->get_cfirst().bif_reciprocal(&cZ);
52  Z->next_ravel_Cell(cZ);
53  Z->check_value(LOC);
54  return Token(TOK_APL_VALUE1, Z);
55  }
56 
57  if (B->get_rank() == 1) // inversion at the unit sphere
58  {
59  const double qct = Workspace::get_CT();
60  const ShapeItem len = B->get_shape_item(0);
61  APL_Complex r2(0.0);
62  loop(l, len)
63  {
64  const APL_Complex b = B->get_cravel(l).get_complex_value();
65  r2 += b*b;
66  }
67 
68  if (r2.real() < qct && r2.real() > -qct &&
69  r2.imag() < qct && r2.imag() > -qct)
70  DOMAIN_ERROR;
71 
72  Value_P Z(len, LOC);
73 
74  if (r2.imag() < qct && r2.imag() > -qct) // real result
75  {
76  loop(l, len)
77  {
78  const APL_Float b = B->get_cravel(l).get_real_value();
79  Z->next_ravel_Float(b / r2.real());
80  }
81  }
82  else // complex result
83  {
84  loop(l, len)
85  {
86  const APL_Complex b = B->get_cravel(l).get_complex_value();
87  Z->next_ravel_Complex(b / r2);
88  }
89  }
90 
91  Z->set_default(*B.get(), LOC);
92  Z->check_value(LOC);
93  return Token(TOK_APL_VALUE1, Z);
94  }
95 
96  if (B->get_rank() > 2) RANK_ERROR;
97 
98 const ShapeItem rows = B->get_shape_item(0);
99 const ShapeItem cols = B->get_shape_item(1);
100  if (cols > rows) LENGTH_ERROR;
101 
102  // create an identity matrix I and call eval_AB(I, B).
103  //
104 const Shape shape_I(rows, rows);
105 Value_P I(shape_I, LOC);
106 
107  loop(y, rows)
108  loop(x, rows) I->next_ravel_Float(y == x ? 1.0 : 0.0);
109 
110 Token result = eval_AB(I, B);
111  return result;
112 }
113 //-----------------------------------------------------------------------------
114 Token
115 Bif_F12_DOMINO::eval_XB(Value_P X, Value_P B) const
116 {
117  if (!X->is_scalar()) RANK_ERROR;
118 const double EPS = X->get_cfirst().get_real_value();
119 
120  if (B->get_rank() != 2) RANK_ERROR;
121 
122  // if rank of A or B is < 2 then treat it as a
123  // 1 by n (or 1 by 1) matrix..
124  //
125 const ShapeItem rows_B = B->get_rows();
126 const ShapeItem cols_B = B->get_cols();
127  if (rows_B < cols_B) LENGTH_ERROR;
128  if (rows_B*cols_B == 0) LENGTH_ERROR;
129 
130 
131 const bool need_complex = B->is_complex(true);
132 Value_P Z(3, LOC);
133  QR_factorization(Z, need_complex, rows_B, cols_B, &B->get_cfirst(), EPS);
134 
135  Z->set_default(*B.get(), LOC); // not needed
136  Z->check_value(LOC);
137  return Token(TOK_APL_VALUE1, Z);
138 }
139 //-----------------------------------------------------------------------------
140 Token
141 Bif_F12_DOMINO::eval_AB(Value_P A, Value_P B) const
142 {
143 ShapeItem rows_A = 1;
144 ShapeItem cols_A = 1;
145 
146 Shape shape_Z;
147 
148  // if rank of A or B is < 2 then treat it as a
149  // 1 by n (or 1 by 1) matrix..
150  //
151 ShapeItem rows_B = 1;
152 ShapeItem cols_B = 1;
153  switch(B->get_rank())
154  {
155  case 0: break;
156 
157  case 1: rows_B = B->get_shape_item(0);
158  break;
159 
160  case 2: cols_B = B->get_shape_item(1);
161  rows_B = B->get_shape_item(0);
162  shape_Z.add_shape_item(cols_B);
163  break;
164 
165  default: RANK_ERROR;
166  }
167 
168  switch(A->get_rank())
169  {
170  case 0: break;
171 
172  case 1: rows_A = A->get_shape_item(0);
173  break;
174 
175  case 2: cols_A = A->get_shape_item(1);
176  rows_A = A->get_shape_item(0);
177  shape_Z.add_shape_item(cols_A);
178  break;
179 
180  default: RANK_ERROR;
181  }
182 
183  if (rows_B < cols_B) LENGTH_ERROR;
184  if (rows_A != rows_B) LENGTH_ERROR;
185 
186 const bool need_complex = A->is_complex(true) || B->is_complex(true);
187 Value_P Z(shape_Z, LOC);
188  divide_matrix(Z.getref(), need_complex, rows_A, cols_A, &A->get_cfirst(),
189  cols_B, &B->get_cfirst());
190 
191  Z->set_default(*B.get(), LOC);
192 
193  Z->check_value(LOC);
194  return Token(TOK_APL_VALUE1, Z);
195 }

269 template<>
270 Value_P Bif_F12_DOMINO::invert_upper_triangle_matrix<false>(const ShapeItem M,
271  const ShapeItem N,
272  double * utm,
273  double * tmp)
274 {
275  // ignore rows ≥ N; they are 0 and then cols ≥ N of the result as well.
276  // the 0s will be filled in at the end.
277 
278 matrix_assert(M >= N);
279 Matrix<false> UTM(utm, N, N); // the matrix to be inverted
280 Matrix<false> AUG(tmp, N, N); // the augmented matrix for the result
281 
282  AUG.init_identity(N); // start with the identity matrix
283 
284 UTM.debug("UTM orig");
285 AUG.debug("AUG orig");
286 
287  // divide every row by its diagonal element, so that the diagonal
288  // elements of UTM become 1.0.
289  //
290  loop(y, N) // for every row y
291  {
292  const double diag_y = UTM.real(y, y);
293  if (diag_y == 0.0) DOMAIN_ERROR;
294 
295  // divide off-diagonal UTM | AUG[y;] by diag_y
296  for (ShapeItem x = y + 1; x < N; ++x) UTM.real(y, x) /= diag_y;
297 
298  // divide diagonal of UTM | AUG by diag_y
299  UTM.real(y, y) = 1.0;
300  AUG.real(y, y) /= diag_y;
301  }
302 
303 UTM.debug("UTM unity " LOC);
304 AUG.debug("AUG unity " LOC);
305 
306  // subtract a multiple of the diagonal element from the items above it
307  // so that the item becomes 0.0
308  //
309  loop(y, N) // for every row y
310  loop (y1, y) // for every row y1 above y
311  {
312 Q1(y) Q1(y1)
313  const double factor = UTM.real(y1, y);
314  if (factor == 0) continue; // already 0.0
315 Q1(factor)
316  for (ShapeItem x1 = y1; x1 < N; ++x1)
317  {
318  // make UTM[y1;x1] zero by subtracting UTM[y;x] × UTM[y;]
319  UTM.real(y1, x1) -= UTM.real(y, x1) * factor;
320  AUG.real(y1, x1) -= AUG.real(y, x1) * factor;
321  }
322 UTM.debug("UTM at " LOC);
323 AUG.debug("AUG at " LOC);
324  }
325 
326 AUG.debug("AUG at " LOC);
327 
328  // create the result value
329 const Shape shape_INV(N, M);
330 Value_P INV(shape_INV, LOC);
331  loop(y, N) // for every row y
332  {
333  loop(x, M) // for every column y
334  {
335  const double flt = (x < N) ? AUG.real(y, x) : 0.0;
336  if (!isfinite(flt)) DOMAIN_ERROR;
337  INV->next_ravel_Float(flt);
338  }
339  }
340 
341  INV->check_value(LOC);
342  return INV;
343 }
344 //-----------------------------------------------------------------------------
346 template<>
347 Value_P Bif_F12_DOMINO::invert_upper_triangle_matrix<true>(ShapeItem M,
348  ShapeItem N,
349  double * utm,
350  double * tmp)
351 {
352  // ignore rows ≥ N; they are 0 and then cols ≥ N of the result as well.
353  // the 0s will be filled in at the end.
354 
355 matrix_assert(M >= N);
356 Matrix<true> UTM(utm, N, N); // the matrix to be inverted
357 Matrix<true> AUG(tmp, N, N); // the augmented matrix for the result
358 
359  AUG.init_identity(N); // start with the identity matrix
360 
361 UTM.debug("UTM orig");
362 AUG.debug("AUG orig");
363 
364  // divide every row by its diagonal element, so that the diagonal
365  // elements of UTM become 1.0.
366  //
367  loop(y, N) // for every row y
368  {
369  const complex<double> diag_y = UTM.get_Z(y, y);
370  if (diag_y.real() == 0.0 && diag_y.imag()) DOMAIN_ERROR;
371 
372  // divide off-diagonal UTM | AUG[y;] by diag_y
373  for (ShapeItem x = y + 1; x < N; ++x)
374  UTM.div_Z(y, x, diag_y);
375 
376  // divide diagonal of UTM | AUG by diag_y
377  UTM.set_Z(y, y, complex<double>(1.0, 0.0));
378  AUG.div_Z(y, y, diag_y);
379  }
380 
381 UTM.debug("UTM unity " LOC);
382 AUG.debug("AUG unity " LOC);
383 
384  // subtract a multiple of the diagonal element from the items above it
385  // so that the item becomes 0.0
386  //
387  loop(y, N) // for every row y
388  loop (y1, y) // for every row y1 above y
389  {
390 Q1(y) Q1(y1)
391  const complex<double> factor = UTM.get_Z(y1, y);
392  if (factor.real() == 0 && factor.imag() == 0) continue;
393 Q1(factor)
394  for (ShapeItem x1 = y1; x1 < N; ++x1)
395  {
396  // make UTM[y1;x1] zero by subtracting UTM[y;x] × UTM[y;]
397  UTM.sub_Z(y1, x1, UTM.get_Z(y, x1) * factor);
398  AUG.sub_Z(y1, x1, AUG.get_Z(y, x1) * factor);
399  }
400 UTM.debug("UTM at " LOC);
401 AUG.debug("AUG at " LOC);
402  }
403 
404 AUG.debug("AUG at " LOC);
405 
406  // create the result value
407 const Shape shape_INV(N, M);
408 Value_P INV(shape_INV, LOC);
409  loop(y, N) // for every row y
410  {
411  loop(x, M) // for every column y
412  {
413  if (x < N) // diagonal or above
414  {
415  const double re = AUG.real(y, x);
416  const double im = AUG.imag(y, x);
417  if (!(isfinite(re) && isfinite(im))) DOMAIN_ERROR;
418  INV->next_ravel_Complex(re, im);
419  }
420  else // below diagonal
421  {
422  INV->next_ravel_Complex(0.0, 0.0);
423  }
424  }
425  }
426 
427  INV->check_value(LOC);
428  return INV;
429 }
430 //-----------------------------------------------------------------------------
431 void
432 Bif_F12_DOMINO::QR_factorization(Value_P Z, bool need_complex, ShapeItem rows,
433  ShapeItem cols, const Cell * cB, double EPS)
434 {
435  /* We want to store all floating point variables (including complex ones)
436  in a single double[]. Before and after each variable we leave one double
437  for storing numbers 42.0, 43.0, ...51.0. These numbers are used to check
438  for overrides of the allocated space.
439 
440  Complex numbers are stored as real followed by imag part.
441 
442  The variables are B, Q, and R with B = Q +.× R, with Q real orthogonal
443  and R real or complex upper triangular. Since being R is computed after
444  Q, it can be used as a temporary variable in the computation of Q (i,e,
445  in function householder()).
446  */
447 
448  // start with the base addresses of the variables. All variables are
449  // allocated as rows * rows so that we can freely rorate them...
450  //
451 const int CPLX = need_complex ? 2 : 1; // number of doubles per variable item
452 const ShapeItem len_B = rows * cols;
453 const ShapeItem len = rows * rows;
454 const ShapeItem base_B = 1;
455 const ShapeItem base_Q = base_B + CPLX*len + 2;
456 const ShapeItem base_Qi = base_Q + CPLX*len + 2;
457 const ShapeItem base_R = base_Qi + CPLX*len + 2;
458 const ShapeItem base_S = base_R + CPLX*len + 2;
459 const ShapeItem end = base_S + CPLX*len + 2;
460 
461 double * data = new double[end*CPLX]; if (data == 0) WS_FULL;
462  memset(data, 0, end*sizeof(double));
463  data[base_B - 1] = 42.0; data[base_B + CPLX*len_B] = 43.0;
464  data[base_Q - 1] = 44.0; data[base_Q + CPLX*len] = 45.0;
465  data[base_Qi - 1] = 46.0; data[base_Qi + CPLX*len] = 47.0;
466  data[base_R - 1] = 48.0; data[base_R + CPLX*len] = 49.0;
467  data[base_S - 1] = 50.0; data[base_S + CPLX*len] = 51.0;
468 
469  if (need_complex) // complex B
470  {
471  setup_complex_B(cB, data + base_B, len_B);
472  double * Q = householder<true>(data + base_B, rows, cols, data + base_Q,
473  data + base_Qi, data + base_R, data + base_S, EPS);
474 
475  setup_complex_B(cB, data + base_B, len_B); // restore B
476  const Matrix<true> Bm(data + base_B, rows, cols);
477  Matrix<true> Qm(Q, rows, rows);
478  Qm.transpose(rows);
479  Matrix<true> Rm(data + base_R, rows, cols);
480  Rm.init_inner_product(Qm, Bm);
481  Qm.debug("final Q");
482  Rm.debug("final R");
483  }
484  else // real B
485  {
486  Assert(data[base_B + CPLX*len_B] == 43.0);
487  setup_real_B(cB, data + base_B, len_B);
488  Assert(data[base_B + CPLX*len_B] == 43.0);
489  double * Q = householder<false>(data + base_B, rows,cols, data + base_Q,
490  data + base_Qi, data + base_R, data + base_S, EPS);
491 
492  setup_real_B(cB, data + base_B, len_B); // restore B
493  const Matrix<false> Bm(data + base_B, rows, cols);
494  Matrix<false> Qm(Q, rows, rows);
495  Qm.transpose(rows);
496  Matrix<false> Rm(data + base_R, rows, cols);
497  Rm.init_inner_product(Qm, Bm);
498  Assert(data[base_B + CPLX*len_B] == 43.0);
499  }
500 
501  // check that the memory areas were not overridden
502  Assert(data[base_B - 1] == 42.0);
503  Assert(data[base_B + CPLX*len_B] == 43.0);
504  Assert(data[base_Q - 1] == 44.0);
505  Assert(data[base_Q + CPLX*len] == 45.0);
506  Assert(data[base_Qi - 1] == 46.0);
507  Assert(data[base_Qi + CPLX*len] == 47.0);
508  Assert(data[base_R - 1] == 48.0);
509  Assert(data[base_R + CPLX*len] == 49.0);
510  Assert(data[base_S - 1] == 50.0);
511  Assert(data[base_S + CPLX*len] == 51.0);
512 
513  {
514  const Shape Q_shape(rows, rows);
515  Value_P Qv(Q_shape, LOC);
516  if (need_complex)
517  {
518  loop(q, len)
519  {
520  const double re = data[base_Q + 2*q];
521  const double im = data[base_Q + 2*q + 1];
522  if (!(isfinite(re) && isfinite(im))) DOMAIN_ERROR;
523  Qv->next_ravel_Complex(re, im);
524  }
525  }
526  else
527  {
528  loop(q, len)
529  {
530  const double re = data[base_Q + q];
531  if (!isfinite(re)) DOMAIN_ERROR;
532  Qv->next_ravel_Float(re);
533  }
534  }
535  Qv->check_value(LOC);
536  Z->next_ravel_Pointer(Qv.get());
537  }
538  {
539  const Shape shape_R(rows, cols);
540  Value_P vR(shape_R, LOC);
541  if (need_complex)
542  {
543  loop(r, rows*cols)
544  {
545  const double re = data[base_R + 2*r];
546  const double im = data[base_R + 2*r + 1];
547  if (!(isfinite(re) && isfinite(im))) DOMAIN_ERROR;
548  vR->next_ravel_Complex(re, im);
549  }
550  }
551  else
552  {
553  loop(r, rows*cols)
554  {
555  const double re = data[base_R + r];
556  if (!isfinite(re)) DOMAIN_ERROR;
557  vR->next_ravel_Float(re);
558  }
559  }
560  vR->check_value(LOC);
561  Z->next_ravel_Pointer(vR.get());
562 
563  if (need_complex)
564  {
565  Value_P INV = invert_upper_triangle_matrix<true>(rows, cols,
566  data + base_R,
567  data + base_Q);
568  Z->next_ravel_Pointer(INV.get());
569  }
570  else
571  {
572  Value_P INV = invert_upper_triangle_matrix<false>(rows, cols,
573  data + base_R,
574  data + base_Q);
575  Z->next_ravel_Pointer(INV.get());
576  }
577  }
578 
579  delete[] data;
580 }

609 //-----------------------------------------------------------------------------
610 template<bool cplx>
611 double *
612 Bif_F12_DOMINO::householder(double * pB, ShapeItem rows, ShapeItem cols,
613  double * pQ, double * pQi, double * pT, double * pS,
614  double EPS)
615 {
616  // pB is the matrix to be factorized, pQ, pQi, and pT were initialized to 0
617  //
618  // the algorithm is essentially the one described in Garry Helzer's paper
619  // "THE HOUSEHOLDER ALGORITHM AND APPLICATIONS" but using complex numbers
620  // when needed.
621 
622 const double qct = Workspace::get_CT();
623 const double qct2 = qct*qct;
624 double BMAX = 0.0;
625 
626 Matrix<cplx> mT (pT, rows, rows); // temporary storage
627 Matrix<cplx> mQ (pQ, rows, rows); // keeps size
628 Matrix<cplx> mB (pB, rows, cols); // shrinks
629 Matrix<cplx> mQi(pQi, rows, rows); // keeps size
630

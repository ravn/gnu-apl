37 void
38 Function::get_attributes(int mode, Cell * dest) const
39 {
40  switch(mode)
41  {
42  case 1: // valences
43  new (dest + 0) IntCell(has_result() ? 1 : 0);
44  new (dest + 1) IntCell(get_fun_valence());
45  new (dest + 2) IntCell(get_oper_valence());
46  return;
47 
48  case 2: // creation time (7⍴0 for system functions)
49  {
50  const YMDhmsu created(get_creation_time());
51  new (dest + 0) IntCell(created.year);
52  new (dest + 1) IntCell(created.month);
53  new (dest + 2) IntCell(created.day);
54  new (dest + 3) IntCell(created.hour);
55  new (dest + 4) IntCell(created.minute);
56  new (dest + 5) IntCell(created.second);
57  new (dest + 6) IntCell(created.micro/1000);
58  }
59  return;
60 
61  case 3: // execution properties
62  new (dest + 0) IntCell(get_exec_properties()[0]);
63  new (dest + 1) IntCell(get_exec_properties()[1]);
64  new (dest + 2) IntCell(get_exec_properties()[2]);
65  new (dest + 3) IntCell(get_exec_properties()[3]);
66  return;
67 
68  case 4: // 4 ⎕DR functions is always 0 0
69  new (dest + 0) IntCell(0);
70  new (dest + 1) IntCell(0);
71  return;
72  }
73 
74  Assert(0 && "Not reached");
75 }

133 Fun_signature
134 Function::get_signature() const
135 {
136 int sig = SIG_FUN;
137  if (has_result()) sig |= SIG_Z;
138  if (has_axis()) sig |= SIG_X;
139 
140  if (get_oper_valence() == 2) sig |= SIG_RO;
141  if (get_oper_valence() >= 1) sig |= SIG_LO;
142 
143  if (get_fun_valence() == 2) sig |= SIG_A;
144  if (get_fun_valence() >= 1) sig |= SIG_B;
145 
146  return Fun_signature(sig);
147 }

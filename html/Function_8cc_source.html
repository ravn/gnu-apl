36 void
37 Function::get_attributes(int mode, Value & Z) const
38 {
39  switch(mode)
40  {
41  case 1: // valences
42  Z.next_ravel_Int(has_result() ? 1 : 0);
43  Z.next_ravel_Int(get_fun_valence());
44  Z.next_ravel_Int(get_oper_valence());
45  return;
46 
47  case 2: // creation time (7⍴0 for system functions)
48  {
49  const YMDhmsu created(get_creation_time());
50  Z.next_ravel_Int(created.year);
51  Z.next_ravel_Int(created.month);
52  Z.next_ravel_Int(created.day);
53  Z.next_ravel_Int(created.hour);
54  Z.next_ravel_Int(created.minute);
55  Z.next_ravel_Int(created.second);
56  Z.next_ravel_Int(created.micro/1000);
57  }
58  return;
59 
60  case 3: // execution properties
61  Z.next_ravel_Int(get_exec_properties()[0]);
62  Z.next_ravel_Int(get_exec_properties()[1]);
63  Z.next_ravel_Int(get_exec_properties()[2]);
64  Z.next_ravel_Int(get_exec_properties()[3]);
65  return;
66 
67  case 4: // 4 ⎕DR for functions is always 0 0
68  Z.next_ravel_Int(0);
69  Z.next_ravel_Int(0);
70  return;
71  }
72 
73  Assert(0 && "Not reached");
74 }

132 Fun_signature
133 Function::get_signature() const
134 {
135 int sig = SIG_FUN;
136  if (has_result()) sig |= SIG_Z;
137  if (has_axis()) sig |= SIG_X;
138 
139  if (get_oper_valence() == 2) sig |= SIG_RO;
140  if (get_oper_valence() >= 1) sig |= SIG_LO;
141 
142  if (get_fun_valence() == 2) sig |= SIG_A;
143  if (get_fun_valence() >= 1) sig |= SIG_B;
144 
145  return Fun_signature(sig);
146 }

59  Shape(uRank rk, const ShapeItem * sh)
60  : rho_rho(0),

62  { loop(r, rk) add_shape_item(sh[r]); }

65  Shape(const Shape & other)
66  : rho_rho(other.rho_rho),
67  volume(other.volume)
68  {
69  /* this may trigger a somewhat bogus -Wmaybe-uninitialized warning but
70  makes some workspaces 3 times faster. The slower code with no warning
71  would be:
72 
73  loop(r, rhorho) rho[r] = other.rho[r];
74 
75  but that prevents loop-unrolling.
76  */
77  loop(r, MAX_RANK) rho[r] = other.rho[r];
78  }

139  void set_shape_item(uAxis r, ShapeItem sh)
140  { Assert(r < rho_rho);
141  if (rho[r]) { volume /= rho[r]; rho[r] = sh; volume *= rho[r]; }
142  else { rho[r] = sh; recompute_volume(); } }
143 
145  void recompute_volume()
146  { volume = 1; loop(r, rho_rho) volume *= rho[r]; }
147 
149  void increment_shape_item(Rank r)
150  { set_shape_item(r, rho[r] + 1); }
151 
153  void set_last_shape_item(ShapeItem sh)
154  { set_shape_item(rho_rho - 1, sh); }
155 
157  void add_shape_item(ShapeItem len)
158  { if (rho_rho >= MAX_RANK) LIMIT_ERROR_RANK;
159  rho[rho_rho++] = len; volume *= len; }
160 
162  void expand_rank(uRank new_rk)
163  { if (rho_rho < new_rk)
164  {
165  const int diff = new_rk - rho_rho;
166  loop(r, rho_rho) rho[new_rk - r - 1] = rho[rho_rho - r - 1];
167  loop(r, diff) rho[r] = 1;
168  rho_rho = new_rk;
169  }
170  }

201  Shape reverse_scan() const
202  { Shape ret; ret.rho_rho = rho_rho;
203  ShapeItem prod = 1;
204  ret.volume = 1;
205  for (Rank r = rho_rho - 1; r >= 0; --r)
206  { ret.rho[r] = prod; ret.volume *= prod; prod *= rho[r]; }
207  return ret; }

235 class Shape3 : public Shape
236 {
237 public:
239  Shape3(const Shape & shape, ShapeItem axis)
240  : Shape(1, 1, 1)
241  {
242  loop(r, shape.get_rank())
243  {
244  if (r < axis) rho[0] *= shape.get_shape_item(r);
245  else if (r > axis) rho[2] *= shape.get_shape_item(r);
246  else rho[1] = shape.get_shape_item(r);
247  }
248  }
249 
251  Shape3(ShapeItem _h, ShapeItem _m, ShapeItem _l)
252  : Shape(_h, _m, _l)
253  {}
254 
256  ShapeItem h() const
257  { return rho[0]; }
258 
260  ShapeItem hml(ShapeItem hh, ShapeItem mm, ShapeItem ll) const
261  { return ll + l()*(mm + m()*hh); }
262 
264  ShapeItem m() const
265  { return rho[1]; }
266 
268  ShapeItem l() const
269  { return rho[2]; }
270 };

47  Shape(ShapeItem rows, ShapeItem cols)
48  : rho_rho(2),

54  : rho_rho(3),

59  Shape(uRank rk, const ShapeItem * sh)
60  : rho_rho(0),
61  volume(1)
62  { loop(r, rk) add_shape_item(sh[r]); }

65  Shape(const Shape & other)
66  : rho_rho(other.rho_rho),
67  volume(other.volume)
68  {
69  /* this may trigger a somewhat bogus -Wmaybe-uninitialized warning but
70  makes some workspaces 3 times faster. The slower code with no warning
71  would be:
72 
73  loop(r, rhorho) rho[r] = other.rho[r];
74 
75  but that prevents loop-unrolling.
76  */
77  loop(r, MAX_RANK) rho[r] = other.rho[r];
78  }

132  ShapeItem get_rows() const
133  { if (rho_rho == 0) return 1; // scalar
134  if (const ShapeItem cols = rho[rho_rho - 1]) return volume / cols;
135  ShapeItem count = 1;
136  loop(r, rho_rho - 1) count *= rho[r];
137  return count; }
138 
140  void set_shape_item(uAxis r, ShapeItem sh)
141  { Assert(r < rho_rho);
142  if (rho[r]) { volume /= rho[r]; rho[r] = sh; volume *= rho[r]; }
143  else { rho[r] = sh; recompute_volume(); } }
144 
146  void recompute_volume()
147  { volume = 1; loop(r, rho_rho) volume *= rho[r]; }
148 
150  void increment_shape_item(Rank r)
151  { set_shape_item(r, rho[r] + 1); }
152 
154  void set_last_shape_item(ShapeItem sh)
155  { set_shape_item(rho_rho - 1, sh); }
156 
158  void add_shape_item(ShapeItem len)
159  { if (rho_rho >= MAX_RANK) LIMIT_ERROR_RANK;
160  rho[rho_rho++] = len; volume *= len; }
161 
163  void expand_rank(uRank new_rank)
164  { if (rho_rho >= new_rank) return; // no need to expand
165  const int diff = new_rank - rho_rho;
166  for (Rank r = rho_rho - 1; r >= 0; --r) rho[r + diff] = rho[r];
167  loop(r, diff) rho[r] = 1;
168  rho_rho = new_rank;
169  }

182  return ret;

196  bool contains_axis(const Axis ax) const

200  bool contains(const Shape & value) const
201  { Assert(rho_rho == value.rho_rho);
202  loop(r, rho_rho) if (value.rho[r] >= rho[r]) return false;
203  return true; }
204 
206  Shape get_weights() const
207  { Shape ret; ret.rho_rho = rho_rho;
208  ShapeItem weight = 1;
209  ret.volume = 1;
210  for (Rank r = rho_rho - 1; r >= 0; --r)
211  {
212  ret.rho[r] = weight;
213  ret.volume *= weight;
214  weight *= rho[r];
215  }
216  return ret;
217  }

229

245 class Shape3 : public Shape
246 {
247 public:
249  Shape3(const Shape & shape, ShapeItem axis)
250  : Shape(1, 1, 1)
251  {
252  loop(r, shape.get_rank())
253  {
254  if (r < axis) rho[0] *= shape.get_shape_item(r);
255  else if (r > axis) rho[2] *= shape.get_shape_item(r);
256  else rho[1] = shape.get_shape_item(r);
257  }
258  }
259 
261  Shape3(ShapeItem _h, ShapeItem _m, ShapeItem _l)
262  : Shape(_h, _m, _l)
263  {}
264 
266  ShapeItem h() const
267  { return rho[0]; }
268 
270  ShapeItem hml(ShapeItem hh, ShapeItem mm, ShapeItem ll) const
271  { return ll + l()*(mm + m()*hh); }
272 
274  ShapeItem m() const
275  { return rho[1]; }
276 
278  ShapeItem l() const
279  { return rho[2]; }
280 };

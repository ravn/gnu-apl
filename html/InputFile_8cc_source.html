64  files_todo[0].from_COPY = false;

67

70  if (files_todo[0].file != stdin)
71  {
72  fclose(files_todo[0].file);
73  files_todo[0].file = 0;
74  files_todo[0].line_no = -1;
75  }
76  }
77  }

80 void
81 InputFile::randomize_files()
82 {
83  {
84  timeval now;
85  gettimeofday(&now, 0);
86  srandom(now.tv_sec + now.tv_usec);
87  }
88 
89  // check that all file are test files
90  //
91  loop(f, files_todo.size())
92  {
93  if (files_todo[f].test == false) // not a test file
94  {
95  CERR << "Cannot randomise testfiles when a mix of -T"
96  " and -f is used (--TR ignored)" << endl;
97  return;
98  }
99  }
100 
101  for (size_t done = 0; done < 4*files_todo.size();)
102  {
103  const int n1 = random() % files_todo.size();
104  const int n2 = random() % files_todo.size();
105  if (n1 == n2) continue;
106 
107  InputFile f1 = files_todo[n1];
108  InputFile f2 = files_todo[n2];
109 
110  const char * ff1 = strrchr(f1.filename.c_str(), '/');
111  if (!ff1++) ff1 = f1.filename.c_str();
112  const char * ff2 = strrchr(f2.filename.c_str(), '/');
113  if (!ff2++) ff2 = f2.filename.c_str();
114 
115  // the order of files named AAA... or ZZZ... shall not be changed
116  //
117  if (strncmp(ff1, "AAA", 3) == 0) continue;
118  if (strncmp(ff1, "ZZZ", 3) == 0) continue;
119  if (strncmp(ff2, "AAA", 3) == 0) continue;
120  if (strncmp(ff2, "ZZZ", 3) == 0) continue;
121 
122  // at this point f1 and f2 shall be swapped.
123  //
124  files_todo[n1] = f2;
125  files_todo[n2] = f1;
126  ++done;
127  }
128 }

137 bool
138 InputFile::check_filter(const UTF8_string & line)
139 {
140 bool ret = in_matched;
141 UCS_string ucs_line(line);
142  ucs_line.remove_leading_and_trailing_whitespaces();
143 
144  if (in_function)
145  {
146  if (ucs_line.size() && ucs_line[0] == UNI_NABLA) // end of a function
147  {
148  in_function = false;
149  in_variable = false;
150  in_matched = false;
151  }
152  return ret;
153  }
154 
155  if (in_variable)
156  {
157  if (ucs_line.size() == 0) // end of a vriable
158  {
159  in_function = false;
160  in_variable = false;
161  in_matched = false;
162  }
163  return ret;
164  }
165 
166  if (ucs_line.size() && (ucs_line[0] == UNI_NABLA)) // new function
167  {
168  in_function = true;
169  in_variable = false;
170  in_matched = false;
171  ucs_line = ucs_line.drop(1);
172  UserFunction_header uh(ucs_line, false);
173  const UCS_string fun_name = uh.get_name();
174  loop(n, object_filter.size())
175  {
176  if (fun_name == object_filter[n])
177  {
178  in_matched = true;
179  return true;
180  }
181  }
182  return false;
183  }
184 
185  if (ucs_line.size() && (Avec::is_quad(ucs_line[0]) ||
186  Avec::is_first_symbol_char(ucs_line[0]))) // new variable
187  {
188  in_function = false;
189  in_variable = true;
190  in_matched = false;
191  loop(u, ucs_line.size())
192  {
193  if (ucs_line[u] != UNI_LEFT_ARROW) continue; // not ←
194 
195  ucs_line.resize(u);
196  ucs_line.remove_leading_and_trailing_whitespaces();
197  loop(n, object_filter.size())
198  {
199  if (ucs_line == object_filter[n])
200  {
201  in_matched = true;
202  return true;
203  }
204  }
205  }
206  }
207 
208  return ret;
209 }

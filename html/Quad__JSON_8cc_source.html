39 Token
40 Quad_JSON::eval_AB(Value_P A, Value_P B) const
41 {
42  if (A->get_rank() > 0) RANK_ERROR;
43 
44 const int function_number = A->get_cfirst().get_int_value();
45  switch(function_number)
46  {
47  case 0: // same as monadic ⎕JSON
48  {
49  return eval_B(B);
50  }
51 
52  case 1: // read and convert a JSOM file
53  {
54  return convert_file(B.getref());
55  }
56 
57  case 2: // read and convert a JSOM file (unsorted)
58  {
59  return Token(TOK_APL_VALUE1, APL_to_JSON(B.getref(), false));
60  }
61 
62  case 3: // read and convert a JSOM file (sorted)
63  {
64  return Token(TOK_APL_VALUE1, APL_to_JSON(B.getref(), true));
65  }
66 
67  }
68 
69  MORE_ERROR() << "A ⎕JSON B: Bad function number A=" << function_number;
70  DOMAIN_ERROR;
71 }

73 Token
74 Quad_JSON::convert_file(const Value & B) const
75 {
76 const UCS_string filename_ucs(B);
77 const UTF8_string filename_utf(filename_ucs);
78 
79  errno = 0;
80 const int fd = open(filename_utf.c_str(), O_RDONLY);
81  if (fd == -1)
82  {
83  MORE_ERROR() << "1 ⎕JSON B: error reading " << B
84  << ": " << strerror(errno);
85  DOMAIN_ERROR;
86  }
87 
88 struct stat st;
89  if (fstat(fd, &st))
90  {
91  MORE_ERROR() << "1 ⎕JSON B: error in fstat(" << B
92  << "): " << strerror(errno);
93  ::close(fd);
94  DOMAIN_ERROR;
95  }
96 
97 UTF8 * buffer = new UTF8[st.st_size];
98  if (buffer == 0)
99  {
100  ::close(fd);
101  WS_FULL;
102  }
103 
104 const ssize_t bytes_read = read(fd, buffer, st.st_size);
105  if (bytes_read != st.st_size)
106  {
107  MORE_ERROR() << "1 ⎕JSON B: error in reading " << B
108  << "): " << strerror(errno);
109  ::close(fd);
110  DOMAIN_ERROR;
111  ::close(fd);
112  delete [] buffer;
113  }
114 
115 const UTF8_string json_string_utf8(buffer, bytes_read);
116  delete[] buffer;
117  ::close(fd);
118 
119 UCS_string json_string_ucs(json_string_utf8);
120 Value_P json_string_value(json_string_ucs, LOC);
121 
122  return eval_B(json_string_value);
123 }

147 void
148 Quad_JSON::APL_to_JSON_string(UCS_string & result, const Cell & cell,
149  bool level, bool sorted)
150 {
151  if (cell.is_integer_cell())
152  {
153  char cc[40];
154  snprintf(cc, sizeof(cc), "%lld", long_long(cell.get_int_value()));
155  result.append_UTF8(cc);
156  return;
157  }
158 
159  if (cell.is_float_cell())
160  {
161  char cc[40];
162  snprintf(cc, sizeof(cc), "%lg", cell.get_real_value());
163  result.append_UTF8(cc);
164  return;
165  }
166 
167  if (cell.is_complex_cell())
168  {
169  char cc[50];
170  snprintf(cc, sizeof(cc), "%lgJ%lg", cell.get_real_value(),
171  cell.get_imag_value());
172  result.append_UTF8(cc);
173  return;
174  }
175 
176  if (!cell.is_pointer_cell())
177  {
178  MORE_ERROR() << "2 ⎕JSON B: Unexpected/unsupported Celltype "
179  << Cell::get_cell_type_name(cell.get_cell_type());
180  DOMAIN_ERROR;
181  }
182 
183  // at this point the cell should end up as char vector. Determint its depth.
184  //
185 const Value * Z = cell.get_pointer_value().get();
186  if (!(Z->get_cfirst().is_pointer_cell() && Z->is_scalar()))
187  {
188  APL_to_JSON_string(result, *Z, level, sorted);
189  return;
190  }
191 
192  Z = Z->get_cfirst().get_pointer_value().get();
193  if (!Z->is_char_vector())
194  {
195 FIXME;
196  }
197 
198 const UCS_string lit_ucs(*Z);
199  if (lit_ucs.compare(UCS_string("null")) == COMP_EQ ||
200  lit_ucs.compare(UCS_string("true")) == COMP_EQ ||
201  lit_ucs.compare(UCS_string("false")) == COMP_EQ)
202  {
203  result.append(lit_ucs);
204  return;
205  }
206 
207  MORE_ERROR() << "2 ⎕JSON B: bad JSON literal ⊂'" << lit_ucs
208  << "'. Expecting ⊂'true', ⊂'false', or ⊂'null'";
209  result.clear(); // indicate error
210 }
211 //-----------------------------------------------------------------------------
212 void
213 Quad_JSON::APL_to_JSON_string(UCS_string & result, const Value & B,
214  bool level, bool sorted)
215 {
216  if (B.is_scalar()) // number or literal
217  {
218  APL_to_JSON_string(result, B.get_cfirst(), level, sorted);
219  return;
220  }
221 
222  if (B.is_char_vector())
223  {
224  const UCS_string ucs_string(B);
225  result += UNI_DOUBLE_QUOTE;
226  loop(u, ucs_string.size())
227  {
228  switch(const Unicode uni = ucs_string[u])
229  {
230  case UNI_BS: result.append_UTF8("\\b"); break;
231  case UNI_HT: result.append_UTF8("\\t"); break;
232  case UNI_LF: result.append_UTF8("\\n"); break;
233  case UNI_FF: result.append_UTF8("\\f"); break;
234  case UNI_CR: result.append_UTF8("\\r"); break;
235  case UNI_DOUBLE_QUOTE: result.append_UTF8("\\\""); break;
236 
237  default: if (uni < UNI_SPACE)
238  {
239  char cc[10];
240  snprintf(cc, sizeof(cc), "\\u%4.4X",
241  int(uni));
242  result.append_UTF8(cc);
243  }
244  else
245  {
246  result += uni;
247  }
248  }
249  }
250  result += UNI_DOUBLE_QUOTE;
251  return;
252  }
253 
254  if (B.get_rank() == 1) // JSON array
255  {
256  UCS_string array("[ ");
257  const ShapeItem ec = B.element_count();
258  loop(e, ec)
259  {
260  if (e) array.append(", ");
261  if (array.size() > 60)
262  {
263  array.back() = UNI_LF;
264  result.append(array);
265  array = UCS_string(2*level + 2, UNI_SPACE);
266  }
267 
268  APL_to_JSON_string(array, B.get_cravel(e), level + 1, sorted);
269  if (array.size() == 0) // error in APL_to_JSON_string()
270  {
271  result.clear(); // indicate error
272  return;
273  }
274  }
275  array.append_UTF8(" ]");
276  result.append(array);
277  return;
278  }
279 
280  if (B.is_structured()) // JSON object
281  {
282  std::vector<ShapeItem> member_indices;
283  B.used_members(member_indices, sorted);
284 
285  loop(m, member_indices.size())
286  {
287  const Cell & member_name = B.get_cravel(2*member_indices[m]);
288  const Cell & member_data = B.get_cravel(2*member_indices[m] + 1);
289 
290  result.append(UCS_string(2*level, UNI_SPACE)); // level indent
291  if (m) result.append_UTF8(" \"");
292  else result.append_UTF8("{ \"");
293  UCS_string member(member_name.get_pointer_value().getref());
294  result.append(member_name);
295  result.append_UTF8("\": ");
296  APL_to_JSON_string(result, member_data, level + 1, sorted);
297  if (result.size() == 0) return; // error converting member_data
298  if (size_t(m) < (member_indices.size() - 1))
299  {
300  result += UNI_COMMA;
301  result += UNI_LF;
302  }
303  else
304  {
305  result += UNI_SPACE;
306  }
307  }
308 
309  result.append_UTF8("}");
310  return;
311  }
312 
313  // unexpected rank for a JSON value
314  //
315  MORE_ERROR() << "⎕JSON B: bad rank " << B.get_rank();
316  result.clear(); // indicate error
317 }
318 //-----------------------------------------------------------------------------
319 size_t
320 Quad_JSON::skip_string(const UCS_string & ucs_B, ShapeItem & b)
321 {
322 const ShapeItem B0 = b++; // the leading "
323  Assert(ucs_B[B0] == UNI_DOUBLE_QUOTE);
324 
325 ShapeItem content_len = 0;
326  for (; b < ucs_B.size(); ++b)
327  {
328  const Unicode uni = ucs_B[b];
329  if (uni == UNI_DOUBLE_QUOTE) return content_len;
330 
331  ++content_len;
332  if (uni == UNI_BACKSLASH) // skip the escaped part...
333  {
334  if (ucs_B[b + 1] == UNI_u) // \uUUUU
335  {
336  const Unicode u1 = decode_UUUU(ucs_B, b);
337  if (u1 == Unicode_0) // decode_UUUU() failed
338  {
339  MORE_ERROR() << "⎕JSON B: bad escape sequence "
340  << ucs_B[b ] << ucs_B[b + 1]
341  << ucs_B[b + 2] << ucs_B[b + 3]
342  << ucs_B[b + 4] << ucs_B[b + 5]
343  << " at " << b << "↓B";
344  DOMAIN_ERROR;
345  }
346 
347  if (is_high_surrogate(u1))
348  {
349  const Unicode u2 = decode_UUUU(ucs_B, b + 6);
350  if (u2 == Unicode_0) // decode_UUUU() failed
351  {
352  MORE_ERROR() << "⎕JSON B: bad escape sequence "
353  << ucs_B[b + 6] << ucs_B[b + 7]
354  << ucs_B[b + 8] << ucs_B[b + 9]
355  << ucs_B[b + 10] << ucs_B[b + 11]
356  << " at " << (b + 6) << "↓B";
357  DOMAIN_ERROR;
358  }
359  b += 11;
360  }
361  else // u1 is not a high surrogate (normal \uUUUU)
362  {
363  b += 5;
364  }
365  }
366  else // 1-character escape
367  {
368  ++b; // the escaped character
369  }
370  }
371  }
372 
373  // end of string reached without seeing "
374  //
375  MORE_ERROR() <<
376  "⎕JSON B: No string end for string starting at " << B0 << "↓B";
377  DOMAIN_ERROR;
378 }
379 //-----------------------------------------------------------------------------
380 size_t
381 Quad_JSON::number_len(const UCS_string & ucs_B, ShapeItem b)
382 {
383 const ShapeItem B0 = b;
384 
385  // leading sign
386  if (ucs_B[b] == UNI_MINUS) ++b;
387  else if (ucs_B[b] == UNI_OVERBAR) ++b;
388 
389  // mandatory integral part
390  while (uint32_t(ucs_B[b] - UNI_0) < 10) ++b;
391 
392  if (ucs_B[b] == UNI_FULLSTOP) // optional fractional part
393  {
394  ++b; // skip UNI_FULLSTOP
395  while (uint32_t(ucs_B[b] - UNI_0) < 10) ++b;
396  }
397 
398  if (ucs_B[b] == UNI_E || ucs_B[b] == UNI_e) // optional exponent
399  {
400  ++b; // skip UNI_E / UNI_e
401  if (ucs_B[b] == UNI_MINUS) ++b;
402  else if (ucs_B[b] == UNI_OVERBAR) ++b;
403  while (uint32_t(ucs_B[b] - UNI_0) < 10) ++b;
404  }
405 
406  return b - B0;
407 }
408 //-----------------------------------------------------------------------------
409 Value_P
410 Quad_JSON::JSON_to_APL(const Value & B)
411 {
412 const ShapeItem len_B = B.element_count();
413 
414 UCS_string ucs_B;
415  ucs_B.reserve(len_B + 1);
416  loop(b, len_B) ucs_B += B.get_cravel(b).get_char_value();
417  ucs_B += Unicode_0; // 0-terminate ucs_B to avoid too many length checks
418 
419  // tokenize ucs_B
420  //
421 std::vector<ShapeItem> tokens_B;
422  tokens_B.reserve(10 + len_B/2);
423 
424  loop(b, ucs_B.size())
425  {
426  const Unicode uni = ucs_B[b];
427  if (uni <= ' ') continue; // skip leading whitespace
428 
429  tokens_B.push_back(b);
430  switch(uni)
431  {
432  case UNI_DOUBLE_QUOTE:
433  skip_string(ucs_B, b);
434  continue;
435 
436  case UNI_COMMA:
437  case UNI_COLON:
438  case UNI_L_BRACK:
439  case UNI_R_BRACK:
440  case UNI_L_CURLY:
441  case UNI_R_CURLY: continue; // the six structural chars
442 
443  case UNI_f: b += 4; continue; // literal false
444  case UNI_n: // literal true
445  case UNI_t: b += 3; continue; // literal null
446 
447  case UNI_MINUS:
448  case UNI_OVERBAR:
449  case UNI_FULLSTOP:
450  case UNI_0 ... UNI_9:
451  b += number_len(ucs_B, b) - 1;
452  continue;
453 
454  default:
455  MORE_ERROR() << "⎕JSON B: Got '" << uni <<
456  "' when expecting a JSON token at " << b << "↓B";
457  DOMAIN_ERROR;
458  MORE_ERROR() << "⎕JSON B: Got '" << uni <<
459  "' when expecting a JSON token at " << b << "↓B";
460  DOMAIN_ERROR;
461  }
462  }
463 
464 Value_P Z(LOC);
465 size_t token0 = 0;
466  parse_value(Z.getref(), ucs_B, tokens_B, token0);
467  Z->check_value(LOC);
468 
469  if (token0 != tokens_B.size())
470  {
471  MORE_ERROR() <<
472  "⎕JSON B: there were extra token in B (tokenized: " << tokens_B.size()
473  << ", but processed: " << token0 <<
474  ").\n The JSON string must be one serialized value.";
475  LENGTH_ERROR;
476  }
477 
478  if (Z->is_simple_scalar())
479  {
480  Assert(Z->get_cfirst().is_numeric());
481  return Z; // number
482  }
483 
484  Assert(Z->is_scalar());
485  Assert(Z->get_cfirst().is_pointer_cell());
486  return Z->get_cfirst().get_pointer_value();
487 }
488 //-----------------------------------------------------------------------------
489 void
490 Quad_JSON::parse_value(Value & Z, const UCS_string & ucs_B,
491  const std::vector<ShapeItem> & tokens_B, size_t & token0)
492 {
493 const ShapeItem b = tokens_B.at(token0);
494  switch(ucs_B[b])
495  {
496  // unexpected token...
497  //
498  case UNI_COMMA:
499  MORE_ERROR() << "⎕JSON: Got ',' when expecting a value";
500  DOMAIN_ERROR;
501 
502  case UNI_COLON:
503  MORE_ERROR() << "⎕JSON: Got ':' when expecting a value";
504  DOMAIN_ERROR;
505 
506  case UNI_R_BRACK:
507  MORE_ERROR() << "⎕JSON: Got ']' when expecting a value";
508  DOMAIN_ERROR;
509 
510  case UNI_R_CURLY:
511  MORE_ERROR() << "⎕JSON: Got '}' when expecting a value";
512  DOMAIN_ERROR;
513 
514  // single token (we increment token0)...
515  //
516  case '-':
517  case UNI_OVERBAR:
518  case UNI_FULLSTOP:
519  case '0' ... '9': parse_number(Z, ucs_B, b); ++token0; break;
520  case UNI_DOUBLE_QUOTE: parse_string(Z, ucs_B, b); ++token0; break;
521  case UNI_f: parse_literal(Z, ucs_B, b, "false"); ++token0; break;
522  case UNI_n: parse_literal(Z, ucs_B, b, "null"); ++token0; break;
523  case UNI_t: parse_literal(Z, ucs_B, b, "true"); ++token0; break;
524 
525  // multi token (they increment token0)...
526  //
527  case UNI_L_BRACK: parse_array (Z, ucs_B, tokens_B, token0); break;
528  case UNI_L_CURLY: parse_object(Z, ucs_B, tokens_B, token0); break;
529 
530  default: FIXME;
531  }
532 }
533 //-----------------------------------------------------------------------------
534 void
535 Quad_JSON::parse_array(Value & Z, const UCS_string & ucs_B,
536  const std::vector<ShapeItem> & tokens_B, size_t & token0)
537 {
538 size_t token_from = token0;
539  Assert(ucs_B[tokens_B[token_from]] == UNI_L_BRACK); // [
540 
541 const size_t commas = comma_count(ucs_B, tokens_B, token0);
542  Assert(ucs_B[tokens_B[token0]] == UNI_R_BRACK); // always ]
543 
544  ++token_from; // skip [
545  if (commas == 0) // [ ] or [ item ]
546  {
547  if ((token0 - token_from) == 0) // [ ]
548  {
549  // CERR << "empty ARRAY" << std::endl;
550  Value_P Zsub = Idx0(LOC);
551  Z.next_ravel_Pointer(Zsub.get());
552  }
553  else
554  {
555  // CERR << "One element ARRAY" << std::endl;
556  Value_P Zsub(1, LOC);
557  parse_value(Zsub.getref(), ucs_B, tokens_B, token_from);
558  Zsub->check_value(LOC);
559  Z.next_ravel_Value(Zsub.get());
560  }
561  ++token_from; // skip ]
562  }
563  else // [ item , item... ]
564  {
565  const size_t len = commas + 1;
566  // CERR << "ARRAY with " << len << " elements" << endl;
567 
568  Value_P Zsub(len, LOC);
569  loop(l, len)
570  {
571  parse_value(Zsub.getref(), ucs_B, tokens_B, token_from);
572  const Unicode uni = ucs_B[tokens_B[token_from]];
573  if (uni == UNI_COMMA) ++token_from;
574  else if (uni == UNI_R_BRACK) ++token_from;
575  else
576  {
577  MORE_ERROR() << "⎕JSON B: Got '" << uni
578  << "' when expecting ',' or ']' at "
579  << token_from << "↓B";
580  DOMAIN_ERROR;
581  }
582  }
583  Zsub->check_value(LOC);
584  Z.next_ravel_Value(Zsub.get());
585  }
586 
587  ++token0; // skip final ]
588  Assert(token0 == token_from);
589 }
590 //-----------------------------------------------------------------------------
591 void
592 Quad_JSON::parse_object(Value & Z, const UCS_string & ucs_B,
593  const std::vector<ShapeItem> & tokens_B,
594  size_t & token0)
595 {
596 size_t token_from = token0;
597  Assert(ucs_B[tokens_B[token_from]] == UNI_L_CURLY); // {
598 
599 const size_t commas = comma_count(ucs_B, tokens_B, token0);
600  Assert(ucs_B[tokens_B[token0]] == UNI_R_CURLY); // always }
601 
602 Value_P assoc_array = EmptyStruct(LOC);
603  Z.next_ravel_Pointer(assoc_array.get());
604 
605  ++token_from; // skip {
606  if (commas == 0) // { } or { 'name' : value }
607  {
608  if ((token0 - token_from) == 0) // { }
609  {
610  // CERR << "empty OBJECT" << std::endl;
611  ++token_from; // skip }
612  }
613  else
614  {
615  // CERR << "One element OBJECT" << std::endl;
616  parse_object_member(assoc_array.getref(), ucs_B, tokens_B,
617  token_from);
618  }
619  }
620  else // { 'name' : value , 'name' : value... }
621  {
622  const size_t items = commas + 1;
623  // CERR << "OBJECT with " << items << " elements" << endl;
624 
625  loop(it, items)
626  {
627  parse_object_member(assoc_array.getref(), ucs_B, tokens_B,
628  token_from);
629  }
630  }
631 
632  ++token0; // skip final }
633  Assert(token0 == token_from);
634 }
635 //-----------------------------------------------------------------------------
636 void
637 Quad_JSON::parse_object_member(Value & Z, const UCS_string & ucs_B,
638  const std::vector<ShapeItem> & tokens_B,
639  size_t & token_from)
640 {
641 const size_t B_start = tokens_B[token_from];
642  if (ucs_B[B_start] != UNI_DOUBLE_QUOTE)
643  {
644  MORE_ERROR() << "⎕JSON B: Got '" << ucs_B[B_start] <<
645  "' when expecting (double-quoted) object member name at " <<
646  B_start << "↓B";
647  DOMAIN_ERROR;
648  }
649  ++token_from; // skip "
650 
651  // parse "member-name" :
652  //
653 UCS_string member_name;
654  {
655  const ShapeItem ucs_B_name = B_start + 1;
656 
657  const ShapeItem ucs_B_colon = tokens_B[token_from];
658  if (ucs_B[ucs_B_colon] != UNI_COLON)
659  {
660  MORE_ERROR() <<
661  "⎕JSON B: Got '" << ucs_B[ucs_B_colon] <<
662  "' when expecting ':' after object member name at " <<
663  B_start << "↓B";
664  DOMAIN_ERROR;
665  }
666  ++token_from; // skip :
667 
668  // find trailing " of "member-name" : by searching backwards from ':'
669  //
670  ShapeItem name_len = ucs_B_colon - ucs_B_name;
671  while (name_len && ucs_B[ucs_B_name + name_len - 1] != UNI_DOUBLE_QUOTE)
672  --name_len;
673  --name_len; // skip the trailing "
674  if (name_len < 1)
675  {
676  FIXME; // since skip_string() should have failed
677  }

716 //-----------------------------------------------------------------------------
717 void
718 Quad_JSON::parse_number(Value & Z, const UCS_string & ucs_B, ShapeItem b)
719 {
720  enum { MAX_NUMLEN = 90 };
721 char cc[MAX_NUMLEN + 10];
722 size_t cc_len = 0;
723 bool need_fract = false;
724 bool have_expo = false;
725 double dval = 0;
726 
727  if (ucs_B[b] == UNI_MINUS || ucs_B[b] == UNI_OVERBAR)
728  { ++b; cc[cc_len++] = '-'; }
729 
730  // copy integer part to cc...
731  //
732  while (ucs_B[b] >= UNI_0 && ucs_B[b] <= UNI_9)
733  {
734  if (cc_len >= MAX_NUMLEN) goto number_too_long;
735  cc[cc_len++] = ucs_B[b++];
736  }
737 
738  // maybe copy fractional part to cc...
739  //
740  if (ucs_B[b] == UNI_FULLSTOP) // fractional part
741  {
742  need_fract = true; // force FloatCell
743  if (cc_len >= MAX_NUMLEN) goto number_too_long;
744  cc[cc_len++] = '.';
745  ++b;
746  while (ucs_B[b] >= UNI_0 && ucs_B[b] <= UNI_9)
747  {
748  if (cc_len >= MAX_NUMLEN) goto number_too_long;
749  cc[cc_len++] = ucs_B[b++];
750  }
751  }
752 
753  // maybe copy exponent part to cc...
754  //
755  if (ucs_B[b] == UNI_E || ucs_B[b] == UNI_e) // exponent part
756  {
757  have_expo = true;
758  if (cc_len >= MAX_NUMLEN) goto number_too_long;
759  cc[cc_len++] = 'e';
760  ++b;
761  if (ucs_B[b] == UNI_MINUS || ucs_B[b] == UNI_OVERBAR) // negative expo
762  {
763  need_fract = true;
764  ++b; cc[cc_len++] = '-'; }
765  while (ucs_B[b] >= UNI_0 && ucs_B[b] <= UNI_9)
766  {
767  if (cc_len >= MAX_NUMLEN) goto number_too_long;
768  cc[cc_len++] = ucs_B[b++];
769  }
770  }
771  cc[cc_len++] = 0;
772 
773  // always scan the double value
774  //
775  dval = strtod(cc, 0);
776 
777  if (dval > LARGE_INT || dval < SMALL_INT) need_fract = true;
778 
779  if (need_fract)
780  {
781  Z.next_ravel_Float(dval);
782  return;
783  }
784  else if (have_expo) // so strtoll wont work
785  {
786  if (dval < 0) Z.next_ravel_Int(dval - 0.5); // round negative → 0
787  else Z.next_ravel_Int(dval + 0.5); // round positive → 0
788  return;
789  }
790  else
791  {
792  const long long int ival = strtoll(cc, 0, 10);
793  Z.next_ravel_Int(ival);
794  return;
795  }
796 
797 number_too_long:
798  MORE_ERROR() << "⎕JSON B: number too long at " << b
799  << "↓B (max. length is " << MAX_NUMLEN << ")";
800  DOMAIN_ERROR;
801 }
802 //-----------------------------------------------------------------------------
803 void
804 Quad_JSON::parse_string(Value & Z, const UCS_string & ucs_B, ShapeItem b)
805 {
806 ShapeItem bb = b + 1; // NOTE: skip_string() will increment b
807 const ShapeItem content_len = skip_string(ucs_B, b);
808 
809 Value_P Zsub(content_len, LOC);
810  loop(l, content_len)
811  {
812  Unicode uni = ucs_B[bb++];
813  if (uni == UNI_BACKSLASH)
814  {
815  Unicode surr = Unicode_0;
816 
817  uni = ucs_B[bb++]; // default: escaped = character itself
818  switch(uni)
819  {
820  case UNI_b: uni = UNI_BS; break; // \b
821  case UNI_f: uni = UNI_FF; break; // \f
822  case UNI_n: uni = UNI_LF; break; // \n
823  case UNI_r: uni = UNI_CR; break; // \r
824  case UNI_t: uni = UNI_HT; break; // \t
825  case UNI_u: // \uUUUU
826  uni = decode_UUUU(ucs_B, bb - 2);
827  surr = decode_UUUU(ucs_B, bb + 4);
828  if (is_high_surrogate(uni) && is_low_surrogate(surr))
829  {
830  uni = Unicode(0x10000 + (surr & 0x03FF)
831  + ((uni & 0x03FF) << 10));
832  bb += 10;
833  }
834  else if (is_high_surrogate(uni))
835  {
836  MORE_ERROR() <<
837  "⎕JSON B: No low surrogate at " << bb << "↓B";
838  DOMAIN_ERROR;
839  }
840  else if (uni == Unicode_0) // decode_UUUU() failed
841  {
842  FIXME; // since skip_string() should have failed
843  }
844  else // normal (non-surrogate) \uUUUU
845  {
846  bb += 4;
847  }
848  break; // case UNI_u
849 
850  default: break;
851  }
852  }
853 
854  Zsub->next_ravel_Char(uni);
855  }
856 
857  Zsub->check_value(LOC);
858  Z.next_ravel_Pointer(Zsub.get());
859 }
860 //-----------------------------------------------------------------------------
861 Unicode
862 Quad_JSON::decode_UUUU(const UCS_string & ucs_B, ShapeItem b)
863 {
864  if (ucs_B[b++] != UNI_BACKSLASH) return Unicode_0;
865  if (ucs_B[b++] != UNI_u) return Unicode_0;
866 
867 char cc[5];
868  cc[0] = ucs_B[b++];
869  cc[1] = ucs_B[b++];
870  cc[2] = ucs_B[b++];
871  cc[3] = ucs_B[b++];
872  cc[4] = 0;
873 
874  return Unicode(strtol(cc, 0, 16));
875 }

905 size_t
906 Quad_JSON::comma_count(const UCS_string & ucs_B,
907  const std::vector<ShapeItem> & tokens_B, size_t & token0)
908 {
909 const Unicode start = ucs_B[tokens_B[token0]];
910 Unicode end;
911  if (start == UNI_L_BRACK) end = UNI_R_BRACK;
912  else if (start == UNI_L_CURLY) end = UNI_R_CURLY;
913  else FIXME;
914 
915 UCS_string stack(end);
916 size_t commas = 0;
917 bool expect_comma = false;
918 bool expect_colon = true;
919 
920  for (++token0; token0 < tokens_B.size(); ++token0)
921  {
922  const Unicode uni = ucs_B[tokens_B[token0]];
923  switch(uni)
924  {
925  case UNI_L_BRACK:
926  stack += UNI_R_BRACK;
927  continue;
928 
929  case UNI_L_CURLY:
930  stack += UNI_R_CURLY;
931  continue;
932 
933  case UNI_COMMA:
934  if (stack.size() > 1) continue;
935  if (!expect_comma)
936  {
937  MORE_ERROR() << "⎕JSON B: Got unexpected ',' at "
938  << tokens_B[token0] << "↓B";
939  DOMAIN_ERROR;
940  }
941 
942  ++commas;
943  expect_comma = false;
944  expect_colon = true;
945  continue;
946 
947  case UNI_COLON:
948  if (stack.size() > 1) continue;
949  if (!expect_comma)
950  {
951  MORE_ERROR() << "⎕JSON B: Got unexpected ':' at "
952  << tokens_B[token0] << "↓B";
953  DOMAIN_ERROR;
954  }
955 
956  expect_comma = false;
957  if (!expect_colon)
958  {
959  MORE_ERROR() << "⎕JSON B: Got ':' "
960  "instead of ',' at "
961  << tokens_B[token0] << "↓B";
962  DOMAIN_ERROR;
963  }
964  expect_colon = false;
965  continue;
966 
967  case UNI_R_BRACK:
968  //
969  // UNI_R_BRACK must match stack.back()
970  //
971  if (stack.back() != UNI_R_BRACK)
972  {
973  MORE_ERROR() << "Mismatch ]";
974  DOMAIN_ERROR;
975  }
976  stack.pop_back();
977  if (stack.size() == 0) return commas;
978  continue;
979 
980  case UNI_R_CURLY:
981  //
982  // UNI_R_CURLY must match stack.back()
983  //
984  if (stack.back() != UNI_R_CURLY)
985  {
986  MORE_ERROR() << "Mismatch ]";
987  DOMAIN_ERROR;
988  }
989  stack.pop_back();
990  if (stack.size() == 0) return commas;
991  continue;
992 
993  default:
994  if (stack.size() > 1) continue; // not of interest
995  if (expect_comma)
996  {
997  MORE_ERROR() << "⎕JSON B: Got '" << uni
998  << "' when expecting ',' at "
999  << tokens_B[token0] << "↓B";
1000  DOMAIN_ERROR;
1001  }
1002  expect_comma = true;
1003  }
1004 
1005  }
1006 
1007  MORE_ERROR() << "⎕JSON B: No matching " << end << " for " << start
1008  << " at " << tokens_B[token0] << "↓B ";

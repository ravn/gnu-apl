91 Token
92 Quad_AF::eval_B(Value_P B)
93 {
94 const ShapeItem ec = B->element_count();
95 Value_P Z(B->get_shape(), LOC);
96 
97  loop(v, ec)
98  {
99  const Cell & cell_B = B->get_ravel(v);
100 
101  if (cell_B.is_character_cell()) // Unicode to AV index
102  {
103  const Unicode uni = cell_B.get_char_value();
104  int32_t pos = Avec::find_av_pos(uni);
105  if (pos < 0) Z->next_ravel_Int(MAX_AV);
106  else Z->next_ravel_Int(pos);
107  continue;
108  }
109 
110  if (cell_B.is_integer_cell())
111  {
112  const APL_Integer idx = cell_B.get_near_int();
113  Z->next_ravel_Char(Quad_AV::indexed_at(idx));
114  continue;
115  }
116 
117  DOMAIN_ERROR;
118  }
119 
120  Z->check_value(LOC);
121  return Token(TOK_APL_VALUE1, Z);
122 }

124 Token
125 Quad_AT::eval_AB(Value_P A, Value_P B)
126 {
127  // A should be an integer scalar 1, 2, 3, or 4
128  //
129  if (A->get_rank() > 0) RANK_ERROR;
130 const APL_Integer mode = A->get_ravel(0).get_near_int();
131  if (mode < 1) DOMAIN_ERROR;
132  if (mode > 4) DOMAIN_ERROR;
133 
134 const ShapeItem cols = B->get_cols();
135 const ShapeItem rows = B->get_rows();
136  if (rows == 0) LENGTH_ERROR;
137 
138 const int mode_vec[] = { 3, 7, 4, 2 };
139 const int mode_len = mode_vec[mode - 1];
140 Shape shape_Z(rows);
141  shape_Z.add_shape_item(mode_len);
142 
143 Value_P Z(shape_Z, LOC);
144 
145  loop(r, rows)
146  {
147  // get the symbol name by stripping trailing spaces
148  //
149  const ShapeItem b = r*cols; // start of the symbol name
150  UCS_string symbol_name;
151  loop(c, cols)
152  {
153  const Unicode uni = B->get_ravel(b + c).get_char_value();
154  if (uni == UNI_ASCII_SPACE) break;
155  symbol_name.append(uni);
156  }
157 
158  NamedObject * obj = Workspace::lookup_existing_name(symbol_name);
159  if (obj == 0) Error::throw_symbol_error(symbol_name, LOC);
160 
161  const Function * function = obj->get_function();
162  if (function) // user defined or system function.
163  {
164  function->get_attributes(mode, &Z->get_ravel(r*mode_len));
165  continue;
166  }
167 
168  Symbol * symbol = obj->get_symbol();
169  if (symbol) // user defined or system var.
170  {
171  symbol->get_attributes(mode, &Z->get_ravel(r*mode_len));
172  continue;
173  }
174 
175  // neither function nor variable (e.g. unused name)
176  VALUE_ERROR;
177 
178  if (Avec::is_quad(symbol_name[0])) // system function or variable
179  {
180  int l;
181  const Token t(Workspace::get_quad(symbol_name, l));
182  if (t.get_Class() == TC_SYMBOL) // system variable
183  {
184  // Assert() if symbol_name is not a function
185  t.get_sym_ptr();
186  }
187  else
188  {
189  // throws SYNTAX_ERROR if t is not a function
190  t.get_function();
191  }
192  }
193  else // user defined
194  {
195  Symbol * symbol = Workspace::lookup_existing_symbol(symbol_name);
196  if (symbol == 0) VALUE_ERROR;
197 
198  symbol->get_attributes(mode, &Z->get_ravel(r*mode_len));
199  }
200  }
201 
202  Z->check_value(LOC);
203  return Token(TOK_APL_VALUE1, Z);
204 }

206 Token
207 Quad_DL::eval_B(Value_P B)
208 {
209 const APL_time_us start = now();
210 
211  // B should be an integer or real scalar
212  //
213  if (B->get_rank() > 0) RANK_ERROR;
214  if (!B->get_ravel(0).is_real_cell()) DOMAIN_ERROR;
215 
216 const APL_time_us end = start + 1000000 * B->get_ravel(0).get_real_value();
217  if (end < start) DOMAIN_ERROR;
218  if (end > start + 31*24*60*60*1000000LL) DOMAIN_ERROR; // > 1 month
219 
220  for (;;)
221  {
222  // compute time remaining.
223  //
224  const APL_time_us remaining_us = end - now();
225  if (remaining_us <= 0) break;
226 
227  const int wait_sec = remaining_us/1000000;
228  const int wait_usec = remaining_us%1000000;
229  timeval tv = { wait_sec, wait_usec };
230  if (select(0, 0, 0, 0, &tv) == 0) break;
231  }
232 
233  // return time elapsed.
234  //
235 Value_P Z(LOC);
236  Z->next_ravel_Float(0.000001*(now() - start));
237 
238  Z->check_value(LOC);
239  return Token(TOK_APL_VALUE1, Z);
240 }

340 void
341 Quad_EC::eoc(Token & result)
342 {
343  // set Token result to an APL value Z = (Z1 Z2 Z3) where:
344  //
345  // Z1 is an integer scalar 0-5 (return code):
346  //
347  // 0: Error
348  // 1: Value
349  // 2: Committed Value
350  // 3: Void
351  // 4: Branch
352  // 5: Escape
353  //
354  // Z2 is a two-element integer vector with the value that ⎕ET would have,
355  // ⎕ET is not set while in ⎕EC.
356  //
357  // Z3 is a value that depends on Z1:
358  //
359  // 0: ⎕EM
360  // 1 or 2: the value
361  // 3 or 5: 0⍴0
362  // 4: the branch line
363 
364 Value_P Z(3, LOC);
365  if (result.get_tag() == TOK_ERROR)
366  {
367  StateIndicator * si = Workspace::SI_top();
368  si->clear_safe_execution();
369 
370  const Error & err = StateIndicator::get_error(si);
371  const ErrorCode ec = ErrorCode(result.get_int_val());
372 
373  PrintBuffer pb;
374  pb.append_ucs(Error::error_name(ec));
375  pb.append_ucs(UCS_string(UTF8_string(err.get_error_line_2())));
376  pb.append_ucs(err.get_error_line_3());
377 
378  Value_P Z2(2, LOC);
379  Z2->next_ravel_Int(Error::error_major(ec));
380  Z2->next_ravel_Int(Error::error_minor(ec));
381  Z2->check_value(LOC);
382 
383  Value_P Z3(pb, LOC); // 3 line message like ⎕EM
384 
385  Z->next_ravel_Int(0);
386  Z->next_ravel_Pointer(Z2.get());
387  Z->next_ravel_Pointer(Z3.get());
388 
389  Z->check_value(LOC);
390  result.move_2(Token(TOK_APL_VALUE1, Z), LOC);
391  return;
392  }
393 
394  // all other cases have Z2 = 0 0
395  //
396 Value_P Z2(2, LOC);
397  Z2->next_ravel_Int(0);
398  Z2->next_ravel_Int(0);
399  Z2->check_value(LOC);
400 
401  switch(result.get_tag())
402  {
403  case TOK_APL_VALUE1:
404  case TOK_APL_VALUE3:
405  Z->next_ravel_Int(1);
406  Z->next_ravel_Pointer(Z2.get());
407  Z->next_ravel()->init_from_value(result.get_apl_val().get(),
408  Z.getref(), LOC);
409  break;
410 
411  case TOK_APL_VALUE2:
412  Z->next_ravel_Int(2);
413  Z->next_ravel_Pointer(Z2.get());
414  Z->next_ravel()->init_from_value(result.get_apl_val().get(),
415  Z.getref(), LOC);
416  break;
417 
418  case TOK_NO_VALUE:
419  case TOK_VOID:
420  Z->next_ravel_Int(3);
421  Z->next_ravel_Pointer(Z2.get());
422  Z->next_ravel_Pointer(Idx0_0(LOC).get());
423  break;
424 
425  case TOK_BRANCH:
426  Z->next_ravel_Int(4);
427  Z->next_ravel_Pointer(Z2.get());
428  Z->next_ravel_Int(result.get_int_val());
429  break;
430 
431  case TOK_ESCAPE:
432  Z->next_ravel_Int(5);
433  Z->next_ravel_Pointer(Z2.get());
434  Z->next_ravel_Pointer(Idx0_0(LOC).get());
435  break;
436 
437  default: CERR << "unexpected result tag " << result.get_tag()
438  << " in Quad_EC::eoc()" << endl;
439  Assert(0);
440  }
441 
442  Z->check_value(LOC);
443  result.move_2(Token(TOK_APL_VALUE1, Z), LOC);
444 }

446 Token
447 Quad_ENV::eval_B(Value_P B)
448 {
449  if (!B->is_char_string()) DOMAIN_ERROR;
450 
451 const ShapeItem ec_B = B->element_count();
452 
453 std::vector<const char *> evars;
454 
455  for (char **e = environ; *e; ++e)
456  {
457  const char * env = *e;
458 
459  // check if env starts with B.
460  //
461  bool match = true;
462  loop(b, ec_B)
463  {
464  if (B->get_ravel(b).get_char_value() != Unicode(env[b]))
465  {
466  match = false;
467  break;
468  }
469  }
470 
471  if (match) evars.push_back(env);
472  }
473 
474 const Shape sh_Z(evars.size(), 2);
475 Value_P Z(sh_Z, LOC);
476 
477  loop(e, evars.size())
478  {
479  const char * env = evars[e];
480  UCS_string ucs;
481  while (*env)
482  {
483  if (*env != '=') ucs.append(Unicode(*env++));
484  else break;
485  }
486  ++env; // skip '='
487 
488  Value_P varname(ucs, LOC);
489 
490  ucs.clear();
491  while (*env) ucs.append(Unicode(*env++));
492 
493  Value_P varval(ucs, LOC);
494 
495  Z->next_ravel_Pointer(varname.get());
496  Z->next_ravel_Pointer(varval.get());
497  }
498 
499  Z->set_default_Spc();
500  Z->check_value(LOC);
501  return Token(TOK_APL_VALUE1, Z);
502 }

525 //-----------------------------------------------------------------------------
526 Token
527 Quad_ES::event_simulate(const UCS_string * A, Value_P B, Error & error)
528 {
529  // B is empty: no action
530  //
531  if (B->element_count() == 0) return Token();
532 
533 const ErrorCode ec = get_error_code(B);
534  if (ec == E_QUAD_ES_BRA) return Token(TOK_QUAD_ES_BRA, B->clone(LOC));
535  if (ec == E_QUAD_ES_COM) return Token(TOK_QUAD_ES_COM, B->clone(LOC));
536  if (ec == E_QUAD_ES_ERR) return Token(TOK_QUAD_ES_ERR, B->clone(LOC));
537  if (ec == E_QUAD_ES_ESC) return Token(TOK_QUAD_ES_ESC, B->clone(LOC));
538 
539  new (&error) Error(ec, error.get_throw_loc());
540 
541  if (error.get_error_code() == E_NO_ERROR) // B = 0 0: reset ⎕ET and ⎕EM.
542  {
543  Workspace::clear_error(LOC);
544  return Token();
545  }
546 
547  if (error.get_error_code() == E_ASSERTION_FAILED) // B = 0 ASSERTION_FAILED
548  {
549  Assert(0 && "simulated ASSERTION_FAILED in ⎕ES");
550  }
551 
552  // at this point we shall throw the error. Add some error details.
553  //
554  // set up error message 1
555  //
556  if (A) // A ⎕ES B
557  {
558  UCS_string msg1_ucs(*A);
559  UTF8_string msg1_utf(msg1_ucs);
560  error.set_error_line_1(msg1_utf.c_str());
561  }
562  else if (error.get_error_code() == E_USER_DEFINED_ERROR) // ⎕ES with character B
563  {
564  UCS_string msg1_ucs(*B.get());
565  UTF8_string msg1_utf(msg1_ucs);
566  error.set_error_line_1(msg1_utf.c_str());
567  }
568  else if (error.is_known()) // ⎕ES B with known major/minor B
569  {
570  /* error_message_1 already OK */ ;
571  }
572  else // ⎕ES B with unknown major/minor B
573  {
574  char cc[58];
575  snprintf(cc, sizeof(cc), "Unkown error (major %d, minor %d) in ⎕ES B",
576  error.get_error_code() >> 16, error.get_error_code() & 0xFFFF);
577  error.set_error_line_1(cc);
578  }
579 
580  error.set_show_locked(true);
581 
582  Assert(Workspace::SI_top());
583  if (StateIndicator * si = Workspace::SI_top()->get_parent())
584  {
585  const UserFunction * ufun = si->get_executable()->get_ufun();
586  if (ufun)
587  {
588  // lrm p 282: When ⎕ES is executed from within a defined function
589  // and B is not empty, the event action is generated as though
590  // the function were primitive.
591  //
592  UCS_string ufun_name(" ");
593  ufun_name.append(ufun->get_name());
594  error.set_error_line_2(ufun_name, 6, -1);
595  Workspace::pop_SI(LOC);
596  StateIndicator::get_error(Workspace::SI_top()) = error;
597  error.print_em(UERR, LOC);
598  return Token();
599  }
600  }
601 
602  error.update_error_info(Workspace::SI_top());
603  return Token();
604 }
605 //-----------------------------------------------------------------------------
606 ErrorCode
607 Quad_ES::get_error_code(Value_P B)
608 {
609  // B shall be one of: → Error code
610  //
611  // 1. empty, or → No error
612  // 2. a 2 element integer vector, or → B[0]/N[1]
613  // 3. a simple character vector. → user defined
614  //
615  // Otherwise, throw an error
616 
617  // We extend the standard error codes by a markers E_QUAD_ES_* (10-13) which
618  // are only used by the Macros Z__A_Quad_EA_B and Z__A_Quad_EB_B in order to
619  // return branches and the like as a result.
620 
621  if (B->get_rank() > 1) RANK_ERROR;
622 
623  if (B->element_count() == 0) return E_NO_ERROR;
624  if (B->is_char_string()) return E_USER_DEFINED_ERROR;
625 
626 const APL_Integer err = (B->get_ravel(0).get_near_int() << 16)
627  | (B->get_ravel(1).get_near_int());
628 
629  if ((err >> 16) == (E_QUAD_ES_BRA >> 16)) // one of the ⎕EA or ⎕EB events
630  {
631  const ShapeItem len_B = B->element_count();
632  if (err == E_QUAD_ES_COM && len_B == 3) return E_QUAD_ES_COM;
633  if (err == E_QUAD_ES_ERR && len_B == 5) return E_QUAD_ES_ERR;
634  if (err == E_QUAD_ES_BRA && len_B == 3) return E_QUAD_ES_BRA;
635  if (err == E_QUAD_ES_ESC && len_B == 2) return E_QUAD_ES_ESC;
636  DOMAIN_ERROR;
637  }
638 
639  if (B->element_count() != 2) LENGTH_ERROR;
640 
641  return ErrorCode(err);
642 }

938 //-----------------------------------------------------------------------------
939 void
940 Quad_INP::split_strings()
941 {
942 UCS_string empty;
943  loop(r, raw_lines.size())
944  {
945  UCS_string & line = raw_lines[r];
946  const ShapeItem epos = line.substr_pos(esc1);
947  if (esc1.size() == 0 || epos == -1) // no escape in this line
948  {
949  prefixes.push_back(line);
950  escapes.push_back(empty);
951  suffixes.push_back(empty);
952  continue;
953  }
954 
955  // at this point line did contain an exec string
956  //
957  UCS_string pref = line;
958  pref.resize(epos);
959  prefixes.push_back(pref);
960 
961  line = line.drop(epos + esc1.size()); // skip prefix and esc1
962  if (esc2.size()) // end defined
963  {
964  const ShapeItem eend = line.substr_pos(esc2);
965  if (eend == -1) // no exec end in this line
966  {
967  escapes.push_back(line);
968  suffixes.push_back(empty);
969  continue;
970  }
971  else // found an exec end in this line
972  {
973  UCS_string exec = line;
974  exec.resize(eend);
975  escapes.push_back(exec);
976  line = line.drop(eend + esc2.size()); // skip exec and esc2
977  suffixes.push_back(line);
978  continue;
979  }
980  }
981  else // no end defined
982  {
983  escapes.push_back(line);
984  suffixes.push_back(empty);
985  }
986  }
987 
988  Assert(prefixes.size() >= raw_lines.size());
989  Assert(prefixes.size() == escapes.size());
990  Assert(prefixes.size() == suffixes.size());
991 }

1044  //

1067 Quad_NL::do_quad_NL(Value_P A, Value_P B)
1068 {
1069  if (!!A && !A->is_char_string()) DOMAIN_ERROR;
1070  if (B->get_rank() > 1) RANK_ERROR;
1071  if (B->element_count() == 0) // nothing requested
1072  {
1073  return Token(TOK_APL_VALUE1, Str0_0(LOC));
1074  }
1075 
1076 UCS_string first_chars;
1077  if (!!A) first_chars = UCS_string(*A);
1078 
1079  // 1. create a bitmap of ⎕NC values requested in B
1080  //
1081 int requested_NCs = 0;
1082  {
1083  loop(b, B->element_count())
1084  {
1085  const APL_Integer bb = B->get_ravel(b).get_near_int();
1086  if (bb < 1) DOMAIN_ERROR;
1087  if (bb > 6) DOMAIN_ERROR;
1088  requested_NCs |= 1 << bb;
1089  }
1090  }
1091 
1092  // 2, build a name table, starting with user defined names
1093  //
1094 UCS_string_vector names;
1095  {
1096  std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
1097 
1098  loop(s, symbols.size())
1099  {
1100  const Symbol * symbol = symbols[s];
1101  if (symbol->is_erased()) continue;
1102 
1103  NameClass nc = symbol->get_nc();
1104  if (nc == NC_SHARED_VAR) nc = NC_VARIABLE;
1105 
1106  if (!(requested_NCs & 1 << nc)) continue; // name class not in B
1107 
1108  if (first_chars.size())
1109  {
1110  const Unicode first_char = symbol->get_name()[0];
1111  if (!first_chars.contains(first_char)) continue;
1112  }
1113 
1114  names.push_back(symbol->get_name());
1115  }
1116  }
1117 
1118  // 3, append ⎕-vars and ⎕-functions to name table (unless prevented by A)
1119  //
1120  if (first_chars.size() == 0 || first_chars.contains(UNI_Quad_Quad))
1121  {
1122 #define ro_sv_def(x, _str, _txt) \
1123  { Symbol * symbol = &Workspace::get_v_ ## x(); \
1124  if ((requested_NCs & 1 << 5) && symbol->get_nc() != 0) \
1125  names.push_back(symbol->get_name()); }
1126 
1127 #define rw_sv_def(x, _str, _txt) \
1128  { Symbol * symbol = &Workspace::get_v_ ## x(); \
1129  if ((requested_NCs & 1 << 5) && symbol->get_nc() != 0) \
1130  names.push_back(symbol->get_name()); }
1131 
1132 #define sf_def(x, _str, _txt) \
1133  { if (requested_NCs & 1 << 6) names.push_back((x::fun->get_name())); }
1134 #include "SystemVariable.def"
1135  }
1136 
1137 
1138  // 4. compute length of longest name
1139  //
1140 ShapeItem longest = 0;
1141  loop(n, names.size())
1142  {
1143  if (longest < names[n].size())
1144  longest = names[n].size();
1145  }
1146 
1147 const Shape shZ(names.size(), longest);
1148 Value_P Z(shZ, LOC);
1149

1151  // (or should) not be a perfomance // critical function, so we can
1152  // use a simple (find smallest and print) // approach.
1153  //
1154  for (int count = names.size(); count; --count)
1155  {
1156  // find smallest.
1157  //
1158  uint32_t smallest = count - 1;
1159  loop(t, count - 1)
1160  {
1161  if (names[smallest].compare(names[t]) > 0) smallest = t;
1162  }
1163  // copy name to result, padded with spaces.
1164  //
1165  loop(l, longest)
1166  {

1168  Z->next_ravel_Char(l < ucs.size() ? ucs[l] : UNI_ASCII_SPACE);
1169  }
1170 
1171  // remove smalles from table
1172  //
1173  names[smallest] = names[count - 1];
1174  }
1175

1177  Z->check_value(LOC);
1178  return Token(TOK_APL_VALUE1, Z);

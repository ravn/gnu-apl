36 //-----------------------------------------------------------------------------
37 Token::Token(TokenTag tg, IndexExpr & idx)
38 {
39  if (tg == TOK_PINDEX)
40  {
41  // this token is a partial index in the prefix parser.
42  // use it as is.
43  //
44  tag = TOK_PINDEX;
45  value.index_val = &idx;
46  }
47  else
48  {
49  // this token is a complete index
50  //
51  Assert(tg == TOK_INDEX);
52  if (idx.value_count() < 2) // [idx] or []
53  {
54  tag = TOK_AXES;
55  if (idx.value_count() == 0) // []
56  {
57  new (&value.apl_val) Value_P;
58  }
59  else // [x]
60  {
61  new (&value.apl_val) Value_P(idx.extract_value(0));
62  }
63  }
64  else // [idx; ...]
65  {
66  tag = TOK_INDEX;
67  value.index_val = &idx;
68  }
69  }
70 }

121 //-----------------------------------------------------------------------------
122 ostream &
123 operator << (ostream & out, const Token & token)
124 {
125  if (token.get_tag() == TOK_CHARACTER)
126  {
127  out << "CHAR«" << token.get_char_val() << "»";
128  return out;
129  }
130 
131  if (token.get_tag() == TOK_APL_VALUE1 ||
132  token.get_tag() == TOK_APL_VALUE2 ||
133  token.get_tag() == TOK_APL_VALUE3)
134  {
135  if (token.get_tag() == TOK_APL_VALUE1) out << "VALUE1";
136  else if (token.get_tag() == TOK_APL_VALUE2) out << "VALUE2";
137  else if (token.get_tag() == TOK_APL_VALUE3) out << "VALUE3";
138  else out << "VALUE???";
139  Value_P value = token.get_apl_val();
140  Assert(!!value);
141  const APL_types::Depth depth = value->compute_depth();
142  out << "«";
143  for (APL_types::Depth d = 0; d < depth; ++d) out << "≡";
144 
145  if (value->get_rank()) out << value->get_shape();
146 
147  const PrintContext pctx(PR_APL);
148  PrintBuffer pb(*value, pctx, 0);
149  bool more = pb.get_height() > 1;
150  if (pb.get_height() > 0)
151  {
152  UCS_string ucs = pb.get_line(0).no_pad();
153  if (ucs.size() > 20)
154  {
155  ucs.resize(20);
156  more = true;
157  }
158  out << ucs;
159  }
160  if (more) out << "...";
161  return out << "»";
162  }
163 
164  if (token.get_tag() == TOK_ERROR)
165  {
166  return out << Error::error_name(ErrorCode(token.get_int_val()));
167  }
168 
169  if (token.get_tag() == TOK_BRANCH)
170  {
171  return out << token.get_Id() << token.get_int_val();
172  }
173 
174  if (token.get_tag() == TOK_NOBRANCH)
175  {
176  return out << token.get_Id() << token.get_int_val();
177  }
178 
179  if (token.get_Id() > ID_No_ID2) return out << token.get_Id();
180 
181  switch(token.get_Class())
182  {
183  case TC_END: return out << "END";
184  case TC_RETURN:
185  if (token.get_tag() == TOK_RETURN_EXEC)
186  return out << "RETURN ⍎";
187  if (token.get_tag() == TOK_RETURN_STATS)
188  return out << "RETURN ◊";
189  if (token.get_tag() == TOK_RETURN_VOID)
190  return out << "RETURN ∇FUN";
191  if (token.get_tag() == TOK_RETURN_SYMBOL)
192  return out << "RETURN Z←FUN";
193  return out << "RETURN ???";
194 
195  case TC_VALUE: return token.print_value(out);
196 
197  case TC_INDEX:
198  case TC_PINDEX:
199  if (token.get_tag() == TOK_INDEX)
200  return out << token.get_index_val();
201  else if (token.get_tag() == TOK_AXES)
202  return out << *token.get_axes();
203  else
204  FIXME;
205 
206  case TC_SYMBOL:
207  if (token.get_tag() == TOK_LSYMB)
208  {
209  token.get_sym_ptr()->print(out << "'");
210  out << UNI_LEFT_ARROW;
211  }
212  else if (token.get_tag() == TOK_LSYMB2)
213  {
214  token.get_sym_ptr()->print(out << "'(... ");
215  out << ")←";
216  }
217  else
218  {
219  token.get_sym_ptr()->print(out << "'");
220  }
221  return out;
222 
223  case TC_FUN0:
224  case TC_FUN12:
225  case TC_OPER1:
226  case TC_OPER2: return token.print_function(out);
227  case TC_LINE: return out << "\n[" << token.get_fun_line() << "] ";
228  case TC_NUMERIC: return out << "Numeric";
229  case TC_NEWLINE: return out << "{LFeed}";
230 
231  default: break;
232  }
233 
234  return out << "{-unknown Token " << token.get_tag() << "-}";
235 }

387 //-----------------------------------------------------------------------------
388 ostream &
389 Token::print_value(ostream & out) const
390 {
391  switch(tag)
392  {
393  case TOK_VARIABLE: return value.sym_ptr->print(out);
394  case TOK_FUN0: return out << "{ fun/0 }";
395  case TOK_CHARACTER: return out << value.char_val;
396  case TOK_INTEGER: return out << value.int_vals[0];
397  case TOK_REAL: return out << value.float_vals[0];
398 
399  case TOK_COMPLEX: return out << value.float_vals[0] << "J"
400  << value.float_vals[1];
401 
402  case TOK_APL_VALUE1:
403  case TOK_APL_VALUE3:
404  {
405  const Value * v = value.apl_val.get();
406  if (v->get_rank() == 0) out << "''";
407  loop(r, v->get_rank())
408  {
409  if (r) out << " ";
410  out << v->get_shape_item(r);
411  }
412  out << UNI_RHO;
413  loop(e, v->element_count())
414  {
415  if (e == 4)
416  {
417  out << "...";
418  break;
419  }
420 
421  const PrintContext pctx(PR_APL, 2, 80);
422  out << UCS_string(v->get_ravel(e)
423  .character_representation(pctx), 0, 80)
424  << " ";
425  }
426  return out;
427  }
428 
429  case TOK_Quad_Quad: return out << UNI_Quad_Quad;
430 
431  case TOK_Quad_AI:

433  case TOK_Quad_CT:

463 //-----------------------------------------------------------------------------
464 void
465 Token::show_trace(ostream & out, const UCS_string & fun_name,
466  Function_Line line) const
467 {
468 UCS_string fn = fun_name;
469  fn.append_UTF8("[");
470  fn.append_number(line);
471  fn.append_UTF8("] ");
472 
473  out << fn;
474 
475  switch(get_tag())
476  {
477  case TOK_APL_VALUE1:
478  case TOK_APL_VALUE2:
479  case TOK_APL_VALUE3:
480  break; // continue below
481 
482  case TOK_BRANCH:
483  out << "→" << get_int_val() << endl;
484  return;
485 
486  case TOK_NOBRANCH:
487  out << "→⍬" << endl;
488  return;
489 
490  case TOK_ESCAPE:
491  out << "→" << endl;
492  return;
493 
494  case TOK_VOID:
495  case TOK_NO_VALUE:
496  out << endl;
497  return;
498 
499  default: Q1(*this)
500  FIXME;
501  }
502 
503  // print a value
504 
505 PrintContext pctx = Workspace::get_PrintContext(PR_APL);
506 const Value & val = *get_apl_val();
507  if (val.get_rank() == 0) // scalar
508  {
509  pctx.set_style(PR_APL_MIN);
510  }
511  else if (val.get_rank() == 1) // vector
512  {
513  if (val.element_count() == 0 && // empty vector
514  val.get_ravel(0).is_simple_cell())
515  {
516  out << endl;
517  return;
518  }
519 
520  pctx.set_style(PR_APL_MIN);
521  }
522  else // matrix or higher
523  {
524  pctx.set_style(PrintStyle(pctx.get_style() | PST_NO_FRACT_0));
525  }
526 
527 PrintBuffer pb(val, pctx, 0);
528 const UCS_string indent(fn.size(), UNI_ASCII_SPACE);
529  loop(l, pb.get_height())
530  {
531  if (l) out << indent;
532  out << pb.get_line(l).no_pad() << endl;
533  }
534 
535  if (pb.get_height() == 0) out << endl;
536 }

543 //-----------------------------------------------------------------------------
544 UCS_string
545 Token::canonical(PrintStyle style) const
546 {
547 UCS_string ucs;
548 
549  switch(get_Class())
550  {
551  case TC_ASSIGN:
552  ucs.append(UNI_LEFT_ARROW);
553  break;
554 
555  case TC_R_ARROW:
556  ucs.append(UNI_RIGHT_ARROW);
557  break;
558 
559  case TC_L_BRACK:
560  if (get_tag() == TOK_L_BRACK)
561  ucs.append(UNI_ASCII_L_BRACK);
562  else if (get_tag() == TOK_SEMICOL)
563  ucs.append(UNI_ASCII_SEMICOLON);
564  else
565  Assert(0);
566  break;
567 
568  case TC_R_BRACK:
569  ucs.append(UNI_ASCII_R_BRACK);
570  break;
571 
572  case TC_END:
573  ucs.append(UNI_DIAMOND);
574  break;
575 
576  case TC_RETURN: break;
577  case TC_LINE:
578  ucs.append(UNI_ASCII_LF);
579  break;
580 
581  case TC_VALUE:
582  {
583  PrintContext pctx(style, DEFAULT_Quad_PP, DEFAULT_Quad_PW);
584  PrintBuffer pbuf(*get_apl_val(), pctx, 0);
585  if (pbuf.get_height() == 0) return ucs;
586  return pbuf.l1();
587  }
588 
589  case TC_SYMBOL:
590  ucs.append(get_sym_ptr()->get_name());
591  break;
592 
593 
594  case TC_R_PARENT:
595  case TC_L_PARENT:
596  case TC_R_CURLY:
597  case TC_L_CURLY:
598  return ID::get_name_UCS(get_Id());
599 
600  case TC_FUN0:
601  case TC_FUN12:
602  case TC_OPER1:
603  case TC_OPER2:
604  if (get_Id() == ID_No_ID) return get_function()->get_name();
605  return ID::get_name_UCS(get_Id());
606 
607  default:
608  CERR << "Token: " << HEX4(tag) << " " << *this
609  << " at " << LOC << endl;
610  Q1(get_Class())
611  Backtrace::show(__FILE__, __LINE__);
612  FIXME;
613  }
614 
615  return ucs;
616 }

632 //-----------------------------------------------------------------------------
633 int
634 Token::error_info(UCS_string & ucs) const
635 {
636 UCS_string can = canonical(PR_APL_FUN).remove_pad();
637 
638 const Unicode c1 = ucs.back();
639 const Unicode c2 = can.size() ? can[0] : Invalid_Unicode;
640 
641  // conditions when we don't need a space
642  //
643 bool need_space = ! (Avec::no_space_after(c1) || Avec::no_space_before(c2));
644 
645  if (need_space) ucs.append(UNI_ASCII_SPACE);
646 
647  ucs.append(can);
648  return need_space ? -can.size() : can.size();
649 }

487 int
488 main(int argc, const char *argv[])
489 {
490  if (const int ret = init_apl(argc, argv)) return ret;
491 
492  if (uprefs.eval_exprs.size())
493  {
494  loop(e, uprefs.eval_exprs.size())
495  {
496  const char * expr = uprefs.eval_exprs[e];
497  const UTF8_string expr_utf(expr);
498  UCS_string expr_ucs(expr_utf);
499  Command::process_line(expr_ucs);
500  }
501  Command::cmd_OFF(0);
502  return 0;
503  }
504 
505 #if HAVE_PTHREAD_SETNAME_NP
506  pthread_setname_np(pthread_self(), "apl/main");
507 #endif
508 
509  for (;;)
510  {
511  Token t = Workspace::immediate_execution(
512  IO_Files::test_mode == IO_Files::TM_EXIT_AFTER_ERROR);
513  if (t.get_tag() == TOK_OFF) Command::cmd_OFF(0);
514  }
515 
516  return 0;
517 }

35 Token
36 Bif_REDUCE::replicate(Value_P A, Value_P B, uAxis axis)
37 {
38  // turn scalar B into ,B
39  //
40 Shape shape_B = B->get_shape();
41  if (shape_B.get_rank() == 0)
42  {
43  shape_B.add_shape_item(1);
44  axis = 0;
45  }
46 
47  if (A->get_rank() > 1) RANK_ERROR;
48  if (axis >= shape_B.get_rank()) INDEX_ERROR;
49 
50 const ShapeItem len_B = shape_B.get_shape_item(axis);
51 ShapeItem len_A = A->element_count();
52 ShapeItem len_Z = 0;
53 std::vector<ShapeItem> rep_counts;
54  rep_counts.reserve(len_B);
55  if (len_A == 1) // single a -> a a ... a (len_B times)
56  {
57  len_A = len_B;
58  APL_Integer rep_A = A->get_ravel(0).get_near_int();
59  loop(a, len_A) rep_counts.push_back(rep_A);
60  if (rep_A > 0) len_Z = rep_A*len_B;
61  else if (rep_A < 0) len_Z = -rep_A*len_B;
62  }
63  else
64  {
65  ShapeItem geq_A = 0; // number of items >= 0 in A
66  loop(a, len_A)
67  {
68  APL_Integer rep_A = A->get_ravel(a).get_near_int();
69  rep_counts.push_back(rep_A);
70  if (rep_A > 0) { len_Z += rep_A; ++geq_A; }
71  else if (rep_A < 0) len_Z -= rep_A;
72  else ++geq_A;
73  }
74 
75  if (len_B != 1 && geq_A != len_B) LENGTH_ERROR;
76  }
77 
78 Shape shape_Z(shape_B);
79  shape_Z.set_shape_item(axis, len_Z);
80 
81 Value_P Z(shape_Z, LOC);
82 
83 const Shape3 shape_B3(shape_B, axis);
84 
85  loop(h, shape_B3.h())
86  {
87  ShapeItem bm = 0;
88  loop(m, rep_counts.size())
89  {
90  const ShapeItem rep = rep_counts[m];
91 
92  if (rep >= 0) // copy l*rep items
93  {
94  loop(r, rep)
95  loop(l, shape_B3.l())
96  {
97  const ShapeItem src = shape_B3.hml(h, bm, l);
98  Z->next_ravel()->init(B->get_ravel(src), Z.getref(),LOC);
99  }
100  if (shape_B3.m() > 1) ++bm;
101  }
102  else // init l*-rep items with the fill item
103  {
104  loop(r, -rep)
105  loop(l, shape_B3.l())
106  {
107  const ShapeItem src = shape_B3.hml(h, 0, l);
108  Z->next_ravel()
109  ->init_type(B->get_ravel(src), Z.getref(), LOC);
110  }
111 
112  // cB is not incremented when fill item is used.
113  }
114  }
115  }
116 
117  Z->set_default(*B.get(), LOC);
118  Z->check_value(LOC);
119  return Token(TOK_APL_VALUE1, Z);
120 }

122 Token
123 Bif_REDUCE::reduce(Token & _LO, Value_P B, uAxis axis)
124 {
125 Function * LO = _LO.get_function();
126  Assert1(LO);
127  if (!LO->has_result()) DOMAIN_ERROR;
128  if (LO->get_fun_valence() != 2) SYNTAX_ERROR;
129 
130  // if B is a scalar, then Z is B.
131  //
132  if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, B->clone(LOC));
133 
134  if (axis >= B->get_rank()) AXIS_ERROR;
135 
136 const ShapeItem m_len = B->get_shape_item(axis);
137 
138 const Shape shape_Z = B->get_shape().without_axis(axis);
139 
140  if (m_len == 0) // apply the identity function
141  {
142  return LO->eval_identity_fun(B, axis);
143  }
144 
145  if (m_len == 1) return Bif_F12_RHO::do_reshape(shape_Z, *B);
146 
147  // non-trivial reduce (len > 1)
148  //
149 const Shape3 B3(B->get_shape(), axis);
150  if (LO->may_push_SI()) // user defined LO
151  {
152  Value_P X4(4, LOC);
153  new (X4->next_ravel()) IntCell(axis + Workspace::get_IO());
154  new (X4->next_ravel()) IntCell(B3.h());
155  new (X4->next_ravel()) IntCell(B3.m());
156  new (X4->next_ravel()) IntCell(B3.l());
157  X4->check_value(LOC);
158  return Macro::get_macro(Macro::MAC_Z__LO_REDUCE_X4_B)
159  ->eval_LXB(_LO, X4, B);
160  }
161 
162  if (shape_Z.is_empty()) return LO->eval_identity_fun(B, axis);
163 
164 const Shape3 Z3(B3.h(), 1, B3.l());
165  return do_reduce(shape_Z, Z3, B3.m(), LO, B, B->get_shape_item(axis));
166 }

168 Token
169 Bif_REDUCE::reduce_n_wise(Value_P A, Token & _LO, Value_P B, uAxis axis)
170 {
171 Function * LO = _LO.get_function();
172  Assert(LO);
173  if (!LO->has_result()) DOMAIN_ERROR;
174 
175  if (A->element_count() != 1) LENGTH_ERROR;
176 const APL_Integer A0 = A->get_ravel(0).get_int_value();
177 const int n_wise = A0 < 0 ? -A0 : A0; // the number of items (= M1 in iso)
178 
179  if (B->is_scalar())
180  {
181  if (n_wise > 2) DOMAIN_ERROR;
182  if (n_wise == 0)
183  {
184  Token ident = LO->eval_identity_fun(B, axis);
185  Value_P val(2, LOC);
186  val->next_ravel()->init(ident.get_apl_val()->get_ravel(0),
187  val.getref(), LOC);
188  val->next_ravel()->init(ident.get_apl_val()->get_ravel(0),
189  val.getref(), LOC);
190  return Token(TOK_APL_VALUE1, val);
191  }
192 
193  // n_wise is 1 or 2: return (2 - n_wise) ⍴ B
194  //
195  Shape sh;
196  if (n_wise == 2) sh.add_shape_item(0);
197  return Bif_F12_RHO::do_reshape(sh, B.getref());
198  }
199  else
200  {
201  if (n_wise > (1 + B->get_shape_item(axis))) DOMAIN_ERROR;
202  }
203 
204  Assert1(LO);
205 
206  if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, B->clone(LOC));
207 
208  if (axis >= B->get_rank()) AXIS_ERROR;
209 
210  if (n_wise == 0) // apply the identity function
211  {
212  Shape shape_B1 = B->get_shape().insert_axis(axis, 0);
213  shape_B1.increment_shape_item(axis + 1);
214  Value_P val(shape_B1, LOC);
215  val->get_ravel(0).init(B->get_ravel(0), val.getref(), LOC); // prototype
216 
217  Token result = LO->eval_identity_fun(val, axis);
218  return result;
219  }
220 
221  if (n_wise == (1 + B->get_shape_item(axis))) // empty result
222  {
223  // the examples in ISO/IEC 13751 (2000) p. 116, as well as lrm,
224  // suggest an empty result, while the algorithm on pp. 115/116 seems
225  // not to work for n_wise == (1 + B->get_shape_item(axis)).
226  // We follow the examples and lrm.
227  //
228  Shape shape_Z = B->get_shape();
229  shape_Z.set_shape_item(axis, 0);
230  Value_P Z(shape_Z, LOC);
231  Z->get_ravel(0).init(B->get_ravel(0), Z.getref(), LOC);
232  Z->check_value(LOC);
233  return Token(TOK_APL_VALUE1, Z);
234  }
235 
236 Shape shape_Z(B->get_shape());
237  shape_Z.set_shape_item(axis, shape_Z.get_shape_item(axis) - n_wise + 1);
238  if (shape_Z.is_empty()) return LO->eval_identity_fun(B, axis);
239 
240  if (n_wise == 1) return Bif_F12_RHO::do_reshape(shape_Z, *B);
241 
242 const Shape3 Z3(shape_Z, axis);
243 const Shape3 B3(B->get_shape(), axis);
244  if (LO->may_push_SI()) // user defined LO
245  {
246  Value_P A1 = IntScalar(A0 < 0 ? A0 + 1 : 1 - A0, LOC);
247  Value_P vsh_Z(LOC, &shape_Z);
248  Value_P vsh_Z3(LOC, &Z3);
249  Value_P vsh_B3(LOC, &B3);
250  Value_P X4(4, LOC);
251  new (X4->next_ravel()) IntCell(axis + Workspace::get_IO()); // X
252  new (X4->next_ravel()) PointerCell(vsh_Z.get(), X4.getref()); // ⍴Z
253  new (X4->next_ravel()) PointerCell(vsh_Z3.get(), X4.getref()); // ⍴Z3
254  new (X4->next_ravel()) PointerCell(vsh_B3.get(), X4.getref()); // ⍴B3
255  X4->check_value(LOC);
256  if (A0 < 0) return Macro::get_macro(Macro::MAC_Z__nA_LO_REDUCE_X4_B)
257  ->eval_ALXB(A1,_LO,X4,B);
258  else return Macro::get_macro(Macro::MAC_Z__pA_LO_REDUCE_X4_B)
259  ->eval_ALXB(A1,_LO,X4,B);
260  }
261 
262  return do_reduce(shape_Z, Z3, A0, LO, B, B->get_shape_item(axis));
263 }

265 Token
266 Bif_REDUCE::do_reduce(const Shape & shape_Z, const Shape3 & Z3, ShapeItem nwise,
267  Function * LO, Value_P B, ShapeItem bm)
268 {
269 Value_P Z(shape_Z, LOC);
270 
271  // constants that are valid for the entire reduction
272  //
273 const ShapeItem len_Z = Z->element_count();
274 const ShapeItem len_L = Z3.get_last_shape_item();
275 const ShapeItem len_BML = bm * len_L;
276 const ShapeItem len_ZM = Z3.m();
277 const ShapeItem len_ZML = len_L * len_ZM;
278 prim_f2 scalar_LO = LO->get_scalar_f2();
279 
280  loop(z, len_Z)
281  {
282  // break down z into H, M, and L components
283  //
284  const ShapeItem z_L = z % len_L;
285  const ShapeItem z_M = (z / len_L) % len_ZM;
286  const ShapeItem z_H = z / len_ZML;
287 
288  // compute LO_count (= the number of LO-reductions), and b (= the
289  // starting point of the LO-reductions). They depend on z and nwise
290  //
291  const ShapeItem LO_count = (nwise == -1) ? z_M
292  : (nwise < -1) ? - nwise - 1
293  : nwise - 1;
294 
295  ShapeItem b = z_L + z_H * len_BML // start of row in B
296  + LO_count * len_L; // end of beam in B
297 
298  // for reverse (i,e. nwise < -1) reduction direction go to the
299  // start of the beam instead of the end of the beam.
300  //
301  if (nwise < -1) b += (nwise + 1) * len_L;
302  if (nwise != -1) b += z_M * len_L; // start of beam
303 
304  // Z[z] = B[b]
305  //
306  Cell * cZ = Z->next_ravel();
307  cZ->init(B->get_ravel(b), Z.getref(), LOC);
308 
309  loop(lo, LO_count)
310  {
311  if (nwise < -1) b += len_L; // move forward in B
312  else b -= len_L; // move backward in B
313 
314  // one reduction step (one call of LO)
315  //
316  const Cell & cB = B->get_ravel(b);
317  if (scalar_LO && cZ->is_simple_cell() && cB.is_simple_cell())
318  {
319  ErrorCode ec = (cZ->*scalar_LO)(cZ, &cB);
320  if (ec) throw_apl_error(ec, LOC);
321  }
322  else
323  {
324  Value_P LO_A = cB.to_value(LOC);
325  Value_P LO_B = cZ->to_value(LOC);
326  Token result = LO->eval_AB(LO_A, LO_B);
327  cZ->release(LOC);
328 
329  if (result.get_tag() == TOK_ERROR)
330  throw_apl_error(result.get_ErrorCode(), LOC);
331 
332  Assert(result.get_Class() == TC_VALUE);
333  Value_P ZZ = result.get_apl_val();
334  cZ->init_from_value(ZZ.get(), Z.getref(), LOC);
335  }
336  }
337  }
338 
339  Z->set_default(*B.get(), LOC);
340  Z->check_value(LOC);
341  return Token(TOK_APL_VALUE1, Z);
342 }

43 UserFunction::UserFunction(const UCS_string txt, const char * loc,
44  const UTF8_string & _creator, bool tolerant,
45  bool macro)
46  : Function(ID_USER_SYMBOL, TOK_FUN2),
47  Executable(txt, true, PM_FUNCTION, loc),
48  header(txt, macro),
49  creator(_creator),
50  error_line(0), // assume header is wrong
51  error_info("Unspecified")
52 {
53  if (header.get_error())
54  {
55  error_info = header.get_error_info();
56  return;
57  }
58 
59  set_creation_time(now());
60 
61  exec_properties[0] = 0;
62  exec_properties[1] = 0;
63  exec_properties[2] = 0;
64  exec_properties[3] = 0;
65 
66  line_starts.push_back(Function_PC_0); // will be set later.
67 
68  if (header.get_error() != E_NO_ERROR) // bad header
69  {
70  error_info = header.get_error_info();
71  return;
72  }
73 
74  // set Function::tag
75  //
76  if (header.RO()) tag = TOK_OPER2;
77  else if (header.LO()) tag = TOK_OPER1;
78  else if (header.A()) tag = TOK_FUN2;
79  else if (header.B()) tag = TOK_FUN1;
80  else tag = TOK_FUN0;
81 
82  parse_body(loc, tolerant, macro);
83  if (error_line > 0)
84  {
85  error_info = "Error in function body";
86  return;
87  }
88 
89  if (uprefs.discard_indentation) // really ?
90  {
91  loop(l, get_text_size())
92  {
93  UCS_string line = get_text(l);
94  line.remove_leading_and_trailing_whitespaces();
95  set_text(l, line);
96  }
97  }
98 
99  error_line = -1; // no error
100  error_info = 0;
101 }

612 //-----------------------------------------------------------------------------
613 ErrorCode
614 UserFunction::transform_multi_line_strings()
615 {
616  /* convert a set of lines like
617 
618  [k+1] PREFIX "L1
619  [k+2] L2 ...
620  [k+N] Ln" SUFFIX
621 
622  into:
623 
624  [k+1] PREFIX "L1" "L2" ... "LN" SUFFIX
625  [...] (empty)
626  [k+N] (empty)
627  */
628 
629 enum Line_status
630  {
631  Function_header = 0,
632  APL_text = 1,
633  Start_of_string = 2,
634  Inside_string = 3,
635  End_of_string = 4
636  };
637 
638 char status[get_text_size()]; status[0] = Function_header;
639 Line_status current = APL_text;
640 
641  // determine line status of each line...
642  //
643  for (int l = 1; l < get_text_size(); ++l)
644  {
645  const int count = get_text(l).double_quote_count(false);
646  if (count & 1) // start or end of string
647  {
648  if (current == APL_text) // start of multi-line string
649  {
650  status[l] = Start_of_string;
651  current = Inside_string;
652  }
653  else // end of multi-line string
654  {
655  status[l] = End_of_string;
656  current = APL_text;
657  }
658  }
659  else // no status change
660  {
661  status[l] = current;
662  }
663  }
664 
665  if (current == Start_of_string || current == Inside_string)
666  {
667  // multi-line string started but not ended.
668  //
669  return E_DEFN_ERROR;
670  }
671 
672  // modify lines...
673  //
674  for (int l = 1; l < get_text_size();)
675  {
676  if (status[l] == APL_text) { ++l; continue; }
677  const int start = l;
678  Assert1(status[l] == Start_of_string);
679  UCS_string accu = get_text(l++);

766  }

774  }
775

777  //

797  //

800 
801 // CERR << endl;

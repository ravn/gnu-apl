32 /*
33  general comment: for operations between an IntCell and a "higher" numeric
34  Cell type (i.e. FloatCell or ComplexCell), we try to call the corresponding
35  function (or its opposite) of the higher type.
36 
37  For example:
38  INT ≠ FLOAT --> FLOAT ≠ INT
39  INT ≤ FLOAT --> FLOAT ≥ INT (≤ is the opposite of ≥)
40 
41  */
42 //-----------------------------------------------------------------------------
43 CellType
44 IntCell::get_cell_subtype() const
45 {
46  if (value.ival < 0) // negative integer (only fits in signed containers)
47  {
48  if (-value.ival <= 0x80)
49  return CellType(CT_INT | CTS_S8 | CTS_S16 | CTS_S32 | CTS_S64);
50 
51  if (-value.ival <= 0x8000)
52  return CellType(CT_INT | CTS_S16 | CTS_S32 | CTS_S64);
53 
54  if (-value.ival <= 0x80000000)
55  return CellType(CT_INT | CTS_S32 | CTS_S64);
56 
57  return CellType(CT_INT | CTS_S64);
58  }
59 
60  // positive integer
61  //
62  if (value.ival == 0) // 0: bit (fits in all containers)
63  return CellType(CT_INT | CTS_BIT | CTS_X8 | CTS_S8 | CTS_U8 |
64  CTS_X16 | CTS_S16 | CTS_U16 |
65  CTS_X32 | CTS_S32 | CTS_U32 |
66  CTS_X64 | CTS_S64 | CTS_U64);
67 
68  if (value.ival == 1) // 1: bit (fits in all containers)
69  return CellType(CT_INT | CTS_BIT | CTS_X8 | CTS_S8 | CTS_U8 |
70  CTS_X16 | CTS_S16 | CTS_U16 |
71  CTS_X32 | CTS_S32 | CTS_U32 |
72  CTS_X64 | CTS_S64 | CTS_U64);
73 
74  if (value.ival <= 0x7F)
75  return CellType(CT_INT | CTS_X8 | CTS_S8 | CTS_U8 |
76  CTS_X16 | CTS_S16 | CTS_U16 |
77  CTS_X32 | CTS_S32 | CTS_U32 |
78  CTS_X64 | CTS_S64 | CTS_U64);
79 
80  if (value.ival <= 0xFF)
81  return CellType(CT_INT | CTS_U8 |
82  CTS_X16 | CTS_S16 | CTS_U16 |
83  CTS_X32 | CTS_S32 | CTS_U32 |
84  CTS_X64 | CTS_S64 | CTS_U64);
85 
86  if (value.ival <= 0x7FFF)
87  return CellType(CT_INT | CTS_X16 | CTS_S16 | CTS_U16 |
88  CTS_X32 | CTS_S32 | CTS_U32 |
89  CTS_X64 | CTS_S64 | CTS_U64);
90 
91  if (value.ival <= 0xFFFF)
92  return CellType(CT_INT | CTS_U16 |
93  CTS_X32 | CTS_S32 | CTS_U32 |
94  CTS_X64 | CTS_S64 | CTS_U64);
95 
96  if (value.ival <= 0x7FFFFFFF)
97  return CellType(CT_INT | CTS_X32 | CTS_S32 | CTS_U32 |
98  CTS_X64 | CTS_S64 | CTS_U64);
99 
100  if (value.ival <= 0xFFFFFFFF)
101  return CellType(CT_INT | CTS_U32 |
102  CTS_X64 | CTS_S64 | CTS_U64);
103 
104  if (value.ival <= 0x7FFFFFFFFFFFFFFFLL) // note: this is always the case
105  return CellType(CT_INT | CTS_X64 | CTS_S64 | CTS_U64);
106 
107  return CellType(CT_INT | CTS_U64);
108 }

110 bool
111 IntCell::equal(const Cell & other, double qct) const
112 {
113  if (other.is_integer_cell()) return value.ival == other.get_int_value();
114  if (other.is_numeric()) return other.equal(*this, qct);
115  return false;
116 }

326 ErrorCode
327 IntCell::bif_factorial(Cell * Z) const
328 {
329 const APL_Integer iv = value.ival;
330 
331  // N! fits into:
332  // 32-bit signed integer for N <= 12
333  // 64-bit signed integer for N <= 20
334  // 64-bit float for N <= 170
335  //
336 
337  if (iv < 0) return E_DOMAIN_ERROR;
338 
339  if (iv < int_fact_max) // integer result that fits into a 64-bit int
340  {
341  new (Z) IntCell(int_factorials[iv]);
342  return E_NO_ERROR;
343  }
344 
345  if (iv < float_fact_max) // float result that fits into a 64 bit double
346  {
347  new (Z) FloatCell(float_factorials[iv]);
348  return E_NO_ERROR;
349  }
350 
351  return E_DOMAIN_ERROR;
352 }

360 //-----------------------------------------------------------------------------
361 ErrorCode
362 IntCell::bif_negative(Cell * Z) const
363 {
364  if (uint64_t(value.ival) == 0x8000000000000000LL) // integer overflow
365  new (Z) FloatCell(- value.ival);
366  else
367  new (Z) IntCell(- value.ival);
368  return E_NO_ERROR;

372 IntCell::bif_direction(Cell * Z) const
373 {

375  else if (value.ival < 0) new (Z) IntCell(-1);
376  else new (Z) IntCell( 0);

387 //-----------------------------------------------------------------------------
388 ErrorCode
389 IntCell::bif_reciprocal(Cell * Z) const
390 {
391  switch(value.ival)
392  {
393  case 0: return E_DOMAIN_ERROR;
394  case 1: return IntCell::zv(Z, 1);
395  case -1: return IntCell::zv(Z, -1);
396  }
397 
398 #ifdef RATIONAL_NUMBERS_WANTED
399  if (value.ival < 0)
400  return FloatCell::zv(Z, -1, -value.ival);
401  else
402  return FloatCell::zv(Z, 1, value.ival);
403 #endif
404 
405  return FloatCell::zv(Z, 1.0/value.ival);
406 }

409 IntCell::bif_roll(Cell * Z) const
410 {

416  new (Z) IntCell(qio + (rnd % set_size));
417  return E_NO_ERROR;
418 }

472 //-----------------------------------------------------------------------------
473 // dyadic built-in functions...
474 //
475 // where possible a function with non-int A is delegated to the corresponding
476 // member function of A. For numeric cells that is the FloatCell or ComplexCell
477 // function and otherwise the default function (that returns E_DOMAIN_ERROR.
478 //
479 //-----------------------------------------------------------------------------
480 ErrorCode
481 IntCell::bif_add(Cell * Z, const Cell * A) const
482 {
483  if (A->is_integer_cell()) // integer A + integer B
484  {
485  // both cells are integers.
486  //
487  const APL_Integer a = A->get_int_value();
488  const APL_Integer b = get_int_value();
489  const APL_Integer z = a + b;
490  if (sum_overflow(z, a, b))
491  return FloatCell::zv(Z, APL_Float(a) + APL_Float(b));
492  else
493  return zv(Z,z);
494  }
495 
496  // delegate to A
497  //
498  return A->bif_add(Z, this);
499 }
500 //-----------------------------------------------------------------------------
501 ErrorCode
502 IntCell::bif_add_inverse(Cell * Z, const Cell * A) const
503 {
504  return A->bif_subtract(Z, this);
505 }
506 //-----------------------------------------------------------------------------
507 ErrorCode
508 IntCell::bif_subtract(Cell * Z, const Cell * A) const
509 {
510  if (A->is_integer_cell()) // integer A - integer B
511  {
512  // both cells are integers.
513  //
514  const APL_Integer a = A->get_int_value();
515  const APL_Integer b = get_int_value();
516  const APL_Integer z = a - b;
517 
518  if (diff_overflow(z, a, b))
519  return FloatCell::zv(Z, APL_Float(a) + APL_Float(b));
520  else
521  return zv(Z,z);
522  }
523 
524  // delegate to A
525  //
526  this->bif_negative(Z); // Z = -B
527  return A->bif_add(Z, Z); // Z = A + Z = A + -B
528 }
529 //-----------------------------------------------------------------------------
530 ErrorCode
531 IntCell::bif_divide(Cell * Z, const Cell * A) const
532 {
533  if (!A->is_numeric()) return E_DOMAIN_ERROR;
534 
535 APL_Integer b = get_int_value();
536 const bool A_zero = A->is_near_zero();
537  if (b == 0) // division by 0
538  {
539  if (A_zero) return IntCell::z1(Z); // 0÷0 is 1 in APL
540  return E_DOMAIN_ERROR;
541  }
542 
543  // at this point B is non-zero
544  //
545  if (A->is_integer_cell())
546  {
547  // both cells are integers.
548  //
549  APL_Integer a = A->get_int_value();
550 
551 #ifdef RATIONAL_NUMBERS_WANTED
552  if (b < 0) // make denominator positive
553  {
554  a = -a;
555  b = -b;
556  }
557  const APL_Integer gcd = FloatCell::gcd(b, a);
558  if (b == gcd) return IntCell::zv(Z, a/gcd);
559  return FloatCell::zv(Z, a/gcd, b/gcd);
560 #else
561  const APL_Float i_quot = a / b;
562  const APL_Float r_quot = a / APL_Float(b);
563 
564  if (a != i_quot * b) return FloatCell::zv(Z, r_quot);
565  else return IntCell::zv(Z, i_quot);
566 #endif
567  }
568 
569  // delegate to A
570  //
571 const ErrorCode ec = this->bif_reciprocal(Z); // Z = ÷B
572  return ec ? ec : A->bif_multiply(Z, Z); // Z = A × Z = A × ÷B
573 }

602 //-----------------------------------------------------------------------------
603 ErrorCode
604 IntCell::bif_power(Cell * Z, const Cell * A) const
605 {
606  // some A to the integer B-th power
607  //
608  if (!A->is_numeric()) return E_DOMAIN_ERROR;
609 
610 APL_Integer b = get_int_value();
611 const bool invert_Z = b < 0;
612  if (invert_Z) b = - b;
613 
614  // at this point, b ≥ 0
615  //
616  if (b <= 1) // special cases A⋆1, A⋆0, and A⋆¯1
617  {
618  if (b == 0) return z1(Z); // A⋆0 is 1
619 
620  if (invert_Z) return A->bif_reciprocal(Z);
621  if (A->is_real_cell()) return A->bif_conjugate(Z);
622  new (Z) ComplexCell(A->get_complex_value());
623  return E_NO_ERROR;
624  }
625 
626  if (A->is_integer_cell())
627  {
628  APL_Integer a = A->get_int_value();
629  if (a == 0)
630  {
631  if (invert_Z) return E_DOMAIN_ERROR;
632 
633  return z0(Z); // 0^b = 0
634  }
635  if (a == 1) return z1(Z); // 1^b = 1
636  if (a == -1) // -1^b = 1 or -1
637  {
638  if (b & 1) return z_1(Z); // -1^(2N+2) = -1
639  else return z1(Z); // -1^(2N+1) = 1
640  }
641 
642  // try to compute A⋆B as integers
643  //
644  const bool negate_Z = (a < 0) && (b & 1);
645  if (a < 0) a = -a;
646 
647  // z is an int small enough for int_64
648  // a⋆b = a⋆(bn + bn-1 + ... + b0) with bj = 0 or 2⋆j
649  // = a⋆bn × a⋆bn-1 × ... × a⋆b0 with a⋆bj = 1 or a⋆2⋆j
650  //
651  bool overflow = false;
652  APL_Integer a_2_n = a; // a^(2^n) = a^(2^(n-1)) × a^(2^(n-1))
653  APL_Integer zi = 1;
654  for (int b1 = b; b1; b1 >>= 1)
655  {
656  if (b1 & 1)
657  {
658  if (uint64_t(zi) >= 0x7FFFFFFFFFFFFFFFULL/a_2_n)
659  {
660  overflow = true;
661  break;
662  }
663  zi *= a_2_n;
664  if (b1 == 1) break;
665  }
666 
667  if (a_2_n >= 100000000LL)
668  {
669  overflow = true;
670  break;
671  }
672 
673  a_2_n *= a_2_n;
674  }
675 
676  if (!overflow)
677  {
678  if (negate_Z) zi = - zi;
679  if (invert_Z)
680  {
681  if (zi == 0) return E_DOMAIN_ERROR;
682  new (Z) FloatCell(1.0 / zi);
683  }
684  else
685  {
686  new (Z) IntCell(zi);
687  }
688  return E_NO_ERROR;
689  }
690 
691  // A and B are integers, but Z is too big for integers
692  //
693  const APL_Float af(a);
694  const APL_Float bf(b);
695  APL_Float z = pow(af, bf);
696 
697  if (negate_Z) z = - z;
698  if (invert_Z) z = 1.0 / z;
699  new (Z) FloatCell(z);
700  return E_NO_ERROR;
701  }
702 
703  if (A->is_float_cell())
704  {
705  APL_Float af = A->get_real_value();
706  const APL_Float bf(b);
707  const bool negate_Z = (af < 0.0) && (b & 1);
708  if (af < 0.0) af = -af;
709 
710  APL_Float z = pow(af, bf);
711  if (negate_Z) z = - z;
712  if (invert_Z)
713  {
714  if (z == 0.0) return E_DOMAIN_ERROR;
715  z = 1.0 / z;
716  }
717  return FloatCell::zv(Z, z);
718  }
719 
720  // complex A to the B-th power
721  //
722  {
723  APL_Complex ca = A->get_complex_value();
724  APL_Complex cb(APL_Float(b), 0.0);
725 
726  APL_Complex z = complex_power(ca, cb);
727  const APL_Float denom = ComplexCell::mag2(z);
728  if (!invert_Z) return ComplexCell::zv(Z, z);
729 
730  if (denom == 0.0) return E_DOMAIN_ERROR;
731  return ComplexCell::zv(Z, z.real() / denom, - z.imag() / denom);
732  }
733 }

735 ErrorCode
736 IntCell::bif_residue(Cell * Z, const Cell * A) const
737 {
738  if (!A->is_numeric()) return E_DOMAIN_ERROR;
739 
740  if (A->get_imag_value() != 0.0) // complex A
741  {
742  ComplexCell B(value.ival, 0);
743  return B.bif_residue(Z, A);
744  }
745 
746  // if A is zero , return B
747  //
748 const APL_Float a = A->get_real_value();
749  if (a == 0.0) return zv(Z, value.ival);
750  if (a > (BIG_INT64_F - 1E10) || a < (1E10 - BIG_INT64_F))
751  {
752  // for large a compute a⍝b as for floats
753  //
754  const FloatCell cB(value.ival);
755  return cB.bif_residue(Z, A);
756  }
757 
758 
759  // IBM: if B is zero , return B
760  //
761  if (value.ival == 0) return IntCell::z0(Z);
762 
763  // if ⎕CT != 0 and B ÷ A is close to an integer within ⎕CT then return 0.
764  //
765 const APL_Float quot = value.ival / a;
766 const APL_Float qf = floor(quot);
767 const double qct = Workspace::get_CT();
768  if (qct != 0)
769  {
770  const APL_Float qc = ceil(quot);
771  if (quot > qc - qct) return IntCell::z0(Z);
772  if (quot < qf + qct) return IntCell::z0(Z);
773  }
774 
775  // Otherwise return B mod A
776  //
777  // ISO: R←B-(×B)×|A×⌊|B÷A and return R if (×A)=×B or R+A otherwise
778  // IBM: if A > 0 then 0 ≤ Z < A
779  // else 0 ≥ Z > A
780  //
781  if (A->is_integer_cell())
782  {
783  APL_Integer rest = value.ival % A->get_int_value();
784  if (A->get_int_value() < 0) // A negative -> Z negative
785  {
786  if (rest > 0) rest += A->get_int_value(); // += since A < 0
787  }
788  else // A positive -> Z positive
789  {
790  if (rest < 0) rest += A->get_int_value();
791  }
792  return zv(Z, rest);
793  }
794 
795 APL_Float z = value.ival - a * qf;
796  if (Cell::is_near_zero(z)) return zv(Z, 0);
797 
798  // make ×Z == ×A
799  //
800  if (a < 0.0 && z > 0.0) z = z + a; // += since a < 0
801  else if (a > 0.0 && z < 0.0) z = z + a;
802 
803  return FloatCell::zv(Z, z);
804 }

893  if (comp == COMP_EQ) return this > &other;
894  return (comp == COMP_GT);

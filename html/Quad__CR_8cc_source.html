141 //-----------------------------------------------------------------------------
142 Token
143 Quad_CR::do_eval_B(const Value & B, bool remove_extra_spaces)
144 {
145 UCS_string symbol_name(B);
146 
147  // remove trailing whitespaces in B
148  //
149  while (symbol_name.back() <= ' ') symbol_name.pop_back();
150 
151  /* return an empty character matrix, if:
152  * 1) symbol_name is not known, or
153  * 2) symbol_name is not user defined, or
154  * 3) symbol_name is a function, or
155  * 4) symbol_name is not displayable
156  */
157 
158  if (symbol_name.size() == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
159 
160 const Function * function = 0;
161  if (symbol_name[0] == UNI_MUE) // macro
162  {
163  loop(m, Macro::MAC_COUNT)
164  {
165  const Macro * macro =
166  Macro::get_macro(Macro::Macro_num(m));
167  if (symbol_name == macro->get_name())
168  {
169  function = macro;
170  break;
171  }
172  }
173 
174  }
175  else // maybe user defined function
176  {
177  const NamedObject * obj = Workspace::lookup_existing_name(symbol_name);
178  if (obj && obj->is_user_defined())
179  {
180  function = obj->get_function();
181  if (function && function->get_exec_properties()[0]) function = 0;
182  }
183  }
184  if (function == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
185 
186  // show the function...
187  //
188 const UCS_string ucs = function->canonical(false);
189 UCS_string_vector tlines;
190  ucs.to_vector(tlines);
191 int max_len = 0;
192  loop(row, tlines.size())
193  {
194  if (remove_extra_spaces)
195  tlines[row].remove_leading_and_trailing_whitespaces();
196 
197  if (max_len < tlines[row].size()) max_len = tlines[row].size();
198  }
199 
200 Shape shape_Z;
201  shape_Z.add_shape_item(tlines.size());
202  shape_Z.add_shape_item(max_len);
203 
204 Value_P Z(shape_Z, LOC);
205  loop(row, tlines.size())
206  {
207  const UCS_string & line = tlines[row];
208  loop(col, line.size())
209  Z->next_ravel_Char(line[col]);
210 
211  loop(col, max_len - line.size())
212  Z->next_ravel_Char(UNI_SPACE);
213  }
214 
215  Z->check_value(LOC);
216  return Token(TOK_APL_VALUE1, Z);
217 }

268 Quad_CR::do_CR(APL_Integer a, const Value * B, PrintContext pctx)
269 {
270  // some functions a have an inverse (which has its own number, but can
271  // also be specified as -a
272  //
273  if (a < 0) switch(a)
274  {
275  case -5: a = 13; break;
276  case -6: a = 13; break;
277  case -13: a = 6; break;
278  case -12: a = 11; break;
279  case -11: a = 12; break;
280  case -14: a = 15; break;
281  case -15: a = 14; break;
282  case -16: a = 17; break;
283  case -17: a = 16; break;
284  case -18: a = 19; break;
285  case -19: a = 18; break;
286  case -33: a = 34; break;
287  case -34: a = 33; break;
288  case -35: a = 36; break;
289  case -36: a = 35; break;
290  case -38: a = 39; break;
291  case -39: a = 38; break;
292  case -40: a = 41; break;
293  case -41: a = 40; break;
294  default: MORE_ERROR() << "A ⎕CR B with invalid A < 0";
295  DOMAIN_ERROR;
296  }
297 
298 bool extra_frame = false;
299  switch(a)
300  {
301  case 4:
302  case 8:
303  case 23:
304  case 24:
305  case 25:
306  case 29: extra_frame = true;
307 
308  }
309 
310  switch(a)
311  {
312  case 0: pctx.set_style(PR_APL); break;
313  case 1: pctx.set_style(PR_APL_FUN); break;
314  case 2: pctx.set_style(PR_BOXED_CHAR); break;
315  case 3: pctx.set_style(PR_BOXED_GRAPHIC); break;
316  case 4: pctx.set_style(PR_BOXED_GRAPHIC); break;
317  case 5: return do_CR5_6("0123456789ABCDEF", *B); // byte-vector → HEX
318  case 6: return do_CR5_6("0123456789abcdef", *B); // byte-vector → hex
319  case 7: pctx.set_style(PR_BOXED_GRAPHIC1); break;
320  case 8: pctx.set_style(PR_BOXED_GRAPHIC1); break;
321  case 9: pctx.set_style(PR_BOXED_GRAPHIC2); break;
322  case 10: return do_CR10(*B);
323  case 11: return do_CR11(*B); // Value → CDR conversion
324  case 12: return do_CR12(*B); // CDR → Value conversion
325  case 13: return do_CR13(*B); // hex → byte-vector
326  case 14: return do_CR14(*B); // Value → CDR → hex conversion
327  case 15: return do_CR15(*B); // hex → CDR → Value conversion
328  case 16: return do_CR16(*B); // byte vector → base64 (RFC 4648)
329  case 17: return do_CR17(*B); // base64 → byte vector (RFC 4648)
330  case 18: return do_CR18(*B); // UCS → UTF8 byte vector
331  case 19: return do_CR19(*B); // UTF8 byte vector → UCS string
332  case 20: pctx.set_style(PR_NARS); break;
333  case 21: pctx.set_style(PR_NARS1); break;
334  case 22: pctx.set_style(PR_NARS2); break;
335  case 23: pctx.set_style(PR_NARS); break;
336  case 24: pctx.set_style(PR_NARS1); break;
337  case 25: pctx.set_style(PR_NARS2); break;
338  case 26: return do_CR26(*B); // Cell types
339  case 27: return do_CR27_28(true, *B); // value as int
340  case 28: return do_CR27_28(false, *B); // value2 as int
341  case 29: pctx.set_style(PR_BOXED_GRAPHIC3); break;
342  case 30: return do_CR30(*B); // conform B (for ⍤ macro)
343  case 31: return do_CR31_32(true, *B); // ⎕INP helper
344  case 32: return do_CR31_32(false, *B); // ⎕INP helper
345  case 33: return do_CR33(*B); // TV to TLV byte vector
346  case 34: return do_CR34(*B); // TLV byte vector to TV
347  case 35: return do_CR35(*B); // lines to nested strings
348  case 36: return do_CR36(*B); // nested strings to lines
349  case 37: return do_CR37(*B); // ⎕CR B with extra spaces kept
350  case 38: return do_CR38(*B); // plain → structure
351  case 39: return do_CR39(*B); // structure → plain
352  case 40: return do_CR40(*B); // boolean → packed
353  case 41: return do_CR41(*B); // packed → boolean
354 
355  default: MORE_ERROR() << "A ⎕CR B with invalid A";
356  DOMAIN_ERROR;
357  }
358 
359  // common code for ⎕CR variants that only differ by print style...
360  //
361  if (extra_frame && !B->is_simple_scalar())
362  {
363  Value_P Z(LOC);
364  new (&Z->get_wproto()) PointerCell(B->clone(LOC).get(), Z.getref());
365  Z->check_value(LOC);
366  PrintBuffer pb(*Z, pctx, 0);
367  return Value_P(pb, LOC);
368  }
369  else // no frame
370  {
371  PrintBuffer pb(*B, pctx, 0);
372  return Value_P(pb, LOC);
373  }
374 }

426 //-----------------------------------------------------------------------------
427 void
428 Quad_CR::do_CR10(UCS_string_vector & result, const Value & B)
429 {
430 const UCS_string symbol_name(B);
431 const Symbol * symbol = Workspace::lookup_existing_symbol(symbol_name);
432  if (symbol == 0) DOMAIN_ERROR;
433 
434  switch(symbol->get_nc())
435  {
436  case NC_VARIABLE:
437  {
438  const Value & value = *symbol->get_apl_value().get();
439  do_CR10_variable(result, symbol_name, value);
440  return;
441  }
442 
443  case NC_FUNCTION:
444  case NC_OPERATOR:
445  {
446  const Function & ufun = *symbol->get_function();
447  const UCS_string text = ufun.canonical(false);
448  if (ufun.is_lambda())
449  {
450  UCS_string res = symbol->get_name();
451  res.append(UNI_LEFT_ARROW);
452  res.append(UNI_L_CURLY);
453  int t = 0;
454  while (t < text.size()) // skip λ header
455  {
456  const Unicode uni = text[t++];
457  if (uni == UNI_LF) break;
458  }
459 
460 
461  while (t < text.size()) // copy body
462  {
463  const Unicode uni = text[t++];
464  if (uni == UNI_LF) break;
465  res.append(uni);
466  }
467  res.append(UNI_R_CURLY);
468 
469  result.push_back(res);
470  }
471  else
472  {
473  UCS_string res("∇");
474 
475  loop(u, text.size())
476  {
477  if (text[u] == '\n')
478  {
479  result.push_back(res);
480  res.clear();
481  UCS_string next(text, u+1, text.size()-(u+1));
482  if (!next.is_comment_or_label() &&
483  u < (text.size() - 1))
484  res.append(UNI_SPACE);
485  }
486  else
487  {
488  res.append(text[u]);
489  }
490  }
491  res.append(ufun.get_exec_properties()[0]
492  ? UNI_DEL_TILDE : UNI_NABLA);
493  result.push_back(res);
494  }
495  return;
496  }
497 
498  default: DOMAIN_ERROR;
499  }
500 }

524 //-----------------------------------------------------------------------------
525 void
526 Quad_CR::do_CR10_value(UCS_string_vector & result, const Value & value,
527  Picker & picker, ShapeItem pidx)
528 {
529  /*
530  recursively encode var_name with value.
531 
532  we emit one of 2 formats:
533 
534  short format: dest ← (⍴ value) ⍴ value
535 
536  long format: dest ← (⍴ value) ⍴ 0 " prolog "
537  dest[] ← partial value ...
538 
539  short format (the default) requires a reasonably short value
540 
541  If value is nested then the short or long format is followed
542  by recursive do_CR10_value() calls for the sub-values.
543  */
544 
545  picker.push(value.get_shape(), pidx);
546 
547 UCS_string left;
548  picker.get(left);
549 
550  // compute shape_rho which is "" for scalars and true vectors
551  // of length > 1, "," for true vectors of length 1, and
552  // "n1 n2 ... ⍴" otherwise
553  //
554 UCS_string shape_rho;
555  {
556  if (value.get_rank() == 1) // true vector
557  {
558  if (value.element_count() == 0) shape_rho.append_UTF8("0⍴");
559  else if (value.element_count() == 1) shape_rho.append_UTF8(",");
560  }
561  else if (value.get_rank() > 1) // matrix or higher
562  {
563  shape_rho.append_shape(value.get_shape());
564  shape_rho.append_UTF8("⍴");
565  }
566  }
567 
568  if (value.element_count() == 0) // empty value
569  {
570  UCS_string reshape(2*(picker.get_level() - 1), UNI_SPACE);
571  Value_P proto = value.prototype(LOC);
572 
573  // emit one line for the prototype
574  UCS_string proto_name;
575  picker.get(proto_name);
576  picker.pop();
577  do_CR10_value(result, *proto, picker, pidx);
578  result.back().append_UTF8(" ⍝ proto 1");
579 
580  // and then another line to reshape the prototype
581  //
582  reshape.append(proto_name);
583  reshape.append_UTF8("←");
584  reshape.append(shape_rho);
585  reshape.append(proto_name);
586  result.push_back(reshape);
587  result.back().append_UTF8(" ⍝ proto 2");
588  return;
589  }
590 
591 bool nested = false;
592 
593  // step 1: top-level ravel. Try short (one-line) ravel and use long
594  // (multi-line ravel) if that fails.
595  //
596  if (short_ravel(result, nested, value, picker, left, shape_rho))
597  {
598  result.push_back(compute_prolog(picker.get_level(), left, value));
599 
600  ShapeItem pos = 0;
601  V_mode mode = Vm_NONE;
602  UCS_string rhs;
603  ShapeItem count = 0; // the number of items on this line
604  ShapeItem todo = value.nz_element_count(); // the number of items to do
605 
606  loop(p, todo)
607  {
608  // recompute line (which changes because count changes)
609  //
610  UCS_string line(2*(picker.get_level() - 1), UNI_SPACE);
611  picker.get_indexed(line, pos, count);
612  line.append_UTF8("←");
613 
614  // compute next item
615  //
616  UCS_string item;
617  const Cell & cell = value.get_cravel(p);
618  if (cell.is_pointer_cell()) nested = true;
619  const bool may_quote = use_quote(mode, value, p);
620  const V_mode next_mode = do_CR10_item(item, cell, mode, may_quote);
621 
622  if (count && (rhs.size() + item.size() + line.size()) > 76)
623  {
624  // next item does not fit in this line. Close the line,
625  // append it to result, and start a new line.
626  //
627  close_mode(rhs, mode);
628 
629  line.append(rhs);
630  pos += count;
631  count = 0;
632 
633  result.push_back(line);
634  rhs.clear();
635  mode = Vm_NONE;
636  }
637 
638  item_separator(rhs, mode, next_mode);
639 
640  mode = next_mode;
641  rhs.append(item);
642  ++count;
643  }
644 
645  // all items done: close mode
646  //
647  close_mode(rhs, mode);
648 
649 UCS_string line(2*(picker.get_level() - 1), UNI_SPACE);
650  picker.get_indexed(line, pos, count);
651  line.append_UTF8("←");
652  line.append(rhs);
653 
654  result.push_back(line);
655  }
656 
657  // step 2: nested items
658  //
659  if (nested)
660  {
661  loop(p, value.element_count())
662  {
663  const Cell & cell = value.get_cravel(p);
664  if (!cell.is_pointer_cell()) continue;
665 
666  const Value & sub_value = *cell.get_pointer_value().get();
667  do_CR10_value(result, sub_value, picker, p);
668  }
669  }
670 
671  picker.pop();
672 }

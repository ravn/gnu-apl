94 Quad_CR::eval_B(Value_P B)
95 {
96 UCS_string symbol_name(*B.get());
97  if (symbol_name.size() == 0) // ⎕CR '' : print help
98  {
99  return list_functions(COUT);
100  }
101 
102  // remove trailing whitespaces in B
103  //
104  while (symbol_name.back() <= ' ') symbol_name.pop_back();
105 
106  /* return an empty character matrix, if:
107  * 1) symbol_name is not known, or
108  * 2) symbol_name is not user defined, or
109  * 3) symbol_name is a function, or
110  * 4) symbol_name is not displayable
111  */
112 
113  if (symbol_name.size() == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
114 
115 const Function * function = 0;
116  if (symbol_name[0] == UNI_MUE) // macro
117  {
118  loop(m, Macro::MAC_COUNT)
119  {
120  const Macro * macro =
121  Macro::get_macro(Macro::Macro_num(m));
122  if (symbol_name == macro->get_name())
123  {
124  function = macro;
125  break;
126  }
127  }
128 
129  }
130  else // maybe user defined function
131  {
132  NamedObject * obj = Workspace::lookup_existing_name(symbol_name);
133  if (obj && obj->is_user_defined())
134  {
135  function = obj->get_function();
136  if (function && function->get_exec_properties()[0]) function = 0;
137  }
138  }
139  if (function == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
140 
141  // show the function...
142  //
143 const UCS_string ucs = function->canonical(false);
144 UCS_string_vector tlines;
145  ucs.to_vector(tlines);
146 int max_len = 0;
147  loop(row, tlines.size())
148  {
149  if (max_len < tlines[row].size()) max_len = tlines[row].size();
150  }
151 
152 Shape shape_Z;
153  shape_Z.add_shape_item(tlines.size());
154  shape_Z.add_shape_item(max_len);
155 
156 Value_P Z(shape_Z, LOC);
157  loop(row, tlines.size())
158  {
159  const UCS_string & line = tlines[row];
160  loop(col, line.size())
161  new (Z->next_ravel()) CharCell(line[col]);
162 
163  loop(col, max_len - line.size())
164  new (Z->next_ravel()) CharCell(UNI_ASCII_SPACE);
165  }
166 
167  Z->check_value(LOC);
168  return Token(TOK_APL_VALUE1, Z);
169 }

186 Quad_CR::do_CR(APL_Integer a, const Value * B, PrintContext pctx)
187 {
188  // some functions a have an inverse (which has its own number, but can
189  // also be specified as -a
190  //
191  if (a < 0) switch(a)
192  {
193  case -5: a = 13; break;
194  case -6: a = 13; break;
195  case -13: a = 6; break;
196  case -12: a = 11; break;
197  case -11: a = 12; break;
198  case -14: a = 15; break;
199  case -15: a = 14; break;
200  case -16: a = 17; break;
201  case -17: a = 16; break;
202  case -18: a = 19; break;
203  case -19: a = 18; break;
204  case -33: a = 34; break;
205  case -34: a = 33; break;
206  case -35: a = 36; break;
207  case -36: a = 35; break;
208  default: MORE_ERROR() << "A ⎕CR B with invalid A < 0";
209  DOMAIN_ERROR;
210  }
211 
212 bool extra_frame = false;
213  switch(a)
214  {
215  case 4:
216  case 8:
217  case 23:
218  case 24:
219  case 25:
220  case 29: extra_frame = true;
221 
222  }
223 
224  switch(a)
225  {
226  case 0: pctx.set_style(PR_APL); break;
227  case 1: pctx.set_style(PR_APL_FUN); break;
228  case 2: pctx.set_style(PR_BOXED_CHAR); break;
229  case 3: pctx.set_style(PR_BOXED_GRAPHIC); break;
230  case 4: pctx.set_style(PR_BOXED_GRAPHIC); break;
231  case 5: return do_CR5_6("0123456789ABCDEF", *B); // byte-vector → HEX
232  case 6: return do_CR5_6("0123456789abcdef", *B); // byte-vector → hex
233  case 7: pctx.set_style(PR_BOXED_GRAPHIC1); break;
234  case 8: pctx.set_style(PR_BOXED_GRAPHIC1); break;
235  case 9: pctx.set_style(PR_BOXED_GRAPHIC2); break;
236  case 10: return do_CR10(*B);
237  case 11: return do_CR11(*B); // Value → CDR conversion
238  case 12: return do_CR12(*B); // CDR → Value conversion
239  case 13: return do_CR13(*B); // hex → byte-vector
240  case 14: return do_CR14(*B); // Value → CDR → hex conversion
241  case 15: return do_CR15(*B); // hex → CDR → Value conversion
242  case 16: return do_CR16(*B); // byte vector → base64 (RFC 4648)
243  case 17: return do_CR17(*B); // base64 → byte vector (RFC 4648)
244  case 18: return do_CR18(*B); // UCS → UTF8 byte vector
245  case 19: return do_CR19(*B); // UTF8 byte vector → UCS string
246  case 20: pctx.set_style(PR_NARS); break;
247  case 21: pctx.set_style(PR_NARS1); break;
248  case 22: pctx.set_style(PR_NARS2); break;
249  case 23: pctx.set_style(PR_NARS); break;
250  case 24: pctx.set_style(PR_NARS1); break;
251  case 25: pctx.set_style(PR_NARS2); break;
252  case 26: return do_CR26(*B); // Cell types
253  case 27: return do_CR27_28(true, *B); // value as int
254  case 28: return do_CR27_28(false, *B); // value2 as int
255  case 29: pctx.set_style(PR_BOXED_GRAPHIC3); break;
256  case 30: return do_CR30(*B); // conform B (for ⍤ macro)
257  case 31: return do_CR31_32(true, *B); // ⎕INP helper
258  case 32: return do_CR31_32(false, *B); // ⎕INP helper
259  case 33: return do_CR33(*B); // TV to TLV byte vector
260  case 34: return do_CR34(*B); // TLV byte vector to TV
261  case 35: return do_CR35(*B); // lines to nested strings
262  case 36: return do_CR36(*B); // nested strings to lines
263 
264  default: MORE_ERROR() << "A ⎕CR B with invalid A";
265  DOMAIN_ERROR;
266  }
267 
268  // common code for ⎕CR variants that only differ by print style...
269  //
270  if (extra_frame && !B->is_simple_scalar())
271  {
272  Value_P B1(LOC);
273  new (&B1->get_ravel(0)) PointerCell(B->clone(LOC).get(), B1.getref());
274  B1->check_value(LOC);
275  PrintBuffer pb(*B1, pctx, 0);
276  return Value_P(pb, LOC);
277  }
278  else // no frame
279  {
280  PrintBuffer pb(*B, pctx, 0);
281  return Value_P(pb, LOC);
282  }

336 void
337 Quad_CR::do_CR10(UCS_string_vector & result, const Value & B)
338 {
339 const UCS_string symbol_name(B);
340 const Symbol * symbol = Workspace::lookup_existing_symbol(symbol_name);
341  if (symbol == 0) DOMAIN_ERROR;
342 
343  switch(symbol->get_nc())
344  {
345  case NC_VARIABLE:
346  {
347  const Value & value = *symbol->get_apl_value().get();
348  do_CR10_var(result, symbol_name, value);
349  return;
350  }
351 
352  case NC_FUNCTION:
353  case NC_OPERATOR:
354  {
355  const Function & ufun = *symbol->get_function();
356  const UCS_string text = ufun.canonical(false);
357  if (ufun.is_lambda())
358  {
359  UCS_string res = symbol->get_name();
360  res.append(UNI_LEFT_ARROW);
361  res.append(UNI_ASCII_L_CURLY);
362  int t = 0;
363  while (t < text.size()) // skip λ header
364  {
365  const Unicode uni = text[t++];
366  if (uni == UNI_ASCII_LF) break;
367  }
368 
369 
370  while (t < text.size()) // copy body
371  {
372  const Unicode uni = text[t++];
373  if (uni == UNI_ASCII_LF) break;
374  res.append(uni);
375  }
376  res.append(UNI_ASCII_R_CURLY);
377 
378  result.push_back(res);
379  }
380  else
381  {
382  UCS_string res("∇");
383 
384  loop(u, text.size())
385  {
386  if (text[u] == '\n')
387  {
388  result.push_back(res);
389  res.clear();
390  UCS_string next(text, u+1, text.size()-(u+1));
391  if (!next.is_comment_or_label() &&
392  u < (text.size() - 1))
393  res.append(UNI_ASCII_SPACE);
394  }
395  else
396  {
397  res.append(text[u]);
398  }
399  }
400  res.append(ufun.get_exec_properties()[0]
401  ? UNI_DEL_TILDE : UNI_NABLA);
402  result.push_back(res);
403  }
404  return;
405  }
406 
407  default: DOMAIN_ERROR;
408  }
409 }

418 //-----------------------------------------------------------------------------
419 void
420 Quad_CR::do_CR10_rec(UCS_string_vector & result, const Value & value,
421  Picker & picker, ShapeItem pidx)
422 {
423  /*
424  recursively encode var_name with value.
425 
426  we emit one of 2 formats:
427 
428  short format: dest ← (⍴ value) ⍴ value
429 
430  long format: dest ← (⍴ value) ⍴ 0 " prolog "
431  dest[] ← partial value ...
432 
433  short format (the default) requires a reasonably short value
434 
435  If value is nested then the short or long format is followed
436  by recursive do_CR10_rec() calls for the sub-values.
437  */
438 
439  picker.push(value.get_shape(), pidx);
440 
441 UCS_string left;
442  picker.get(left);
443 
444  // compute shape_rho which is "" for scalars and true vectors
445  // of length > 1, "," for true vectors of length 1, and
446  // "n1 n2 ... ⍴" otherwise
447  //
448 UCS_string shape_rho;
449  {
450  if (value.get_rank() == 1) // true vector
451  {
452  if (value.element_count() == 0) shape_rho.append_UTF8("0⍴");
453  else if (value.element_count() == 1) shape_rho.append_UTF8(",");
454  }
455  else if (value.get_rank() > 1) // matrix or higher
456  {
457  shape_rho.append_shape(value.get_shape());
458  shape_rho.append_UTF8("⍴");
459  }
460  }
461 
462  if (value.element_count() == 0) // empty value
463  {
464  UCS_string reshape(2*(picker.get_level() - 1), UNI_ASCII_SPACE);
465  Value_P proto = value.prototype(LOC);
466 
467  // emit one line for the prototype
468  UCS_string proto_name;
469  picker.get(proto_name);
470  picker.pop();
471  do_CR10_rec(result, *proto, picker, pidx);
472  result.back().append_UTF8(" ⍝ proto 1");
473 
474  // and then another line to reshape the prototype
475  //
476  reshape.append(proto_name);
477  reshape.append_UTF8("←");
478  reshape.append(shape_rho);
479  reshape.append(proto_name);
480  result.push_back(reshape);
481  result.back().append_UTF8(" ⍝ proto 2");
482  return;
483  }
484 
485 bool nested = false;
486 
487  // step 1: top-level ravel. Try short (one-line) ravel and use long
488  // (multi-line ravel) if that fails.
489  //
490  if (short_ravel(result, nested, value, picker, left, shape_rho))
491  {
492  result.push_back(compute_prolog(picker.get_level(), left, value));
493 
494  ShapeItem pos = 0;
495  V_mode mode = Vm_NONE;
496  UCS_string rhs;
497  ShapeItem count = 0; // the number of items on this line
498  ShapeItem todo = value.nz_element_count(); // the number of items to do
499 
500  loop(p, todo)
501  {
502  // recompute line (which changes because count changes)
503  //
504  UCS_string line(2*(picker.get_level() - 1), UNI_ASCII_SPACE);
505  picker.get_indexed(line, pos, count);
506  line.append_UTF8("←");
507 
508  // compute next item
509  //
510  UCS_string item;
511  const Cell & cell = value.get_ravel(p);
512  if (cell.is_pointer_cell()) nested = true;
513  const bool may_quote = use_quote(mode, value, p);
514  const V_mode next_mode = do_CR10_item(item, cell, mode, may_quote);
515 
516  if (count && (rhs.size() + item.size() + line.size()) > 76)
517  {
518  // next item does not fit in this line. Close the line,
519  // append it to result, and start a new line.
520  //
521  close_mode(rhs, mode);
522 
523  line.append(rhs);
524  pos += count;
525  count = 0;
526 
527  result.push_back(line);
528  rhs.clear();
529  mode = Vm_NONE;
530  }
531 
532  item_separator(rhs, mode, next_mode);
533 
534  mode = next_mode;
535  rhs.append(item);
536  ++count;
537  }
538 
539  // all items done: close mode
540  //
541  close_mode(rhs, mode);
542 
543  UCS_string line(2*(picker.get_level() - 1), UNI_ASCII_SPACE);
544  picker.get_indexed(line, pos, count);
545  line.append_UTF8("←");
546  line.append(rhs);
547 
548  result.push_back(line);
549  }
550 
551  // step 2: nested items
552  //
553  if (nested)
554  {
555  loop(p, value.element_count())
556  {
557  const Cell & cell = value.get_ravel(p);
558  if (!cell.is_pointer_cell()) continue;
559 
560  const Value & sub_value = *cell.get_pointer_value().get();
561  do_CR10_rec(result, sub_value, picker, p);
562  }
563  }
564 
565  picker.pop();
566 }
567 //-----------------------------------------------------------------------------
568 bool
569 Quad_CR::short_ravel(UCS_string_vector & result, bool & nested,
570  const Value & value, const Picker & picker,
571  const UCS_string & left, const UCS_string & shape_rho)
572 {
573  // some quick checks beforehand to avoid wasting time attempting to
574  // create short ravel
575  //
576  enum { MAX_SHORT_RAVEL_LEN = 70 };
577 
578 const ShapeItem value_len = value.nz_element_count();
579  if (value_len >= MAX_SHORT_RAVEL_LEN) return true; // too long
580 int len = 0;
581  loop(v, value_len)
582  {
583  const Cell & cell = value.get_ravel(v);
584  if (cell.is_integer_cell()) len += 2; // min. integer (+ separator)
585  else if (cell.is_character_cell()) len += 1;
586  else if (cell.is_pointer_cell()) len += 1;
587  else if (cell.is_complex_cell()) len += 3;
588  else len += 2;
589 
590  if (len >= MAX_SHORT_RAVEL_LEN) return true;
591  }
592 
593  // at this point, a short ravel MAY be possible. Try it
594  //
595 UCS_string line(2*(picker.get_level() - 1), UNI_ASCII_SPACE); // indent
596 
597  line.append(left);
598  line.append_UTF8("←");
599  line.append(shape_rho);
600 
601 V_mode mode = Vm_NONE;
602  loop(v, value_len)
603  {
604  // compute next item
605  //
606  UCS_string item;
607  const Cell & cell = value.get_ravel(v);
608  if (cell.is_pointer_cell()) nested = true;
609  const bool may_quote = use_quote(mode, value, v);

100 //-----------------------------------------------------------------------------
101 Token
102 Quad_CR::do_eval_B(const Value & B, bool remove_extra_spaces)
103 {
104 UCS_string symbol_name(B);
105 
106  // remove trailing whitespaces in B
107  //
108  while (symbol_name.back() <= ' ') symbol_name.pop_back();
109 
110  /* return an empty character matrix, if:
111  * 1) symbol_name is not known, or
112  * 2) symbol_name is not user defined, or
113  * 3) symbol_name is a function, or
114  * 4) symbol_name is not displayable
115  */
116 
117  if (symbol_name.size() == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
118 
119 const Function * function = 0;
120  if (symbol_name[0] == UNI_MUE) // macro
121  {
122  loop(m, Macro::MAC_COUNT)
123  {
124  const Macro * macro =
125  Macro::get_macro(Macro::Macro_num(m));
126  if (symbol_name == macro->get_name())
127  {
128  function = macro;
129  break;
130  }
131  }
132 
133  }
134  else // maybe user defined function
135  {
136  NamedObject * obj = Workspace::lookup_existing_name(symbol_name);
137  if (obj && obj->is_user_defined())
138  {
139  function = obj->get_function();
140  if (function && function->get_exec_properties()[0]) function = 0;
141  }
142  }
143  if (function == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
144 
145  // show the function...
146  //
147 const UCS_string ucs = function->canonical(false);
148 UCS_string_vector tlines;
149  ucs.to_vector(tlines);
150 int max_len = 0;
151  loop(row, tlines.size())
152  {
153  if (remove_extra_spaces)
154  tlines[row].remove_leading_and_trailing_whitespaces();
155 
156  if (max_len < tlines[row].size()) max_len = tlines[row].size();
157  }
158 
159 Shape shape_Z;
160  shape_Z.add_shape_item(tlines.size());
161  shape_Z.add_shape_item(max_len);
162 
163 Value_P Z(shape_Z, LOC);
164  loop(row, tlines.size())
165  {
166  const UCS_string & line = tlines[row];
167  loop(col, line.size())
168  new (Z->next_ravel()) CharCell(line[col]);
169 
170  loop(col, max_len - line.size())
171  new (Z->next_ravel()) CharCell(UNI_ASCII_SPACE);
172  }
173 
174  Z->check_value(LOC);
175  return Token(TOK_APL_VALUE1, Z);
176 }

191 //-----------------------------------------------------------------------------
192 Value_P
193 Quad_CR::do_CR(APL_Integer a, const Value * B, PrintContext pctx)
194 {
195  // some functions a have an inverse (which has its own number, but can
196  // also be specified as -a
197  //
198  if (a < 0) switch(a)
199  {
200  case -5: a = 13; break;
201  case -6: a = 13; break;
202  case -13: a = 6; break;
203  case -12: a = 11; break;
204  case -11: a = 12; break;
205  case -14: a = 15; break;
206  case -15: a = 14; break;
207  case -16: a = 17; break;
208  case -17: a = 16; break;
209  case -18: a = 19; break;
210  case -19: a = 18; break;
211  case -33: a = 34; break;
212  case -34: a = 33; break;
213  case -35: a = 36; break;
214  case -36: a = 35; break;
215  default: MORE_ERROR() << "A ⎕CR B with invalid A < 0";
216  DOMAIN_ERROR;
217  }
218 
219 bool extra_frame = false;
220  switch(a)
221  {
222  case 4:
223  case 8:
224  case 23:
225  case 24:
226  case 25:
227  case 29: extra_frame = true;
228 
229  }
230 
231  switch(a)
232  {
233  case 0: pctx.set_style(PR_APL); break;
234  case 1: pctx.set_style(PR_APL_FUN); break;
235  case 2: pctx.set_style(PR_BOXED_CHAR); break;
236  case 3: pctx.set_style(PR_BOXED_GRAPHIC); break;
237  case 4: pctx.set_style(PR_BOXED_GRAPHIC); break;
238  case 5: return do_CR5_6("0123456789ABCDEF", *B); // byte-vector → HEX
239  case 6: return do_CR5_6("0123456789abcdef", *B); // byte-vector → hex
240  case 7: pctx.set_style(PR_BOXED_GRAPHIC1); break;
241  case 8: pctx.set_style(PR_BOXED_GRAPHIC1); break;
242  case 9: pctx.set_style(PR_BOXED_GRAPHIC2); break;
243  case 10: return do_CR10(*B);
244  case 11: return do_CR11(*B); // Value → CDR conversion
245  case 12: return do_CR12(*B); // CDR → Value conversion
246  case 13: return do_CR13(*B); // hex → byte-vector
247  case 14: return do_CR14(*B); // Value → CDR → hex conversion
248  case 15: return do_CR15(*B); // hex → CDR → Value conversion
249  case 16: return do_CR16(*B); // byte vector → base64 (RFC 4648)
250  case 17: return do_CR17(*B); // base64 → byte vector (RFC 4648)
251  case 18: return do_CR18(*B); // UCS → UTF8 byte vector
252  case 19: return do_CR19(*B); // UTF8 byte vector → UCS string
253  case 20: pctx.set_style(PR_NARS); break;
254  case 21: pctx.set_style(PR_NARS1); break;
255  case 22: pctx.set_style(PR_NARS2); break;
256  case 23: pctx.set_style(PR_NARS); break;
257  case 24: pctx.set_style(PR_NARS1); break;
258  case 25: pctx.set_style(PR_NARS2); break;
259  case 26: return do_CR26(*B); // Cell types
260  case 27: return do_CR27_28(true, *B); // value as int
261  case 28: return do_CR27_28(false, *B); // value2 as int
262  case 29: pctx.set_style(PR_BOXED_GRAPHIC3); break;
263  case 30: return do_CR30(*B); // conform B (for ⍤ macro)
264  case 31: return do_CR31_32(true, *B); // ⎕INP helper
265  case 32: return do_CR31_32(false, *B); // ⎕INP helper
266  case 33: return do_CR33(*B); // TV to TLV byte vector
267  case 34: return do_CR34(*B); // TLV byte vector to TV
268  case 35: return do_CR35(*B); // lines to nested strings
269  case 36: return do_CR36(*B); // nested strings to lines
270  case 37: return do_CR37(*B); // ⎕CR B with extra spaces kept
271 
272  default: MORE_ERROR() << "A ⎕CR B with invalid A";
273  DOMAIN_ERROR;
274  }
275 
276  // common code for ⎕CR variants that only differ by print style...
277  //
278  if (extra_frame && !B->is_simple_scalar())
279  {
280  Value_P B1(LOC);
281  new (&B1->get_ravel(0)) PointerCell(B->clone(LOC).get(), B1.getref());
282  B1->check_value(LOC);
283  PrintBuffer pb(*B1, pctx, 0);
284  return Value_P(pb, LOC);
285  }
286  else // no frame
287  {
288  PrintBuffer pb(*B, pctx, 0);
289  return Value_P(pb, LOC);
290  }
291 }

294 Quad_CR::do_CR5_6(const char * alpha, const Value & B)
295 {
296 Shape shape_Z(B.get_shape());
297  if (shape_Z.get_rank() == 0) // scalar B
298  {
299  shape_Z.add_shape_item(2);
300  }
301  else
302  {
303  shape_Z.set_shape_item(B.get_rank() - 1, B.get_cols()*2);
304  }
305 
306 Value_P Z(shape_Z, LOC);
307 
308 const Cell * cB = &B.get_ravel(0);
309  loop(b, B.element_count())
310  {
311  const int val = cB++->get_byte_value() & 0x00FF;
312  const int h = alpha[val >> 4];
313  const int l = alpha[val & 0x0F];
314  new (Z->next_ravel()) CharCell(Unicode(h));
315  new (Z->next_ravel()) CharCell(Unicode(l));
316  }
317 
318  Z->set_default_Spc();
319  Z->check_value(LOC);
320  return Z;
321 }

343 //-----------------------------------------------------------------------------
344 void
345 Quad_CR::do_CR10(UCS_string_vector & result, const Value & B)
346 {
347 const UCS_string symbol_name(B);
348 const Symbol * symbol = Workspace::lookup_existing_symbol(symbol_name);
349  if (symbol == 0) DOMAIN_ERROR;
350 
351  switch(symbol->get_nc())
352  {
353  case NC_VARIABLE:
354  {
355  const Value & value = *symbol->get_apl_value().get();
356  do_CR10_var(result, symbol_name, value);
357  return;
358  }
359 
360  case NC_FUNCTION:
361  case NC_OPERATOR:
362  {
363  const Function & ufun = *symbol->get_function();
364  const UCS_string text = ufun.canonical(false);
365  if (ufun.is_lambda())
366  {
367  UCS_string res = symbol->get_name();
368  res.append(UNI_LEFT_ARROW);
369  res.append(UNI_ASCII_L_CURLY);
370  int t = 0;
371  while (t < text.size()) // skip λ header
372  {
373  const Unicode uni = text[t++];
374  if (uni == UNI_ASCII_LF) break;
375  }
376 
377 
378  while (t < text.size()) // copy body
379  {
380  const Unicode uni = text[t++];
381  if (uni == UNI_ASCII_LF) break;
382  res.append(uni);
383  }
384  res.append(UNI_ASCII_R_CURLY);
385 
386  result.push_back(res);
387  }
388  else
389  {
390  UCS_string res("∇");
391 
392  loop(u, text.size())
393  {
394  if (text[u] == '\n')
395  {
396  result.push_back(res);
397  res.clear();
398  UCS_string next(text, u+1, text.size()-(u+1));
399  if (!next.is_comment_or_label() &&
400  u < (text.size() - 1))
401  res.append(UNI_ASCII_SPACE);
402  }
403  else
404  {
405  res.append(text[u]);
406  }
407  }
408  res.append(ufun.get_exec_properties()[0]
409  ? UNI_DEL_TILDE : UNI_NABLA);
410  result.push_back(res);
411  }
412  return;
413  }
414 
415  default: DOMAIN_ERROR;
416  }
417 }

426 //-----------------------------------------------------------------------------
427 void
428 Quad_CR::do_CR10_rec(UCS_string_vector & result, const Value & value,
429  Picker & picker, ShapeItem pidx)
430 {
431  /*
432  recursively encode var_name with value.
433 
434  we emit one of 2 formats:
435 
436  short format: dest ← (⍴ value) ⍴ value
437 
438  long format: dest ← (⍴ value) ⍴ 0 " prolog "
439  dest[] ← partial value ...
440 
441  short format (the default) requires a reasonably short value
442 
443  If value is nested then the short or long format is followed
444  by recursive do_CR10_rec() calls for the sub-values.
445  */
446 
447  picker.push(value.get_shape(), pidx);
448 
449 UCS_string left;
450  picker.get(left);
451 
452  // compute shape_rho which is "" for scalars and true vectors
453  // of length > 1, "," for true vectors of length 1, and
454  // "n1 n2 ... ⍴" otherwise
455  //
456 UCS_string shape_rho;
457  {
458  if (value.get_rank() == 1) // true vector
459  {
460  if (value.element_count() == 0) shape_rho.append_UTF8("0⍴");
461  else if (value.element_count() == 1) shape_rho.append_UTF8(",");
462  }
463  else if (value.get_rank() > 1) // matrix or higher
464  {
465  shape_rho.append_shape(value.get_shape());
466  shape_rho.append_UTF8("⍴");
467  }
468  }
469 
470  if (value.element_count() == 0) // empty value
471  {
472  UCS_string reshape(2*(picker.get_level() - 1), UNI_ASCII_SPACE);
473  Value_P proto = value.prototype(LOC);
474 
475  // emit one line for the prototype
476  UCS_string proto_name;
477  picker.get(proto_name);
478  picker.pop();
479  do_CR10_rec(result, *proto, picker, pidx);
480  result.back().append_UTF8(" ⍝ proto 1");
481 
482  // and then another line to reshape the prototype
483  //
484  reshape.append(proto_name);
485  reshape.append_UTF8("←");
486  reshape.append(shape_rho);
487  reshape.append(proto_name);
488  result.push_back(reshape);
489  result.back().append_UTF8(" ⍝ proto 2");
490  return;
491  }
492 
493 bool nested = false;
494 
495  // step 1: top-level ravel. Try short (one-line) ravel and use long
496  // (multi-line ravel) if that fails.
497  //
498  if (short_ravel(result, nested, value, picker, left, shape_rho))
499  {
500  result.push_back(compute_prolog(picker.get_level(), left, value));
501 
502  ShapeItem pos = 0;
503  V_mode mode = Vm_NONE;
504  UCS_string rhs;
505  ShapeItem count = 0; // the number of items on this line
506  ShapeItem todo = value.nz_element_count(); // the number of items to do
507 
508  loop(p, todo)
509  {
510  // recompute line (which changes because count changes)
511  //
512  UCS_string line(2*(picker.get_level() - 1), UNI_ASCII_SPACE);
513  picker.get_indexed(line, pos, count);
514  line.append_UTF8("←");
515 
516  // compute next item
517  //
518  UCS_string item;
519  const Cell & cell = value.get_ravel(p);
520  if (cell.is_pointer_cell()) nested = true;
521  const bool may_quote = use_quote(mode, value, p);
522  const V_mode next_mode = do_CR10_item(item, cell, mode, may_quote);
523 
524  if (count && (rhs.size() + item.size() + line.size()) > 76)
525  {
526  // next item does not fit in this line. Close the line,
527  // append it to result, and start a new line.
528  //
529  close_mode(rhs, mode);
530 
531  line.append(rhs);
532  pos += count;
533  count = 0;
534 
535  result.push_back(line);
536  rhs.clear();
537  mode = Vm_NONE;
538  }
539 
540  item_separator(rhs, mode, next_mode);
541 
542  mode = next_mode;
543  rhs.append(item);
544  ++count;
545  }
546 
547  // all items done: close mode
548  //
549  close_mode(rhs, mode);
550 
551  UCS_string line(2*(picker.get_level() - 1), UNI_ASCII_SPACE);
552  picker.get_indexed(line, pos, count);
553  line.append_UTF8("←");
554  line.append(rhs);
555 
556  result.push_back(line);
557  }
558 
559  // step 2: nested items
560  //
561  if (nested)
562  {
563  loop(p, value.element_count())
564  {
565  const Cell & cell = value.get_ravel(p);
566  if (!cell.is_pointer_cell()) continue;
567 
568  const Value & sub_value = *cell.get_pointer_value().get();
569  do_CR10_rec(result, sub_value, picker, p);
570  }
571  }
572 
573  picker.pop();
574 }
575 //-----------------------------------------------------------------------------
576 bool
577 Quad_CR::short_ravel(UCS_string_vector & result, bool & nested,
578  const Value & value, const Picker & picker,
579  const UCS_string & left, const UCS_string & shape_rho)
580 {
581  // some quick checks beforehand to avoid wasting time attempting to
582  // create short ravel
583  //
584  enum { MAX_SHORT_RAVEL_LEN = 70 };
585 
586 const ShapeItem value_len = value.nz_element_count();
587  if (value_len >= MAX_SHORT_RAVEL_LEN) return true; // too long
588 int len = 0;
589  loop(v, value_len)
590  {
591  const Cell & cell = value.get_ravel(v);
592  if (cell.is_integer_cell()) len += 2; // min. integer (+ separator)
593  else if (cell.is_character_cell()) len += 1;
594  else if (cell.is_pointer_cell()) len += 1;
595  else if (cell.is_complex_cell()) len += 3;
596  else len += 2;
597 
598  if (len >= MAX_SHORT_RAVEL_LEN) return true;
599  }
600 
601  // at this point, a short ravel MAY be possible. Try it
602  //
603 UCS_string line(2*(picker.get_level() - 1), UNI_ASCII_SPACE); // indent
604 
605  line.append(left);
606  line.append_UTF8("←");
607  line.append(shape_rho);
608 
609 V_mode mode = Vm_NONE;
610  loop(v, value_len)
611  {
612  // compute next item
613  //

921  const int b2 = cB++->get_char_value() & 0x00FF;
922  const int b3 = cB++->get_char_value() & 0x00FF;

933  }

936  //

940

233  static APL_Integer gcd(APL_Integer m, APL_Integer n)
234  {
235  if (m == 0) return n;
236  if (n == 0) return m;
237 
238  if (m < 0) m = -m;
239  if (n < 0) n = -n;
240 
241  // E1: Initialize
242  APL_Integer a_ = 1;
243  APL_Integer b = 1;
244  APL_Integer a = 0;
245  APL_Integer b_ = 0;
246  APL_Integer c = m;
247  APL_Integer d = n;
248  for (;;)
249  {
250  // E2: divide
251  const APL_Integer r = c%d;
252 
253  // E3: remainder zero?
254  if (r == 0) return d;
255 
256  const APL_Integer q = c/d; // from E2
257 
258  // E4: recycle
259  const APL_Integer ta_ = a_;
260  const APL_Integer tb_ = b_;
261  c = d;
262  d = r;
263  a_ = a;
264  a = ta_ - q*a;
265  b_ = b;
266  b = tb_ - q*b;
267  }
268  }

101 bool
102 StateIndicator::uses_function(const UserFunction * ufun) const
103 {
104 const Executable * uexec = ufun;
105 
106  for (const StateIndicator * si = this; si; si = si->get_parent())
107  {
108  // case 1: ufun is the currently executing function
109  //
110  if (uexec == si->get_executable()) return true;
111 
112  // case 2: ufun is used on the prefix parser stack
113  //
114  if (si->current_stack.uses_function(ufun)) return true;
115 
116  // case 3: ufun is in the body of si->get_executable(). This crashes
117  // )SAVE since the symbol is already resolved and points to the old
118  // function.
119  //
120  uexec = si->get_executable();
121  Assert(uexec);
122  const Token_string & body = uexec->get_body();
123  loop(b, body.size())
124  {
125  const Token & tok = body[b];
126  if (tok.get_tag() != TOK_SYMBOL) continue;
127 
128  if (ufun->get_name().compare(tok.get_sym_ptr()->get_name())
129  == COMP_EQ) return true;
130  }
131  }
132  return false;
133 }

312 Token
313 StateIndicator::jump(Value_P value)
314 {
315  // perform a jump. We either remain in the current function (and then
316  // return TOK_VOID), or we (want to) jump into back into the calling
317  // function (and then return TOK_BRANCH.). The jump itself (if any)
318  // is executed in Prefix.cc.
319  //
320  if (value->get_rank() > 1) RANK_ERROR;
321 
322  if (value->element_count() == 0) // â''
323  {
324  // ââ¬ in immediate execution means resume (retry) suspended function
325  // ââ¬ on â or defined functions means do nothing
326  //
327  if (get_parse_mode() == PM_STATEMENT_LIST)
328  return Token(TOK_BRANCH, int64_t(Function_Retry));
329 
330  return Token(TOK_NOBRANCH); // stay in context
331  }
332 
333 const Function_Line line = value->get_line_number();
334 
335 const UserFunction * ufun = get_executable()->get_ufun();
336 
337  if (ufun) // âN in user defined function
338  {
339  set_PC(ufun->pc_for_line(line)); // âN to valid line in user function
340  return Token(TOK_VOID); // stay in context
341  }
342 
343  // âN in â or â
344  //
345  return Token(TOK_BRANCH, int64_t(line < 0 ? Function_Line_0 : line));
346 }

494 void
495 StateIndicator::statement_result(Token & result, bool trace)
496 {
497  Log(LOG_StateIndicator__enter_leave)
498  CERR << "StateIndicator::statement_result(pmode="
499  << get_parse_mode_name() << ", result=" << result << endl;
500 
501  if (trace)
502  {
503  const UserFunction * ufun = executable->get_ufun();
504  if (ufun && (ufun->get_exec_properties()[0] == 0))
505  {
506  const Function_Line line =
507  executable->get_line(Function_PC(get_PC() - 1));
508  result.show_trace(COUT, ufun->get_name(), line);
509  }
510  }
511 
512  fun_oper_cache.reset();
513 
514 // if (get_parse_mode() == PM_EXECUTE) return;
515 
516  // if result is a value then print it, unless it is a committed value
517  // (i.e. TOK_APL_VALUE2)
518  //
519  if (result.get_ValueType() != TV_VAL)
520  {
521 #ifdef WANT_PYTHON
522  python_result_callback(result);
523 #endif
524 
525  return;
526  }
527 
528 const TokenTag tag = result.get_tag();
529 Value_P B(result.get_apl_val());
530  Assert(!!B);
531 
532  // print TOK_APL_VALUE and TOK_APL_VALUE1, but not TOK_APL_VALUE2
533  //
534 bool print_value = tag == TOK_APL_VALUE1 || tag == TOK_APL_VALUE3;
535 
536 #ifdef WANT_LIBAPL
537  if (res_callback) // callback installed
538  {
539  // the callback decides whether the value shall be printed (even
540  // if it was committed)
541  //
542  print_value = res_callback(B.get(), !print_value);
543  }
544 #endif
545 
546 #ifdef WANT_PYTHON
547  print_value = python_result_callback(result);
548 #endif
549 
550  if (!print_value) return;
551 
552  Quad_QUOTE::done(false, LOC);
553 
554 const int boxing_format = Command::get_boxing_format();
555  if (boxing_format == 0) // no boxing
556  {
557  if (Quad_SYL::print_length_limit &&
558  B->element_count() >= Quad_SYL::print_length_limit)
559  {
560  // the value exceeds print_length_limit.
561  // We cut the longest dimension in half until we are below the
562  // limit
563  //
564  Shape sh(B->get_shape());
565  while (sh.get_volume() >= Quad_SYL::print_length_limit)
566  {
567  Rank longest = 0;
568  loop(r, sh.get_rank())
569  {
570  if (sh.get_shape_item(r) > sh.get_shape_item(longest))
571  longest = r;
572  }
573 
574  sh.set_shape_item(longest, sh.get_shape_item(longest) / 2);
575  }
576 
577  Value_P B1 = Bif_F12_TAKE::do_take(sh, B);
578  B1->print(COUT);
579 
580  CERR << " *** display of value was truncated (limit "
581  "âSYL[âIO + " << Quad_SYL::SYL_PRINT_LIMIT
582  << "] reached) ***" << endl;
583  }
584  else // no print length limit or small B
585  {
586  B->print(COUT);
587  }
588  }
589  else if (boxing_format < 0)
590  {
591  const PrintContext pctx = Workspace::get_PrintContext(PST_NONE);
592  Value_P B1 = Quad_CR::do_CR(-boxing_format, B.get(), pctx);
593  if (B1->get_cols() >= Workspace::get_PW()) // too large

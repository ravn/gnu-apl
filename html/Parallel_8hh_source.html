255 class Parallel
257 {
258 public:
260  static void acquire_lock(volatile _Atomic_word & lock)
261  {
262  // chances are low that the lock is held. Therefore we try a simple
263  // atomic_fetch_add() first and return on success.
264  // This should not harm the lock because we do this only once per
265  // thread and acquire_lock()
266  //
267  if (atomic_fetch_add(lock, 1) == 0) return;
268  atomic_add(lock, -1); // undo the atomic_fetch_add()
269 
270  // the lock was busy
271  //
272  for (;;)
273  {
274  // Wait to see a 0 on the lock. This is to avoid that the
275  // atomic_fetch_add() lock attempts occupy the lock without
276  // actually obtaining the lock. Waiting for 0 guarantees that
277  // at least one thread succeeds below.
278  //
279  if (atomic_read(lock)) continue; // not 0: try again
280 
281  if (atomic_fetch_add(lock, 1) == 0) return;
282  }
283  }
284 
286  static void release_lock(volatile _Atomic_word & lock)
287  {
288  atomic_add(lock, -1);
289  }

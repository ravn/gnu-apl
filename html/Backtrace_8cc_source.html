57 const char *
58 Backtrace::find_src(int64_t pc)
59 {
60  if (pc == -1LL) return 0; // no pc
61 
62  if (pc_2_src.size()) // database was properly set up.
63  {
64  if (const PC_src * posp = Heapsort<Backtrace::PC_src>
65  ::search<const int64_t &>(pc, pc_2_src, &pc_cmp, 0))
66  return posp->src_loc; // found
67  }
68 
69  return 0; // not found
70 }
71 //-----------------------------------------------------------------------------
72 void
73 Backtrace::open_lines_file()
74 {
75  if (lines_status != LINES_not_checked) return;
76 
77  // we are here for the first time.
78  // Assume that apl.lines is outdated until proven otherwise.
79  //
80  lines_status = LINES_outdated;
81 
82 struct stat st;
83  if (stat("apl", &st)) return; // stat apl failed
84 
85 const time_t apl_time = st.st_mtime;
86 
87 const int fd = open("apl.lines", O_RDONLY);
88  if (fd == -1) return;
89 
90  if (fstat(fd, &st)) // stat apl.lines failed
91  {
92  close(fd);
93  return;
94  }
95 
96 const time_t apl_lines_time = st.st_mtime;
97 
98  if (apl_lines_time < apl_time)
99  {
100 #ifndef Makefile__srcdir
101 # define Makefile__srcdir "src"
102 #endif
103  close(fd);
104  get_CERR() << endl
105  << "Cannot show line numbers, since apl.lines is older "
106  "than apl." << endl
107  << "Consider running 'make apl.lines' in directory " << endl
108  << Makefile__srcdir " to fix this" << endl;
109  return;
110  }
111 
112  pc_2_src.reserve(100000);
113 char buffer[1000];
114 FILE * file = fdopen(fd, "r");
115  assert(file);
116 
117 const char * src_line = 0;
118 bool new_line = false;
119 int64_t prev_pc = -1LL;
120 
121  for (;;)
122  {
123  const char * s = fgets(buffer, sizeof(buffer) - 1, file);
124  if (s == 0) break; // end of file.
125 
126  buffer[sizeof(buffer) - 1] = 0;
127  int slen = strlen(buffer);
128  if (slen && buffer[slen - 1] == '\n') buffer[--slen] = 0;
129  if (slen && buffer[slen - 1] == '\r') buffer[--slen] = 0;
130 
131  // we look for 2 types of line: abolute source file paths
132  // and code lines.
133  //
134  if (s[0] == '/') // source file path
135  {
136  src_line = strdup(strrchr(s, '/') + 1);
137  new_line = true;
138  continue;
139  }
140 
141  if (!new_line) continue;
142 
143  if (s[8] == ':') // opcode address
144  {
145  long long pc = -1LL;
146  if (1 == sscanf(s, " %llx:", &pc))
147  {
148  if (pc < prev_pc)
149  {
150  assert(0 && "apl.lines not ordered by PC");
151  break;
152  }
153 
154  PC_src pcs = { pc, src_line };
155  pc_2_src.push_back(pcs);
156  prev_pc = pc;
157  new_line = false;
158  }
159  continue;
160  }
161  }
162 
163  fclose(file); // also closes fd
164 
165  cerr << "read " << pc_2_src.size() << " line numbers" << endl;
166  lines_status = LINES_valid;
167 }

169 void
170 Backtrace::show_item(int idx, char * s)
171 {
172 #ifdef HAVE_EXECINFO_H
173  //
174  // s looks like this:
175  //
176  // ./apl(_ZN10APL_parser9nextTokenEv+0x1dc) [0x80778dc]
177  //
178 
179  // split off address from s.
180  //
181 const char * addr = "????????";
182 int64_t pc = -1LL;
183  {
184  char * space = strchr(s, ' ');
185  if (space)
186  {
187  *space = 0;
188  addr = space + 2;
189  char * e = strchr(space + 2, ']');
190  if (e) *e = 0;
191  pc = strtoll(addr, 0, 16);
192  }
193  }
194 
195 const char * src_loc = find_src(pc);
196 
197  // split off function from s.
198  //
199 char * fun = 0;
200  {
201  char * opar = strchr(s, '(');
202  if (opar)
203  {
204  *opar = 0;
205  fun = opar + 1;
206  char * e = strchr(opar + 1, ')');
207  if (e) *e = 0;
208  }
209  }
210 
211  // split off offset from fun.
212  //
213 int offs = 0;
214  if (fun)
215  {
216  char * plus = strchr(fun, '+');
217  if (plus)
218  {
219  *plus++ = 0;
220  sscanf(plus, "%X", &offs);
221  }
222  }
223 
224 char obuf[200] = "@@@@";
225  if (fun)
226  {
227  strncpy(obuf, fun, sizeof(obuf) - 1);
228  obuf[sizeof(obuf) - 1] = 0;
229 
230  size_t obuflen = sizeof(obuf) - 1;
231  int status = 0;
232 // cerr << "mangled fun is: " << fun << endl;
233  __cxxabiv1::__cxa_demangle(fun, obuf, &obuflen, &status);
234  switch(status)
235  {
236  case 0: // demangling succeeded
237  break;
238 
239  case -2: // not a valid name under the C++ ABI mangling rules.
240  break;
241 
242  default:
243  cerr << "__cxa_demangle() returned " << status << endl;
244  break;
245  }
246  }
247 
248 // cerr << setw(2) << idx << ": ";
249 
250  cerr << HEX(pc);
251 
252 // cerr << left << setw(20) << s << right << " ";
253 
254  // indent.
255  for (int i = -1; i < idx; ++i) cerr << " ";
256 
257  cerr << obuf;
258 
259 // if (offs) cerr << " +" << offs;
260 
261  if (src_loc) cerr << " at " << src_loc;
262  cerr << endl;
263 #endif
264 }

297  for (int s = b + 1; s < size; ++s) cerr << " ";
298  backtrace_symbols_fd(buffer + b, 1, STDERR_FILENO);
299  }
300  cerr << "========================================" << endl;
301  return;
302  }
303 
304  for (int i = 1; i < size - 1; ++i)

306  // make a copy of strings[i] that can be

310  char cc[strlen(si) + 1];
311  strcpy(cc, si);
312  cc[sizeof(cc) - 1] = 0;
313  show_item(i - 1, cc);
314  }
315  cerr << "========================================" << endl;
316 
317 #if 0
318  // crashes at times
319  free(strings); // but not strings[x] !
320 #endif

322 #endif

326 int
327 Backtrace::demangle_line(char * result, size_t result_max, const char * buf)
328 {
329 std::string tmp;
330  tmp.reserve(result_max + 1);
331  for (const char * b = buf; *b && b < (buf + result_max); ++b)
332  tmp += *b;
333  tmp.append(0);
334 
335 char * e = 0;
336 int status = 3;
337 char * p = strchr(&tmp[0], '(');
338  if (p == 0) goto error;
339  else ++p;
340 
341  e = strchr(p, '+');
342  if (e == 0) goto error;
343  else *e = 0;
344 
345 // cerr << "mangled fun is: " << p << endl;
346  __cxxabiv1::__cxa_demangle(p, result, &result_max, &status);
347  if (status) goto error;
348  return 0;
349 
350 error:
351  strncpy(result, buf, result_max);
352  result[result_max - 1] = 0;
353  return status;
354 }

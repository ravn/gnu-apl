31 template <typename T, size_t SYMBOL_COUNT>
32 class SymbolTableBase
33 {
34 public:
35  enum { max_symbol_count = SYMBOL_COUNT };
36 
38  SymbolTableBase()
39  { memset(symbol_table, 0, sizeof(symbol_table)); }
40 
41  ~SymbolTableBase()
42  {
43  loop(s, SYMBOL_COUNT)
44  for (T * sym = symbol_table[s]; sym;)
45  {
46  T * del = sym;
47  sym = sym->next;
48  delete del;
49  }
50  }
51 
53  T * lookup_existing_symbol(const UCS_string & name)
54  {
55  const uint32_t hash = compute_hash(name);
56  for (T * sym = symbol_table[hash]; sym; sym = sym->next)
57  {
58  if (!sym->equal(name)) continue; // name mismatch
59  if (!sym->is_erased()) return sym;
60  }
61 
62  return 0;
63  }
64 
66  const T * lookup_existing_symbol(const UCS_string & name) const
67  {
68  const uint32_t hash = compute_hash(name);
69  for (const T * sym = symbol_table[hash]; sym; sym = sym->next)
70  {
71  if (!sym->equal(name)) continue; // name mismatch
72  if (!sym->is_erased()) return sym;
73  }
74 
75  return 0;
76  }
77 
79  UCS_string find_lambda_name(const UserFunction * lambda);
80 
83  void add_symbol(T * sym)
84  {
85  const uint32_t hash = compute_hash(sym->get_name());
86  if (symbol_table[hash] == 0) // unused slot
87  {
88  symbol_table[hash] = sym;
89  sym->next = 0;
90  return;
91  }
92 
93  for (T * t = symbol_table[hash]; ; t = t->next)
94  {
95  if (t->next == 0) // append new_symbol at the end
96  {
97  t->next = sym;
98  sym->next = 0;
99  return;
100  }
101  }
102  }
103 
105  static uint32_t compute_hash(const UCS_string & name)
106  {
107  uint32_t hash = name.FNV_hash();
108  hash = (((hash >> 16) ^ hash) & 0x0000FFFF) % SYMBOL_COUNT;
109 
110  Log(LOG_SYMBOL_lookup_symbol)
111  {
112  CERR << "name[len=" << name.size() << "] " << name
113  << " has hash " << HEX(hash)
114  << " in table of size " << SYMBOL_COUNT << endl;
115  }
116 
117  return hash;
118  }
119 
120 protected:
122  T * symbol_table[SYMBOL_COUNT];
123 };

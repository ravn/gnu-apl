53  T * lookup_existing_symbol(const UCS_string & name)
54  {
55  const uint32_t hash = compute_hash(name);
56  for (T * sym = symbol_table[hash]; sym; sym = sym->next)
57  {

83  void add_symbol(T * sym)
84  {
85  const uint32_t hash = compute_hash(sym->get_name());
86  if (symbol_table[hash] == 0) // unused slot
87  {
88  symbol_table[hash] = sym;
89  sym->next = 0;
90  return;
91  }
92 
93  for (T * t = symbol_table[hash]; ; t = t->next)
94  {
95  if (t->next == 0) // append new_symbol at the end
96  {
97  t->next = sym;
98  sym->next = 0;
99  return;
100  }
101  }
102  }

105  static uint32_t compute_hash(const UCS_string & name)
106  {
107  // Parameters for the FNV-1 hash.
108  enum {
109  FNV_Offset_32 = 0x811C9DC5,
110  FNV_Prime_32 = 16777619
111  };
112 
113  uint32_t hash = FNV_Offset_32;
114  for (int s = 0; s < name.size(); ++s)
115  hash = (hash * FNV_Prime_32) ^ name[s];
116 
117  hash = (((hash >> 16) ^ hash) & 0x0000FFFF) % SYMBOL_COUNT;
118 
119  Log(LOG_SYMBOL_lookup_symbol)
120  {
121  CERR << "name[len=" << name.size() << "] " << name
122  << " has hash " << HEX(hash)
123  << " in table of size " << SYMBOL_COUNT << endl;
124  }
125 
126  return hash;
127  }

129 protected:
131  T * symbol_table[SYMBOL_COUNT];

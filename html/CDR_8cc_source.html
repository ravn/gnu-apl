31 void
32 CDR::to_CDR(CDR_string & result, const Value & value)
33 {
34 const CDR_type type = value.get_CDR_type();
35 const int len = value.total_size_brutto(type);
36  result.reserve(len + 1);
37  result.clear();
38 
39  fill(result, type, len, value);
40 }
41 //-----------------------------------------------------------------------------
42 void
43 CDR::fill(CDR_string & result, int type, int len, const Value & val)
44 {
45  Assert((len & 0x0F) == 0);
46 
47 /*
48  struct tf3_header
49  {
50  uint32_t ptr; // byte reversed
51  uint32_t nb; // byte reversed
52  uint32_t nelm; // byte reversed
53  uint8_t type;
54  uint8_t rank;
55  uint8_t fill[2];
56  uint32_t dim[1]; // byte reversed
57  };
58 */
59 
60  // ptr
61  //
62 const uint32_t ptr = 0x00002020;
63  result.push_back(Unicode(ptr >> 24 & 0xFF));
64  result.push_back(Unicode(ptr >> 16 & 0xFF));
65  result.push_back(Unicode(ptr >> 8 & 0xFF));
66  result.push_back(Unicode(ptr & 0xFF));
67 
68  // nb (header + data + sub-values + padding)
69  //
70  result.push_back(Unicode(len >> 24 & 0xFF));
71  result.push_back(Unicode(len >> 16 & 0xFF));
72  result.push_back(Unicode(len >> 8 & 0xFF));
73  result.push_back(Unicode(len & 0xFF));
74 
75  // nelm
76  //
77 const uint32_t nelm = val.element_count();
78  result.push_back(Unicode(nelm >> 24 & 0xFF));
79  result.push_back(Unicode(nelm >> 16 & 0xFF));
80  result.push_back(Unicode(nelm >> 8 & 0xFF));
81  result.push_back(Unicode(nelm & 0xFF));
82 
83  // type, rank, 0, 0
84  //
85  result.push_back(Unicode(type));
86  result.push_back(Unicode(val.get_rank()));
87  result.push_back(Unicode_0);
88  result.push_back(Unicode_0);
89 
90  // shape
91  //
92  loop(r, val.get_rank())
93  {
94  const uint32_t sh = val.get_shape_item(r);
95  result.push_back(Unicode(sh >> 24 & 0xFF));
96  result.push_back(Unicode(sh >> 16 & 0xFF));
97  result.push_back(Unicode(sh >> 8 & 0xFF));
98  result.push_back(Unicode(sh & 0xFF));
99  }
100 
101  // body
102  //
103  if (type == 0) // packed bit vector
104  {
105  int accu = 0;
106 
107  loop(e, nelm)
108  {
109  const int bit = e%8;
110  const APL_Integer i = val.get_ravel(e).get_int_value();
111  Assert(i == 0 || i == 1);
112  if (i) accu |= 0x80 >> bit;
113  if (bit == 7)
114  {
115  result.push_back(Unicode(accu));
116  accu = 0;
117  }
118  }
119 
120  if (nelm % 8) // partly filled
121  {
122  result.push_back(Unicode(accu));
123  }
124 
125  }
126  else if (type == 1) // 4 byte ints vector
127  {
128  loop(e, nelm)
129  {
130  uint64_t i = val.get_ravel(e).get_int_value();
131  result.push_back(Unicode(i & 0xFF)); i >>= 8;
132  result.push_back(Unicode(i & 0xFF)); i >>= 8;
133  result.push_back(Unicode(i & 0xFF)); i >>= 8;
134  result.push_back(Unicode(i & 0xFF)); i >>= 8;
135  Assert(i == 0 || i == 0xFFFFFFFF);
136  }
137  }
138  else if (type == 2) // 8 byte float vector
139  {
140  loop(e, nelm)
141  {
142  const APL_Float v = val.get_ravel(e).get_real_value();
143  const APL_Float * dv = &v;
144  uint64_t i = *reinterpret_cast<const uint64_t *>(dv);
145  result.push_back(Unicode(i & 0xFF)); i >>= 8;
146  result.push_back(Unicode(i & 0xFF)); i >>= 8;
147  result.push_back(Unicode(i & 0xFF)); i >>= 8;
148  result.push_back(Unicode(i & 0xFF)); i >>= 8;
149  result.push_back(Unicode(i & 0xFF)); i >>= 8;
150  result.push_back(Unicode(i & 0xFF)); i >>= 8;
151  result.push_back(Unicode(i & 0xFF)); i >>= 8;
152  result.push_back(Unicode(i & 0xFF)); i >>= 8;
153  }
154  }
155  else if (type == 3) // 16 byte complex vector
156  {
157  loop(e, nelm)
158  {
159  APL_Float v = val.get_ravel(e).get_real_value();
160  const APL_Float * dv = &v;
161  uint64_t i = *reinterpret_cast<const uint64_t *>(dv);
162  result.push_back(Unicode(i & 0xFF)); i >>= 8;
163  result.push_back(Unicode(i & 0xFF)); i >>= 8;
164  result.push_back(Unicode(i & 0xFF)); i >>= 8;
165  result.push_back(Unicode(i & 0xFF)); i >>= 8;
166  result.push_back(Unicode(i & 0xFF)); i >>= 8;
167  result.push_back(Unicode(i & 0xFF)); i >>= 8;
168  result.push_back(Unicode(i & 0xFF)); i >>= 8;
169  result.push_back(Unicode(i & 0xFF)); i >>= 8;
170 
171  v = val.get_ravel(e).get_imag_value();
172  dv = &v;
173  i = *reinterpret_cast<const uint64_t *>(dv);
174  result.push_back(Unicode(i & 0xFF)); i >>= 8;
175  result.push_back(Unicode(i & 0xFF)); i >>= 8;
176  result.push_back(Unicode(i & 0xFF)); i >>= 8;
177  result.push_back(Unicode(i & 0xFF)); i >>= 8;
178  result.push_back(Unicode(i & 0xFF)); i >>= 8;
179  result.push_back(Unicode(i & 0xFF)); i >>= 8;
180  result.push_back(Unicode(i & 0xFF)); i >>= 8;
181  result.push_back(Unicode(i & 0xFF)); i >>= 8;
182  }
183  }
184  else if (type == 4) // 1 byte char vector
185  {
186  loop(e, nelm)
187  {
188  const Unicode uni = val.get_ravel(e).get_char_value();
189  Assert(uni >= 0);
190  Assert(uni < 256);
191  result.push_back(uni);
192  }
193  }
194  else if (type == 5) // 4 byte UNICODE vector
195  {
196  loop(e, nelm)
197  {
198  uint32_t i = val.get_ravel(e).get_char_value();
199  result.push_back(Unicode(i & 0xFF)); i >>= 8;
200  result.push_back(Unicode(i & 0xFF)); i >>= 8;
201  result.push_back(Unicode(i & 0xFF)); i >>= 8;
202  result.push_back(Unicode(i & 0xFF)); i >>= 8;
203  }
204  }
205  else if (type == 7) // packed vector with 4 bytes offsets.
206  {
207  // the first offset is after header + offset-vector and then increases
208  // by the size of each sub-value. I.e.
209  //
210  // +------------------+
211  // | TOP-LEVEL-HEADER | 16 bytes + shape
212  // +------------------+
213  // | OFFSET-1 | --------+ 4 bytes
214  // +------------------+ |
215  // | OFFSET-2 | --------|--+ 4 bytes
216  // +------------------+ | |
217  // | ... | | | ...
218  // +------------------+ | |
219  // | OFFSET-nelm | --------|--|--+ 4 bytes
220  // +------------------+ | | |
221  // | SUB-HEADER-1 | <-------+ | | 16 bytes + shape
222  // | VALUE-1 | | |
223  // +------------------+ | |
224  // | SUB-HEADER-2 | <----------+ | 16 bytes + shape
225  // | VALUE-2 | |
226  // +------------------+ |
227  // | ... | | ...
228  // +------------------+ |
229  // | SUB-HEADER-nelm | <-------------+ 16 bytes + shape
230  // | VALUE-nelm |
231  // +------------------+
232  //
233 
234  uint32_t offset = 16 + 4*val.get_rank() + 4*nelm;
235  while (offset & 0x0F) ++offset;
236  loop(e, nelm)
237  {
238  result.push_back(Unicode(offset & 0xFF));
239  result.push_back(Unicode(offset >> 8 & 0xFF));
240  result.push_back(Unicode(offset >> 16 & 0xFF));
241  result.push_back(Unicode(offset >> 24 & 0xFF));
242 
243  const Cell & cell = val.get_ravel(e);
244  if (cell.is_simple_cell())
245  {
246  // a non-pointer sub value: 16 byte header,
247  // and 1-16 data bytes, padded up to 16 bytes,
248  //
249  offset += 32;
250  }
251  else if (cell.is_pointer_cell())
252  {
253  const Value & sub_val = *cell.get_pointer_value();
254  const CDR_type sub_type = sub_val.get_CDR_type();
255  offset += sub_val.total_size_brutto(sub_type);
256  }
257  else
258  DOMAIN_ERROR;
259  }
260 
261  // pad top level to 16 bytes. end is aligned to 16 byte so we
262  // case use it as reference.
263  //
264  while (result.size() & 0x0F) result.push_back(Unicode_0);
265 
266  // append sub-values.
267  //
268  loop(e, nelm)
269  {
270  const Cell & cell = val.get_ravel(e);
271  if (cell.is_simple_cell())
272  {
273  Value_P sub_val(LOC);
274  sub_val->get_ravel(0).init(cell, sub_val.getref(), LOC);
275 
276  const CDR_type sub_type = sub_val->get_CDR_type();
277  const int sub_len = sub_val->total_size_brutto(sub_type);
278  fill(result, sub_type, sub_len, *sub_val);
279  }
280  else
281  {
282  Value_P sub_val = cell.get_pointer_value();
283 
284  const CDR_type sub_type = sub_val->get_CDR_type();
285  const int sub_len = sub_val->total_size_brutto(sub_type);
286  fill(result, sub_type, sub_len, *sub_val);
287  }
288  }
289 
290  // since we padded the top level to 16 and sub-values have a padded
291  // length as well, we should be properly padded here.
292  //
293  }
294  else Assert(0 && "Bad/unsupported CDR type");
295 
296  // final padding
297  //
298  while (result.size() & 15) result.push_back(Unicode_0);
299 }
300 //-----------------------------------------------------------------------------
301 Value_P
302 CDR::from_CDR(const CDR_string & cdr, const char * loc)
303 {
304  // read header...
305  //
306 /*
307 struct tf3_header
308  {
309  uint32_t ptr; // 0: byte reversed
310  uint32_t nb; // 4: byte reversed
311  uint32_t nelm; // 8: byte reversed
312  uint8_t type; // 12:
313  uint8_t rank; // 13:
314  uint8_t fill[2]; // 14:
315  uint32_t dim[1]; // 16: byte reversed
316  }
317 */
318 
319  if (cdr.size() < 20)
320  {
321  MORE_ERROR() << "CDR strings must have a header of at least 20 bytes";
322  LENGTH_ERROR;
323  }
324 
325 const uint8_t * data = cdr.get_items();
326 
327 const uint32_t nelm = get_4_be(data + 8);
328 const CDR_type vtype = CDR_type(data[12]);
329 const Rank rank = data[13];
330 Shape shape;
331  loop(r, rank)
332  {
333  const ShapeItem sh = get_4_be(data + 16 + 4*r);
334  shape.add_shape_item(sh);
335  }
336 
337 Value_P ret(shape, loc);
338 
339 const uint8_t * ravel = data + 16 + 4*rank;
340 
341  if (vtype == CDR_BOOL1) // packed bit vector
342  {
343  loop(n, nelm)
344  {
345  const int bit = ravel[n >> 3] & (0x80 >> (n & 7));
346  if (bit) new (&ret->get_ravel(n)) IntCell(1);
347  else new (&ret->get_ravel(n)) IntCell(0);
348  }
349  }
350  else if (vtype == CDR_INT32) // 4 byte ints vector
351  {
352  loop(n, nelm)
353  {
354  APL_Integer d = *reinterpret_cast<const uint32_t *>(ravel + 4*n);
355  if (d & 0x80000000) d |= 0xFFFFFFFF00000000ULL;
356  new (&ret->get_ravel(n)) IntCell(d);
357  }
358  }
359  else if (vtype == CDR_FLT64) // 8 byte float vector
360  {
361  loop(n, nelm)
362  {
363  const APL_Float v =
364  *reinterpret_cast<const APL_Float *>(ravel + 8*n);
365  new (&ret->get_ravel(n)) FloatCell(v);
366  }
367  }
368  else if (vtype == CDR_CPLX128) // 16 byte complex vector
369  {
370  loop(n, nelm)
371  {
372  const APL_Float vr =
373  *reinterpret_cast<const APL_Float *>(ravel + 16*n);
374  const APL_Float vi =
375  *reinterpret_cast<const APL_Float *>(ravel + 16*n + 8);
376  new (&ret->get_ravel(n)) ComplexCell(vr, vi);
377  }
378  }
379  else if (vtype == CDR_CHAR8) // 1 byte char vector
380  {
381  loop(n, nelm)
382  {
383  uint32_t d = ravel[n];
384  if (d & 0x80) d |= 0xFFFFFFFFFFFFFF00ULL;
385  new (&ret->get_ravel(n)) CharCell(Unicode(d));
386  }
387  }
388  else if (vtype == CDR_CHAR32) // 4 byte UNICODE vector
389  {
390  loop(n, nelm)
391  {
392  const uint32_t d = *reinterpret_cast<const uint32_t *>(ravel+4*n);
393  new (&ret->get_ravel(n)) CharCell(Unicode(d));
394  }
395  }
396  else if (vtype == CDR_NEST32) // packed vector with 4 bytes offsets.
397  {
398  loop(n, nelm)
399  {
400  APL_Integer offset =
401  *reinterpret_cast<const uint32_t *>(ravel + 4*n);
402  const uint8_t * sub_data = data + offset;
403  const uint32_t sub_vtype = sub_data[12];
404  const Rank sub_rank = sub_data[13];
405  const uint8_t * sub_ravel = sub_data + 16 + 4*sub_rank;
406 
407  // if the sub-item is a non-nested scalar then we append it to
408  // ret directly; otherwise we append a pointer to it,
409  //
410  if (sub_rank == 0) // scalar
411  {
412  if (sub_vtype == 0) // bit
413  {
414  new (&ret->get_ravel(n))
415  IntCell((*sub_ravel & 0x80) ? 1 : 0);
416  continue; // next n
417  }
418 
419  if (sub_vtype == 1) // 4 byte int
420  {
421  APL_Integer d = *reinterpret_cast<const uint32_t *>
422  (sub_ravel);
423  if (d & 0x80000000) d |= 0xFFFFFFFF00000000ULL;
424  new (&ret->get_ravel(n)) IntCell(d);
425  continue; // next n
426  }
427 
428  if (sub_vtype == 2) // 8 byte real
429  {
430  new (&ret->get_ravel(n))
431  FloatCell(*reinterpret_cast<const APL_Float *>
432  (sub_ravel));
433  continue; // next n
434  }
435 
436  if (sub_vtype == 3) // 16 byte complex
437  {
438  new (&ret->get_ravel(n))
439  ComplexCell(*reinterpret_cast<const APL_Float *>
440  (sub_ravel),
441  *reinterpret_cast<const APL_Float *>
442  (sub_ravel + 8));
443  continue; // next n
444  }
445 
446  if (sub_vtype == 4) // 1 byte char
447  {
448  new (&ret->get_ravel(n))
449  CharCell((Unicode(*sub_ravel)));
450  continue; // next n
451  }
452 
453  if (sub_vtype == 5) // 4 byte Unicode
454  {
455  new (&ret->get_ravel(n))
456  CharCell(Unicode(get_4_be(sub_ravel)));
457  continue; // next n
458  }
459 
460  // sub_vtype 6 (progression vector) can't be scalar
461  }
462 
463  // at this point the sub-item is not a scalar
464  //
465  if (sub_vtype == 6) // arithmetic progression vector
466  {
467  // arithmetic progression vectors (APVs) are not well
468  // described. Normally a progression vector is a triple
469  //
470  // { initial-value, element-count, increment }
471  //
472  // But: we have only 8 bytes = 2 integers. We therefore
473  // assume that element-count == nelm so that the two
474  // integers are initial-value and increment.
475  //
476  Shape sh;
477  loop(r, sub_rank)
478  sh.add_shape_item(get_4_be(sub_data + 16));
479 
480  Value_P sub_val(sh, LOC);
481  const APL_Integer qio = Workspace::get_IO();
482  loop(v, sh.get_volume())
483  {
484  new (&sub_val->get_ravel(v)) IntCell(v + qio);
485  }
486 
487  continue; // next n
488  }
489 
490  const uint32_t sub_cdr_len = get_4_be(sub_data + 4);
491  CDR_string sub_cdr(sub_data, sub_cdr_len);
492  Value_P sub_val = from_CDR(sub_cdr, LOC);
493  Assert(!!sub_val);
494  new (&ret->get_ravel(n)) PointerCell(sub_val.get(), ret.getref());
495  }
496  }
497  else
498  {
499  CERR << "Unsupprted CDR type " << vtype << endl;
500  Assert(0 && "Bad/unsupported CDR type");
501  }
502 
503  ret->check_value(LOC);
504  return ret;
505 }

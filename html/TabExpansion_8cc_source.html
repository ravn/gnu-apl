38  //

53 ExpandResult
54 TabExpansion::expand_user_name(UCS_string & user)
55 {
56 std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
57 
58 UCS_string_vector matches;
59  loop(s, symbols.size())
60  {
61  const Symbol * sym = symbols[s];
62  if (sym->is_erased()) continue;
63 
64  const UCS_string & sym_name = sym->get_name();
65  if (!sym_name.starts_with(user)) continue;
66  matches.push_back(sym_name);
67  }
68 
69  if (matches.size() == 0) return ER_IGNORE; // no match
70 
71  if (matches.size() > 1) // multiple names match user input
72  {
73  const int user_len = user.size();
74  user.clear();
75  return show_alternatives(user, user_len, matches);
76  }
77 
78  // unique match
79  //
80  if (user.size() < matches[0].size())
81  {
82  // the name is longer than user, so we expand it.
83  //
84  user = matches[0];
85  return ER_REPLACE;
86  }
87 
88  return ER_IGNORE;
89 }

91 ExpandResult
92 TabExpansion::expand_APL_command(UCS_string & user)
93 {
94 ExpandHint ehint = EH_NO_PARAM;
95 const char * shint = 0;
96 UCS_string_vector matches;
97 UCS_string cmd = user;
98 UCS_string arg;
99  cmd.split_ws(arg);
100 
101 #define cmd_def(cmd_str, code, arg, hint) \
102  { UCS_string ustr(cmd_str); \
103  if (ustr.starts_iwith(cmd)) \
104  { matches.push_back(ustr); ehint = hint; shint = arg; } }
105 #include "Command.def"
106 
107  // no match was found: ignore the TAB
108  //
109  if (matches.size() == 0) return ER_IGNORE;
110 
111  // if we have multiple matches but the user has provided a command
112  // argument then some matches were wrong. For example:
113  //
114  // LIB 3 matches LIB and LIBS
115  //
116  // remove wrong matches
117  //
118  if (matches.size() > 1 && arg.size() > 0)
119  {
120  again:
121  if (matches.size() > 1)
122  {
123  loop(m, matches.size())
124  {
125  if (matches[m].size() != cmd.size()) // wrong match
126  {
127  matches[m].swap(matches.back());
128  matches.pop_back();
129  goto again;
130  }
131  }
132  }
133  }
134 
135  // if multiple matches were found then either expand the common part
136  // or list all matches
137  //
138  if (matches.size() > 1) // multiple commands match cmd
139  {
140  user.clear();
141  return show_alternatives(user, cmd.size(), matches);
142  }
143 
144  // unique match
145  //
146  if (cmd.size() < matches[0].size())
147  {
148  // the command is longer than user, so we expand it.
149  //
150  user = matches[0];
151  if (ehint != EH_NO_PARAM) user.append(UNI_ASCII_SPACE);
152  return ER_REPLACE;
153  }
154 
155  if (cmd.size() == matches[0].size() &&
156  ehint != EH_NO_PARAM &&
157  arg.size() == 0 &&
158  !have_trailing_blank) // no blank entered
159  {
160  // the entire command was entered but without a blank. If the
161  // command has arguments then append a space to indicate that.
162  // Otherwiese fall throught to expand_command_arg();
163  //
164  user = matches[0];
165  user.append(UNI_ASCII_SPACE);
166  return ER_REPLACE;
167  }
168 
169  return expand_command_arg(user, ehint, shint, matches[0], arg);
170 }

201 ExpandResult
202 TabExpansion::expand_help_topics(UCS_string & user)
203 {
204  if (user.size() <= 5) return expand_help_topics(); // initial display
205 
206 const UCS_string help(user, 0, 6);
207 UCS_string prefix(user, 5, user.size() - 5); // the name prefix
208  prefix.remove_leading_whitespaces();
209 
210 std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
211 
212 UCS_string_vector matches;
213  loop(s, symbols.size())
214  {
215  const Symbol * sym = symbols[s];
216  if (sym->is_erased()) continue;
217 
218  const UCS_string & sym_name = sym->get_name();
219  if (!sym_name.starts_with(prefix)) continue;
220  matches.push_back(sym_name);
221  }
222 
223  if (matches.size() == 0) return ER_IGNORE; // no match
224  if (matches.size() > 1) // multiple matches
225  {
226  matches.sort();
227 
228  const int common_len = compute_common_length(prefix.size(), matches);
229  if (common_len > prefix.size())
230  {
231  // all matches can be extended in a unique way
232  //
233  matches[0].resize(common_len);
234  user = help + matches[0];
235  return ER_REPLACE;
236  }
237 
238  // all possible extensions are different
239  //
240  return show_alternatives(user, prefix.size(), matches);
241  }
242 
243  // unique match
244  //
245  user = help + matches[0];
246  return ER_REPLACE;
247 }

251 {
252  CIN << "\n";
253  CERR << "Help topics (APL primitives and user-defined names) are:" << endl;
254 
255  // show APL primotives (but only once). For that Help.def must be sorted.
256  //

258 int col = 0;

260 
261 #define help_def(_ar, prim, _name, _title, _descr) \
262  if (strcmp(prim, last)) \
263  { CERR << " " << (last = prim); col += 2; \
264  if (col > max_col) { CERR << endl; col = 0; } \
265  }
266 #include "Help.def"
267 
268 std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
269 
270 UCS_string_vector names;
271  loop(s, symbols.size())
272  {
273  const Symbol * sym = symbols[s];
274  if (sym->is_erased()) continue;
275 
276  const UCS_string & sym_name = sym->get_name();
277  names.push_back(sym_name);
278  }
279 
280  names.sort();
281 
282  // see if printing full names takes more than 5 lines. If so then
283  // only print first characters
284  //
285 int rows = 1; // the current row
286 int c1 = col;
287  loop(n, names.size())
288  {
289  const int len = 1 + names[n].size();
290  c1 += len;
291  if (c1 > max_col) { ++rows; c1 = len; }
292  }
293 
294  if (names.size() == 0) /* nothing to do */;
295  else if (rows < 6)
296  {
297  loop(n, names.size())
298  {
299  const int len = 1 + names[n].size();
300  col += len;
301  if (col > max_col) { CERR << endl; col = len; }

303  }
304  }
305  else

308  CERR << " " << uni;
309  for (ShapeItem n = 1; n < ShapeItem(names.size()); ++n)
310  {
311  if (names[n][0] == uni) continue; // same first character
312  uni = names[n][0];
313  CERR << " " << uni; col += 2;
314  if (col > max_col) { CERR << endl; col = 0; }
315  }
316  }
317 
318  CERR << endl;
319  return ER_AGAIN;
320 }

404 ExpandResult
405 TabExpansion::expand_filename(UCS_string & user,
406  ExpandHint ehint, const char * shint,
407  const UCS_string cmd, UCS_string arg)
408 {
409  if (arg.size() == 0)
410  {
411  // the user has entered a command but nothing else.
412  // If the command accepts a library number then we propose one of
413  // the existing libraries.
414  //
415  if (ehint == EH_oLIB_WSNAME || ehint == EH_DIR_OR_LIB)
416  {
417  // the command accepts a library reference number
418  //
419  UCS_string libs_present;
420  loop(lib, 10)
421  {
422  if (!LibPaths::is_present(LibRef(lib)))
423  continue;
424 
425  libs_present.append(UNI_ASCII_SPACE);
426  libs_present.append(Unicode(UNI_ASCII_0 + lib));
427  }
428  if (libs_present.size() == 0) goto nothing;
429 
430  CIN << endl;
431  CERR << cmd << libs_present << " <workspace-name>" << endl;
432  return ER_AGAIN;
433  }
434 
435  goto nothing;
436  }
437 
438  if (arg[0] >= '0' && arg[0] <= '9')
439  {
440  // library number 0-9. EH_DIR_OR_LIB is complete already so
441  // EH_oLIB_WSNAME is the only expansion case left
442  //
443  if (ehint != EH_oLIB_WSNAME) goto nothing;
444 
445  const LibRef lib = LibRef(arg[0] - '0');
446 
447  if (arg.size() == 1 && !have_trailing_blank) // no space yet
448  {
449  user = cmd;
450  user.append(UNI_ASCII_SPACE);
451  user.append(Unicode(arg[0]));
452  user.append(UNI_ASCII_SPACE);
453  return ER_REPLACE;
454  }
455 
456  // discard library reference number
457  //
458  if (arg.size() == 1) arg.erase(0);
459  if (arg.size()) arg.erase(0);
460  return expand_wsname(user, cmd, lib, arg);
461  }
462 
463  // otherwise: real file name
464  {
465  UCS_string dir_ucs;
466  const bool slash_at_1 = arg.size() > 1 && arg[1] == UNI_ASCII_SLASH;
467  const bool tilde_at_0 = arg[0] == UNI_ASCII_TILDE ||
468  arg[0] == UNI_TILDE_OPERATOR;
469 
470  if (arg[0] == UNI_ASCII_SLASH) // absolute path /xxx
471  {
472  dir_ucs = arg;
473  }
474  else if (arg[0] == UNI_ASCII_FULLSTOP && slash_at_1) // relative path ./xxx
475  {
476  const char * pwd = getenv("PWD");
477  if (pwd == 0) goto nothing;
478  dir_ucs = UCS_string(pwd);
479  dir_ucs.append(arg.drop(1));
480  }
481  else if (tilde_at_0 && slash_at_1) // user's home ~/
482  {
483  const char * home = getenv("HOME");
484  if (home == 0) goto nothing;
485  dir_ucs = UCS_string(home);
486  dir_ucs.append(arg.drop(1));
487  }
488  else if (tilde_at_0) // somebody's home
489  {
490  dir_ucs = UCS_string("/home/");
491  dir_ucs.append(arg.drop(1));
492  }
493  else goto nothing;
494 
495  UTF8_string dir_utf = UTF8_string(dir_ucs);
496 
497  const char * dir_dirname = dir_utf.c_str();
498  const char * dir_basename = strrchr(dir_dirname, '/');
499  if (dir_basename == 0) goto nothing;
500 
501  UTF8_string base_utf = UTF8_string(++dir_basename);
502  dir_utf.resize(dir_basename - dir_dirname);
503  dir_ucs = UCS_string(dir_utf);
504 
505  DIR * dir = opendir(dir_utf.c_str());
506  if (dir == 0) goto nothing;
507 
508  UCS_string_vector matches;
509  read_matching_filenames(dir, dir_utf, base_utf, ehint, matches);
510  closedir(dir);
511 
512  if (matches.size() == 0)
513  {
514  CIN << endl;
515  CERR << " no matching filesnames" << endl;
516  return ER_AGAIN;
517  }
518 
519  if (matches.size() > 1)
520  {
521  UCS_string prefix(base_utf);
522  user = cmd; // e.g. )LOAD
523  user.append(UNI_ASCII_SPACE);
524  user.append(dir_ucs); // e.g. )LOAD /usr/apl/
525  return show_alternatives(user, prefix.size(), matches);
526  }
527 
528  // unique match
529  //
530  user = cmd;
531  user.append(UNI_ASCII_SPACE);
532  user.append(dir_ucs);
533  user.append(matches[0]);
534  return ER_REPLACE;
535  }
536 
537 nothing:
538  CIN << endl;
539  CERR << cmd << " " << shint << endl;
540  return ER_AGAIN;
541 }

605 TabExpansion::compute_common_length(int len, const UCS_string_vector & matches)
606 {
607  // we assume that all matches have the same case
608 
609  for (;; ++len)
610  {
611  loop(m, matches.size())
612  {
613  if (len >= matches[m].size()) return matches[m].size();
614  if (matches[0][len] != matches[m][len]) return len;
615  }
616  }

618 //-----------------------------------------------------------------------------
619 void
620 TabExpansion::read_matching_filenames(DIR * dir, UTF8_string dirname,
621  UTF8_string prefix, ExpandHint ehint,
622  UCS_string_vector & matches)
623 {
624 const bool only_workspaces = (ehint == EH_oLIB_WSNAME) ||
625  (ehint == EH_WSNAME ) ||
626  (ehint == EH_oWSNAME );
627 
628  for (;;)
629  {
630  struct dirent * dent = readdir(dir);
631  if (dent == 0) break;
632 
633  const size_t dlen = strlen(dent->d_name);
634  if (dlen == 1 && dent->d_name[0] == '.') continue;
635  if (dlen == 2 && dent->d_name[0] == '.'
636  && dent->d_name[1] == '.') continue;
637 
638  if (strncmp(dent->d_name, prefix.c_str(), prefix.size())) continue;
639 
640  UCS_string name(dent->d_name);
641 
642  const bool is_dir = Command::is_directory(dent, dirname);
643  if (is_dir) name.append(UNI_ASCII_SLASH);
644  else if (only_workspaces)
645  {
646  const UTF8_string filename(dent->d_name);
647  bool is_wsname = false;
648  if (filename.ends_with(".apl")) is_wsname = true;
649  if (filename.ends_with(".xml")) is_wsname = true;
650  if (!is_wsname) continue;
651  }
652 
653  matches.push_back(name);
654  }
655 }

662

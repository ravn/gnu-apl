45 void
46 Parallel::init(bool logit)
47 {
48  // supposedly called only once...
49  //
50  Assert(!init_done);
51  init_done = true;
52 
53 #if PARALLEL_ENABLED
54 
55  // init global semaphores
56  //
57  __sem_init(print_sema, /* shared */ 0, /* value */ 1);
58  __sem_init(pthread_create_sema, /* shared */ 0, /* value */ 0);
59 
60  // compute number of cores available and set total_CPU_count accordingly
61  //
62  CPU_pool::init(logit);
63 
64  // initialize thread contexts
65  //
66  Thread_context::init_parallel(CPU_pool::get_count(), logit);
67 
68  // create threads...
69  //
70  Thread_context::get_context(CNUM_MASTER)->thread = pthread_self();
71  for (int w = CNUM_WORKER1; w < CPU_pool::get_count(); ++w)
72  {
73  Thread_context * tctx = Thread_context::get_context(CoreNumber(w));
74  const int result = pthread_create(&(tctx->thread), /* attr */ 0,
75  worker_main, tctx);
76  if (result)
77  {
78  CERR << "pthread_create() failed at " << LOC
79  << " : " << strerror(result) << endl;
80  Thread_context::set_active_core_count(CCNT_1);
81  return;
82  }
83 
84  // wait until new thread has reached its work loop
85  sem_wait(pthread_create_sema);
86  }
87 
88  // bind threads to cores
89  //
90  loop(c, CPU_pool::get_count())
91  {
92  const CPU_Number cpu = CPU_pool::get_CPU(c);
93  Thread_context::get_context(CoreNumber(c))->bind_to_cpu(cpu, logit);
94  }
95 
96  if (logit) Thread_context::print_all(CERR);
97 
98  // the threads above start in state locked. Wake them up...
99  //
100 # if CORE_COUNT_WANTED == -3
101  CPU_pool::change_core_count(CCNT_1, logit);
102 # else
103  CPU_pool::change_core_count(CPU_pool::get_count(), logit);
104 # endif
105 
106 #else // not PARALLEL_ENABLED
107 
108  Thread_context::init_sequential(logit);
109  CPU_pool::add_CPU(CPU_0);
110 
111 #endif // PARALLEL_ENABLED
112 }

120 bool
121 CPU_pool::change_core_count(CoreCount new_count, bool logit)
122 {
123  // this function is called from âSYL[26]ânew_countm but also from
124  // other places. It returns true on error.
125  //
126  Log(LOG_Parallel || logit)
127  get_CERR() << "change_core_count(" << new_count << ")" << endl;
128 
129  if (new_count < CCNT_0) return true; // error
130  if (new_count > CPU_pool::get_count()) return true; // error
131 
132  Parallel::run_parallel = new_count != CCNT_0;
133  if (new_count == CCNT_0) new_count = CCNT_1;
134 
135 const CoreCount current_count = Thread_context::get_active_core_count();
136  if (new_count == current_count)
137  {
138  Log(LOG_Parallel || logit)
139  {
140  CERR <<
141  "Parallel::change_core_count(): keeping current core count of "
142  << Thread_context::get_active_core_count() << endl;
143  Thread_context::print_all(CERR);
144  }
145  }
146  else if (new_count > current_count)
147  {
148  Log(LOG_Parallel || logit)
149  CERR << "Parallel::change_core_count(): increasing core count from "
150  << Thread_context::get_active_core_count()
151  << " to " << new_count << endl;
152 
153  lock_pool(logit);
154  Thread_context::set_active_core_count(new_count);
155  for (int c = 1; c < new_count; ++c)
156  Thread_context::get_context(CoreNumber(c))->job_number =
157  Thread_context::get_master().job_number;
158  unlock_pool(logit);
159 
160  Log(LOG_Parallel || logit) Thread_context::print_all(CERR);
161  }
162  else // new_count < current_count
163  {
164  Log(LOG_Parallel || logit)
165  CERR << "Parallel::change_core_count(): decreasing core count from "
166  << Thread_context::get_active_core_count()
167  << " to " << new_count << endl;
168  lock_pool(logit);
169  Thread_context::set_active_core_count(new_count);
170  unlock_pool(logit);
171 
172  Log(LOG_Parallel || logit) Thread_context::print_all(CERR);
173  }
174 
175  return false; // no error
176 }

178 #if PARALLEL_ENABLED
179 void
180 CPU_pool::init(bool logit)
181 {
182  // determine the max number of cores. We first handle the ./configure cases
183  // that can live without pthread_getaffinity_np() and friends
184  //
185 CoreCount count = CoreCount(CORE_COUNT_WANTED);
186 
187 #if CORE_COUNT_WANTED >= 1 // static parallel
188 
189  Parallel::run_parallel = true;
190  loop(c, CORE_COUNT_WANTED) add_CPU(CPU_Number(c));
191  return;
192 
193 #elif CORE_COUNT_WANTED == 0 // sequential
194 
195  Parallel::run_parallel = false;
196  add_CPU(CPU_0);
197  return;
198 
199 #elif CORE_COUNT_WANTED == -1 // handled below
200 # if ! HAVE_AFFINITY_NP
201 # error "CORE_COUNT_WANTED == -1 cannot be used on platforms without pthread_getaffinity_np"
202 # endif
203 
204 #elif CORE_COUNT_WANTED == -2 // --cc N
205 
206  Parallel::run_parallel = uprefs.requested_cc > 0;
207  loop(c, uprefs.requested_cc) add_CPU(CPU_Number(c));
208  return;
209 
210 #elif CORE_COUNT_WANTED == -3 // handled below
211 # if ! HAVE_AFFINITY_NP
212 # error "CORE_COUNT_WANTED == -3 cannot be used on platforms without pthread_getaffinity_np"
213 # endif
214 
215 #else // CORE_COUNT_WANTED == xxx
216 
217 # error "Bad CORE_COUNT_WANTED value in ./configure"
218 
219 #endif // CORE_COUNT_WANTED == xxx
220 
221  // at this point CORE_COUNT_WANTED is -1 (all cores) or -3 (âSYL)
222  // Figure how many cores we have.
223  //
224 #if ! HAVE_AFFINITY_NP
225 
226  Log(LOG_Parallel || logit)
227  {
228  CERR <<
229  "The number of cores could not be detected because function\n"
230  "pthread_getaffinity_np() is not provided by your platform.\n"
231  "Assuming a maximum of 64 cores." << endl;
232  }
233 
234  Parallel::run_parallel = true;
235  loop(c, 64) add_CPU(CPU_Number(c));
236  return;
237 
238 #endif
239 
240 cpu_set_t CPUs;
241  CPU_ZERO(&CPUs);
242 
243 const int err = pthread_getaffinity_np(pthread_self(), sizeof(CPUs), &CPUs);
244  if (err)
245  {
246  CERR << "pthread_getaffinity_np() failed with error "
247  << err << endl;
248  add_CPU(CPU_0);
249  return;
250  }
251 
252  {
253  const CoreCount CPU_count = CoreCount(CPU_COUNT(&CPUs));
254  // start with even CPUs followed by odd CPUs. This is to avoid that
255  // a secondart CPU thread is used before all primary threads have been
256  // exhausted.
257  //
258  for (size_t c = 0; c < 8*sizeof(cpu_set_t); c += 2) //even
259  {
260  if (CPU_ISSET(c, &CPUs))
261  {
262  add_CPU(CPU_Number(c));
263  if (get_count() == CPU_count) break; // all CPUs found
264  }
265  }
266 
267  for (size_t c = 1; c < 8*sizeof(cpu_set_t); c += 2) //odd
268  {
269  if (CPU_ISSET(c, &CPUs))
270  {
271  add_CPU(CPU_Number(c));
272  if (get_count() == CPU_count) break; // all CPUs found
273  }
274  }
275  }
276 
277  if (get_count() == 0)
278  {
279  CERR << "*** no cores detected, assuming at least one! "
280  << err << endl;
281  add_CPU(CPU_0);
282  return;
283  }
284 
285  // at this point we know the max. number of CPUs and can map cases
286  // -1 and -3 to the max. number of CPUs.
287  //
288  if (count < 0) count = get_count();
289 
290  // if there are more CPUs than requested then limit all_CPUs accordingly
291  if (get_count() > count) resize(count);
292 
293  Log(LOG_Parallel || logit)
294  {
295  CERR << "detected " << get_count() << " cores:";
296  loop(cc, get_count()) CERR << " #" << get_CPU(cc);
297  CERR << endl;
298  }
299 }

311 void
312 CPU_pool::lock_pool(bool logit)
313 {
314  if (Thread_context::get_active_core_count() <= 1) return;
315 
316  Thread_context::get_master().M_lock_pool();
317  Thread_context::M_join();
318 
319  Log(LOG_Parallel || logit)
320  {
321  PRINT_LOCKED(
322  CERR << "Parallel::lock_pool() : pool state is now :" << endl; )
323  Thread_context::print_all(CERR);
324  }
325 }
326 //-----------------------------------------------------------------------------
327 void
328 CPU_pool::unlock_pool(bool logit)
329 {
330  if (Thread_context::get_active_core_count() <= 1) return;
331 
332  Log(LOG_Parallel || logit)
333  {
334  for (int a = 1; a < Thread_context::get_active_core_count(); ++a)
335  {
336  PRINT_LOCKED(CERR << "Parallel::unlock_pool() : " << endl; )
337  Thread_context * tc = Thread_context::get_context(CoreNumber(a));
338  sem_post(&tc->pool_sema);
339  PRINT_LOCKED(
340  CERR << " pool_sema of thread #" << a << " incremented."
341  << endl;)
342  }
343  }
344  else
345  {
346  for (int a = 1; a < Thread_context::get_active_core_count(); ++a)
347  sem_post(&Thread_context::get_context(CoreNumber(a))->pool_sema);
348  }
349 }
350 //-----------------------------------------------------------------------------
351 void *
352 Parallel::worker_main(void * arg)
353 {
354 Thread_context & tctx = *reinterpret_cast<Thread_context *>(arg);
355 
356  Log(LOG_Parallel)
357  {
358  PRINT_LOCKED(CERR << "worker #" << tctx.get_N() << " started" << endl)
359  }
360 
361  // tell the creator that we have started
362  //
363  sem_post(pthread_create_sema);
364 
365  // start in state locked
366  //
367  tctx.blocked = true;
368  sem_wait(&tctx.pool_sema);
369  tctx.blocked = false;
370 
371  Log(LOG_Parallel)
372  PRINT_LOCKED(CERR << "thread #" << tctx.get_N()
373  << " was unblocked (initially) from pool_sema" << endl)
374 
375  for (;;)
376  {
377  tctx.PF_fork();
378  tctx.job_name = Thread_context::get_master().job_name;
379  tctx.do_join = true;
380  Thread_context::do_work(tctx);
381  if (tctx.do_join) tctx.PF_join();
382  }
383 
384  /* not reached */
385  return 0;
386 }

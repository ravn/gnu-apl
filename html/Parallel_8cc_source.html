43 void
44 Parallel::init(bool logit)
45 {
46 #if PARALLEL_ENABLED
47 
48  // init global semaphores
49  //
50  __sem_init(print_sema, /* shared */ 0, /* value */ 1);
51  __sem_init(pthread_create_sema, /* shared */ 0, /* value */ 0);
52 
53  // compute number of cores available and set total_CPU_count accordingly
54  //
55  init_all_CPUs(logit);
56 
57  // initialize thread contexts
58  //
59  Thread_context::init_parallel(get_max_core_count(), logit);
60 
61  // create threads...
62  //
63  Thread_context::get_context(CNUM_MASTER)->thread = pthread_self();
64  for (int w = CNUM_WORKER1; w < get_max_core_count(); ++w)
65  {
66  Thread_context * tctx = Thread_context::get_context(CoreNumber(w));
67  const int result = pthread_create(&(tctx->thread), /* attr */ 0,
68  worker_main, tctx);
69  if (result)
70  {
71  CERR << "pthread_create() failed at " << LOC
72  << " : " << strerror(result) << endl;
73  Thread_context::set_active_core_count(CCNT_1);
74  return;
75  }
76 
77  // wait until new thread has reached its work loop
78  sem_wait(pthread_create_sema);
79  }
80 
81  // bind threads to cores
82  //
83  loop(c, get_max_core_count())
84  Thread_context::get_context(CoreNumber(c))
85  ->bind_to_cpu(all_CPUs[c], logit);
86 
87  if (logit) Thread_context::print_all(CERR);
88 
89  // the threads above start in state locked. Wake them up...
90  //
91 #if CORE_COUNT_WANTED == -3
92  set_core_count(CCNT_1, logit);
93 #else
94  set_core_count(get_max_core_count(), logit);
95 #endif
96 
97 #else // not PARALLEL_ENABLED
98 
99  Thread_context::init_sequential(logit);
100  all_CPUs.push_back(CPU_0);
101 
102 #endif // PARALLEL_ENABLED
103 }

111 bool
112 Parallel::set_core_count(CoreCount new_count, bool logit)
113 {
114  // this function is called from ⎕SYL[26]←
115  //
116  if (new_count < CCNT_0) return true; // error
117  if (new_count > get_max_core_count()) return true; // error
118 
119  run_parallel = new_count != CCNT_0;
120  if (new_count == CCNT_0) new_count = CCNT_1;
121 
122  if (new_count == Thread_context::get_active_core_count())
123  {
124  Log(LOG_Parallel || logit)
125  {
126  CERR <<
127  "Parallel::set_core_count(): keeping current core count of "
128  << Thread_context::get_active_core_count() << endl;
129  Thread_context::print_all(CERR);
130  }
131 
132  return false;
133  }
134 
135  if (new_count > Thread_context::get_active_core_count())
136  {
137  Log(LOG_Parallel || logit)
138  CERR << "Parallel::set_core_count(): increasing core count from "
139  << Thread_context::get_active_core_count()
140  << " to " << new_count << endl;
141 
142  lock_pool(logit);
143  Thread_context::set_active_core_count(new_count);
144  for (int c = 1; c < new_count; ++c)
145  Thread_context::get_context(CoreNumber(c))->job_number =
146  Thread_context::get_master().job_number;
147  unlock_pool(logit);
148 
149  Log(LOG_Parallel || logit) Thread_context::print_all(CERR);
150  }
151  else
152  {
153  Log(LOG_Parallel || logit)
154  CERR << "Parallel::set_core_count(): decreasing core count from "
155  << Thread_context::get_active_core_count()
156  << " to " << new_count << endl;
157  lock_pool(logit);
158  Thread_context::set_active_core_count(new_count);
159  unlock_pool(logit);
160 
161  Log(LOG_Parallel || logit) Thread_context::print_all(CERR);
162  }
163 
164  return false; // no error
165 }
166 //-----------------------------------------------------------------------------
167 void
168 Parallel::init_all_CPUs(bool logit)
169 {
170  // determine the max number of cores. We first handle the ./configure cases
171  // that can live without pthread_getaffinity_np() and friends
172  //
173 int count = CORE_COUNT_WANTED;
174 
175 #if CORE_COUNT_WANTED == 0
176 
177  run_parallel = false;
178  all_CPUs.push_back(CPU_0);
179  return;
180 
181 #elif CORE_COUNT_WANTED == 1
182 
183  run_parallel = true;
184  all_CPUs.push_back(CPU_0);
185  return;
186 
187 #elif CORE_COUNT_WANTED == -2 // --cc N
188 
189 #elif ! HAVE_AFFINITY_NP
190 
191  if (uprefs.requested_cc < 1) // serial or 1 core
192  {
193  run_parallel = uprefs.requested_cc > 0;
194  all_CPUs.push_back(CPU_0);
195  return;
196  }
197  else
198  {
199  count = uprefs.requested_cc;
200  }
201 #endif
202 
203  // at this point count is -1, -3, or > 1. Figure how many cores we have.
204  //
205 #if HAVE_AFFINITY_NP
206 
207 cpu_set_t CPUs;
208  CPU_ZERO(&CPUs);
209 
210 const int err = pthread_getaffinity_np(pthread_self(), sizeof(CPUs), &CPUs);
211  if (err)
212  {
213  CERR << "pthread_getaffinity_np() failed with error "
214  << err << endl;
215  all_CPUs.push_back(CPU_0);
216  return;
217  }
218 
219  {
220  const size_t CPU_count = CPU_COUNT(&CPUs);
221  loop(c, 8*sizeof(cpu_set_t))
222  {
223  if (CPU_ISSET(c, &CPUs))
224  {
225  all_CPUs.push_back(CPU_Number(c));
226  if (all_CPUs.size() == CPU_count) break; // all CPUs found
227  }
228  }
229  }
230 
231  if (all_CPUs.size() == 0)
232  {
233  CERR << "*** no cores detected, assuming at least one! "
234  << err << endl;
235  all_CPUs.push_back(CPU_0);
236  return;
237  }
238 
239  // at this point we know the max. number of CPUs and can map cases
240  // -1 and -3 to the max. number of CPUs.
241  //
242  if (count < 0) count = all_CPUs.size();
243 
244  // if there are more CPUs than requested then limit all_CPUs accordingly
245  if (all_CPUs.size() > size_t(count)) all_CPUs.resize(count);
246 
247  Log(LOG_Parallel || logit)
248  {
249  CERR << "detected " << all_CPUs.size() << " cores:";
250  loop(cc, all_CPUs.size()) CERR << " #" << all_CPUs[cc];
251  CERR << endl;
252  }
253 
254 #else // ! HAVE_AFFINITY_NP
255 
256  if (count < 0) count = 64;
257 
258  loop(c, CORE_COUNT_WANTED) all_CPUs.push_back(CPU_Number(c));
259 
260  Log(LOG_Parallel || logit)
261  {
262  CERR <<
263 "The precise number of cores could not be detected because function\n"
264 "pthread_getaffinity_np() is not provided by your platform. Assuming a\n"
265 " maximum of " << all_CPUs.size() << " cores." << endl;
266  }
267 
268 #endif // HAVE_AFFINITY_NP
269 }

272 Parallel::lock_pool(bool logit)
273 {
274  if (Thread_context::get_active_core_count() <= 1) return;
275 
276  Thread_context::get_master().M_lock_pool();
277  Thread_context::M_join();
278 
279  Log(LOG_Parallel || logit)

284  }

286 //-----------------------------------------------------------------------------
287 void
288 Parallel::unlock_pool(bool logit)
289 {
290  if (Thread_context::get_active_core_count() <= 1) return;
291 
292  Log(LOG_Parallel || logit)
293  {
294  for (int a = 1; a < Thread_context::get_active_core_count(); ++a)
295  {
296  PRINT_LOCKED(CERR << "Parallel::unlock_pool() : " << endl; )
297  Thread_context * tc = Thread_context::get_context(CoreNumber(a));
298  sem_post(tc->pool_sema);
299  PRINT_LOCKED(
300  CERR << " pool_sema of thread #" << a << " incremented."
301  << endl;)
302  }
303  }
304  else
305  {
306  for (int a = 1; a < Thread_context::get_active_core_count(); ++a)
307  sem_post(Thread_context::get_context(CoreNumber(a))->pool_sema);
308  }
309 }
310 //-----------------------------------------------------------------------------
311 void *
312 Parallel::worker_main(void * arg)
313 {
314 Thread_context & tctx = *reinterpret_cast<Thread_context *>(arg);
315 
316  Log(LOG_Parallel)
317  {
318  PRINT_LOCKED(CERR << "worker #" << tctx.get_N() << " started" << endl)
319  }
320 
321  // tell the creator that we have started
322  //
323  sem_post(pthread_create_sema);
324 
325  // start in state locked
326  //
327  tctx.blocked = true;
328  sem_wait(tctx.pool_sema);
329  tctx.blocked = false;
330 
331  Log(LOG_Parallel)
332  PRINT_LOCKED(CERR << "thread #" << tctx.get_N()
333  << " was unblocked (initially) from pool_sema" << endl)
334 
335  for (;;)
336  {
337  tctx.PF_fork();
338  tctx.job_name = Thread_context::get_master().job_name;
339  tctx.do_join = true;
340  Thread_context::do_work(tctx);
341  if (tctx.do_join) tctx.PF_join();
342  }
343 
344  /* not reached */
345  return 0;
346 }

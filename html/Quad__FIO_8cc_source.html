194 //-----------------------------------------------------------------------------
195 void
196 Quad_FIO::do_sprintf(UCS_string & UZ, const UCS_string & A_format,
197  const Value * B, int b)
198 {
199  // A is the format string, B is the nested APL values for each % field in A.
200  // Result is the formatted string.
201  //
202 char numbuf[50];
203 
204  for (int f = 0; f < A_format.size(); /* no f++ */ )
205  {
206  const Unicode uni = A_format[f++];
207  if (uni != UNI_ASCII_PERCENT) // not %
208  {
209  UZ.append(uni);
210  continue;
211  }
212 
213  // % seen. copy the field in format to fmt and fprintf it with
214  // the next argument. That is for format = eg. "...%42.42llf..."
215  // we want fmt to be "%42.42llf"
216  //
217  char fmt[40];
218  unsigned int fm = 0; // an index into fmt;
219  fmt[fm++] = '%'; // copy the '%'
220  for (;;)
221  {
222  if (f >= A_format.size())
223  {
224  // end of format string reached without seeing the
225  // format char. We print the string and are done.
226  // (This was a mal-formed format string from the user)
227  //
228  UCS_string ufmt(fmt);
229  UZ.append(ufmt);
230  return;
231  }
232 
233  if (fm >= sizeof(fmt))
234  {
235  // end of fmt reached without seeing the format char.
236  // We print the string and are done.
237  // (This may or may not be a mal-formed format string
238  // from the user; we assume it is)
239  //
240  //
241  UCS_string ufmt(fmt);
242  UZ.append(ufmt);
243  goto field_done;
244  }
245 
246  const Unicode un1 = A_format[f++];
247  switch(un1)
248  {
249  // flag chars and field width/precision
250  //
251  case '#':
252  case '0' ... '9':
253  case '-':
254  case ' ':
255  case '+':
256  case '\'': // SUSE
257  case 'I': // glibc
258  case '.':
259 
260  // length modifiers
261  //
262  case 'h':
263  case 'l':
264  case 'L':
265  case 'q':
266  case 'j':
267  case 'z':
268  case 't': fmt[fm++] = un1; fmt[fm] = 0;
269  continue;
270 
271  // conversion specifiers
272  //
273  case 'd': case 'i': case 'o':
274  case 'u': case 'x': case 'X': case 'p':
275  {
276  const Cell & cell = B->get_ravel(b++);
277  APL_Integer iv;
278  if (cell.is_integer_cell())
279  {
280  iv = cell.get_int_value();
281  }
282  else
283  {
284  const double fv = cell.get_real_value();
285  if (fv < 0.0) iv = -int(-fv);
286  else iv = int(fv);
287  }
288  fmt[fm++] = un1; fmt[fm] = 0;
289  sprintf(numbuf, fmt, iv);
290  UZ.append_UTF8(numbuf);
291  }
292  goto field_done;
293 
294  case 'e': case 'E': case 'f': case 'F':
295  case 'g': case 'G': case 'a': case 'A':
296  {
297  const APL_Float fv =
298  B->get_ravel(b++).get_real_value();
299  fmt[fm++] = un1; fmt[fm] = 0;
300  sprintf(numbuf, fmt, fv);
301  UZ.append_UTF8(numbuf);
302  }
303  goto field_done;
304 
305  case 's': // string or char
306  if (B->get_ravel(b).is_character_cell()) goto cval;
307  {
308  Value_P str =
309  B->get_ravel(b++).get_pointer_value();
310  UCS_string ucs(*str.get());
311  UZ.append(ucs);
312  }
313  goto field_done;
314 
315  case 'c': // single char
316  cval:
317  UZ.append(B->get_ravel(b++).get_char_value());
318  goto field_done;
319 
320  case 'm':
321  sprintf(numbuf, "%s", strerror(errno));
322  UZ.append_UTF8(numbuf);
323  goto field_done;
324 
325  case '%':
326  if (fm == 0) // %% is %
327  {
328  UZ.append(UNI_ASCII_PERCENT);
329  goto field_done;
330  }
331  /* no break */
332 
333  default:
334  MORE_ERROR() << "invalid format character " << un1
335  << " in function 22 (aka. printf())"
336  " in module file_io:: ";
337  DOMAIN_ERROR; // bad format char
338  }
339  }
340  field_done: ;
341  }

345 Quad_FIO::get_random(APL_Integer mode, APL_Integer len)
346 {
347 const int device = open("/dev/urandom", O_RDONLY);
348  if (device == -1)
349  {
350  MORE_ERROR() << "⎕FIO[60] failed when trying to open /dev/urandom: "
351  << strerror(errno);
352  DOMAIN_ERROR;
353  }
354 
355 unsigned char buffer[32]; // caller has checked that len <= 32
356 const ssize_t bytes = read(device, buffer, len);
357  close(device);
358 
359  if (bytes == -1)
360  {
361  MORE_ERROR() << "⎕FIO[60] failed when trying to read /dev/urandom: "
362  << strerror(errno);
363  DOMAIN_ERROR;
364  }
365 
366  if (mode == 0) // single integer result (len ≤ 8)
367  {
368  APL_Integer result = 0;
369  loop(l, len) result = result << 8 | (buffer[l] & 0xFF);
370  return IntScalar(result, LOC);
371  }
372  else if (mode == 1) // byte vector result
373  {
374  Value_P Z(len, LOC);
375  loop(l, len) new (Z->next_ravel()) IntCell(buffer[l] & 0xFF);
376  return Z;
377  }
378 
379  MORE_ERROR() << "Invalid mode A in A ⎕FIO[60] B";
380  DOMAIN_ERROR;

383 Unicode
384 Quad_FIO::fget_utf8(FILE * file, ShapeItem & fget_count)
385 {
386 const int b0 = fgetc(file);
387  if (b0 == EOF) return UNI_EOF;
388 
389  ++fget_count;
390  if (!(b0 & 0x80)) return Unicode(b0); // ASCII
391 
392 int len,bx;
393  if ((b0 & 0xE0) == 0xC0) { len = 2; bx = b0 & 0x1F; }
394  else if ((b0 & 0xF0) == 0xE0) { len = 3; bx = b0 & 0x0F; }
395  else if ((b0 & 0xF8) == 0xF0) { len = 4; bx = b0 & 0x0E; }
396  else if ((b0 & 0xFC) == 0xF8) { len = 5; bx = b0 & 0x0E; }
397  else if ((b0 & 0xFE) == 0xFC) { len = 6; bx = b0 & 0x0E; }
398  else return UNI_EOF;
399 
400 uint32_t uni = 0;
401  loop(l, len - 1)
402  {
403  const int subc = fgetc(file);
404  if (subc == EOF) return UNI_EOF;
405  bx <<= 6;
406  uni <<= 6;
407  uni |= subc & 0x3F;
408  }
409 
410 
411  return Unicode(bx | uni);
412 }

572 //-----------------------------------------------------------------------------
573 Token
574 Quad_FIO::do_scanf(File_or_String & input, const UCS_string & format)
575 {
576 ShapeItem count = 0;
577  if (format.size() == 0) LENGTH_ERROR;
578 
579  // we take the total number of % as an upper bound fir the number of items
580  //
581  loop(f, format.size() - 1)
582  {
583  if (format[f] == UNI_ASCII_PERCENT) ++count;
584  }
585 
586 Value_P Z(count, LOC);
587  loop(z, count) new (Z->next_ravel()) IntCell(0);
588 ShapeItem z = 0;
589 
590 Unicode lookahead = input.get_next();
591  if (lookahead == UNI_EOF) goto out;
592 
593  loop(f, format.size())
594  {
595  const Unicode fmt_ch = format[f];
596  if (fmt_ch == UNI_ASCII_SPACE)
597  {
598  // one space in the format matches 0 or more spaces in the input
599  //
600  while (lookahead == UNI_ASCII_SPACE)
601  {
602  lookahead = input.get_next();
603  if (lookahead == UNI_EOF) goto out;
604  }
605  continue;
606  }
607 
608  if (fmt_ch != UNI_ASCII_PERCENT)
609  {
610  // normal character in format: match with input
611  //
612  match:
613  if (fmt_ch != lookahead) goto out;
614  lookahead = input.get_next();
615  if (lookahead == UNI_EOF) goto out;
616  continue;
617  }
618 
619  ++f; // skip (first) %
620  const Unicode fmt_ch1 = format[f];
621  if (fmt_ch1 == UNI_ASCII_PERCENT) goto match; // double % is %
622 
623  Unicode conv = Unicode_0;
624  int conv_len = 0;
625  bool suppress = false;
626  for (;f < format.size(); ++f)
627  {
628  const Unicode cc = format[f];
629  if (cc == UNI_ASCII_ASTERISK) // assignment suppression
630  {
631  suppress = true;
632  continue;
633  }
634 
635  if (strchr("hjlLmqtz", cc)) // type modifier
636  {
637  // we provide our own conversion modifiers and ignore
638  // conversion modifiers given by the user
639  //
640  continue;
641  }
642 
643  if (strchr("0123456789", cc)) // field length
644  {
645  conv_len = 10*conv_len + (cc - '0');
646  continue;
647  }
648 
649  if (strchr("cdDfFeginousxX", cc))
650  {
651  conv = cc;
652  break; // conversion
653  }
654  }
655 
656  if (conv == Unicode_0)
657  {
658  MORE_ERROR() << "expecting conversion character "
659  "%, c, d, f, i, n, o, u, s, or x after %";
660  DOMAIN_ERROR;
661  }
662 
663  if (strchr("dDiouxX", conv)) // integer conversion
664  {
665  input.unget(lookahead); // let scanf_long_long() read it
666  const char fmt[] = { '%', 'l', 'l', char(conv), 0 };
667  long long val = 0;
668  const int count = input.scanf_long_long(fmt, & val);
669  lookahead = input.get_next();
670  if (lookahead == UNI_EOF) goto out;
671  if (count != 1) goto out;
672 
673  if (!suppress) new (&Z->get_ravel(z++)) IntCell(val);
674  }
675  else if (strchr("fFeg", conv)) // float conversion
676  {
677  input.unget(lookahead); // let scanf_double() read it
678  const char fmt[] = { '%', 'l', char(conv), 0 };
679  APL_Float val = 0;
680  const int count = input.scanf_double(fmt, &val);
681  lookahead = input.get_next();
682  if (lookahead == UNI_EOF) goto out;
683  if (count != 1) goto out;
684 
685  if (!suppress) new (&Z->get_ravel(z++)) FloatCell(val);
686  }
687  else if (conv == UNI_ASCII_c) // char(s)
688  {
689  if (conv_len == 0) // default: single char
690  {
691  if (!suppress) new (&Z->get_ravel(z++)) CharCell(lookahead);
692  lookahead = input.get_next();
693  if (lookahead == UNI_EOF) goto out;
694  }
695  else
696  {
697  UCS_string ucs;
698  loop(c, conv_len)
699  {
700  ucs.append(lookahead);
701  lookahead = input.get_next();
702  if (lookahead == UNI_EOF) break;
703  }
704 
705  if (!suppress)
706  {
707  Value_P ZZ(ucs, LOC);
708  new (&Z->get_ravel(z++))
709  PointerCell(ZZ.get(), Z.getref());
710  }
711  }
712  }
713  else if (conv == UNI_ASCII_s) // string
714  {
715  UCS_string ucs;
716  while (lookahead > UNI_ASCII_SPACE)
717  {
718  ucs.append(lookahead);
719  if (conv_len && ucs.size() >= conv_len) break;
720  lookahead = input.get_next();
721  if (lookahead == UNI_EOF) break;
722  }
723 
724  if (!suppress)
725  {
726  Value_P ZZ(ucs, LOC);
727  new (&Z->get_ravel(z++)) PointerCell(ZZ.get(), Z.getref());
728  }
729  }
730  else if (conv == UNI_ASCII_n) // characters consumed thus far
731  {
732  if (!suppress) new (&Z->get_ravel(z++))
733  IntCell(input.get_count());
734  }
735  }
736 
737 out:
738  // shorten Z to the actual number of converted items
739  //

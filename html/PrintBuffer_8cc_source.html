54 //-----------------------------------------------------------------------------
55 PrintBuffer::PrintBuffer(const Value & value, const PrintContext & _pctx,
56  ostream * out)
57  : complete(false)
58 {
59 PERFORMANCE_START(start_0)
60 
61  // Note: if ostream is non-0 then this value may be incomplete
62  // (as indicated by member complete if it is huge). This is to speed
63  // up printing if the value is discarded after having been printed
64 
65 const PrintStyle outer_style = _pctx.get_style();
66 const bool framed = outer_style & (PST_CS_MASK | PST_CS_OUTER);
67 PrintContext pctx(_pctx);
68  pctx.set_style(PrintStyle(outer_style &~ PST_CS_OUTER));
69 
70 const ShapeItem ec = value.element_count();
71 
72  if (value.is_scalar())
73  {
74  PERFORMANCE_START(start_1)
75  const Cell & cell = value.get_ravel(0);
76  PrintContext pctx1(pctx);
77  if (cell.need_scaling(pctx)) pctx1.set_scaled();
78 
79  *this = cell.character_representation(pctx1);
80 
81  // pad the value unless it is framed
82  if (value.compute_depth() > 1 && !framed)
83  {
84  pad_l(UNI_iPAD_U4, 1);
85  pad_r(UNI_iPAD_U5, 1);
86  }
87 
88  add_outer_frame(outer_style);
89 
90  if (out)
91  {
92  UCS_string ucs(*this, value.get_rank(), _pctx.get_PW());
93  if (ucs.size()) *out << ucs << endl;
94  }
95  complete = true;
96  PERFORMANCE_END(fs_PrintBuffer1_B, start_1, ec)
97  return;
98  }
99 
100  if (pctx.get_style() & PST_QUOTE_CHARS)
101  {
102  if (value.is_char_vector())
103  {
104  UCS_string ucs;
105  ucs.append(UNI_ASCII_DOUBLE_QUOTE);
106  loop(v, ec) ucs.append(value.get_ravel(v).get_char_value());
107  ucs.append(UNI_ASCII_DOUBLE_QUOTE);
108  append_ucs(ucs);
109  update_info();
110  complete = true;
111  return;
112  }
113 
114  if (value.is_char_array())
115  {
116  pctx.set_style(PR_BOXED_GRAPHIC2);
117  new (this) PrintBuffer(value, pctx, out);
118  return;
119  }
120  }
121 
122  if (ec == 0) // empty value of any dimension
123  {
124  pb_empty(value, pctx, outer_style);
125  if (out)
126  {
127  UCS_string ucs(*this, value.get_rank(), _pctx.get_PW());
128  if (ucs.size()) *out << ucs << endl;
129  }
130  update_info();
131  complete = true;
132  return;
133  }
134 
135  if (pctx.get_style() == PR_APL_FUN)
136  {
137  pb_for_function(value, pctx, outer_style);
138  if (out)
139  {
140  UCS_string ucs(*this, value.get_rank(), _pctx.get_PW());
141  if (ucs.size()) *out << ucs << endl;
142  }
143  update_info();
144  complete = true;
145  return;
146  }
147 
148  // non-trivial PrintBuffer
149  //
150 const ShapeItem cols = value.get_last_shape_item();
151 
152 vector<bool> scaling;
153  scaling.reserve(cols);
154  loop(c, cols) scaling.push_back(false);
155 
156 vector<PrintBuffer> pcols;
157  pcols.reserve(cols);
158  loop(c, cols) pcols.push_back(PrintBuffer());
159 
160 PrintBuffer * item_matrix = 0;
161  try { item_matrix = new PrintBuffer[ec]; }
162  catch (...)
163  {
164  MORE_ERROR() << "value too large to print ("
165  << cols << " columns, " << ec << " items)";
166  WS_FULL;
167  }
168 
169  do_PrintBuffer(value, pctx, out, outer_style, scaling, pcols, item_matrix);
170 
171  delete [] item_matrix;
172 
173  PERFORMANCE_END(fs_PrintBuffer_B, start_0, ec)
174 }
175 //-----------------------------------------------------------------------------
176 void
177 PrintBuffer::do_PrintBuffer(const Value & value, const PrintContext & pctx,
178  ostream * out, PrintStyle outer_style,
179  vector<bool> & scaling, vector<PrintBuffer> & pcols,
180  PrintBuffer * item_matrix)
181 {
182 const bool framed = outer_style & (PST_CS_MASK | PST_CS_OUTER);
183 const ShapeItem ec = value.element_count();
184 const uint64_t ii_count = interrupt_count;
185 const bool huge = out && ec > 10000;
186 const bool nested = !value.is_simple();
187 
188  {
189  // 1. create a vector with a bool per column that tells if the
190  // column needs scaling (i.e. exponential format) or not
191  //
192  const ShapeItem cols = value.get_last_shape_item();
193  const ShapeItem rows = ec/cols;
194 
195  loop(x, cols)
196  {
197  bool need_scaling = false;
198  loop(y, rows)
199  {
200  if (huge && (ii_count != interrupt_count)) goto interrupted;
201  if (value.get_ravel(x + y*cols).need_scaling(pctx))
202  {
203  need_scaling = true;
204  break;
205  }
206  }
207  scaling[x] = need_scaling;
208  }
209 
210  // 2. create a matrix of items.
211  // An item of the matrix is a top-level cell (possibly nested),
212  // therefore we have (â´,value) items. Items are rectangular.
213  //
214  PERFORMANCE_START(start_2)
215  vector<Rank> max_row_ranks;
216  max_row_ranks.reserve(rows);
217  loop(y, rows)
218  {
219  ShapeItem max_row_height = 0;
220  max_row_ranks.push_back(0);
221 
222  loop(x, cols)
223  {
224  if (huge && (ii_count != interrupt_count)) goto interrupted;
225 
226  PrintBuffer & item = item_matrix[y*cols + x];
227  PrintContext pctx1 = pctx;
228  if (scaling[x]) pctx1.set_scaled();
229  const Cell & cell = value.get_ravel(x + y*cols);
230  item = cell.character_representation(pctx1);
231  if (!item.get_height())
232  {
233  UCS_string empty;
234  item.append_ucs(empty);
235  }
236 
237  if (cell.is_pointer_cell())
238  {
239  Value_P sub_val = cell.get_pointer_value();
240  const Rank sub_rank = sub_val->get_rank();
241  if (max_row_ranks.back() < sub_rank)
242  max_row_ranks.back() = sub_rank;
243  }
244 
245  if (max_row_height < item.get_height())
246  max_row_height = item.get_height();
247 
248  Assert1(item.is_rectangular());
249  }
250 
251  // pad all items to the same height
252  //
253  loop(x, cols)
254  {
255  PrintBuffer & item = item_matrix[y*cols + x];
256  if (huge && (ii_count != interrupt_count)) goto interrupted;
257 
258  item.pad_height(UNI_iPAD_U6, max_row_height);
259  Assert1(item.is_rectangular());
260  }
261  }
262  PERFORMANCE_END(fs_PrintBuffer2_B, start_2, ec)
263 
264  // 3. align all columns (which pads them to the same width).
265  //
266  PERFORMANCE_START(start_3)
267  loop(x, cols)
268  {
269  ColInfo col_info_x;
270  loop(y, rows)
271  {
272  if (huge && (ii_count != interrupt_count)) goto interrupted;
273  PrintBuffer * item_row = item_matrix + y*cols;
274  col_info_x.consider(item_row[x].get_info());
275  }
276  if (col_info_x.real_len<(col_info_x.int_len + col_info_x.denom_len))
277  col_info_x.real_len = col_info_x.int_len + col_info_x.denom_len;
278 
279  loop(y, rows)
280  {
281  if (huge && (ii_count != interrupt_count)) goto interrupted;
282  PrintBuffer * item_row = item_matrix + y*cols;
283  item_row[x].align(col_info_x);
284  }
285  }
286  PERFORMANCE_END(fs_PrintBuffer3_B, start_3, ec)
287 
288  // 4. combine columms, then rows.
289  //
290  PERFORMANCE_START(start_4)
291 
292  int last_col_spacing = 0; // the col_spacing of the previous column
293  bool last_notchar = false; // the notchar property of the previous column
294 
295  loop(x, cols)
296  {
297  // merge the different items in column x of item_matrix into one
298  // PrintBuffer pcol. Insert separator rows as needed.
299  //
300  PrintBuffer & dest = pcols[x];
301  pcols[x] = PrintBuffer();
302 
303  // compute the final height of dest and reserve enough rows as to
304  // avoid unnecessary copies
305  //
306  {
307  ShapeItem dest_height = 0;
308  loop(y, rows)
309  {
310  const Rank Rk_y_1 = y ? max_row_ranks[y - 1] : 0;
311  const ShapeItem srows = separator_rows(y, value, nested,
312  max_row_ranks[y],
313  Rk_y_1);
314  const ShapeItem irows = item_matrix[y*cols + x].get_height();
315  Assert(irows);
316  dest_height += srows + irows;
317  }
318  dest.buffer.reserve(dest_height);
319  }
320 
321  loop(y, rows)
322  {
323  if (huge && (ii_count != interrupt_count)) goto interrupted;
324 
325  // insert separator row(s)
326  //
327  const ShapeItem srows = separator_rows(y, value, nested,
328  max_row_ranks[y],
329  y ? max_row_ranks[y - 1] : 0);
330  if (srows) // unless first row
331  {
332  const UCS_string sepa_row(dest.get_width(0), UNI_iPAD_L0);
333  loop(r, srows) dest.append_ucs(sepa_row);
334  }
335 
336  const PrintBuffer & src = item_matrix[y*cols + x];
337  dest.add_row(src);
338  }
339 
340  bool not_char = false; // determined by get_col_spacing()
341  const int col_spacing = value.get_col_spacing(not_char, x, framed);
342 
343  const int32_t max_spacing = (col_spacing > last_col_spacing)
344  ? col_spacing : last_col_spacing;
345  int32_t not_char_spaces = 0;
346 
347  if (huge && (ii_count != interrupt_count)) goto interrupted;
348 
349  if (x) // subsequent column
350  {
351  if (last_notchar)
352  {
353  // the previous column was 'notchar' so we add one pad
354  // char to it.
355  //
356  pcols[x - 1].pad_r(UNI_iPAD_U2, 1);
357  ++not_char_spaces;
358  }
359  else if (not_char)
360  {
361  // the current column is 'notchar' so we add one pad
362  // char to it.
363  //
364  dest.pad_l(UNI_iPAD_U3, 1);
365  ++not_char_spaces;
366  }
367 
368  // we want a total spacing of 'max_spacing'
369  // but we deduct the 'not_char_spaces' chars Â² and Â³
370  // that were appended above.
371  //
372  const int u7_pad_len = max_spacing - not_char_spaces;
373  if (u7_pad_len)
374  {
375  pcols[x - 1].pad_r(UNI_iPAD_U7, u7_pad_len);
376  }
377  }
378 
379  if (huge && (ii_count != interrupt_count)) goto interrupted;
380 
381  last_col_spacing = col_spacing;
382  last_notchar = not_char;
383  }
384 
385  // combine columns
386  //
387  for (int dx = 1; dx < cols; dx += dx)
388  for (int xx = dx; xx < cols; xx += dx)
389  {
390  pcols[xx - dx].append_col(pcols[xx]);
391  }
392  *this = pcols[0];
393 
394  if (value.compute_depth() > 1 && !framed)
395  {
396  pad_l(UNI_iPAD_U8, 1);
397  pad_r(UNI_iPAD_U9, 1);
398  }
399 
400  if (!is_rectangular()) // should not happen
401  {
402  Q1(get_height())
403  loop(h, get_height()) CERR << "w=" << get_width(h) << "*" << endl;
404  loop(h, get_height()) CERR << "*" << get_line(h) << "*" << endl;
405  }
406 
407  Assert(is_rectangular());
408  add_outer_frame(outer_style);
409 
410  PERFORMANCE_END(fs_PrintBuffer4_B, start_4, ec)
411  }
412 
413  {
414  PERFORMANCE_START(start_5)
415 
416  if (huge) // ergo: out
417  {
418  print_interruptible(*out, value.get_rank(), pctx.get_PW());
419  }
420  else if (out)
421  {
422  UCS_string ucs(*this, value.get_rank(), pctx.get_PW());
423  if (ucs.size()) *out << ucs << endl;
424  }
425 
426  PERFORMANCE_END(fs_PrintBuffer5_B, start_5, ec)
427  }
428 
429  complete = true;
430  update_info();
431  return;
432 
433 interrupted:
434  clear_attention_raised(LOC);
435  clear_interrupt_raised(LOC);
436  *out << endl << "INTERRUPT" << endl;
437 }
438 //-----------------------------------------------------------------------------
439 void
440 PrintBuffer::print_interruptible(ostream & out, Rank rank, int quad_PW)
441 {
442  if (get_height() == 0) return; // empty PrintBuffer
443 
444 const int total_width = get_width(0);
445 const int max_breaks = 2*total_width/quad_PW;
446 
447 ShapeItem * del = 0;
448 ShapeItem __breakpoints[PB_MAX_BREAKPOINTS];
449 ShapeItem * breakpoints = __breakpoints;
450  if (max_breaks >= PB_MAX_BREAKPOINTS)
451  breakpoints = del = new ShapeItem[max_breaks];
452 ShapeItem bp_len = 0;
453 
454  // print rows, breaking at breakpoints
455  //
456  loop(row, get_height())
457  {
458  if (row) out << endl; // end previous row
459  int col = 0;
460  int b = 0;
461 
462  while (col < total_width)
463  {
464  int chunk_len;
465  if (row == 0) // first row: set up breakpoints
466  {
467  chunk_len = get_line(0).compute_chunk_length(quad_PW, col);
468  breakpoints[bp_len++] = chunk_len;
469  }
470  else // subsequent row: re-use breakpoints
471  {
472  chunk_len = breakpoints[b++];
473  }
474 
475  if (col) out << endl << " ";
476  UCS_string trow(get_line(row), col, chunk_len);
477  trow.remove_trailing_padchars();
478 
479  loop(t, trow.size())
480  {
481  const Unicode uni = trow[t];
482  if (is_iPAD_char(uni)) out << " ";
483  else out << uni;
484  }
485  col += chunk_len;
486 
487  if (interrupt_is_raised())
488  {
489  out << endl << "INTERRUPT" << endl;
490  clear_attention_raised(LOC);
491  clear_interrupt_raised(LOC);
492  delete del;
493  return;
494  }
495  }
496  }
497 
498  out << endl;
499  delete del;
500 }
501 //-----------------------------------------------------------------------------
502 void
503 PrintBuffer::pb_empty(const Value & value, PrintContext pctx,
504  PrintStyle outer_style)
505 {
506  if (value.get_rank() == 1) // vector: 1 line
507  {
508  if (pctx.get_style() == PR_APL_FUN)
509  {
510  if (value.get_ravel(0).is_character_cell()) // ''
511  {
512  UCS_string ucs("''");
513  ColInfo ci;
514  *this = PrintBuffer(ucs, ci);
515  add_outer_frame(outer_style);
516  return;
517  }
518 
519  if (value.get_ravel(0).is_numeric()) // â¬
520  {
521  UTF8_string utf("â¬");
522  UCS_string ucs(utf);
523  ColInfo ci;
524  *this = PrintBuffer(ucs, ci);
525  add_outer_frame(outer_style);
526  return;
527  }
528  }
529 
530  UCS_string ucs; // empty
531  append_ucs(ucs);
532  add_outer_frame(outer_style);
533  return; // 1 row
534  }
535 
536 const Shape sh = value.get_shape().without_axis(value.get_rank()-1);
537  if (sh.get_volume() <= 1) // empty vector
538  {
539  add_outer_frame(outer_style);
540  return; // 0 rows
541  }
542 
543  // value has > 0 rows. Compute how many lines we need.
544  //
545 ShapeItem lines = sh.get_volume();
546  loop(s, sh.get_rank())
547  lines += s * (sh.get_shape_item(sh.get_rank() - s - 1));
548 
549  buffer.resize(lines);
550  add_outer_frame(outer_style);
551 }
552 //-----------------------------------------------------------------------------
553 void
554 PrintBuffer::pb_for_function(const Value & value, PrintContext pctx,
555  PrintStyle outer_style)
556 {
557 const ShapeItem ec = value.element_count();
558 UCS_string ucs;
559 
560  if (value.is_char_vector())
561  {
562  ucs.append(UNI_SINGLE_QUOTE);
563  loop(e, ec)
564  {
565  const Unicode uni = value.get_ravel(e).get_char_value();
566  ucs.append(uni);
567  if (uni == UNI_SINGLE_QUOTE) ucs.append(uni); // ' -> ''
568  }
569  ucs.append(UNI_SINGLE_QUOTE);
570  }
571  else
572  {
573  loop(e, ec)
574  {
575  PrintBuffer pb = value.get_ravel(e)
576  .character_representation(pctx);
577  if (e) ucs.append(UNI_ASCII_SPACE);
578  ucs.append(UCS_string(pb, 0, pctx.get_PW()));
579  }
580  }
581 
582 ColInfo ci;
583  *this = PrintBuffer(ucs, ci);
584  add_outer_frame(outer_style);
585 }
586 //-----------------------------------------------------------------------------
587 ShapeItem
588 PrintBuffer::separator_rows(ShapeItem y, const Value & value, bool nested,
589  Rank rk1, Rank rk2)
590 {
591  if (y == 0) return 0;
592 
593 const Shape & shape = value.get_shape();
594 
595 ShapeItem prod = 1;
596 ShapeItem ret = 0;
597  loop(r, shape.get_rank() - 2)
598  {
599  prod *= shape.get_shape_item(shape.get_rank() - r - 2);
600  if (y % prod == 0) ++ret;
601  else break;
602  }
603 
604  if (nested) // see lrm p. 138
605  {
606  const int max_rk = rk1 > rk2 ? rk1 : rk2;
607  if (max_rk > 1) ret += max_rk - 1;
608  }
609 
610  return ret;
611 }

628 //-----------------------------------------------------------------------------
629 void
630 PrintBuffer::pad_l(Unicode pad, ShapeItem count)
631 {
632  if (count == 1)
633  {
634  loop(y, get_height()) buffer[y].prepend(pad);
635  }
636  else
637  {
638  UCS_string pads(count, pad);
639  loop(y, get_height()) buffer[y] = pads + buffer[y];
640  }
641 }
642 //-----------------------------------------------------------------------------
643 void
644 PrintBuffer::pad_r(Unicode pad, ShapeItem count)
645 {
646 UCS_string ucs(count, pad);
647  loop(y, get_height()) buffer[y].append(ucs);
648 }
649 //-----------------------------------------------------------------------------
650 void
651 PrintBuffer::pad_height(Unicode pad, ShapeItem height)
652 {
653  if (height > get_height())
654  {
655  UCS_string ucs(get_width(0), pad);
656  while (height > get_height()) buffer.push_back(ucs);
657  }
658 }
659 //-----------------------------------------------------------------------------
660 void
661 PrintBuffer::pad_to_spaces()
662 {
663  loop(y, get_height())
664  loop(x, get_width(y))
665  if (is_iPAD_char(get_char(x, y))) set_char(x, y, UNI_ASCII_SPACE);
666 }
667 //-----------------------------------------------------------------------------
668 void
669 PrintBuffer::pad_height_above(Unicode pad, ShapeItem height)
670 {
671  if (height > get_height())
672  {
673  UCS_string ucs(get_width(0), pad);
674  while (height > get_height()) buffer.insert(buffer.begin(), ucs);
675  }
676 }
677 //-----------------------------------------------------------------------------
678 void
679 PrintBuffer::get_frame_chars(PrintStyle pst,
680  Unicode & HORI, Unicode & VERT,
681  Unicode & NW, Unicode & NE,
682  Unicode & SE, Unicode & SW)
683 {
684  switch(pst & PST_CS_MASK)
685  {
686  case PST_CS_ASCII:
687  HORI = UNI_ASCII_MINUS;
688  VERT = UNI_ASCII_BAR;
689  NW = UNI_ASCII_FULLSTOP;
690  NE = UNI_ASCII_FULLSTOP;
691  SE = UNI_SINGLE_QUOTE;
692  SW = UNI_SINGLE_QUOTE;
693  break;
694 
695  case PST_CS_THIN:
696  HORI = UNI_LINE_HORI;
697  VERT = UNI_LINE_VERT;
698  NW = UNI_LINE_DOWN_LEFT;
699  NE = UNI_LINE_DOWN_RIGHT;
700  SE = UNI_LINE_UP_RIGHT;
701  SW = UNI_LINE_UP_LEFT;
702  break;
703 
704  case PST_CS_THICK:
705  HORI = UNI_LINE_HORI2;
706  VERT = UNI_LINE_VERT2;
707  NW = UNI_LINE_DOWN2_LEFT2;
708  NE = UNI_LINE_DOWN2_RIGHT2;
709  SE = UNI_LINE_UP2_RIGHT2;
710  SW = UNI_LINE_UP2_LEFT2;
711  break;
712 
713  case PST_CS_DOUBLE:
714  HORI = UNI_LINE2_HORI;
715  VERT = UNI_LINE2_VERT;
716  NW = UNI_LINE2_DOWN_LEFT;
717  NE = UNI_LINE2_DOWN_RIGHT;
718  SE = UNI_LINE2_UP_RIGHT;
719  SW = UNI_LINE2_UP_LEFT;
720  break;
721 
722  default:
723  HORI = Unicode_0;
724  VERT = Unicode_0;
725  NW = Unicode_0;
726  NE = Unicode_0;
727  SE = Unicode_0;

963 //-----------------------------------------------------------------------------
964 void
965 PrintBuffer::append_aligned(const UCS_string & ucs, Unicode align)
966 {
967  Assert(is_rectangular());
968 
969 int ucs_pos = -1;
970  loop(u, ucs.size())
971  {
972  if (ucs[u] == align)
973  {
974  ucs_pos = u;
975  break;
976  }
977  }
978 
979 int this_pos = -1;
980  loop(y, buffer.size())
981  {
982  const UCS_string & row = buffer[y];
983  loop(u, row.size())
984  {
985  if (row[u] == align)
986  {
987  this_pos = u;
988  break;
989  }
990  }
991  if (this_pos != -1) break;
992  }
993 
994 const int ucs_w = ucs.size();
995 const int this_w = get_width(0);
996 
997 int ucs_l = 0; // padding left of ucs
998 int ucs_r = 0; // padding right of ucs
999 int this_l = 0; // padding left of this
1000 int this_r = 0; // padding right of this
1001 
1002  if (ucs_pos == -1) // no align char in ucs
1003  if (this_pos == -1) // no align char in this pb
1004  {
1005  // no align char at all: align at right border
1006  //
1007  // TTTTttt
1008  // UUUUuuu
1009  //
1010  if (this_w > ucs_w) ucs_l = this_w - ucs_w;
1011  else this_l = ucs_w - this_w;
1012  }
1013  else // align char only in this pb
1014  {
1015  //
1016  // TTTTttt.ttt
1017  // UUUUuuu
1018  //
1019  ucs_r = this_w - this_pos;
1020  if (this_pos > ucs_w) ucs_l = this_pos - ucs_w;
1021  else this_l = ucs_w - this_pos;
1022  }
1023  else // align char in ucs
1024  if (this_pos == -1) // no align char in this pb
1025  {
1026  //
1027  // TTTTttt
1028  // UUUUuuu.uuu
1029  //
1030  this_r = ucs_w - ucs_pos;
1031  if (ucs_pos > this_w) this_l = ucs_pos - this_w;
1032  else ucs_l = this_w - ucs_pos;
1033  }
1034  else // align char in this pb
1035  {
1036  //
1037  // TTTTttt.tttTTTT
1038  // UUUUuuu.uuuUUUU
1039  //
1040  if (ucs_pos > this_pos) this_l = ucs_pos - this_pos;
1041  else ucs_l = this_pos - ucs_pos;
1042  const int uu = ucs_w - ucs_pos;
1043  const int tt = this_w - this_pos;
1044  if (uu > tt) this_r = uu - tt;
1045  else ucs_r = tt - uu;
1046  }
1047 
1048  Assert(ucs_l >= 0);
1049  Assert(ucs_r >= 0);
1050  Assert(this_l >= 0);
1051  Assert(this_r >= 0);
1052 
1053 UCS_string ucs1;
1054 
1055  if (ucs_l > 0) ucs1.append(UCS_string(ucs_l, UNI_ASCII_SPACE));
1056  ucs1.append(ucs);
1057  if (ucs_r > 0) ucs1.append(UCS_string(ucs_r, UNI_ASCII_SPACE));
1058 
1059  if (this_l > 0) pad_l(UNI_ASCII_SPACE, this_l);
1060  if (this_r > 0) pad_r(UNI_ASCII_SPACE, this_r);
1061 
1062  buffer.push_back(ucs1);
1063 
1064  Assert(is_rectangular());
1065 }

1092 //-----------------------------------------------------------------------------
1093 void
1094 PrintBuffer::align(ColInfo & cols)
1095 {
1096  // this PrintBuffer is one (possibly nested) APL value.
1097  // Align the buffer:
1098  //
1099  // to the J (in a column containing complex numbers), or
1100  // to the decimal point (in a column containing non-complex numbers), or
1101  // to the left (in a column containing text or nested values).
1102  //
1103  // make sure that the item is (and remains) rectangular).
1104  //
1105  Assert(is_rectangular());
1106 
1107  if (cols.flags & has_j) align_j(cols);
1108  else if ((cols.flags & CT_NUMERIC)) align_dot(cols);
1109  else align_left(cols);
1110 
1111  Assert(is_rectangular());
1112 }
1113 //-----------------------------------------------------------------------------
1114 void
1115 PrintBuffer::align_dot(ColInfo & COL_INFO)
1116 {
1117  // align this PrintBuffer (one value) at the decimal dot of COL_INFO.
1118  //
1119  // COL_INFO is the desired ColInfo of the entire column while
1120  // col_info (a member of this PrintBuffer) is the (smaller) item
1121  // being aligned
1122  //
1123  Log(LOG_printbuf_align)
1124  {
1125  CERR << "before align_dot():" << endl
1126  << "desired COL_INFO = "
1127  "i-" << COL_INFO.int_len
1128  << " f-" << COL_INFO.fract_len
1129  << " r-" << COL_INFO.real_len
1130  << " Ã·" << COL_INFO.denom_len << endl
1131  << "this row = "
1132  "i-" << col_info.int_len
1133  << " f-" << col_info.fract_len
1134  << " r-" << col_info.real_len
1135  << " Ã·" << col_info.denom_len << endl;
1136  debug(CERR, 0);
1137  }
1138 
1139  Assert(buffer.size() > 0);
1140 
1141  // make sure that consider() has worked.
1142  //
1143  Assert(COL_INFO.int_len >= col_info.int_len);
1144  Assert(COL_INFO.fract_len >= col_info.fract_len);
1145  Assert(COL_INFO.real_len >= col_info.real_len);
1146  Assert(COL_INFO.denom_len >= col_info.denom_len);
1147 
1148  if (col_info.flags & CT_NUMERIC)
1149  {
1150  // numeric items are aligned at the decimal dot. First pad the
1151  // integer part with spaces to the left
1152  //
1153  if (COL_INFO.int_len > col_info.int_len)
1154  {
1155  const size_t diff = COL_INFO.int_len - col_info.int_len;
1156  pad_l(UNI_iPAD_L3, diff);
1157  col_info.real_len += diff;
1158  col_info.int_len += diff;
1159  }
1160 
1161  if (col_info.denom_len) // quotient: maybe pad right with spaces
1162  {
1163  const size_t diff = COL_INFO.real_len - col_info.real_len;
1164  if (diff)
1165  {
1166  pad_r(UNI_iPAD_L4, diff);
1167  col_info.real_len += diff;
1168  }
1169  }
1170  else
1171  {
1172  if (COL_INFO.fract_len > col_info.fract_len)
1173  {
1174  pad_fraction(COL_INFO.fract_len, COL_INFO.have_expo());
1175  }
1176 
1177  if (COL_INFO.real_len > col_info.real_len)
1178  {
1179  const size_t diff = COL_INFO.real_len - col_info.real_len;
1180  if (!(COL_INFO.flags & (real_has_E | imag_has_E))
1181  || col_info.have_expo()) // or has one already
1182  {
1183  pad_r(UNI_iPAD_L4, diff);
1184  }
1185  else // no expo yet: create one
1186  {
1187  Assert1(diff >= 2);
1188  pad_r(UNI_ASCII_E, 1);
1189  pad_r(UNI_ASCII_0, 1);
1190  pad_r(UNI_iPAD_L4, diff - 2);
1191  }
1192  col_info.real_len = COL_INFO.real_len;
1193  }
1194  }
1195  }
1196  else
1197  {
1198  // char items are right aligned
1199  //
1200  size_t LEN = COL_INFO.total_len();
1201  size_t len = col_info.total_len();
1202  if (LEN > len)
1203  {
1204  const size_t diff = LEN - len;
1205  pad_l(UNI_iPAD_L5, diff);
1206  col_info.int_len = COL_INFO.int_len;
1207  col_info.fract_len = COL_INFO.fract_len;
1208  col_info.real_len = COL_INFO.real_len;
1209  }
1210  }
1211 
1212  Log(LOG_printbuf_align) debug(CERR, "after align_dot()");
1213 }

1270 //-----------------------------------------------------------------------------
1271 void
1272 PrintBuffer::pad_fraction(int wanted_fract_len, bool want_expo)
1273 {
1274 const int diff = wanted_fract_len - col_info.fract_len;
1275  Assert1(diff > 0);
1276 
1277  // copy integer part of this PrintBuffer to to new_buf
1278  //
1279 UCS_string new_buf(buffer[0], 0, col_info.int_len);
1280 
1281  // copy fractional part to new_buf. If the number has no exponent part,
1282  // then we fill with spaces. Otherwise fill with '0', possibly inserting
1283  // a decimal point.
1284  //
1285  loop(f, col_info.fract_len)
1286  new_buf.append(buffer[0][col_info.int_len + f]);
1287  if (!want_expo) // no exponent, e.g. 1,0
1288  {
1289  loop(d, diff) new_buf.append(UNI_iPAD_L4);
1290  }
1291  else if (col_info.fract_len == 0) // no fractional part (yet), e.g. 1E2
1292  {
1293  new_buf.append(UNI_ASCII_FULLSTOP);
1294  loop(d, diff - 1) new_buf.append(UNI_ASCII_0);
1295  }
1296  else
1297  {
1298  loop(d, diff) new_buf.append(UNI_ASCII_0);
1299  }
1300 
1301  // copy exponent part
1302  //
1303  for (int ex = col_info.int_len + col_info.fract_len;
1304  ex < buffer[0].size(); ++ex)
1305  new_buf.append(buffer[0][ex]);
1306 
1307  col_info.fract_len = wanted_fract_len;
1308  col_info.real_len += diff;
1309 
1310  buffer[0] = new_buf;
1311 
1312  // if buffer is multi-len then pad remaining line to new length
1313  //
1314  for (ShapeItem h = 1; h < get_height(); ++h)
1315  {
1316  const int diff = new_buf.size() - get_width(h);
1317  if (diff > 0)
1318  {
1319  const UCS_string ucs(diff, UNI_iPAD_L4);
1320  buffer[h].append(ucs);
1321  }
1322  }
1323 }

1367 //-----------------------------------------------------------------------------
1368 void
1369 ColInfo::consider(const ColInfo & item)
1370 {
1371  // this is the collective ColInfo of an entire column, and item
1372  // is a new, not yet considered item in the columns.
1373  //
1374  flags |= item.flags;
1375  if (item.imag_len) flags |= has_j;
1376 
1377  if (int_len < item.int_len)
1378  {
1379  real_len += item.int_len - int_len;
1380  int_len = item.int_len;
1381  }
1382 
1383  if (item.denom_len)
1384  {

54 //-----------------------------------------------------------------------------
55 PrintBuffer::PrintBuffer(const Value & value, const PrintContext & _pctx,
56  ostream * out)
57  : complete(false)
58 {
59 PERFORMANCE_START(start_0)
60 
61  // Note: if ostream is non-0 then this value may be incomplete
62  // (as indicated by member complete if it is huge). This is to speed
63  // up printing if the value is discarded after having been printed
64 
65 const PrintStyle outer_style = _pctx.get_style();
66 const bool framed = outer_style & (PST_CS_MASK | PST_CS_OUTER);
67 PrintContext pctx(_pctx);
68  pctx.set_style(PrintStyle(outer_style &~ PST_CS_OUTER));
69 
70 const ShapeItem ec = value.element_count();
71 
72  if (value.is_scalar())
73  {
74  PERFORMANCE_START(start_1)
75  const Cell & cell = value.get_ravel(0);
76  PrintContext pctx1(pctx);
77  if (cell.need_scaling(pctx)) pctx1.set_scaled();
78 
79  *this = cell.character_representation(pctx1);
80 
81  // pad the value unless it is framed
82  if (value.compute_depth() > 1 && !framed)
83  {
84  pad_l(UNI_iPAD_U4, 1);
85  pad_r(UNI_iPAD_U5, 1);
86  }
87 
88  add_outer_frame(outer_style);
89 
90  if (out)
91  {
92  UCS_string ucs(*this, value.get_rank(), _pctx.get_PW());
93  if (ucs.size()) *out << ucs << endl;
94  }
95  complete = true;
96  PERFORMANCE_END(fs_PrintBuffer1_B, start_1, ec)
97  return;
98  }
99 
100  if (pctx.get_style() & PST_QUOTE_CHARS)
101  {
102  if (value.is_char_vector())
103  {
104  UCS_string ucs;
105  ucs.append(UNI_ASCII_DOUBLE_QUOTE);
106  loop(v, ec) ucs.append(value.get_ravel(v).get_char_value());
107  ucs.append(UNI_ASCII_DOUBLE_QUOTE);
108  append_ucs(ucs);
109  update_info();
110  complete = true;
111  return;
112  }
113 
114  if (value.is_char_array())
115  {
116  pctx.set_style(PR_BOXED_GRAPHIC2);
117  new (this) PrintBuffer(value, pctx, out);
118  return;
119  }
120  }
121 
122  if (ec == 0) // empty value of any dimension
123  {
124  pb_empty(value, pctx, outer_style);
125  if (out)
126  {
127  UCS_string ucs(*this, value.get_rank(), _pctx.get_PW());
128  if (ucs.size()) *out << ucs << endl;
129  }
130  update_info();
131  complete = true;
132  return;
133  }
134 
135  if (pctx.get_style() == PR_APL_FUN)
136  {
137  pb_for_function(value, pctx, outer_style);
138  if (out)
139  {
140  UCS_string ucs(*this, value.get_rank(), _pctx.get_PW());
141  if (ucs.size()) *out << ucs << endl;
142  }
143  update_info();
144  complete = true;
145  return;
146  }
147 
148  // non-trivial PrintBuffer
149  //
150 const ShapeItem cols = value.get_last_shape_item();
151 
152 vector<bool> scaling;
153  scaling.reserve(cols);
154  loop(c, cols) scaling.push_back(false);
155 
156 vector<PrintBuffer> pcols;
157  pcols.reserve(cols);
158  loop(c, cols) pcols.push_back(PrintBuffer());
159 
160 PrintBuffer * item_matrix = 0;
161  try { item_matrix = new PrintBuffer[ec]; }
162  catch (...)
163  {
164  MORE_ERROR() << "value too large to print ("
165  << cols << " columns, " << ec << " items)";
166  WS_FULL;
167  }
168 
169  do_PrintBuffer(value, pctx, out, outer_style, scaling, pcols, item_matrix);
170 
171  delete [] item_matrix;
172 
173  PERFORMANCE_END(fs_PrintBuffer_B, start_0, ec)
174 }
175 //-----------------------------------------------------------------------------
176 void
177 PrintBuffer::do_PrintBuffer(const Value & value, const PrintContext & pctx,
178  ostream * out, PrintStyle outer_style,
179  vector<bool> & scaling, vector<PrintBuffer> & pcols,
180  PrintBuffer * item_matrix)
181 {
182 const bool framed = outer_style & (PST_CS_MASK | PST_CS_OUTER);
183 const ShapeItem ec = value.element_count();
184 const uint64_t ii_count = interrupt_count;
185 const bool huge = out && ec > 10000;
186 const bool nested = !value.is_simple();
187 
188  {
189  // 1. create a vector with a bool per column that tells if the
190  // column needs scaling (i.e. exponential format) or not
191  //
192  const ShapeItem cols = value.get_last_shape_item();
193  const ShapeItem rows = ec/cols;
194 
195  loop(x, cols)
196  {
197  bool need_scaling = false;
198  loop(y, rows)
199  {
200  if (huge && (ii_count != interrupt_count)) goto interrupted;
201  if (value.get_ravel(x + y*cols).need_scaling(pctx))
202  {
203  need_scaling = true;
204  break;
205  }
206  }
207  scaling[x] = need_scaling;
208  }
209 
210  // 2. create a matrix of items.
211  // An item of the matrix is a top-level cell (possibly nested),
212  // therefore we have (⍴,value) items. Items are rectangular.
213  //
214  PERFORMANCE_START(start_2)
215  vector<Rank> max_row_ranks;
216  max_row_ranks.reserve(rows);
217  loop(y, rows)
218  {
219  ShapeItem max_row_height = 0;
220  max_row_ranks.push_back(0);
221 
222  loop(x, cols)
223  {
224  if (huge && (ii_count != interrupt_count)) goto interrupted;
225 
226  PrintBuffer & item = item_matrix[y*cols + x];
227  PrintContext pctx1 = pctx;
228  if (scaling[x]) pctx1.set_scaled();
229  const Cell & cell = value.get_ravel(x + y*cols);
230  item = cell.character_representation(pctx1);
231 
232  if (cell.is_pointer_cell())
233  {
234  Value_P sub_val = cell.get_pointer_value();
235  const Rank sub_rank = sub_val->get_rank();
236  if (max_row_ranks.back() < sub_rank)
237  max_row_ranks.back() = sub_rank;
238  }
239 
240  if (max_row_height < item.get_height())
241  max_row_height = item.get_height();
242 
243  Assert1(item.is_rectangular());
244  }
245 
246  // pad all items to the same height
247  //
248  loop(x, cols)
249  {
250  PrintBuffer & item = item_matrix[y*cols + x];
251  if (huge && (ii_count != interrupt_count)) goto interrupted;
252 
253  item.pad_height(UNI_iPAD_U6, max_row_height);
254  Assert1(item.is_rectangular());
255  }
256  }
257  PERFORMANCE_END(fs_PrintBuffer2_B, start_2, ec)
258 
259  // 3. align all columns (which pads them to the same width).
260  //
261  PERFORMANCE_START(start_3)
262  loop(x, cols)
263  {
264  ColInfo col_info_x;
265  loop(y, rows)
266  {
267  if (huge && (ii_count != interrupt_count)) goto interrupted;
268  PrintBuffer * item_row = item_matrix + y*cols;
269  col_info_x.consider(item_row[x].get_info());
270  }
271  if (col_info_x.real_len<(col_info_x.int_len + col_info_x.denom_len))
272  col_info_x.real_len = col_info_x.int_len + col_info_x.denom_len;
273 
274  loop(y, rows)
275  {
276  if (huge && (ii_count != interrupt_count)) goto interrupted;
277  PrintBuffer * item_row = item_matrix + y*cols;
278  item_row[x].align(col_info_x);
279  }
280  }
281  PERFORMANCE_END(fs_PrintBuffer3_B, start_3, ec)
282 
283  // 4. combine columms, then rows.
284  //
285  PERFORMANCE_START(start_4)
286 
287  int last_col_spacing = 0; // the col_spacing of the previous column
288  bool last_notchar = false; // the notchar property of the previous column
289 
290  loop(x, cols)
291  {
292  // merge the different items in column x of item_matrix into one
293  // PrintBuffer pcol. Insert separator rows as needed.
294  //
295  PrintBuffer & dest = pcols[x];
296  pcols[x] = PrintBuffer();
297 
298  // compute the final height of dest and reserve enough rows as to
299  // avoid unnecessary copies
300  //
301  {
302  ShapeItem dest_height = 0;
303  loop(y, rows)
304  {
305  dest_height += separator_rows(y, value, nested,
306  max_row_ranks[y],
307  y ? max_row_ranks[y - 1] : 0);
308  dest_height += item_matrix[y*cols + x].get_height();
309  }
310  dest.buffer.reserve(dest_height);
311  }
312 
313  loop(y, rows)
314  {
315  if (huge && (ii_count != interrupt_count)) goto interrupted;
316 
317  // insert separator row(s)
318  //
319  const ShapeItem srows = separator_rows(y, value, nested,
320  max_row_ranks[y],
321  y ? max_row_ranks[y - 1] : 0);
322  if (srows) // unless first row
323  {
324  const UCS_string sepa_row(dest.get_width(0), UNI_iPAD_L0);
325  loop(r, srows) dest.append_ucs(sepa_row);
326  }
327 
328  const PrintBuffer & src = item_matrix[y*cols + x];
329  dest.add_row(src);
330  }
331 
332  bool not_char = false; // determined by get_col_spacing()
333  const int col_spacing = value.get_col_spacing(not_char, x, framed);
334 
335  const int32_t max_spacing = (col_spacing > last_col_spacing)
336  ? col_spacing : last_col_spacing;
337  int32_t not_char_spaces = 0;
338 
339  if (huge && (ii_count != interrupt_count)) goto interrupted;
340 
341  if (x) // subsequent column
342  {
343  if (last_notchar)
344  {
345  // the previous column was 'notchar' so we add one pad
346  // char to it.
347  //
348  pcols[x - 1].pad_r(UNI_iPAD_U2, 1);
349  ++not_char_spaces;
350  }
351  else if (not_char)
352  {
353  // the current column is 'notchar' so we add one pad
354  // char to it.
355  //
356  dest.pad_l(UNI_iPAD_U3, 1);
357  ++not_char_spaces;
358  }
359 
360  // we want a total spacing of 'max_spacing'
361  // but we deduct the 'not_char_spaces' chars ² and ³
362  // that were appended above.
363  //
364  const int u7_pad_len = max_spacing - not_char_spaces;
365  if (u7_pad_len)
366  {
367  pcols[x - 1].pad_r(UNI_iPAD_U7, u7_pad_len);
368  }
369  }
370 
371  if (huge && (ii_count != interrupt_count)) goto interrupted;
372 
373  last_col_spacing = col_spacing;
374  last_notchar = not_char;
375  }
376 
377  // combine columns
378  //
379  for (int dx = 1; dx < cols; dx += dx)
380  for (int xx = dx; xx < cols; xx += dx)
381  {
382  pcols[xx - dx].append_col(pcols[xx]);
383  }
384  *this = pcols[0];
385 
386  if (value.compute_depth() > 1 && !framed)
387  {
388  pad_l(UNI_iPAD_U8, 1);
389  pad_r(UNI_iPAD_U9, 1);
390  }
391 
392  if (!is_rectangular()) // should not happen
393  {
394  Q1(get_height())
395  loop(h, get_height()) CERR << "w=" << get_width(h) << "*" << endl;
396  loop(h, get_height()) CERR << "*" << get_line(h) << "*" << endl;
397  }
398 
399  Assert(is_rectangular());
400  add_outer_frame(outer_style);
401 
402  PERFORMANCE_END(fs_PrintBuffer4_B, start_4, ec)
403  }
404 
405  {
406  PERFORMANCE_START(start_5)
407 
408  if (huge) // ergo: out
409  {
410  print_interruptible(*out, value.get_rank(), pctx.get_PW());
411  }
412  else if (out)
413  {
414  UCS_string ucs(*this, value.get_rank(), pctx.get_PW());
415  if (ucs.size()) *out << ucs << endl;
416  }
417 
418  PERFORMANCE_END(fs_PrintBuffer5_B, start_5, ec)
419  }
420 
421  complete = true;
422  update_info();
423  return;
424 
425 interrupted:
426  clear_attention_raised(LOC);
427  clear_interrupt_raised(LOC);
428  *out << endl << "INTERRUPT" << endl;
429 }
430 //-----------------------------------------------------------------------------
431 void
432 PrintBuffer::print_interruptible(ostream & out, Rank rank, int quad_PW)
433 {
434  if (get_height() == 0) return; // empty PrintBuffer
435 
436 const int total_width = get_width(0);
437 const int max_breaks = 2*total_width/quad_PW;
438 
439 ShapeItem * del = 0;
440 ShapeItem __breakpoints[PB_MAX_BREAKPOINTS];
441 ShapeItem * breakpoints = __breakpoints;
442  if (max_breaks >= PB_MAX_BREAKPOINTS)
443  breakpoints = del = new ShapeItem[max_breaks];
444 ShapeItem bp_len = 0;
445 
446  // print rows, breaking at breakpoints
447  //
448  loop(row, get_height())
449  {
450  if (row) out << endl; // end previous row
451  int col = 0;
452  int b = 0;
453 
454  while (col < total_width)
455  {
456  int chunk_len;
457  if (row == 0) // first row: set up breakpoints
458  {
459  chunk_len = get_line(0).compute_chunk_length(quad_PW, col);
460  breakpoints[bp_len++] = chunk_len;
461  }
462  else // subsequent row: re-use breakpoints
463  {
464  chunk_len = breakpoints[b++];
465  }
466 
467  if (col) out << endl << " ";
468  UCS_string trow(get_line(row), col, chunk_len);
469  trow.remove_trailing_padchars();
470 
471  loop(t, trow.size())
472  {
473  const Unicode uni = trow[t];
474  if (is_iPAD_char(uni)) out << " ";
475  else out << uni;
476  }
477  col += chunk_len;
478 
479  if (interrupt_is_raised())
480  {
481  out << endl << "INTERRUPT" << endl;
482  clear_attention_raised(LOC);
483  clear_interrupt_raised(LOC);
484  delete del;
485  return;
486  }
487  }
488  }
489 
490  out << endl;
491  delete del;
492 }
493 //-----------------------------------------------------------------------------
494 void
495 PrintBuffer::pb_empty(const Value & value, PrintContext pctx,
496  PrintStyle outer_style)
497 {
498  if (value.get_rank() == 1) // vector: 1 line
499  {
500  if (pctx.get_style() == PR_APL_FUN)
501  {
502  if (value.get_ravel(0).is_character_cell()) // ''
503  {
504  UCS_string ucs("''");
505  ColInfo ci;
506  *this = PrintBuffer(ucs, ci);
507  add_outer_frame(outer_style);
508  return;
509  }
510 
511  if (value.get_ravel(0).is_numeric()) // ⍬
512  {
513  UCS_string ucs("⍬");
514  ColInfo ci;
515  *this = PrintBuffer(ucs, ci);
516  add_outer_frame(outer_style);
517  return;
518  }
519  }
520 
521  UCS_string ucs; // empty
522  append_ucs(ucs);
523  add_outer_frame(outer_style);
524  return; // 1 row
525  }
526 
527 const Shape sh = value.get_shape().without_axis(value.get_rank()-1);
528  if (sh.get_volume() <= 1) // empty vector
529  {
530  add_outer_frame(outer_style);
531  return; // 0 rows
532  }
533 
534  // value has > 0 rows. Compute how many lines we need.
535  //
536 ShapeItem lines = sh.get_volume();
537  loop(s, sh.get_rank())
538  lines += s * (sh.get_shape_item(sh.get_rank() - s - 1));
539 
540  buffer.resize(lines);
541  add_outer_frame(outer_style);
542 }
543 //-----------------------------------------------------------------------------
544 void
545 PrintBuffer::pb_for_function(const Value & value, PrintContext pctx,
546  PrintStyle outer_style)
547 {
548 const ShapeItem ec = value.element_count();
549 UCS_string ucs;
550 
551  if (value.is_char_vector())
552  {
553  ucs.append(UNI_SINGLE_QUOTE);
554  loop(e, ec)
555  {
556  const Unicode uni = value.get_ravel(e).get_char_value();
557  ucs.append(uni);
558  if (uni == UNI_SINGLE_QUOTE) ucs.append(uni); // ' -> ''
559  }
560  ucs.append(UNI_SINGLE_QUOTE);
561  }
562  else
563  {
564  loop(e, ec)
565  {
566  PrintBuffer pb = value.get_ravel(e)
567  .character_representation(pctx);
568  if (e) ucs.append(UNI_ASCII_SPACE);
569  ucs.append(UCS_string(pb, 0, pctx.get_PW()));
570  }
571  }
572 
573 ColInfo ci;
574  *this = PrintBuffer(ucs, ci);
575  add_outer_frame(outer_style);
576 }
577 //-----------------------------------------------------------------------------
578 ShapeItem
579 PrintBuffer::separator_rows(ShapeItem y, const Value & value, bool nested,
580  Rank rk1, Rank rk2)
581 {
582  if (y == 0) return 0;
583 
584 const Shape & shape = value.get_shape();
585 
586 ShapeItem prod = 1;
587 ShapeItem ret = 0;
588  loop(r, shape.get_rank() - 2)
589  {
590  prod *= shape.get_shape_item(shape.get_rank() - r - 2);
591  if (y % prod == 0) ++ret;
592  else break;
593  }
594 
595  if (nested) // see lrm p. 138
596  {
597  const int max_rk = rk1 > rk2 ? rk1 : rk2;
598  if (max_rk > 1) ret += max_rk - 1;
599  }
600 
601  return ret;
602 }

666  }
667 }
668 //-----------------------------------------------------------------------------
669 void
670 PrintBuffer::get_frame_chars(PrintStyle pst,
671  Unicode & HORI, Unicode & VERT,
672  Unicode & NW, Unicode & NE,
673  Unicode & SE, Unicode & SW)
674 {
675  switch(pst & PST_CS_MASK)
676  {
677  case PST_CS_ASCII:
678  HORI = UNI_ASCII_MINUS;
679  VERT = UNI_ASCII_BAR;
680  NW = UNI_ASCII_FULLSTOP;
681  NE = UNI_ASCII_FULLSTOP;
682  SE = UNI_SINGLE_QUOTE;
683  SW = UNI_SINGLE_QUOTE;
684  break;
685 
686  case PST_CS_THIN:
687  HORI = UNI_LINE_HORI;
688  VERT = UNI_LINE_VERT;
689  NW = UNI_LINE_DOWN_LEFT;
690  NE = UNI_LINE_DOWN_RIGHT;
691  SE = UNI_LINE_UP_RIGHT;
692  SW = UNI_LINE_UP_LEFT;
693  break;
694 
695  case PST_CS_THICK:
696  HORI = UNI_LINE_HORI2;
697  VERT = UNI_LINE_VERT2;
698  NW = UNI_LINE_DOWN2_LEFT2;
699  NE = UNI_LINE_DOWN2_RIGHT2;
700  SE = UNI_LINE_UP2_RIGHT2;
701  SW = UNI_LINE_UP2_LEFT2;
702  break;
703 
704  case PST_CS_DOUBLE:
705  HORI = UNI_LINE2_HORI;
706  VERT = UNI_LINE2_VERT;
707  NW = UNI_LINE2_DOWN_LEFT;
708  NE = UNI_LINE2_DOWN_RIGHT;
709  SE = UNI_LINE2_UP_RIGHT;
710  SW = UNI_LINE2_UP_LEFT;
711  break;
712 
713  default:
714  HORI = Unicode_0;
715  VERT = Unicode_0;
716  NW = Unicode_0;
717  NE = Unicode_0;
718  SE = Unicode_0;
719  SW = Unicode_0;
720  FIXME;
721  }
722 }
723 //-----------------------------------------------------------------------------
724 void
725 PrintBuffer::add_frame(PrintStyle style, const Shape & shape, int depth)
726 {
727  Assert(is_rectangular());
728

954 //-----------------------------------------------------------------------------
955 void
956 PrintBuffer::append_aligned(const UCS_string & ucs, Unicode align)
957 {
958  Assert(is_rectangular());
959 
960 int ucs_pos = -1;
961  loop(u, ucs.size())
962  {
963  if (ucs[u] == align)
964  {
965  ucs_pos = u;
966  break;
967  }
968  }
969 
970 int this_pos = -1;
971  loop(y, buffer.size())
972  {
973  const UCS_string & row = buffer[y];
974  loop(u, row.size())
975  {
976  if (row[u] == align)
977  {
978  this_pos = u;
979  break;
980  }
981  }
982  if (this_pos != -1) break;
983  }
984 
985 const int ucs_w = ucs.size();
986 const int this_w = get_width(0);
987 
988 int ucs_l = 0; // padding left of ucs
989 int ucs_r = 0; // padding right of ucs
990 int this_l = 0; // padding left of this
991 int this_r = 0; // padding right of this
992 
993  if (ucs_pos == -1) // no align char in ucs
994  if (this_pos == -1) // no align char in this pb
995  {
996  // no align char at all: align at right border
997  //
998  // TTTTttt
999  // UUUUuuu
1000  //
1001  if (this_w > ucs_w) ucs_l = this_w - ucs_w;
1002  else this_l = ucs_w - this_w;
1003  }
1004  else // align char only in this pb
1005  {
1006  //
1007  // TTTTttt.ttt
1008  // UUUUuuu
1009  //
1010  ucs_r = this_w - this_pos;
1011  if (this_pos > ucs_w) ucs_l = this_pos - ucs_w;
1012  else this_l = ucs_w - this_pos;
1013  }
1014  else // align char in ucs
1015  if (this_pos == -1) // no align char in this pb
1016  {
1017  //
1018  // TTTTttt
1019  // UUUUuuu.uuu
1020  //
1021  this_r = ucs_w - ucs_pos;
1022  if (ucs_pos > this_w) this_l = ucs_pos - this_w;
1023  else ucs_l = this_w - ucs_pos;
1024  }
1025  else // align char in this pb
1026  {
1027  //
1028  // TTTTttt.tttTTTT
1029  // UUUUuuu.uuuUUUU
1030  //
1031  if (ucs_pos > this_pos) this_l = ucs_pos - this_pos;
1032  else ucs_l = this_pos - ucs_pos;
1033  const int uu = ucs_w - ucs_pos;
1034  const int tt = this_w - this_pos;
1035  if (uu > tt) this_r = uu - tt;
1036  else ucs_r = tt - uu;
1037  }
1038 
1039  Assert(ucs_l >= 0);
1040  Assert(ucs_r >= 0);
1041  Assert(this_l >= 0);
1042  Assert(this_r >= 0);
1043 
1044 UCS_string ucs1;
1045 
1046  if (ucs_l > 0) ucs1.append(UCS_string(ucs_l, UNI_ASCII_SPACE));
1047  ucs1.append(ucs);
1048  if (ucs_r > 0) ucs1.append(UCS_string(ucs_r, UNI_ASCII_SPACE));
1049 
1050  if (this_l > 0) pad_l(UNI_ASCII_SPACE, this_l);
1051  if (this_r > 0) pad_r(UNI_ASCII_SPACE, this_r);
1052 
1053  buffer.push_back(ucs1);
1054 
1055  Assert(is_rectangular());
1056 }

1083 //-----------------------------------------------------------------------------
1084 void
1085 PrintBuffer::align(ColInfo & cols)
1086 {
1087  // this PrintBuffer is one (possibly nested) APL value.
1088  // Align the buffer:
1089  //
1090  // to the J (in a column containing complex numbers), or
1091  // to the decimal point (in a column containing non-complex numbers), or
1092  // to the left (in a column containing text or nested values).
1093  //
1094  // make sure that the item is (and remains) rectangular).
1095  //
1096  Assert(is_rectangular());
1097 
1098  if (cols.flags & has_j) align_j(cols);
1099  else if ((cols.flags & CT_NUMERIC)) align_dot(cols);
1100  else align_left(cols);
1101 
1102  Assert(is_rectangular());
1103 }
1104 //-----------------------------------------------------------------------------
1105 void
1106 PrintBuffer::align_dot(ColInfo & COL_INFO)
1107 {
1108  // align this PrintBuffer (one value) at the decimal dot of COL_INFO.
1109  //
1110  // COL_INFO is the desired ColInfo of the entire column while
1111  // col_info (a member of this PrintBuffer) is the (smaller) item
1112  // being aligned
1113  //
1114  Log(LOG_printbuf_align)
1115  {
1116  CERR << "before align_dot():" << endl
1117  << "desired COL_INFO = "
1118  "i-" << COL_INFO.int_len
1119  << " f-" << COL_INFO.fract_len
1120  << " r-" << COL_INFO.real_len
1121  << " ÷" << COL_INFO.denom_len << endl
1122  << "this row = "
1123  "i-" << col_info.int_len
1124  << " f-" << col_info.fract_len
1125  << " r-" << col_info.real_len
1126  << " ÷" << col_info.denom_len << endl;
1127  debug(CERR, 0);
1128  }
1129 
1130  Assert(buffer.size() > 0);
1131 
1132  // make sure that consider() has worked.
1133  //
1134  Assert(COL_INFO.int_len >= col_info.int_len);
1135  Assert(COL_INFO.fract_len >= col_info.fract_len);
1136  Assert(COL_INFO.real_len >= col_info.real_len);
1137  Assert(COL_INFO.denom_len >= col_info.denom_len);
1138 
1139  if (col_info.flags & CT_NUMERIC)
1140  {
1141  // numeric items are aligned at the decimal dot. First pad the
1142  // integer part with spaces to the left
1143  //
1144  if (COL_INFO.int_len > col_info.int_len)
1145  {
1146  const size_t diff = COL_INFO.int_len - col_info.int_len;
1147  pad_l(UNI_iPAD_L3, diff);
1148  col_info.real_len += diff;
1149  col_info.int_len += diff;
1150  }
1151 
1152  if (col_info.denom_len) // quotient: maybe pad right with spaces
1153  {
1154  const size_t diff = COL_INFO.real_len - col_info.real_len;
1155  if (diff)
1156  {
1157  pad_r(UNI_iPAD_L4, diff);
1158  col_info.real_len += diff;
1159  }
1160  }
1161  else
1162  {
1163  if (COL_INFO.fract_len > col_info.fract_len)
1164  {
1165  pad_fraction(COL_INFO.fract_len, COL_INFO.have_expo());
1166  }
1167 
1168  if (COL_INFO.real_len > col_info.real_len)
1169  {
1170  const size_t diff = COL_INFO.real_len - col_info.real_len;
1171  if (!(COL_INFO.flags & (real_has_E | imag_has_E))
1172  || col_info.have_expo()) // or has one already
1173  {
1174  pad_r(UNI_iPAD_L4, diff);
1175  }
1176  else // no expo yet: create one
1177  {
1178  Assert1(diff >= 2);
1179  pad_r(UNI_ASCII_E, 1);
1180  pad_r(UNI_ASCII_0, 1);
1181  pad_r(UNI_iPAD_L4, diff - 2);
1182  }
1183  col_info.real_len = COL_INFO.real_len;
1184  }
1185  }
1186  }
1187  else
1188  {
1189  // char items are right aligned
1190  //
1191  size_t LEN = COL_INFO.total_len();
1192  size_t len = col_info.total_len();
1193  if (LEN > len)
1194  {
1195  const size_t diff = LEN - len;
1196  pad_l(UNI_iPAD_L5, diff);
1197  col_info.int_len = COL_INFO.int_len;
1198  col_info.fract_len = COL_INFO.fract_len;
1199  col_info.real_len = COL_INFO.real_len;
1200  }
1201  }
1202 
1203  Log(LOG_printbuf_align) debug(CERR, "after align_dot()");
1204 }

1261 //-----------------------------------------------------------------------------
1262 void
1263 PrintBuffer::pad_fraction(int wanted_fract_len, bool want_expo)
1264 {
1265 const int diff = wanted_fract_len - col_info.fract_len;
1266  Assert1(diff > 0);
1267 
1268  // copy integer part of this PrintBuffer to to new_buf
1269  //
1270 UCS_string new_buf(buffer[0], 0, col_info.int_len);
1271 
1272  // copy fractional part to new_buf. If the number has no exponent part,
1273  // then we fill with spaces. Otherwise fill with '0', possibly inserting
1274  // a decimal point.
1275  //
1276  loop(f, col_info.fract_len)
1277  new_buf.append(buffer[0][col_info.int_len + f]);
1278  if (!want_expo) // no exponent, e.g. 1,0
1279  {
1280  loop(d, diff) new_buf.append(UNI_iPAD_L4);
1281  }
1282  else if (col_info.fract_len == 0) // no fractional part (yet), e.g. 1E2
1283  {
1284  new_buf.append(UNI_ASCII_FULLSTOP);
1285  loop(d, diff - 1) new_buf.append(UNI_ASCII_0);
1286  }
1287  else
1288  {
1289  loop(d, diff) new_buf.append(UNI_ASCII_0);
1290  }
1291 
1292  // copy exponent part
1293  //
1294  for (int ex = col_info.int_len + col_info.fract_len;
1295  ex < buffer[0].size(); ++ex)
1296  new_buf.append(buffer[0][ex]);
1297 
1298  col_info.fract_len = wanted_fract_len;
1299  col_info.real_len += diff;
1300 
1301  buffer[0] = new_buf;
1302 
1303  // if buffer is multi-len then pad remaining line to new length
1304  //
1305  for (ShapeItem h = 1; h < get_height(); ++h)
1306  {
1307  const int diff = new_buf.size() - get_width(h);
1308  if (diff > 0)
1309  {
1310  const UCS_string ucs(diff, UNI_iPAD_L4);
1311  buffer[h].append(ucs);
1312  }
1313  }
1314 }

1358 //-----------------------------------------------------------------------------
1359 void
1360 ColInfo::consider(const ColInfo & item)
1361 {
1362  // this is the collective ColInfo of an entire column, and item
1363  // is a new, not yet considered item in the columns.
1364  //
1365  flags |= item.flags;
1366  if (item.imag_len) flags |= has_j;
1367 
1368  if (int_len < item.int_len)
1369  {
1370  real_len += item.int_len - int_len;
1371  int_len = item.int_len;
1372  }
1373 
1374  if (item.denom_len)
1375  {
1376  if (denom_len < item.denom_len) denom_len = item.denom_len;
1377  if (real_len < int_len + denom_len) real_len = int_len + denom_len;
1378  }
1379  else
1380  {
1381  const int EXPO_LEN = real_len - fract_len - int_len;
1382  const int expo_len = item.real_len - item.fract_len - item.int_len;

27 Token
28 Bif_OPER1_COMMUTE::eval_LB(Token & LO, Value_P B) const
29 {
30  if (!LO.is_function()) SYNTAX_ERROR;
31 Function_P fun_LO = LO.get_function();
32 
33 /*
34  Under "normal" circumstances, eval_LB() is called with a dyadic function LO
35  which is then called with value B as both left and right argument. I.e.
36 
37  LO ⍨ B is: B LO B or
38  e.g: + ⍨ 1 2 3 is: 1 2 3 + 1 2 3
39 
40  Unfortunately a special case occurs when:
41 
42  1. LO is an operator, and
43  2. LO accepts a value
44 
45  In that case one might expect (and e.g. Dyalog SPL does) that
46 
47  A LO ⍨ B would be: B LO A because ⍨ should call LO with A and B exchanged.
48 
49  However, what actually happens is that A LO ⍨ B is parsed as (parentheses
50  added to show the binding strengths):
51 
52  ((A LO) ⍨) B
53 
54  The parser binds A to LO, creating a derived function, say A_LO.
55  The parser then binds the derived function A_LO to operator ~ and
56  creates another derived function, say A_LO_COMMUTE.
57  Finally, A_LO_COMMUTE B is evaluated (monadically!) which brings us here.
58  It should have come to Bif_OPER1_COMMUTE::eval_ALB() below, but it cannot
59  see the left argument A anymore (which is hidden in the derived A_LO).
60 
61  In this special situation we dig into the (derived) argument LO (which is
62  A_LO above) and split it into A and LO. LO is then called syadically with
63  left (!) argument B and right (!) argument A.
64 */
65 
66  if (fun_LO->is_derived()) // maybe special case
67  {
68  const DerivedFunction * A_LO =
69  reinterpret_cast<const DerivedFunction *>(fun_LO);
70  Value_P A = A_LO->get_bound_LO_value();
71  if (+A) // definitelt special case
72  {
73  Function_P LO = A_LO->get_OPER();
74  return LO->eval_AB(B, A);
75  }
76  }
77 
78  // normal case
79  //
80  return fun_LO->eval_AB(B, B);
81 }

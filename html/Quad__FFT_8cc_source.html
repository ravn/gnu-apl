40 void
41 Quad_FFT::init_in(void * _in, Value_P B, window_function win)
42 {
43 fftw_complex * in = reinterpret_cast<fftw_complex *>(_in);
44 const APL_Integer N = B->element_count();
45 
46  if (N < 2)
47  {
48  in[0][0] = B->get_ravel(0).get_real_value();
49  in[0][1] = B->get_ravel(0).get_imag_value();
50  }
51 
52  if (win == 0)
53  {
54  loop(n, N)
55  {
56  in[n][0] = B->get_ravel(n).get_real_value();
57  in[n][1] = B->get_ravel(n).get_imag_value();
58  }
59  }
60  else if (B->get_rank() == 1)
61  {
62  loop(n, N)
63  {
64  const double w = win(n, N);
65  in[n][0] = w * B->get_ravel(n).get_real_value();
66  in[n][1] = w * B->get_ravel(n).get_imag_value();
67  }
68  }
69  else
70  {
71  double * wp = new double[N];
72  if (wp == 0) WS_FULL;
73  fill_window(wp, B->get_shape(), win);
74  loop(n, N)
75  {
76  const double w = wp[n];
77  in[n][0] = w * B->get_ravel(n).get_real_value();
78  in[n][1] = w * B->get_ravel(n).get_imag_value();
79  }
80  delete [] wp;
81  }
82 }

84 Token
85 Quad_FFT::do_fft(int dir, Value_P B, window_function win)
86 {
87  if (!system_wisdom_loaded)
88  {
89  fftw_import_system_wisdom();
90  system_wisdom_loaded = true; // try only once
91  }
92 
93 const APL_Integer N = B->element_count();
94  if (N == 0) LENGTH_ERROR;
95 
96 const ShapeItem io_size = N * sizeof(fftw_complex);
97 
98 fftw_complex * in = reinterpret_cast<fftw_complex *>(fftw_malloc(io_size));
99  if (in == 0) WS_FULL;
100 fftw_complex * out = reinterpret_cast<fftw_complex *>(fftw_malloc(io_size));
101  if (out == 0) { fftw_free(in); WS_FULL; }
102 
103  enum { flags = FFTW_ESTIMATE | FFTW_DESTROY_INPUT };
104 
105  // fill in[] with B
106  //
107  if (B->get_rank() <= 1) // one-dimensional FFT
108  {
109  fftw_plan plan = fftw_plan_dft_1d(N, in, out, dir, flags);
110  if (plan == 0)
111  {
112  fftw_free(in);
113  fftw_free(out);
114  WS_FULL;
115  }
116 
117  init_in(in, B, win); // do this after plan was created
118  fftw_execute(plan);
119  fftw_destroy_plan(plan);
120  }
121  else if (B->get_rank() == 2) // two-dimensional FFT
122  {
123  fftw_plan plan = fftw_plan_dft_2d(B->get_shape_item(0),
124  B->get_shape_item(1),
125  in, out, dir, flags);
126  if (plan == 0)
127  {
128  fftw_free(in);
129  fftw_free(out);
130  WS_FULL;
131  }
132 
133  init_in(in, B, win); // do this after plan was created
134  fftw_execute(plan);
135  fftw_destroy_plan(plan);
136  }
137  else if (B->get_rank() == 3) // two-dimensional FFT
138  {
139  fftw_plan plan = fftw_plan_dft_3d(B->get_shape_item(0),
140  B->get_shape_item(1),
141  B->get_shape_item(2),
142  in, out, dir, flags);
143  if (plan == 0)
144  {
145  fftw_free(in);
146  fftw_free(out);
147  WS_FULL;
148  }
149 
150  init_in(in, B, win); // do this after plan was created
151  fftw_execute(plan);
152  fftw_destroy_plan(plan);
153  }
154  else // k-dimensional FFT
155  {
156  int ish[MAX_RANK];
157  loop(r, B->get_rank()) ish[r] = B->get_shape_item(r);
158 
159  fftw_plan plan = fftw_plan_dft(B->get_rank(), ish, in, out, dir, flags);
160  if (plan == 0)
161  {
162  fftw_free(in);
163  fftw_free(out);
164  WS_FULL;
165  }
166 
167  init_in(in, B, win); // do this after plan was created
168  fftw_execute(plan);
169  fftw_destroy_plan(plan);
170  }
171 
172 Value_P Z(B->get_shape(), LOC);
173 const double norm = sqrt(N);
174  loop(n, N) new (Z->next_ravel())
175  ComplexCell(out[n][0]/norm, out[n][1]/norm);
176 
177  fftw_free(in);
178  fftw_free(out);
179 
180  Z->check_value(LOC);
181  return Token(TOK_APL_VALUE1, Z);
182 }

184 Token
185 Quad_FFT::do_window(Value_P B, window_function win)
186 {
187  Assert(win);
188 
189  if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, IntScalar(1, LOC));
190 
191 const ShapeItem N = B->element_count();
192  if (N < 2) LENGTH_ERROR;
193 
194 Value_P Z(B->get_shape(), LOC);
195  if (B->get_rank() == 1)
196  {
197  loop(n, N)
198  {
199  const double w = win(n, N);
200  const Cell & cell_B = B->get_ravel(n);
201  if (cell_B.is_complex_cell())
202  new (Z->next_ravel()) ComplexCell(w*cell_B.get_real_value(),
203  w*cell_B.get_imag_value());
204  else
205  new (Z->next_ravel()) FloatCell(w * cell_B.get_real_value());
206  }
207  }
208  else
209  {
210  double * wp = new double[N];
211  if (wp == 0) WS_FULL;
212  fill_window(wp, B->get_shape(), win);
213 
214  loop(n, N)
215  {
216  const double w = wp[n];
217  const Cell & cell_B = B->get_ravel(n);
218  if (cell_B.is_complex_cell())
219  new (Z->next_ravel()) ComplexCell(w*cell_B.get_real_value(),
220  w*cell_B.get_imag_value());
221  else
222  new (Z->next_ravel()) FloatCell(w*cell_B.get_real_value());
223  }
224  delete [] wp;
225  }
226 
227  Z->check_value(LOC);
228  return Token(TOK_APL_VALUE1, Z);
229 }

232 Quad_FFT::eval_AB(Value_P A, Value_P B)
233 {
234  if (A->get_rank() > 1) RANK_ERROR;
235  if (A->element_count() != 1) LENGTH_ERROR;
236 
237 const APL_Integer what = A->get_ravel(0).get_int_value();
238  switch(what)
239  {
240  case 15: return do_fft(FFTW_FORWARD, B, &flat_top);
241  case 14: return do_fft(FFTW_FORWARD, B, &blackman_nuttall_window);
242  case 13: return do_fft(FFTW_FORWARD, B, &blackman_harris_window);
243  case 12: return do_fft(FFTW_FORWARD, B, &blackman_window);
244  case 11: return do_fft(FFTW_FORWARD, B, &hamming_window);
245  case 10: return do_fft(FFTW_FORWARD, B, &hann_window);
246 
247  case 0: return do_fft(FFTW_FORWARD, B, 0);
248  case -1: return do_fft(FFTW_BACKWARD, B, 0);
249 
250  case -10: return do_window(B, &hann_window);
251  case -11: return do_window(B, &hamming_window);
252  case -12: return do_window(B, &blackman_window);
253  case -13: return do_window(B, &blackman_harris_window);
254  case -14: return do_window(B, &blackman_nuttall_window);
255  case -15: return do_window(B, &flat_top);
256  }
257 
258  MORE_ERROR() << "Invalid mode A (= " << what
259  << ") of A âFFT B. Valid modes are:\n"
260 " A=Â¯15: no FFT, return (B Ã Flat-Top window)\n"
261 " A=Â¯14: no FFT, return (B Ã Blackman-Nuttal window)\n"
262 " A=Â¯13: no FFT, return (B Ã Blackman-Harris window)\n"
263 " A=Â¯12: no FFT, return (B Ã Blackman window)\n"
264 " A=Â¯11: no FFT, return (B Ã Hamming window)\n"
265 " A=Â¯10: no FFT, return (B Ã Hann window)\n"
266 "\n"
267 " A=Â¯1: inverse FFT\n"
268 " A=0: forward FFT (no window) of B\n"
269 "\n"
270 " A=10: forward FFT(B Ã Hann window)\n"
271 " A=11: forward FFT(B Ã Hamming window)\n"
272 " A=12: forward FFT(B Ã Blackman window)\n"
273 " A=13: forward FFT(B Ã Blackman-Harris window)\n"
274 " A=14: forward FFT(B Ã Blackman-Nuttal window)\n"

280 void
281 Quad_FFT::fill_window(double * result, const Shape & shape, window_function win)
282 {
283 ShapeItem rlen = 1;
284  result[0] = 1.0;
285 
286  for (Rank r = shape.get_rank() - 1; r >= 0; --r)
287  {
288  const ShapeItem axis_len = shape.get_shape_item(r);
289  double * e = result + rlen * axis_len;
290  for (ShapeItem a = axis_len - 1; a >= 0; --a)
291  {
292  const double wa = win(a, axis_len);
293  for (ShapeItem r = rlen - 1; r >= 0; --r)
294  {
295  *--e = wa * result[r];
296  }
297  }
298 
299  rlen *= axis_len;
300  }
301 
302 }

42 void
43 Quad_FFT::init_in(void * _in, Value_P B, window_function win)
44 {
45 fftw_complex * in = reinterpret_cast<fftw_complex *>(_in);
46 const APL_Integer N = B->element_count();
47 
48  if (N < 2)
49  {
50  in[0][0] = B->get_cfirst().get_real_value();
51  in[0][1] = B->get_cfirst().get_imag_value();
52  }
53 
54  if (win == 0)
55  {
56  loop(n, N)
57  {
58  in[n][0] = B->get_cravel(n).get_real_value();
59  in[n][1] = B->get_cravel(n).get_imag_value();
60  }
61  }
62  else if (B->get_rank() == 1)
63  {
64  loop(n, N)
65  {
66  const double w = win(n, N);
67  in[n][0] = w * B->get_cravel(n).get_real_value();
68  in[n][1] = w * B->get_cravel(n).get_imag_value();
69  }
70  }
71  else
72  {
73  double * wp = new double[N];
74  if (wp == 0) WS_FULL;
75  fill_window(wp, B->get_shape(), win);
76  loop(n, N)
77  {
78  const double w = wp[n];
79  in[n][0] = w * B->get_cravel(n).get_real_value();
80  in[n][1] = w * B->get_cravel(n).get_imag_value();
81  }
82  delete [] wp;
83  }
84 }

86 Token
87 Quad_FFT::do_fft(int dir, Value_P B, window_function win)
88 {
89  if (!system_wisdom_loaded)
90  {
91  fftw_import_system_wisdom();
92  system_wisdom_loaded = true; // try only once
93  }
94 
95 const APL_Integer N = B->element_count();
96  if (N == 0) LENGTH_ERROR;
97 
98 const ShapeItem io_size = N * sizeof(fftw_complex);
99 
100 fftw_complex * in = reinterpret_cast<fftw_complex *>(fftw_malloc(io_size));
101  if (in == 0) WS_FULL;
102 fftw_complex * out = reinterpret_cast<fftw_complex *>(fftw_malloc(io_size));
103  if (out == 0) { fftw_free(in); WS_FULL; }
104 
105  enum { flags = FFTW_ESTIMATE | FFTW_DESTROY_INPUT };
106 
107  // fill in[] with B
108  //
109  if (B->get_rank() <= 1) // one-dimensional FFT
110  {
111  fftw_plan plan = fftw_plan_dft_1d(N, in, out, dir, flags);
112  if (plan == 0)
113  {
114  fftw_free(in);
115  fftw_free(out);
116  WS_FULL;
117  }
118 
119  init_in(in, B, win); // do this after plan was created
120  fftw_execute(plan);
121  fftw_destroy_plan(plan);
122  }
123  else if (B->get_rank() == 2) // two-dimensional FFT
124  {
125  fftw_plan plan = fftw_plan_dft_2d(B->get_shape_item(0),
126  B->get_shape_item(1),
127  in, out, dir, flags);
128  if (plan == 0)
129  {
130  fftw_free(in);
131  fftw_free(out);
132  WS_FULL;
133  }
134 
135  init_in(in, B, win); // do this after plan was created
136  fftw_execute(plan);
137  fftw_destroy_plan(plan);
138  }
139  else if (B->get_rank() == 3) // two-dimensional FFT
140  {
141  fftw_plan plan = fftw_plan_dft_3d(B->get_shape_item(0),
142  B->get_shape_item(1),
143  B->get_shape_item(2),
144  in, out, dir, flags);
145  if (plan == 0)
146  {
147  fftw_free(in);
148  fftw_free(out);
149  WS_FULL;
150  }
151 
152  init_in(in, B, win); // do this after plan was created
153  fftw_execute(plan);
154  fftw_destroy_plan(plan);
155  }
156  else // k-dimensional FFT
157  {
158  int ish[MAX_RANK];
159  loop(r, B->get_rank()) ish[r] = B->get_shape_item(r);
160 
161  fftw_plan plan = fftw_plan_dft(B->get_rank(), ish, in, out, dir, flags);
162  if (plan == 0)
163  {
164  fftw_free(in);
165  fftw_free(out);
166  WS_FULL;
167  }
168 
169  init_in(in, B, win); // do this after plan was created
170  fftw_execute(plan);
171  fftw_destroy_plan(plan);
172  }
173 
174 Value_P Z(B->get_shape(), LOC);
175 const double norm = sqrt(N);
176  loop(n, N) Z->next_ravel_Complex(out[n][0]/norm, out[n][1]/norm);
177 
178  fftw_free(in);
179  fftw_free(out);
180 
181  Z->check_value(LOC);
182  return Token(TOK_APL_VALUE1, Z);
183 }
184 //-----------------------------------------------------------------------------
185 Token
186 Quad_FFT::do_window(Value_P B, window_function win)
187 {
188  Assert(win);
189 
190  if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, IntScalar(1, LOC));
191 
192 const ShapeItem N = B->element_count();
193  if (N < 2) LENGTH_ERROR;
194 
195 Value_P Z(B->get_shape(), LOC);
196  if (B->get_rank() == 1)
197  {
198  loop(n, N)
199  {
200  const double w = win(n, N);
201  const Cell & cell_B = B->get_cravel(n);
202  if (cell_B.is_complex_cell())
203  Z->next_ravel_Complex(w*cell_B.get_real_value(),
204  w*cell_B.get_imag_value());
205  else
206  Z->next_ravel_Float(w * cell_B.get_real_value());
207  }
208  }
209  else
210  {
211  double * wp = new double[N];
212  if (wp == 0) WS_FULL;
213  fill_window(wp, B->get_shape(), win);
214 
215  loop(n, N)
216  {
217  const double w = wp[n];
218  const Cell & cell_B = B->get_cravel(n);
219  if (cell_B.is_complex_cell())
220  Z->next_ravel_Complex(w*cell_B.get_real_value(),
221  w*cell_B.get_imag_value());
222  else
223  Z->next_ravel_Float(w*cell_B.get_real_value());
224  }
225  delete [] wp;
226  }
227 
228  Z->check_value(LOC);
229  return Token(TOK_APL_VALUE1, Z);
230 }

232 Token
233 Quad_FFT::eval_AB(Value_P A, Value_P B) const
234 {
235  if (A->get_rank() > 1) RANK_ERROR;
236  if (A->element_count() != 1) LENGTH_ERROR;
237 
238 const APL_Integer what = A->get_cfirst().get_int_value();
239  switch(what)
240  {
241  case 15: return do_fft(FFTW_FORWARD, B, &flat_top);
242  case 14: return do_fft(FFTW_FORWARD, B, &blackman_nuttall_window);
243  case 13: return do_fft(FFTW_FORWARD, B, &blackman_harris_window);
244  case 12: return do_fft(FFTW_FORWARD, B, &blackman_window);
245  case 11: return do_fft(FFTW_FORWARD, B, &hamming_window);
246  case 10: return do_fft(FFTW_FORWARD, B, &hann_window);
247 
248  case 0: return do_fft(FFTW_FORWARD, B, 0);
249  case -1: return do_fft(FFTW_BACKWARD, B, 0);
250 
251  case -10: return do_window(B, &hann_window);
252  case -11: return do_window(B, &hamming_window);
253  case -12: return do_window(B, &blackman_window);
254  case -13: return do_window(B, &blackman_harris_window);
255  case -14: return do_window(B, &blackman_nuttall_window);
256  case -15: return do_window(B, &flat_top);
257  }
258 
259  MORE_ERROR() << "Invalid mode A (= " << what
260  << ") of A ⎕FFT B. Valid modes are:\n"
261 " A=¯15: no FFT, return (B × Flat-Top window)\n"
262 " A=¯14: no FFT, return (B × Blackman-Nuttal window)\n"
263 " A=¯13: no FFT, return (B × Blackman-Harris window)\n"
264 " A=¯12: no FFT, return (B × Blackman window)\n"
265 " A=¯11: no FFT, return (B × Hamming window)\n"
266 " A=¯10: no FFT, return (B × Hann window)\n"
267 "\n"
268 " A=¯1: inverse FFT\n"
269 " A=0: forward FFT (no window) of B\n"
270 "\n"
271 " A=10: forward FFT(B × Hann window)\n"
272 " A=11: forward FFT(B × Hamming window)\n"
273 " A=12: forward FFT(B × Blackman window)\n"
274 " A=13: forward FFT(B × Blackman-Harris window)\n"
275 " A=14: forward FFT(B × Blackman-Nuttal window)\n"
276 " A=15: forward FFT(B × Flat-Top window)\n";
277 
278  DOMAIN_ERROR;
279 }
280 //-----------------------------------------------------------------------------
281 void
282 Quad_FFT::fill_window(double * result, const Shape & shape, window_function win)
283 {
284 ShapeItem rlen = 1;
285  result[0] = 1.0;
286 
287  for (Rank r = shape.get_rank() - 1; r >= 0; --r)
288  {
289  const ShapeItem axis_len = shape.get_shape_item(r);
290  double * e = result + rlen * axis_len;
291  for (ShapeItem a = axis_len - 1; a >= 0; --a)
292  {
293  const double wa = win(a, axis_len);
294  for (ShapeItem r = rlen - 1; r >= 0; --r)
295  {
296  *--e = wa * result[r];
297  }
298  }
299 
300  rlen *= axis_len;
301  }
302 
303 }

92 class TakeDropIterator
93 {
94 public:
96  TakeDropIterator(bool take, const Shape & sh_A, const Shape & sh_B)
97  : ref_B(sh_B),
98  current_offset(0),
99  outside_B(0),
100  has_overtake(false),
101  done(false)
102  {
103  // increase ⍴⍴B if necessary
104  //
105  ref_B.expand_rank(sh_A.get_rank());
106 
107  ShapeItem _weight = 1;
108  loop(r, sh_A.get_rank())
109  {
110  const ShapeItem sA = sh_A.get_rank() == 0
111  ? 1 : sh_A.get_transposed_shape_item(r);
112  const ShapeItem sB = (sh_B.get_rank() == 0 ||
113  r >= sh_B.get_rank())
114  ? 1 : sh_B.get_transposed_shape_item(r);
115 
116  ShapeItem _from, _to;
117  if (take) // sh_A ↑ B
118  {
119  if (sA < 0) // take from end
120  {
121  _to = sB;
122  _from = sB + sA; // + since sA < 0
123  }
124  else // take from start
125  {
126  _from = 0;
127  _to = sA;
128  }
129 
130  if (_from < 0 || _from >= sB) outside_B |= 1ULL << r;
131  if (_from < 0 || _to >= sB) has_overtake = true;
132  }
133  else // sh_A ↓ B
134  {
135  if (sA < 0) // drop from end
136  {
137  _from = 0;
138  _to = sB + sA; // + since sA < 0
139  }
140  else // drop from start
141  {
142  _from = sA;
143  _to = sB;
144  }
145 
146  if (_from >= _to) // over-drop
147  {
148  _from = 0;
149  _to = 0;
150  }
151  }
152 
153  Assert(_from <= _to);
154 
155  _ftwc & ftwc_r = ftwc[r];
156  ftwc_r.from = _from;
157  ftwc_r.to = _to;
158  ftwc_r.weight = _weight;
159  ftwc_r.current = _from;
160  current_offset += _from * _weight;
161 
162  if (_from == _to) done = true; // empty array
163 
164  _weight *= sB;
165  }
166  }
167 
169  void operator++()
170  {
171  loop(r, ref_B.get_rank())
172  {
173  _ftwc & ftwc_r = ftwc[r];
174  ++ftwc_r.current;
175  current_offset += ftwc_r.weight;
176  if (has_overtake)
177  {
178  if (ftwc_r.current < 0 ||
179  ftwc_r.current >= ref_B.get_transposed_shape_item(r))
180  outside_B |= 1ULL << r;
181  else
182  outside_B &= ~(1ULL << r);
183  }
184 
185  if (ftwc_r.current < ftwc_r.to) return;
186 
187  // end of dimension reached: reset this dimension
188  // and increment next dimension
189  //
190  current_offset -= ftwc_r.weight * (ftwc_r.current - ftwc_r.from);
191  ftwc_r.current = ftwc_r.from;
192  if (has_overtake)
193  {
194  if (ftwc_r.current < 0 ||
195  ftwc_r.current >= ref_B.get_transposed_shape_item(r))
196  outside_B |= 1ULL << r;
197  else
198  outside_B &= ~(1ULL << r);
199  }
200  }
201  done = true;
202  }
203 
205  bool more() const { return !done; }
206 
208  ShapeItem operator()() const
209  { return outside_B ? -1 : current_offset; }
210 
211 protected:
213  Shape ref_B;
214 
216  struct _ftwc ftwc[MAX_RANK];
217 
219  ShapeItem current_offset;
220 
222  ShapeItem outside_B;
223 
225  bool has_overtake;
226 
228  bool done;
229 };

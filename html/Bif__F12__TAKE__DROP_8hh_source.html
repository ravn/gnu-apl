93 class TakeDropIterator
94 {
95 public:
97  TakeDropIterator(bool take, const Shape & sh_A, const Shape & sh_B)
98  : ref_B(sh_B),
99  current_offset(0),
100  outside_B(0),
101  has_overtake(false),
102  done(false)
103  {
104  ShapeItem _weight = 1;
105  loop(r, sh_A.get_rank())
106  {
107  const ShapeItem sA = sh_A.get_rank() == 0
108  ? 1 : sh_A.get_transposed_shape_item(r);
109  const ShapeItem sB = sh_B.get_rank() == 0
110  ? 1 : sh_B.get_transposed_shape_item(r);
111 
112  ShapeItem _from, _to;
113  if (take) // sh_A â B
114  {
115  if (sA < 0) // take from end
116  {
117  _to = sB;
118  _from = sB + sA; // + since sA < 0
119  }
120  else // take from start
121  {
122  _from = 0;
123  _to = sA;
124  }
125 
126  if (_from < 0 || _from >= sB) outside_B |= 1ULL << r;
127  if (_from < 0 || _to >= sB) has_overtake = true;
128  }
129  else // sh_A â B
130  {
131  if (sA < 0) // drop from end
132  {
133  _from = 0;
134  _to = sB + sA; // + since sA < 0
135  }
136  else // drop from start
137  {
138  _from = sA;
139  _to = sB;
140  }
141 
142  if (_from >= _to) // over-drop
143  {
144  _from = 0;
145  _to = 0;
146  }
147  }
148 
149  Assert(_from <= _to);
150 
151  _ftwc & ftwc_r = ftwc[r];
152  ftwc_r.from = _from;
153  ftwc_r.to = _to;
154  ftwc_r.weight = _weight;
155  ftwc_r.current = _from;
156  current_offset += _from * _weight;
157 
158  if (_from == _to) done = true; // empty array
159 
160  _weight *= sB;
161  }
162  }
163 
165  void operator++()
166  {
167  loop(r, ref_B.get_rank())
168  {
169  _ftwc & ftwc_r = ftwc[r];
170  ++ftwc_r.current;
171  current_offset += ftwc_r.weight;
172  if (has_overtake)
173  {
174  if (ftwc_r.current < 0 ||
175  ftwc_r.current >= ref_B.get_transposed_shape_item(r))
176  outside_B |= 1ULL << r;
177  else
178  outside_B &= ~(1ULL << r);
179  }
180 
181  if (ftwc_r.current < ftwc_r.to) return;
182 
183  // end of dimension reached: reset this dimension
184  // and increment next dimension
185  //
186  current_offset -= ftwc_r.weight * (ftwc_r.current - ftwc_r.from);
187  ftwc_r.current = ftwc_r.from;
188  if (has_overtake)
189  {
190  if (ftwc_r.current < 0 ||
191  ftwc_r.current >= ref_B.get_transposed_shape_item(r))
192  outside_B |= 1ULL << r;
193  else
194  outside_B &= ~(1ULL << r);
195  }
196  }
197  done = true;
198  }
199 
201  bool more() const { return !done; }
202 
204  ShapeItem operator()() const
205  { return outside_B ? -1 : current_offset; }
206 
207 protected:
209  const Shape & ref_B;
210 
212  struct _ftwc ftwc[MAX_RANK];
213 
215  ShapeItem current_offset;
216 
218  ShapeItem outside_B;
219 
221  bool has_overtake;
222 
224  bool done;
225 };

62 bool
63 IndexExpr::check_range(const Shape & shape) const
64 {
65  loop(r, rank)
66  {
67  Value_P ival = values[rank - r - 1];
68  if (!ival) continue; // elided index
69 
70  const ShapeItem max_idx = shape.get_shape_item(r) + quad_io;
71  loop(i, ival->element_count())
72  {
73  const APL_Integer idx = ival->get_ravel(i).get_near_int();
74  if (idx < quad_io) return true;
75  if (idx >= max_idx) return true;
76  }
77  }
78 
79  return false;
80 }
81 //-----------------------------------------------------------------------------
82 Rank
83 IndexExpr::get_axis(Rank max_axis) const
84 {
85  if (rank != 1) INDEX_ERROR;
86 
87 const APL_Integer qio = Workspace::get_IO();
88 
89 Value_P I = values[0];
90  if (!I->is_scalar_or_len1_vector()) INDEX_ERROR;
91 
92  if (!I->get_ravel(0).is_near_int()) INDEX_ERROR;
93 
94  // if axis becomes (signed) negative then it will be (unsigned) too big.
95  // Therefore we need not test for < 0.
96  //
97 Rank axis = I->get_ravel(0).get_near_int() - qio;
98  if (axis >= max_axis) INDEX_ERROR;
99 
100  return axis;
101 }

103 void
104 IndexExpr::set_value(uAxis axis, Value_P val)
105 {
106  // Note: we expect that all axes have been allocated with add(0)
107  // Note also that IndexExpr is in parsing order, ie. 0 is the lowest axis.
108  //
109  Assert(axis < rank);
110  Assert(!values[rank - axis - 1]);
111  values[rank - axis - 1] = val;
112 }

115 IndexExpr::to_shape() const
116 {
117  if (value_count() != 1) INDEX_ERROR;
118 
119 Value_P vx = values[0];
120  if (!vx) INDEX_ERROR;
121 
122 const ShapeItem xlen = vx->element_count();
123 const APL_Integer qio = Workspace::get_IO();
124 
125 Shape shape;
126  loop(x, xlen)
127  shape.add_shape_item(vx->get_ravel(x).get_near_int() - qio);
128 
129  return shape;
130 }

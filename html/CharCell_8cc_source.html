25 CellType
26 CharCell::get_cell_subtype() const
27 {
28  if (value.ival < 0) // negative char (only fits in signed containers)
29  {
30  if (-value.ival <= 0x80)
31  return CellType(CT_CHAR | CTS_S8 | CTS_S16 | CTS_S32 | CTS_S64);
32 
33  if (-value.ival <= 0x8000)
34  return CellType(CT_CHAR | CTS_S16 | CTS_S32 | CTS_S64);
35 
36  return CellType(CT_CHAR | CTS_S32 | CTS_S64);
37  }
38 
39  // positive char
40  //
41  if (value.ival <= 0x7F)
42  return CellType(CT_CHAR | CTS_X8 | CTS_S8 | CTS_U8 |
43  CTS_X16 | CTS_S16 | CTS_U16 |
44  CTS_X32 | CTS_S32 | CTS_U32 |
45  CTS_X64 | CTS_S64 | CTS_U64);
46 
47  if (value.ival <= 0xFF)
48  return CellType(CT_CHAR | CTS_U8 |
49  CTS_X16 | CTS_S16 | CTS_U16 |
50  CTS_X32 | CTS_S32 | CTS_U32 |
51  CTS_X64 | CTS_S64 | CTS_U64);
52 
53  if (value.ival <= 0x7FFF)
54  return CellType(CT_CHAR | CTS_X16 | CTS_S16 | CTS_U16 |
55  CTS_X32 | CTS_S32 | CTS_U32 |
56  CTS_X64 | CTS_S64 | CTS_U64);
57 
58  if (value.ival <= 0xFFFF)
59  return CellType(CT_CHAR | CTS_U16 |
60  CTS_X32 | CTS_S32 | CTS_U32 |
61  CTS_X64 | CTS_S64 | CTS_U64);
62 
63  return CellType(CT_CHAR | CTS_X32 | CTS_S32 | CTS_U32 |
64  CTS_X64 | CTS_S64 | CTS_U64);
65 }

67 bool
68 CharCell::greater(const Cell & other) const
69 {
70  // char cells are smaller than all others
71  //
72  if (other.get_cell_type() != CT_CHAR) return false;
73 
74 const Unicode this_val = get_char_value();
75 const Unicode other_val = other.get_char_value();
76 
77  // if both chars are the same, compare cell address
78  //
79  if (this_val == other_val) return this > &other;
80 
81  return this_val > other_val;
82 }

91 Comp_result
92 CharCell::compare(const Cell & other) const
93 {
94  if (other.is_character_cell())
95  {
96  if (other.get_char_value() == value.aval) return COMP_EQ;
97  return (value.aval < other.get_char_value()) ? COMP_LT : COMP_GT;
98  }
99 
100  return COMP_LT; // char < everything else
101 }

111 PrintBuffer
112 CharCell::character_representation(const PrintContext & pctx) const
113 {
114 UCS_string ucs;
115 ColInfo info;
116  info.flags |= CT_CHAR;
117 
118 PrintStyle style = pctx.get_style();
119 Unicode uni = get_char_value();
120  if ((style & PST_PRETTY) && uni < UNI_ASCII_SPACE)
121  uni = Unicode(uni + 0x2400);
122 
123  if (style == PR_APL_FUN)
124  {
125  if (uni == UNI_SINGLE_QUOTE)
126  {
127  ucs.append(UNI_SINGLE_QUOTE);
128  ucs.append(uni);
129  ucs.append(uni);
130  ucs.append(UNI_SINGLE_QUOTE);
131  }
132  else
133  {
134  ucs.append(UNI_SINGLE_QUOTE);
135  ucs.append(uni);
136  ucs.append(UNI_SINGLE_QUOTE);
137  }
138  }
139  else
140  {
141  if (style & PST_QUOTE_CHARS)
142  {
143  ucs.append(UNI_SINGLE_QUOTE);
144  ucs.append(uni);
145  ucs.append(UNI_SINGLE_QUOTE);
146  }
147  else
148  {
149  ucs.append(uni);
150  }
151  }
152 
153  info.real_len = info.int_len = ucs.size();
154 
155  return PrintBuffer(ucs, info);
156 }

158 int
159 CharCell::get_byte_value() const
160 {
161  if (value.aval < -128)
162  {
163  char cc[20];
164  snprintf(cc, sizeof(cc), "' = U+%4.4X", value.aval & 0x0FFFF);
165  MORE_ERROR() << "Unicode character '" << value.aval << cc
166  << " is too small (< U+FF80) for a byte value";
167  DOMAIN_ERROR;
168  }
169 
170  if (value.aval > 255)
171  {
172  char cc[20];
173  snprintf(cc, sizeof(cc), "' = U+%4.4X", value.aval & 0x0FFFF);
174  MORE_ERROR() << "Unicode character '" << value.aval << cc
175  << " is too large (> U+00FF) for a byte value";
176  DOMAIN_ERROR;
177  }
178 
179  return value.aval;
180 }

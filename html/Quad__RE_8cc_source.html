55  MORE_ERROR() << "Unknown âRE flag: '" << UCS_string(1, uni)
56  << "'. Valid flags are: Eimsxââ/";

58  }
59  }
60 
61  if (ofcnt > 1)

63  MORE_ERROR() << "Multiple âRE output flags: '" << flags_string

69 static Value_P
70 deep_value(int idx, const PCRE2_SIZE * ovector, int count, const int * parents,
71  const int * child_count, const UCS_string * B)
72 {
73  if (child_count[idx] == 0) // simple RE (no sub-REs)
74  {
75  const PCRE2_SIZE start = ovector[2*idx];
76  const PCRE2_SIZE end = ovector[2*idx + 1];
77  if (B) // string
78  {
79  const UCS_string item(*B, start, end - start);
80  Value_P Z(item, LOC);
81  Z->check_value(LOC);
82  return Z;
83  }
84  else // pos+len
85  {
86  Value_P Z(2, LOC);
87  Z->next_ravel_Int(start);
88  Z->next_ravel_Int(end - start);
89  Z->check_value(LOC);
90  return Z;
91  }
92  }
93 
94 ShapeItem ini = B ? 1 : 2;
95 Value_P Z(ini + child_count[idx], LOC);
96  if (B)
97  {
98  const PCRE2_SIZE start = ovector[2*idx];
99  const PCRE2_SIZE end = ovector[2*idx + 1];
100  const UCS_string item(*B, start, end - start);
101  Value_P sub_value(item, LOC);
102  sub_value->check_value(LOC);
103  Z->next_ravel_Pointer(sub_value.get());
104  }
105  else
106  {
107  Z->next_ravel_Int(ovector[2*idx]);
108  Z->next_ravel_Int(ovector[2*idx + 1] - ovector[2*idx]);
109  }
110 
111  loop(ch, count)
112  {
113  if (parents[ch] != idx) continue; // ch is not a child of idx
114  Value_P CH = deep_value(ch, ovector, count, parents, child_count, B);
115  Z->next_ravel_Pointer(CH.get());
116  }
117 
118  Z->check_value(LOC);
119  return Z;
120 }

122 Value_P
123 Quad_RE::regex_results(const Regexp & A, const Flags & X, const UCS_string & B)
124 {
125 ShapeItem B_offset = 0; // updated by XXX_result() functions
126 
127  switch(X.get_result_type())
128  {
129  case RT_reduce:
130  case RT_partition: return partition_result(A, X, B);
131 
132  case RT_string: if (X.get_global()) break; // continue below
133  return string_result(A, X, B, B_offset);
134 
135  case RT_pos_len: if (X.get_global()) break; // continue below
136  return index_result (A, X, B, B_offset);
137 
138  default: FIXME;
139  }
140 
141  /* At this point the result type is RT_string or RT_pos_len, and the global
142  flag is set (which means that all matches shall be returned).
143 
144  We have not yet called the corresponding string_result() or index_result()
145  function (which creates the result), and therefore do not know how long
146  Z will become.
147 
148  We handle this by:
149 
150  1. starting with a â´Z of of SHORT_VALUE_LENGTH_WANTED, and
151  2. doubling â´Z whenever needed, and finally
152  3. shrinking â´Z to the true number of matches.
153  */
154 
155 Value_P Z(SHORT_VALUE_LENGTH_WANTED, LOC);
156 
157  for (;;)
158  {
159  Value_P ZZ;
160  switch(X.get_result_type())
161  {
162  case RT_string:
163  ZZ = string_result(A, X, B, B_offset);
164  break;
165 
166  case RT_pos_len:
167  ZZ = index_result(A, X, B, B_offset);
168  break;
169 
170  default: FIXME;
171  }
172 
173  if (B_offset == -1) break; // no more matches
174 
175  if (!Z->more()) // Z is full
176  {
177  const ShapeItem ec_Z = Z->element_count();
178  Value_P Z2(2*ec_Z, LOC);
179  loop(z, ec_Z)
180  {
181  Cell & cell = Z->get_wravel(z);
182  Z2->next_ravel_Pointer(cell.get_pointer_value().get());
183  cell.release(LOC);
184  }
185  Z = Z2;
186  }
187 
188  Z->next_ravel_Pointer(ZZ.get());
189  }
190 
191  // most likely, Z is over-allocated at this point and we should init
192  // the not-yet-used Cells before shrinking Z.
193  //
194 const Shape sh_Z(Z->get_valid_item_count());
195  while (Z->more()) Z->next_ravel_Int(0); // init the remaining cells
196  Z->check_value(LOC);
197 
198  Z->set_shape(sh_Z);
199  return Z;
200 }

202 Value_P
203 Quad_RE::partition_result(const Regexp & A, const Flags & X,
204  const UCS_string & B)
205 {
206 const PCRE2_SIZE len = B.size();
207 Value_P Z(len, LOC);
208 
209 PCRE2_SIZE B_offset = 0;
210 ShapeItem match_id_partition = 1;
211 ShapeItem match_id_compress = 1;
212 
213 ShapeItem & match_id = X.get_result_type() == RT_partition
214  ? match_id_partition : match_id_compress;
215 
216  // compress needs a numeric vector like 1 1 1 0 0 0 0 0 1 1 1...
217  // partition needs a numeric vector like 1 1 1 2 2 2 2 2 3 3 3...
218  // where the ranges of equal numbers belong to the same partition.
219  //
220  // We call RegexpMatch() multiple times and create a partition (-range)
221  // for every match.
222  //
223 PCRE2_SIZE last_end = 0;
224  for (; B_offset < len; ++match_id_partition)
225  {
226  RegexpMatch rem(A.get_code(), B, B_offset);
227  if (!rem.is_match()) break;
228 
229  const PCRE2_SIZE * ovector = rem.get_ovector();
230  const PCRE2_SIZE start = ovector[0];
231  const PCRE2_SIZE end = ovector[1];
232 
233  // zeros (if any) between the previous and the current partition
234  loop(z, start - last_end) Z->next_ravel_Int(0);
235  last_end = end;
236 
237  loop (b, end - start) Z->next_ravel_Int(match_id);
238 
239  if (!X.get_global()) break;
240  B_offset = end;
241  }
242 
243  while (Z->more()) Z->next_ravel_Int(0);
244  Z->check_value(LOC);
245  return Z;
246 }

248 Value_P
249 Quad_RE::string_result(const Regexp & A, const Flags & X,
250  const UCS_string & B, ShapeItem & B_offset)
251 {
252 RegexpMatch rem(A.get_code(), B, B_offset);
253  if (!rem.is_match())
254  {
255  B_offset = -1; // indicates an error
256  if (X.get_global()) return Value_P();
257  if (!X.get_error_on_no_match()) return Idx0(LOC);
258  MORE_ERROR() << "No match";
259  DOMAIN_ERROR;
260  }
261 
262  // rem is a match
263  //
264 const PCRE2_SIZE * ovector = rem.get_ovector();
265  B_offset = ovector[1];
266  if (0 && rem.num_matches() == 1) // simple match
267  {
268  // single string
269  //
270  Value_P Z(2, LOC);
271  const PCRE2_SIZE start = ovector[0];
272  const PCRE2_SIZE end = ovector[1];
273  Z->next_ravel_Int(start);
274  Z->next_ravel_Int(end - start);
275  Z->check_value(LOC);
276  return Z;
277  }
278 
279 const uint32_t ovector_count = rem.get_ovector_count();
280  B_offset = ovector[1];
281 int parents[ovector_count];
282 int ccount[ovector_count];
283 
284  loop(o, ovector_count)
285  {
286  parents[o] = -1;
287  ccount[o] = 0;
288  }
289 
290  for (int o = ovector_count - 1; o >= 0; --o)
291  {
292  const PCRE2_SIZE ostart = ovector[2*o];
293  for (int p = o - 1; p >= 0; --p)
294  {
295  if (ovector[2*p] <= ostart && ostart < ovector[2*p + 1])
296  {
297  parents[o] = p;
298  ++ccount[p];
299  break;
300  }
301  }
302  }
303 
304  return deep_value(0, ovector, ovector_count, parents, ccount, &B);
305 }

367 Quad_RE::eval_AXB(Value_P A, Value_P X, Value_P B) const
368 {
369  if (A->get_rank() > 1) RANK_ERROR;
370  if (X->get_rank() > 1) RANK_ERROR;
371 
372  if (!A->is_char_string())
373  {
374  MORE_ERROR() << "left âRE arguments must be a string value";
375  DOMAIN_ERROR;
376  }
377 
378 Flags flags(X->get_UCS_ravel());
379 Regexp regexp(A->get_UCS_ravel(), flags.get_compflags());
380 
381 const Shape & shape = B->get_shape();
382  if (shape.get_rank() == 0)
383  return Token(TOK_APL_VALUE1, Idx0(LOC));
384 
385  if (B->is_char_string())
386  {
387  Value_P Z = regex_results(regexp, flags, B->get_UCS_ravel());
388  Z->check_value(LOC);
389  return Token(TOK_APL_VALUE1, Z);
390  }
391 
392 Value_P Z(shape, LOC);
393  for (ShapeItem i = 0 ; i < shape.get_volume() ; i++)
394  {
395  const Cell & cell = B->get_cravel(i);
396  Value_P B_sub = cell.to_value(LOC);
397  if (!B_sub->is_char_string())
398  {
399  MORE_ERROR() << "Cell does not contain a string";
400  DOMAIN_ERROR;
401  }
402 
403  Value_P Z_sub = regex_results(regexp, flags, B_sub->get_UCS_ravel());
404  Z_sub->check_value(LOC);
405  Z->next_ravel_Pointer(Z_sub.get());
406  }
407 
408  Z->check_value(LOC);
409  return Token(TOK_APL_VALUE1, Z);
410 }

74 Cell::init_type(const Cell & other, Value & cell_owner, const char * loc)
75 {
76  if (other.is_pointer_cell())
77  {
78  Value_P sub = other.get_pointer_value()->clone(loc);
79  Assert(!sub->is_simple_scalar());
80  sub->to_proto();
81  new (this) PointerCell(sub.get(), cell_owner);
82  }
83  else if (other.is_lval_cell())
84  {
85  new (this) LvalCell(other.get_lval_value(),
86  other.cLvalCell().get_cell_owner());
87  }
88  else if (other.is_character_cell())
89  {
90  new (this) CharCell(UNI_ASCII_SPACE);
91  }
92  else // numeric
93  {
94  new (this) IntCell(0);
95  }

175 //-----------------------------------------------------------------------------
176 APL_Integer
177 Cell::near_int(APL_Float value)
178 {
179  if (value >= LARGE_INT) DOMAIN_ERROR;
180  if (value <= SMALL_INT) DOMAIN_ERROR;
181 
182 const APL_Float result = nearbyint(value);
183 const APL_Float diff = value - result;
184  if (diff > INTEGER_TOLERANCE) DOMAIN_ERROR;
185  if (diff < -INTEGER_TOLERANCE) DOMAIN_ERROR;
186 
187  if (result > 0.0) return APL_Integer(0.3 + result);
188  else return - APL_Integer(0.3 - result);
189 }

191 bool
192 Cell::greater_cp(const ShapeItem & A, const ShapeItem & B, const void * ctx)
193 {
194 const ravel_comp_len * rcl = reinterpret_cast<const ravel_comp_len *>(ctx);
195 const Cell * cells = rcl->ravel;
196 const ShapeItem comp_len = rcl->comp_len;
197 const Cell * cell_A = cells + A * comp_len;
198 const Cell * cell_B = cells + B * comp_len;
199 
200  loop(l, comp_len)
201  {
202  if (const Comp_result cr = cell_A++->compare(*cell_B++))
203  return cr == COMP_GT;
204  }
205 
206  // at this point all cells were equal
207  //
208  return A > B;
209 }

283 ShapeItem *
284 Cell::sorted_indices(const Cell * ravel, ShapeItem length, Sort_order order,
285  ShapeItem comp_len)
286 {
287 ShapeItem * indices = new ShapeItem[length];
288  if (indices == 0) return indices;
289 
290  // initialize indices with 0, 1, ... length-1
291  loop(l, length) indices[l] = l;
292 
293 const ravel_comp_len ctx = { ravel, comp_len};
294  if (order == SORT_ASCENDING)
295  Heapsort<ShapeItem>::sort(indices, length, &ctx, &Cell::greater_cp);
296  else
297  Heapsort<ShapeItem>::sort(indices, length, &ctx, &Cell::smaller_cp);
298  return indices;
299 }

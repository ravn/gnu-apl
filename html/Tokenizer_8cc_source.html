60 {

64  }
65  catch (Error err)

67  const int caret_1 = input.size() - rest_1;
68  const int caret_2 = input.size() - rest_2;

75 //-----------------------------------------------------------------------------
78 void
79 Tokenizer::do_tokenize(const UCS_string & input, Token_string & tos)
80 {
81  Log(LOG_tokenize)
82  CERR << "tokenize: input[" << input.size() << "] is: Â«"
83  << input << "Â»" << endl;
84 
85 Unicode_source src(input);
86  while ((rest_1 = rest_2 = src.rest()) != 0)
87  {
88  Unicode uni = *src;
89  if (uni == UNI_COMMENT) break; // â comment
90  if (uni == UNI_ASCII_NUMBER_SIGN) break; // # comment
91 
92  const Token tok = Avec::uni_to_token(uni, LOC);
93 
94  Log(LOG_tokenize)
95  {
96  Unicode_source s1(src, 0, 24);
97  CERR << " tokenize(" << src.rest() << " chars) sees [tag "
98  << tok.tag_name() << " Â«" << uni << "Â»] " << s1;
99  if (src.rest() != s1.rest()) CERR << " ...";
100  CERR << endl;
101  }
102 
103  switch(tok.get_Class())
104  {
105  case TC_END: // chars without APL meaning
106  rest_2 = src.rest();
107  {
108  Log(LOG_error_throw)
109  CERR << endl << "throwing "
110  << Error::error_name(E_NO_TOKEN)
111  << " in Tokenizer" << endl;
112 
113  char cc[20];
114  snprintf(cc, sizeof(cc), "U+%4.4X (", uni);
115  MORE_ERROR() << "Tokenizer: No token for Unicode "
116  << cc << uni << ")\nInput: " << input;
117  Error error(E_NO_TOKEN, LOC);
118  throw error;
119  }
120  break;
121 
122  case TC_RETURN:
123  case TC_LINE:
124  case TC_VALUE:
125  case TC_INDEX:
126  CERR << "Offending token: " << tok.get_tag()
127  << " (" << tok << ")" << endl;
128  if (tok.get_tag() == TOK_CHARACTER)
129  CERR << "Unicode: " << UNI(tok.get_char_val()) << endl;
130  rest_2 = src.rest();
131  Error::throw_parse_error(E_NON_APL_CHAR, LOC, loc);
132  break;
133 
134  case TC_VOID:
135  // Avec::uni_to_token returns TC_VOID for non-apl characters
136  //
137  rest_2 = src.rest();
138  UERR << "Unknown APL character: " << uni
139  << " (" << UNI(uni) << ")" << endl;
140  Error::throw_parse_error(E_NON_APL_CHAR, LOC, loc);
141  break;
142 
143  case TC_SYMBOL:
144  if (Avec::is_quad(uni))
145  {
146  tokenize_quad(src, tos);
147  }
148  else if (uni == UNI_QUOTE_Quad)
149  {
150  ++src;
151  tos.push_back(Token(TOK_Quad_QUOTE,
152  &Workspace::get_v_Quad_QUOTE()));
153  }
154  else if (uni == UNI_ALPHA)
155  {
156  ++src;
157  tos.push_back(Token(TOK_ALPHA,
158  &Workspace::get_v_ALPHA()));
159  }
160  else if (uni == UNI_ALPHA_UNDERBAR)
161  {
162  ++src;
163  tos.push_back(Token(TOK_ALPHA_U,
164  &Workspace::get_v_ALPHA_U()));
165  }
166  else if (uni == UNI_CHI)
167  {
168  ++src;
169  tos.push_back(Token(TOK_CHI,
170  &Workspace::get_v_CHI()));
171  }
172  else if (uni == UNI_LAMBDA)
173  {
174  // this could be Î» like in Î»â ...
175  // or Î»1 or Î»2 or ... as in ... âº Î»1 âµ
176  //
177  if (src.rest() > 1 && Avec::is_digit(src[1])) // Î»n
178  {
179  tokenize_symbol(src, tos);
180  }
181  else // Î»
182  {
183  ++src;
184  tos.push_back(Token(TOK_LAMBDA,
185  &Workspace::get_v_LAMBDA()));
186  }
187  }
188  else if (uni == UNI_OMEGA)
189  {
190  ++src;
191  tos.push_back(Token(TOK_OMEGA,
192  &Workspace::get_v_OMEGA()));
193  }
194  else if (uni == UNI_OMEGA_UNDERBAR)
195  {
196  ++src;
197  tos.push_back(Token(TOK_OMEGA_U,
198  &Workspace::get_v_OMEGA_U()));
199  }
200  else
201  {
202  tokenize_symbol(src, tos);
203  }
204  break;
205 
206  case TC_FUN0:
207  case TC_FUN12:
208  case TC_OPER1:
209  tokenize_function(src, tos);
210  break;
211 
212  case TC_OPER2:
213  if (tok.get_tag() == TOK_OPER2_INNER && src.rest())
214  {
215  // tok is a dot. This could mean that . is either
216  //
217  // the start of a number: e.g. +.3
218  // or an operator: e.g. +.*
219  // or a syntax error: e.g. Done.
220  //
221  if (src.rest() == 1) // syntax error
222  Error::throw_parse_error(E_SYNTAX_ERROR, LOC, loc);
223 
224  Unicode uni_1 = src[1];
225  const Token tok_1 = Avec::uni_to_token(uni_1, LOC);
226  if ((tok_1.get_tag() & TC_MASK) == TC_NUMERIC)
227  tokenize_number(src, tos);
228  else
229  tokenize_function(src, tos);
230  }
231  else
232  {
233  tokenize_function(src, tos);
234  }
235  if (tos.size() >= 2 &&
236  tos.back().get_tag() == TOK_OPER2_INNER &&
237  tos[tos.size() - 2].get_tag() == TOK_JOT)
238  {
239  new (&tos.back()) Token(TOK_OPER2_OUTER,
240  Bif_OPER2_OUTER::fun);
241  }
242 
243  break;
244 
245  case TC_R_ARROW:
246  ++src;
247  if (src.rest()) tos.push_back(tok);
248  else tos.push_back(Token(TOK_ESCAPE));
249  break;
250 
251  case TC_ASSIGN:
252  ++src;
253  {
254  const bool sym = tos.size() >= 1 &&
255  tos[tos.size() - 1].get_tag() == TOK_SYMBOL;
256  const bool dia = tos.size() > 1 &&
257  tos[tos.size() - 2].get_tag() == TOK_DIAMOND;
258  const bool col = tos.size() > 1 &&
259  tos[tos.size() - 2].get_tag() == TOK_COLON;
260 
261  tos.push_back(tok);
262 
263  // change token tag of â if:
264  //
265  // SYM â (at the start of line), or
266  // â SYM â (at the start of statement), or
267  // : SYM â (at the start of statement after label)
268  //
269  if (sym && ((tos.size() == 2) || dia || col))
270  tos.back().ChangeTag(TOK_ASSIGN1);
271  }
272  break;
273 
274  case TC_L_PARENT:
275  case TC_R_PARENT:
276  case TC_L_BRACK:
277  case TC_R_BRACK:
278  case TC_L_CURLY:
279  case TC_R_CURLY:
280  ++src;
281  tos.push_back(tok);
282  break;
283 
284  case TC_DIAMOND:
285  ++src;
286  tos.push_back(tok);
287  break;
288 
289  case TC_COLON:
290  if (pmode != PM_FUNCTION)
291  {
292  rest_2 = src.rest();
293  if (pmode == PM_EXECUTE)
294  Error::throw_parse_error(E_ILLEGAL_COLON_EXEC,
295  LOC, loc);
296  else
297  Error::throw_parse_error(E_ILLEGAL_COLON_STAT,
298  LOC, loc);
299  }
300 
301  ++src;
302  tos.push_back(tok);
303  break;
304 
305  case TC_NUMERIC:
306  tokenize_number(src, tos);
307  break;
308 
309  case TC_SPACE:
310  case TC_NEWLINE:
311  ++src;
312  break;
313 
314  case TC_QUOTE:
315  if (tok.get_tag() == TOK_QUOTE1)
316  tokenize_string1(src, tos);
317  else
318  tokenize_string2(src, tos);
319  break;
320 
321  default:
322  CERR << "Input: " << input << endl
323  << "uni: " << uni << endl
324  << "Token = " << tok.get_tag() << endl;
325 
326  if (tok.get_Id() != ID_No_ID)
327  {
328  CERR << ", Id = " << Id(tok.get_tag() >> 16);
329  }
330  CERR << endl;
331  Assert(0 && "Should not happen");
332  }
333  }
334 
335  Log(LOG_tokenize)
336  CERR << "tokenize() done (no error)" << endl;
337 }

447 void
448 Tokenizer::tokenize_quad(Unicode_source & src, Token_string & tos)
449 {
450  Log(LOG_tokenize)
451  CERR << "tokenize_quad(" << src.rest() << " chars)"<< endl;
452 
453  src.get(); // discard (possibly alternative) â
454 UCS_string ucs(UNI_Quad_Quad);
455  Assert(ucs[0]);
456 
457  if (src.rest() > 0) ucs.append(src[0]);
458  if (src.rest() > 1) ucs.append(src[1]);
459  if (src.rest() > 2) ucs.append(src[2]);
460  if (src.rest() > 3) ucs.append(src[3]);
461  if (src.rest() > 4) ucs.append(src[4]);
462 
463 int len = 0;
464 const Token t = Workspace::get_quad(ucs, len);
465  src.skip(len - 1);
466  tos.push_back(t);
467 }
468 //-----------------------------------------------------------------------------
473 void
474 Tokenizer::tokenize_string1(Unicode_source & src, Token_string & tos)
475 {
476  Log(LOG_tokenize) CERR << "tokenize_string1(" << src << ")" << endl;
477 
478 const Unicode uni = src.get();
479  Assert(Avec::is_single_quote(uni));
480 
481 UCS_string string_value;
482 bool got_end = false;
483 
484  while (src.rest())
485  {
486  const Unicode uni = src.get();
487 
488  if (Avec::is_single_quote(uni))
489  {
490  // a single ' is the end of the string, while a double '
491  // (i.e. '') is a single '.
492  //
493  if ((src.rest() == 0) || !Avec::is_single_quote(*src))
494  {
495  got_end = true;
496  break;
497  }
498 
499  string_value.append(UNI_SINGLE_QUOTE);
500  ++src; // skip the second '
501  }
502  else if (uni == UNI_ASCII_CR)
503  {
504  continue;
505  }
506  else if (uni == UNI_ASCII_LF)
507  {
508  rest_2 = src.rest();
509  Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
510  }
511  else
512  {
513  string_value.append(uni);
514  }
515  }
516 
517  if (!got_end) Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
518 
519  if (string_value.size() == 1) // scalar
520  {
521  tos.push_back(Token(TOK_CHARACTER, string_value[0]));
522  }
523  else
524  {
525  tos.push_back(Token(TOK_APL_VALUE1, Value_P(string_value, LOC)));
526  }
527 }
528 //-----------------------------------------------------------------------------
533 void
534 Tokenizer::tokenize_string2(Unicode_source & src, Token_string & tos)
535 {
536  Log(LOG_tokenize) CERR << "tokenize_string2(" << src << ")" << endl;
537 
538  // skip the leading "
539  {
540  const Unicode uni = src.get();
541  if (uni) { /* do nothing, needed for -Wall */ }
542 
543  Assert1(uni == UNI_ASCII_DOUBLE_QUOTE);
544  }
545 
546 UCS_string string_value;
547 bool got_end = false;
548 
549  while (src.rest())
550  {
551  const Unicode uni = src.get();
552 
553  if (uni == UNI_ASCII_DOUBLE_QUOTE) // terminating "
554  {
555  got_end = true;
556  break;
557  }
558  else if (uni == UNI_ASCII_CR) // ignore CR
559  {
560  continue;
561  }
562  else if (uni == UNI_ASCII_LF) // end of line before "
563  {
564  rest_2 = src.rest();
565  Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
566  }
567  else if (uni == UNI_ASCII_BACKSLASH) // backslash
568  {
569  const Unicode uni1 = src.get();
570  switch(uni1)
571  {
572  case '0': string_value.append(UNI_ASCII_NUL); break;
573  case 'a': string_value.append(UNI_ASCII_BEL); break;
574  case 'b': string_value.append(UNI_ASCII_BS); break;
575  case 't': string_value.append(UNI_ASCII_HT); break;
576  case 'n': string_value.append(UNI_ASCII_LF); break;
577  case 'v': string_value.append(UNI_ASCII_VT); break;
578  case 'f': string_value.append(UNI_ASCII_FF); break;
579  case 'r': string_value.append(UNI_ASCII_CR); break;
580  case '[': string_value.append(UNI_ASCII_ESC); break;
581  case '"': string_value.append(UNI_ASCII_DOUBLE_QUOTE);break;
582  case '\\': string_value.append(UNI_ASCII_BACKSLASH); break;
583  default: string_value.append(uni);
584  string_value.append(uni1);
585  }
586  }
587  else
588  {
589  string_value.append(uni);
590  }
591  }
592 
593  if (!got_end) Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
594 
595  else
596  {
597  tos.push_back(Token(TOK_APL_VALUE1, Value_P(string_value, LOC)));
598  }
599 }
600 //-----------------------------------------------------------------------------
601 void
602 Tokenizer::tokenize_number(Unicode_source & src, Token_string & tos)
603 {
604  Log(LOG_tokenize) CERR << "tokenize_number(" << src << ")" << endl;
605 
606  // numbers:
607  // real
608  // real 'J' real
609  // real 'D' real // magnitude + angle in degrees
610  // real 'R' real // magnitude + angle in radian
611 
612 APL_Float real_flt = 0.0; // always valid
613 APL_Integer real_int = 0; // valid if need_float is false
614 bool real_need_float = false;
615 const bool real_valid = tokenize_real(src, real_need_float, real_flt, real_int);
616  if (!real_need_float) real_flt = real_int;
617  if (!real_valid)
618  {
619  rest_2 = src.rest();
620  Error::throw_parse_error(E_BAD_NUMBER, LOC, loc);
621  }
622 
623  if (src.rest() && (*src == UNI_ASCII_J || *src == UNI_ASCII_j))
624  {
625  ++src; // skip 'J'
626 
627  APL_Float imag_flt = 0.0; // always valid
628  APL_Integer imag_int = 0; // valid if imag_need_float is false
629  bool imag_need_float = false;
630  const bool imag_valid = tokenize_real(src, imag_need_float,
631  imag_flt, imag_int);
632  if (!imag_need_float) imag_flt = imag_int;
633 
634  if (!imag_valid)
635  {
636  --src; // undo skip 'J'
637  if (real_need_float)
638  {
639  tos.push_back(Token(TOK_REAL, real_flt));
640  Log(LOG_tokenize)
641  CERR << " tokenize_number: real " << real_flt << endl;
642  }
643  else
644  {
645  tos.push_back(Token(TOK_INTEGER, real_int));
646  Log(LOG_tokenize)
647  CERR << " tokenize_number: integer " << real_int << endl;
648  }
649  goto done;;
650  }
651 
652  tos.push_back(Token(TOK_COMPLEX, real_flt, imag_flt));
653  Log(LOG_tokenize) CERR << " tokenize_number: complex "
654  << real_flt << "J" << imag_flt << endl;
655  }
656  else if (src.rest() && (*src == UNI_ASCII_D || *src == UNI_ASCII_d))
657  {
658  ++src; // skip 'D'
659 
660  APL_Float degrees_flt = 0; // always valid
661  APL_Integer degrees_int = 0; // valid if imag_floating is true
662  bool imag_need_float = false;
663  const bool imag_valid = tokenize_real(src, imag_need_float,
664  degrees_flt, degrees_int);
665  if (!imag_need_float) degrees_flt = degrees_int;

795 bool
796 Tokenizer::tokenize_real(Unicode_source & src, bool & need_float,
797  APL_Float & flt_val, APL_Integer & int_val)
798 {
799  int_val = 0;
800  need_float = false;
801 
802 UTF8_string int_digits;
803 UTF8_string fract_digits;
804 UTF8_string expo_digits;
805 bool negative = false;
806 bool expo_negative = false;
807 bool skipped_0 = false;
808 bool dot_seen = false;
809 
810  // hexadecimal ?
811  //
812  if (src.rest() > 1 && *src == UNI_ASCII_DOLLAR_SIGN)
813  {
814  src.get(); // skip $
815  if (!Avec::is_hex_digit(*src)) return false; // no hex after $
816  while (src.rest())
817  {
818  int digit;
819  if (*src < UNI_ASCII_0) break;
820  else if (*src <= UNI_ASCII_9) digit = src.get() - '0';
821  else if (*src < UNI_ASCII_A) break;
822  else if (*src <= UNI_ASCII_F) digit = 10 + src.get() - 'A';
823  else if (*src < UNI_ASCII_a) break;
824  else if (*src <= UNI_ASCII_f) digit = 10 + src.get() - 'a';
825  else break;
826  int_val = int_val << 4 | digit;
827  }
828  flt_val = int_val;
829  return true; // OK
830  }
831 
832  // 1. split src into integer, fractional, and exponent parts, removing:
833  // 1a. a leading sign of the integer part
834  // 1b. leading zeros of the integer part
835  // 1c. the . between the integer and fractional parts
836  // 1d. trailing zeros of the fractional part
837  // 1e. the E between the fractional and exponent parts
838  // 1f. a sign of the exponent part
839  //
840  if (src.rest() && *src == UNI_OVERBAR) // 1a.
841  {
842  negative = true;
843  ++src;
844  }
845 
846  // 1b. skip leading zeros in integer part
847  //
848  while (src.rest() && *src == '0') { ++src; skipped_0 = true; }
849 
850  // integer part
851  //
852  while (src.rest() && Avec::is_digit(*src)) int_digits += src.get();
853 
854  // fractional part
855  //
856  if (src.rest() && *src == UNI_ASCII_FULLSTOP) // fract part present
857  {
858  ++src; // 1c. skip '.'
859  dot_seen = true;
860  while (src.rest() && Avec::is_digit(*src))
861  {
862  fract_digits += src.get();
863  }
864 
865  while (fract_digits.size() && fract_digits.back() == '0') // 1d.
866  fract_digits.pop_back();
867  }
868 
869  // require at least one integer or fractional digit
870  //
871  if (int_digits.size() == 0 &&
872  fract_digits.size() == 0 &&
873  !skipped_0) return false; // error
874 
875  // exponent part (but could also be a name starting with E or e)
876  //
877  if (src.rest() >= 2 && // at least E and a digit or Â¯
878  (*src == UNI_ASCII_E || *src == UNI_ASCII_e)) // maybe exponent
879  {
880  expo_negative = (src[1] == UNI_OVERBAR);
881  if (expo_negative &&
882  src.rest() >= 3 &&
883  Avec::is_digit(src[2])) // EÂ¯nnn
884  {
885  need_float = true;
886  ++src; // skip e/E
887  ++src; // skip Â¯
888  while (src.rest() && Avec::is_digit(*src))
889  expo_digits += src.get();
890  }
891  else if (Avec::is_digit(src[1])) // Ennn
892  {
893  need_float = true;
894  ++src; // skip e/E
895  while (src.rest() && Avec::is_digit(*src))
896  expo_digits += src.get();
897  }
898  }
899 
900  // second dot ?
901  if (dot_seen && src.rest() && *src == UNI_ASCII_FULLSTOP)
902  return false; // error
903 
904 int expo = 0;
905  loop(d, expo_digits.size()) expo = 10*expo + (expo_digits[d] - '0');
906  if (expo_negative) expo = - expo;
907  expo += int_digits.size();
908 
909 UTF8_string digits = int_digits;
910  digits.append_UTF8(fract_digits);
911 
912  // at this point, digits is the fractional part ff... of 0.ff... Ã10âexpo
913  // discard leading fractional 0s and adjust expo accordingly
914  //
915  while (digits.size() && digits[0] == '0')
916  {
917  digits.erase(0); // discard '0'
918  --expo;
919  }
920 
921  // at this point, digits is the fractional part ff... of 0.ff... Ã10âexpo
922  // from 0.1000... to 0.9999... Discard digits beyond integer precision.
923  //
924  // The largest 64-bit integer is 0x7FFFFFFFFFFFFFFF = 9223372036854775807
925  // which has 19 decimal digits. Discard all after first 20 digits and
926  // round according to the last digit.
927  //
928  if (digits.size() > MAX_TOKENIZE_DIGITS_1)
929  digits.resize(MAX_TOKENIZE_DIGITS_1);
930 
931  if (digits.size() == MAX_TOKENIZE_DIGITS_1)
932  {
933  if (digits.round_0_1()) ++expo;
934  }
935 
936  // special cases: all digits 0 or very small number
937  //
938  if (digits.size() == 0) return true; // OK: all digits were 0
939  if (expo <= -307) return true; // OK: very small number
940 
941  Assert(digits.size() <= MAX_TOKENIZE_DIGITS);
942 
943  if (expo > MAX_TOKENIZE_DIGITS) need_float = true;
944 
945  // special case: integer between 9223372036854775807 and 9999999999999999999
946  //
947  if (expo == MAX_TOKENIZE_DIGITS && digits[0] == '9')
948  {
949  // a uint64_t compare might overflow, so we compare the digit string
950  const char * maxint = "9223372036854775807";
951  loop(j, digits.size())
952  {
953  if (digits[j] < maxint[j]) break;
954  if (digits[j] == maxint[j]) continue;
955  need_float = true;
956  break;
957  }
958  }
959 
960  if (int(digits.size()) > expo) need_float = true;
961 
962  if (need_float)
963  {
964  if (digits.size() > 17) digits.resize(17);
965 
966  int64_t v = 0;
967  loop(j, digits.size())
968  {
969  v = 10*v + (digits[j] - '0');
970  --expo;
971  }
972 
973  if (expo > 0)
974  {
975  if (expo > 308) return false;
976  if (negative) flt_val = - v * expo_tab[expo];
977  else flt_val = v * expo_tab[expo];
978  return true; // OK
979  }
980  else if (expo < 0)
981  {
982  if (negative) flt_val = - v * nexpo_tab[-expo];
983  else flt_val = v * nexpo_tab[-expo];
984  return true; // OK
985  }
986 
987  if (negative) flt_val = - v;
988  else flt_val = v;
989  return true; // OK
990  }
991  else
992  {
993  int_val = 0;
994  loop(j, digits.size()) int_val = 10*int_val + (digits[j] - '0');
995  if (negative) int_val = - int_val;
996  flt_val = int_val;
997  return true; // OK
998  }
999 }

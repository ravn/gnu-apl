54 inline ostream & operator << (ostream & out, const Unicode_source & src)

56 //-----------------------------------------------------------------------------
59 ErrorCode
60 Tokenizer::tokenize(const UCS_string & input, Token_string & tos) const
61 {
62 size_t rest_2 = 0;
63  try {
64  do_tokenize(input, tos, rest_2);
65  }
66  catch (Error err)
67  {
68  const int caret_1 = input.size();
69  const int caret_2 = input.size() - rest_2;
70  err.set_error_line_2(input, caret_1, caret_2);
71  return err.get_error_code();
72  }
73 
74  return E_NO_ERROR;
75 }
76 //-----------------------------------------------------------------------------
79 void
80 Tokenizer::do_tokenize(const UCS_string & input, Token_string & tos,
81  size_t & rest_2) const
82 {
83  Log(LOG_tokenize)
84  CERR << "tokenize: input[" << input.size() << "] is: «"
85  << input << "»" << endl;
86 
87 Unicode_source src(input);
88  while ((rest_2 = src.rest()) != 0)
89  {
90  Unicode uni = *src;
91  if (uni == UNI_COMMENT) break; // ⍝ comment
92  if (uni == UNI_NUMBER_SIGN) break; // # comment
93 
94  const Token tok = Avec::uni_to_token(uni, LOC);
95 
96  Log(LOG_tokenize)
97  {
98  Unicode_source s1(src, 0, 24);
99  CERR << " tokenize(" << src.rest() << " chars) sees [tag "
100  << tok.tag_name() << " «" << uni << "»] " << s1;
101  if (src.rest() != s1.rest()) CERR << " ...";
102  CERR << endl;
103  }
104 
105  switch(tok.get_Class())
106  {
107  case TC_END: // chars without APL meaning
108  rest_2 = src.rest();
109  {
110  Log(LOG_error_throw)
111  CERR << endl << "throwing "
112  << Error::error_name(E_NO_TOKEN)
113  << " in Tokenizer" << endl;
114 
115  char cc[20];
116  snprintf(cc, sizeof(cc), "U+%4.4X (", uni);
117  MORE_ERROR() << "Tokenizer: No token for Unicode "
118  << cc << uni << ")\nInput: " << input;
119  Error error(E_NO_TOKEN, LOC);
120  throw error;
121  }
122  break;
123 
124  case TC_RETURN:
125  case TC_LINE:
126  case TC_VALUE:
127  case TC_INDEX:
128  CERR << "Offending token: " << tok.get_tag()
129  << " (" << tok << ")" << endl;
130  if (tok.get_tag() == TOK_CHARACTER)
131  CERR << "Unicode: " << UNI(tok.get_char_val()) << endl;
132  rest_2 = src.rest();
133  Error::throw_parse_error(E_NON_APL_CHAR, LOC, loc);
134  break;
135 
136  case TC_VOID:
137  // Avec::uni_to_token returns TC_VOID for non-apl characters
138  //
139  rest_2 = src.rest();
140  UERR << "Unknown APL character: " << uni
141  << " (" << UNI(uni) << ")" << endl;
142  Error::throw_parse_error(E_NON_APL_CHAR, LOC, loc);
143  break;
144 
145  case TC_SYMBOL:
146  if (Avec::is_quad(uni))
147  {
148  tokenize_quad(src, tos);
149  }
150  else if (uni == UNI_QUOTE_Quad)
151  {
152  ++src;
153  tos.push_back(Token(TOK_Quad_QUOTE,
154  &Workspace::get_v_Quad_QUOTE()));
155  }
156  else if (uni == UNI_ALPHA)
157  {
158  ++src;
159  tos.push_back(Token(TOK_ALPHA,
160  &Workspace::get_v_ALPHA()));
161  }
162  else if (uni == UNI_ALPHA_UNDERBAR)
163  {
164  ++src;
165  tos.push_back(Token(TOK_ALPHA_U,
166  &Workspace::get_v_ALPHA_U()));
167  }
168  else if (uni == UNI_CHI)
169  {
170  ++src;
171  tos.push_back(Token(TOK_CHI,
172  &Workspace::get_v_CHI()));
173  }
174  else if (uni == UNI_LAMBDA)
175  {
176  // this could be λ like in λ← ...
177  // or λ1 or λ2 or ... as in ... ⍺ λ1 ⍵
178  //
179  if (src.rest() > 1 && Avec::is_digit(src[1])) // λn
180  {
181  tokenize_symbol(src, tos);
182  }
183  else // λ
184  {
185  ++src;
186  tos.push_back(Token(TOK_LAMBDA,
187  &Workspace::get_v_LAMBDA()));
188  }
189  }
190  else if (uni == UNI_OMEGA)
191  {
192  ++src;
193  tos.push_back(Token(TOK_OMEGA,
194  &Workspace::get_v_OMEGA()));
195  }
196  else if (uni == UNI_OMEGA_UNDERBAR)
197  {
198  ++src;
199  tos.push_back(Token(TOK_OMEGA_U,
200  &Workspace::get_v_OMEGA_U()));
201  }
202  else
203  {
204  tokenize_symbol(src, tos);
205  }
206  break;
207 
208  case TC_FUN0:
209  case TC_FUN12:
210  case TC_OPER1:
211  tokenize_function(src, tos);
212  break;
213 
214  case TC_OPER2:
215  if (tok.get_tag() == TOK_OPER2_INNER && src.rest())
216  {
217  // tok is a dot. This could mean that . is either
218  //
219  // the start of a number: e.g. +.3
220  // or an operator: e.g. +.*
221  // or a syntax error: e.g. Done.
222  //
223  if (src.rest() == 1) // syntax error
224  Error::throw_parse_error(E_SYNTAX_ERROR, LOC, loc);
225 
226  Unicode uni_1 = src[1];
227  const Token tok_1 = Avec::uni_to_token(uni_1, LOC);
228  if ((tok_1.get_tag() & TC_MASK) == TC_NUMERIC)
229  tokenize_number(src, tos, rest_2);
230  else
231  tokenize_function(src, tos);
232  }
233  else
234  {
235  tokenize_function(src, tos);
236  }
237  if (tos.size() >= 2 &&
238  tos.back().get_tag() == TOK_OPER2_INNER &&
239  tos[tos.size() - 2].get_tag() == TOK_JOT)
240  {
241  new (&tos.back()) Token(TOK_OPER2_OUTER,
242  Bif_OPER2_OUTER::fun);
243  }
244 
245  break;
246 
247  case TC_R_ARROW:
248  ++src;
249  if (src.rest()) tos.push_back(tok);
250  else tos.push_back(Token(TOK_ESCAPE));
251  break;
252 
253  case TC_ASSIGN:
254  ++src;
255  {
256  const bool sym = tos.size() >= 1 &&
257  tos[tos.size() - 1].get_tag() == TOK_SYMBOL;
258  const bool dia = tos.size() > 1 &&
259  tos[tos.size() - 2].get_tag() == TOK_DIAMOND;
260  const bool col = tos.size() > 1 &&
261  tos[tos.size() - 2].get_tag() == TOK_COLON;
262 
263  tos.push_back(tok);
264 
265  // change token tag of ← if:
266  //
267  // SYM ← (at the start of line), or
268  // ◊ SYM ← (at the start of statement), or
269  // : SYM ← (at the start of statement after label)
270  //
271  if (sym && ((tos.size() == 2) || dia || col))
272  tos.back().ChangeTag(TOK_ASSIGN1);
273  }
274  break;
275 
276  case TC_L_PARENT:
277  case TC_R_PARENT:
278  case TC_L_BRACK:
279  case TC_R_BRACK:
280  case TC_L_CURLY:
281  case TC_R_CURLY:
282  ++src;
283  tos.push_back(tok);
284  break;
285 
286  case TC_DIAMOND:
287  ++src;
288  tos.push_back(tok);
289  break;
290 
291  case TC_COLON:
292  if (pmode != PM_FUNCTION)
293  {
294  rest_2 = src.rest();
295  if (pmode == PM_EXECUTE)
296  Error::throw_parse_error(E_ILLEGAL_COLON_EXEC,
297  LOC, loc);
298  else
299  Error::throw_parse_error(E_ILLEGAL_COLON_STAT,
300  LOC, loc);
301  }
302 
303  ++src;
304  tos.push_back(tok);
305  break;
306 
307  case TC_NUMERIC:
308  tokenize_number(src, tos, rest_2);
309  break;
310 
311  case TC_SPACE:
312  case TC_NEWLINE:
313  ++src;
314  break;
315 
316  case TC_QUOTE:
317  if (tok.get_tag() == TOK_QUOTE1)
318  tokenize_string1(src, tos, rest_2);
319  else
320  tokenize_string2(src, tos, rest_2);
321  break;
322 
323  default:
324  CERR << "Input: " << input << endl
325  << "uni: " << uni << endl
326  << "Token = " << tok.get_tag() << endl;
327 
328  if (tok.get_Id() != ID_No_ID)
329  {
330  CERR << ", Id = " << Id(tok.get_tag() >> 16);
331  }
332  CERR << endl;
333  Assert(0 && "Should not happen");
334  }
335  }
336 
337  Log(LOG_tokenize)
338  CERR << "tokenize() done (no error)" << endl;
339 }

449 //-----------------------------------------------------------------------------
450 void
451 Tokenizer::tokenize_quad(Unicode_source & src, Token_string & tos) const
452 {
453  Log(LOG_tokenize)
454  CERR << "tokenize_quad(" << src.rest() << " chars)"<< endl;
455 
456  src.get(); // discard (possibly alternative) ⎕
457 UCS_string ucs(UNI_Quad_Quad);
458  Assert(ucs[0]);
459 
460  if (src.rest() > 0) ucs.append(src[0]);
461  if (src.rest() > 1) ucs.append(src[1]);
462  if (src.rest() > 2) ucs.append(src[2]);
463  if (src.rest() > 3) ucs.append(src[3]);
464  if (src.rest() > 4) ucs.append(src[4]);
465 
466 int len = 0;
467 const Token t = Workspace::get_quad(ucs, len);
468  src.skip(len - 1);
469  tos.push_back(t);
470 }
471 //-----------------------------------------------------------------------------
476 void
477 Tokenizer::tokenize_string1(Unicode_source & src, Token_string & tos,
478  size_t & rest_2) const
479 {
480  Log(LOG_tokenize) CERR << "tokenize_string1(" << src << ")" << endl;
481 
482 const Unicode uni = src.get();
483  Assert(Avec::is_single_quote(uni));
484 
485 UCS_string string_value;
486 bool got_end = false;
487 
488  while (src.rest())
489  {
490  const Unicode uni = src.get();
491 
492  if (Avec::is_single_quote(uni))
493  {
494  // a single ' is the end of the string, while a double '
495  // (i.e. '') is a single '.
496  //
497  if ((src.rest() == 0) || !Avec::is_single_quote(*src))
498  {
499  got_end = true;
500  break;
501  }
502 
503  string_value.append(UNI_SINGLE_QUOTE);
504  ++src; // skip the second '
505  }
506  else if (uni == UNI_CR)
507  {
508  continue;
509  }
510  else if (uni == UNI_LF)
511  {
512  rest_2 = src.rest();
513  Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
514  }
515  else
516  {
517  string_value.append(uni);
518  }
519  }
520 
521  if (!got_end) Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
522 
523  if (string_value.size() == 1) // scalar
524  {
525  tos.push_back(Token(TOK_CHARACTER, string_value[0]));
526  }
527  else
528  {
529  tos.push_back(Token(TOK_APL_VALUE1, Value_P(string_value, LOC)));
530  }
531 }
532 //-----------------------------------------------------------------------------
537 void
538 Tokenizer::tokenize_string2(Unicode_source & src, Token_string & tos,
539  size_t & rest_2) const
540 {
541  Log(LOG_tokenize) CERR << "tokenize_string2(" << src << ")" << endl;
542 
543  // skip the leading "
544  {
545  const Unicode uni = src.get();
546  if (uni) { /* do nothing, needed for -Wall */ }
547 
548  Assert1(uni == UNI_DOUBLE_QUOTE);
549  }
550 
551 UCS_string string_value;
552 bool got_end = false;
553 
554  while (src.rest())
555  {
556  const Unicode uni = src.get();
557 
558  if (uni == UNI_DOUBLE_QUOTE) // terminating "
559  {
560  got_end = true;
561  break;
562  }
563  else if (uni == UNI_CR) // ignore CR
564  {
565  continue;
566  }
567  else if (uni == UNI_LF) // end of line before "
568  {
569  rest_2 = src.rest();
570  Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
571  }
572  else if (uni == UNI_BACKSLASH) // backslash
573  {
574  const Unicode uni1 = src.get();
575  switch(uni1)
576  {
577  case '0': string_value.append(UNI_NUL); break;
578  case 'a': string_value.append(UNI_BEL); break;
579  case 'b': string_value.append(UNI_BS); break;
580  case 't': string_value.append(UNI_HT); break;
581  case 'n': string_value.append(UNI_LF); break;
582  case 'v': string_value.append(UNI_VT); break;
583  case 'f': string_value.append(UNI_FF); break;
584  case 'r': string_value.append(UNI_CR); break;
585  case '[': string_value.append(UNI_ESC); break;
586  case '"': string_value.append(UNI_DOUBLE_QUOTE);break;
587  case '\\': string_value.append(UNI_BACKSLASH); break;
588  default: string_value.append(uni);
589  string_value.append(uni1);
590  }
591  }
592  else
593  {
594  string_value.append(uni);
595  }
596  }
597 
598  if (!got_end) Error::throw_parse_error(E_NO_STRING_END, LOC, loc);
599 
600  else
601  {
602  tos.push_back(Token(TOK_APL_VALUE1, Value_P(string_value, LOC)));
603  }
604 }
605 //-----------------------------------------------------------------------------
606 void
607 Tokenizer::tokenize_number(Unicode_source & src, Token_string & tos,
608  size_t & rest_2) const
609 {
610  Log(LOG_tokenize) CERR << "tokenize_number(" << src << ")" << endl;
611 
612  // numbers:
613  // real
614  // real 'J' real
615  // real 'D' real // magnitude + angle in degrees
616  // real 'R' real // magnitude + angle in radian
617 
618 APL_Float real_flt = 0.0; // always valid
619 APL_Integer real_int = 0; // valid if need_float is false
620 bool real_need_float = false;
621 const bool real_valid = tokenize_real(src, real_need_float, real_flt, real_int);
622  if (!real_need_float) real_flt = real_int;
623  if (!real_valid)
624  {
625  rest_2 = src.rest();
626  Error::throw_parse_error(E_BAD_NUMBER, LOC, loc);
627  }
628 
629  if (src.rest() && (*src == UNI_J || *src == UNI_j))
630  {
631  ++src; // skip 'J'
632 
633  APL_Float imag_flt = 0.0; // always valid
634  APL_Integer imag_int = 0; // valid if imag_need_float is false
635  bool imag_need_float = false;
636  const bool imag_valid = tokenize_real(src, imag_need_float,
637  imag_flt, imag_int);
638  if (!imag_need_float) imag_flt = imag_int;
639 
640  if (!imag_valid)
641  {
642  --src; // undo skip 'J'
643  if (real_need_float)
644  {
645  tos.push_back(Token(TOK_REAL, real_flt));
646  Log(LOG_tokenize)
647  CERR << " tokenize_number: real " << real_flt << endl;
648  }
649  else
650  {
651  tos.push_back(Token(TOK_INTEGER, real_int));
652  Log(LOG_tokenize)
653  CERR << " tokenize_number: integer " << real_int << endl;
654  }
655  goto done;;
656  }
657 
658  tos.push_back(Token(TOK_COMPLEX, real_flt, imag_flt));
659  Log(LOG_tokenize) CERR << " tokenize_number: complex "
660  << real_flt << "J" << imag_flt << endl;
661  }
662  else if (src.rest() && (*src == UNI_D || *src == UNI_d))
663  {
664  ++src; // skip 'D'
665 
666  APL_Float degrees_flt = 0; // always valid
667  APL_Integer degrees_int = 0; // valid if imag_floating is true
668  bool imag_need_float = false;
669  const bool imag_valid = tokenize_real(src, imag_need_float,
670  degrees_flt, degrees_int);
671  if (!imag_need_float) degrees_flt = degrees_int;

692  //

694  APL_Float imag = real_flt * sin(M_PI*degrees_flt / 180.0);
695  tos.push_back(Token(TOK_COMPLEX, real, imag));
696  Log(LOG_tokenize) CERR << " tokenize_number: complex " << real

698  }

707  radian_flt, radian_int);
708  if (!radian_need_float) radian_flt = radian_int;

710  if (!imag_valid)
711  {
712  --src; // undo skip 'R'
713  if (real_need_float)
714  {
715  tos.push_back(Token(TOK_REAL, real_flt));
716  Log(LOG_tokenize)
717  CERR << " tokenize_number: real " << real_flt << endl;
718  }
719  else
720  {
721  tos.push_back(Token(TOK_INTEGER, real_int));
722  Log(LOG_tokenize)
723  CERR << " tokenize_number: integer " << real_int << endl;
724  }
725  goto done;;
726  }
727 
728  // real_flt is the magnitude and the angle is in radian.
729  //
730  APL_Float real = real_flt * cos(radian_flt);
731  APL_Float imag = real_flt * sin(radian_flt);
732  tos.push_back(Token(TOK_COMPLEX, real, imag));
733  Log(LOG_tokenize) CERR << " tokenize_number: complex " << real
734  << "J" << imag << endl;
735  }
736  else
737  {
738  if (real_need_float)
739  {
740  tos.push_back(Token(TOK_REAL, real_flt));
741  Log(LOG_tokenize)
742  CERR << " tokenize_number: real " << real_flt << endl;
743  }
744  else
745  {
746  tos.push_back(Token(TOK_INTEGER, real_int));
747  Log(LOG_tokenize)
748  CERR << " tokenize_number: integer " << real_int << endl;
749  }
750  }
751 
752 done:
753  // ISO 13751 requires a space between two numeric scalar literals (page 42),
754  // but not between a numeric scalar literal and an identifier.
755  //
756  // IBM APL2 requires a space in both cases. For example, 10Q10 is tokenized
757  // as 10Q 10
758  //
759  // We follow ISO. The second numeric literal cannot start with 0-9 because
760  // that would have been eaten by the first literal. Therefor the only cases
761  // remaining to be checked are a numeric scalar literal followed by ¯
762  // or by . (page 42)
763  //
764  if (src.rest())
765  {
766  if (*src == UNI_OVERBAR || *src == '.')
767  Error::throw_parse_error(E_BAD_NUMBER, LOC, loc);
768  }

801 bool
802 Tokenizer::tokenize_real(Unicode_source & src, bool & need_float,
803  APL_Float & flt_val, APL_Integer & int_val) const
804 {
805  int_val = 0;
806  need_float = false;
807 
808 UTF8_string int_digits;
809 UTF8_string fract_digits;
810 UTF8_string expo_digits;
811 bool negative = false;
812 bool expo_negative = false;
813 bool skipped_0 = false;
814 bool dot_seen = false;
815 
816  // hexadecimal ?
817  //
818  if (src.rest() > 1 && *src == UNI_DOLLAR_SIGN)
819  {
820  src.get(); // skip $
821  if (!Avec::is_hex_digit(*src)) return false; // no hex after $
822  while (src.rest())
823  {
824  int digit;
825  if (*src < UNI_0) break;
826  else if (*src <= UNI_9) digit = src.get() - '0';
827  else if (*src < UNI_A) break;
828  else if (*src <= UNI_F) digit = 10 + src.get() - 'A';
829  else if (*src < UNI_a) break;
830  else if (*src <= UNI_f) digit = 10 + src.get() - 'a';
831  else break;
832  int_val = int_val << 4 | digit;
833  }
834  flt_val = int_val;
835  return true; // OK
836  }
837 
838  // 1. split src into integer, fractional, and exponent parts, removing:
839  // 1a. a leading sign of the integer part
840  // 1b. leading zeros of the integer part
841  // 1c. the . between the integer and fractional parts
842  // 1d. trailing zeros of the fractional part
843  // 1e. the E between the fractional and exponent parts
844  // 1f. a sign of the exponent part
845  //
846  if (src.rest() && *src == UNI_OVERBAR) // 1a.
847  {
848  negative = true;
849  ++src;
850  }
851 
852  // 1b. skip leading zeros in integer part
853  //
854  while (src.rest() && *src == '0') { ++src; skipped_0 = true; }
855 
856  // integer part
857  //
858  while (src.rest() && Avec::is_digit(*src)) int_digits += src.get();
859 
860  // fractional part
861  //
862  if (src.rest() && *src == UNI_FULLSTOP) // fract part present
863  {
864  ++src; // 1c. skip '.'
865  dot_seen = true;
866  while (src.rest() && Avec::is_digit(*src))
867  {
868  fract_digits += src.get();
869  }
870 
871  while (fract_digits.size() && fract_digits.back() == '0') // 1d.
872  fract_digits.pop_back();
873  }
874 
875  // require at least one integer or fractional digit
876  //
877  if (int_digits.size() == 0 &&
878  fract_digits.size() == 0 &&
879  !skipped_0) return false; // error
880 
881  // exponent part (but could also be a name starting with E or e)
882  //
883  if (src.rest() >= 2 && // at least E and a digit or ¯
884  (*src == UNI_E || *src == UNI_e)) // maybe exponent
885  {
886  expo_negative = (src[1] == UNI_OVERBAR);
887  if (expo_negative &&
888  src.rest() >= 3 &&
889  Avec::is_digit(src[2])) // E¯nnn
890  {
891  need_float = true;
892  ++src; // skip e/E
893  ++src; // skip ¯
894  while (src.rest() && Avec::is_digit(*src))
895  expo_digits += src.get();
896  }
897  else if (Avec::is_digit(src[1])) // Ennn
898  {
899  need_float = true;
900  ++src; // skip e/E
901  while (src.rest() && Avec::is_digit(*src))
902  expo_digits += src.get();
903  }
904  }
905 
906  // second dot ?
907  if (dot_seen && src.rest() && *src == UNI_FULLSTOP)
908  return false; // error
909 
910 int expo = 0;
911  loop(d, expo_digits.size()) expo = 10*expo + (expo_digits[d] - '0');
912  if (expo_negative) expo = - expo;
913  expo += int_digits.size();
914 
915 UTF8_string digits = int_digits;
916  digits.append_UTF8(fract_digits);
917 
918  // at this point, digits is the fractional part ff... of 0.ff... ×10⋆expo
919  // discard leading fractional 0s and adjust expo accordingly
920  //
921  while (digits.size() && digits[0] == '0')
922  {
923  digits.erase(0); // discard '0'
924  --expo;
925  }
926 
927  // at this point, digits is the fractional part ff... of 0.ff... ×10⋆expo
928  // from 0.1000... to 0.9999... Discard digits beyond integer precision.
929  //
930  // The largest 64-bit integer is 0x7FFFFFFFFFFFFFFF = 9223372036854775807
931  // which has 19 decimal digits. Discard all after first 20 digits and
932  // round according to the last digit.
933  //
934  if (digits.size() > MAX_TOKENIZE_DIGITS_1)
935  digits.resize(MAX_TOKENIZE_DIGITS_1);
936 
937  if (digits.size() == MAX_TOKENIZE_DIGITS_1)
938  {
939  if (digits.round_0_1()) ++expo;
940  }
941 
942  // special cases: all digits 0 or very small number
943  //
944  if (digits.size() == 0) return true; // OK: all digits were 0
945  if (expo <= -307) return true; // OK: very small number
946 
947  Assert(digits.size() <= MAX_TOKENIZE_DIGITS);
948 
949  if (expo > MAX_TOKENIZE_DIGITS) need_float = true;
950 
951  // special case: integer between 9223372036854775807 and 9999999999999999999
952  //
953  if (expo == MAX_TOKENIZE_DIGITS && digits[0] == '9')
954  {
955  // a uint64_t compare might overflow, so we compare the digit string
956  const char * maxint = "9223372036854775807";
957  loop(j, digits.size())
958  {
959  if (digits[j] < maxint[j]) break;
960  if (digits[j] == maxint[j]) continue;
961  need_float = true;
962  break;
963  }
964  }
965 
966  if (int(digits.size()) > expo) need_float = true;
967 
968  if (need_float)
969  {
970  if (digits.size() > 17) digits.resize(17);
971 
972  int64_t v = 0;
973  loop(j, digits.size())
974  {
975  v = 10*v + (digits[j] - '0');
976  --expo;
977  }
978 
979  if (expo > 0)
980  {
981  if (expo > 308) return false;
982  if (negative) flt_val = - v * expo_tab[expo];
983  else flt_val = v * expo_tab[expo];
984  return true; // OK
985  }
986  else if (expo < 0)
987  {
988  if (negative) flt_val = - v * nexpo_tab[-expo];
989  else flt_val = v * nexpo_tab[-expo];
990  return true; // OK
991  }
992 
993  if (negative) flt_val = - v;
994  else flt_val = v;
995  return true; // OK
996  }
997  else
998  {
999  int_val = 0;
1000  loop(j, digits.size()) int_val = 10*int_val + (digits[j] - '0');
1001  if (negative) int_val = - int_val;
1002  flt_val = int_val;
1003  return true; // OK
1004  }
1005 }

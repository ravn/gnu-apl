123 Symbol::assign(Value_P new_value, bool clone, const char * loc)
124 {
125  Assert(!!new_value);
126  Assert(value_stack.size());
127 
128  if (!new_value->is_complete())
129  {
130  CERR << "Incomplete value at " LOC << endl;
131  new_value->print_properties(CERR, 0, false);
132  VH_entry::print_history(CERR, new_value.get(), LOC);
133  Assert(0);
134  }
135 
136 ValueStackItem & vs = value_stack.back();
137 
138  switch(vs.name_class)
139  {
140  case NC_UNUSED_USER_NAME:
141  if (clone) new_value = new_value->clone(loc);
142 
143  vs.name_class = NC_VARIABLE;
144  vs.apl_val = new_value;
145  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
146  return;
147 
148  case NC_LABEL:
149  MORE_ERROR() << "attempt to assign a value to label " << get_name();
150  case NC_VARIABLE:
151  if (vs.apl_val == new_value) return; // XâX
152 
153  if (clone) new_value = new_value->clone(loc);
154 
155  vs.apl_val = new_value;
156  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
157  return;
158 
159  case NC_SHARED_VAR:
160  assign_shared_variable(new_value, loc);
161  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
162  return;
163 
164  default: SYNTAX_ERROR;
165  }
166 }
167 //-----------------------------------------------------------------------------
168 void
169 Symbol::assign_indexed(Value_P X, Value_P B) // A[X] â B
170 {
171  // this function is called for A[X}âB when X is one-dimensional, i.e.
172  // an index with no semicolons. If X contains semicolons, then
173  // assign_indexed(IndexExpr IX, ...) is called instead.
174  //
175 const APL_Integer qio = Workspace::get_IO();
176 
177 Value_P A = get_apl_value();
178 const ShapeItem max_idx = A->element_count();
179  if (!!X && X->is_scalar() && B->is_scalar() && A->get_rank() == 1)
180  {
181  const APL_Integer idx = X->get_ravel(0).get_near_int() - qio;
182  if (idx >= 0 && idx < max_idx)
183  {
184  Cell & cell = A->get_ravel(idx);
185  cell.release(LOC);
186  cell.init(B->get_ravel(0), A.getref(), LOC);
187  return;
188  }
189  }
190 
191  if (A->get_rank() != 1) RANK_ERROR;
192 
193  if (!X) // X[] â B
194  {
195  const Cell & src = B->get_ravel(0);
196  loop(a, max_idx)
197  {
198  Cell & dest = A->get_ravel(a);
199  dest.release(LOC); // free sub-values etc (if any)
200  dest.init(src, A.getref(), LOC);
201  }
202  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
203  return;
204  }
205 
206 const ShapeItem ec_B = B->element_count();
207 const ShapeItem ec_X = X->element_count();
208 const int incr_B = (ec_B == 1) ? 0 : 1; // maybe scalar extend B
209 const Cell * cX = &X->get_ravel(0);
210 const Cell * cB = &B->get_ravel(0);
211 
212  if (ec_B != 1 && ec_B != ec_X) LENGTH_ERROR;
213 
214  loop(x, ec_X)
215  {
216  const ShapeItem idx = cX++->get_near_int() - qio;
217  if (idx < 0) INDEX_ERROR;
218  if (idx >= max_idx) INDEX_ERROR;
219  Cell & dest = A->get_ravel(idx);
220  dest.release(LOC); // free sub-values etc (if any)
221  dest.init(*cB, A.getref(), LOC);
222 
223  cB += incr_B;
224  }
225 
226  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
227 }
228 //-----------------------------------------------------------------------------
229 void
230 Symbol::assign_indexed(IndexExpr & IX, Value_P B) // A[IX;...] â B
231 {
232  if (IX.value_count() == 1 && !!IX.values[0]) // one-dimensional index
233  {
234  assign_indexed(IX.values[0], B);
235  return;
236  }
237 
238  // see Value::index() for comments.
239 
240 Value_P A = get_apl_value();
241  if (A->get_rank() != IX.value_count()) RANK_ERROR; // ISO p. 159
242 
243  // B must either be a scalar (and is then scalar extended to the size
244  // of the updated area, or else have the shape of the concatenated index
245  // items for example:
246  //
247  // X: X1 ; X2 ; X3
248  // â´B: b1 b2 b3 b4 b5 b6
249  //
250  if (1 && !B->is_scalar())
251  {
252  // remove dimensions with len 1 from the shapes of X and B...
253  // if we see an empty Xn then we return.
254  //
255  Shape B1;
256  loop(b, B->get_rank())
257  {
258  const ShapeItem sb = B->get_shape_item(b);
259  if (sb != 1) B1.add_shape_item(sb);
260  }
261 
262  Shape IX1;
263  for (ShapeItem ix = IX.value_count() - 1; ix >= 0; --ix)
264  {
265  const Value * ix_val = IX.values[ix].get();
266  if (ix_val) // normal index
267  {
268  if (ix_val->element_count() == 0) return; // empty index
269  loop(xx, ix_val->get_rank())
270  {
271  const ShapeItem sxx = ix_val->get_shape_item(xx);
272  if (sxx != 1) IX1.add_shape_item(sxx);
273  }
274  }
275  else // elided index: add corresponding B dimenssion
276  {
277  if (ix >= A->get_rank()) RANK_ERROR;
278  const ShapeItem sbx =
279  A->get_shape_item(A->get_rank() - ix - 1);
280  if (sbx == 0) return; // empty index
281  if (sbx != 1) IX1.add_shape_item(sbx);
282  }
283  }
284 
285  if (B1 != IX1)
286  {
287  if (B1.get_rank() != IX1.get_rank()) RANK_ERROR;
288  LENGTH_ERROR;
289  }
290  }
291 
292 MultiIndexIterator mult(A->get_shape(), IX);
293 
294 const ShapeItem ec_B = B->element_count();
295 const Cell * cB = &B->get_ravel(0);
296 const int incr_B = (ec_B == 1) ? 0 : 1;
297 
298  while (mult.more())
299  {
300  const ShapeItem offset_A = mult++;
301  if (offset_A < 0) INDEX_ERROR;
302  if (offset_A >= A->element_count()) INDEX_ERROR;
303  Cell & dest = A->get_ravel(offset_A);
304  dest.release(LOC); // free sub-values etc (if any)
305  dest.init(*cB, A.getref(), LOC);
306  cB += incr_B;
307  }
308 
309  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
310 }

312 bool
313 Symbol::assign_named_lambda(Function * lambda, const char * loc)
314 {
315 ValueStackItem & vs = value_stack.back();
316 UserFunction * ufun = lambda->get_ufun1();
317  Assert(ufun);
318 const Executable * uexec = ufun;
319  Assert(uexec);
320 
321  switch(vs.name_class)
322  {
323  case NC_FUNCTION:
324  case NC_OPERATOR:
325  {
326  Function * old_fun = vs.sym_val.function;
327  Assert(old_fun);
328  if (!old_fun->is_lambda()) SYNTAX_ERROR;
329  UserFunction * old_ufun = old_fun->get_ufun1();
330  Assert(old_ufun);
331  for (StateIndicator * si = Workspace::SI_top();
332  si; si = si->get_parent())
333  {
334 // if (uexec == si->get_executable())
335  if (si->uses_function(old_ufun))
336  {
337  MORE_ERROR() << "function " << get_name()
338  << " is suspended or used";
339  return true;
340  }
341  }
342 
343  vs.sym_val.function->get_ufun1()->decrement_refcount(LOC);
344  }
345 
346  /* fall through */
347 
348  case NC_UNUSED_USER_NAME:
349  if (lambda->is_operator()) vs.name_class = NC_OPERATOR;
350  else vs.name_class = NC_FUNCTION;
351 
352  vs.sym_val.function = ufun;
353  ufun->increment_refcount(LOC);
354  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
355  return false;
356 
357  default: SYNTAX_ERROR;
358  }
359 
360  return false;
361 }

580 //-----------------------------------------------------------------------------
581 void
582 Symbol::get_attributes(int mode, Cell * dest) const
583 {
584 const ValueStackItem & vs = value_stack.back();
585 bool has_result = false;
586 int fun_valence = 0;
587 int oper_valence = 0;
588 double created = 0.0;
589 static int exec_prop[] = { 0, 0, 0, 0 };
590 const int * exec_properties = exec_prop;
591 
592  switch(vs.name_class)
593  {
594  case NC_LABEL:
595  case NC_VARIABLE:
596  has_result = true;
597  break;
598 
599  case NC_FUNCTION:
600  has_result = vs.sym_val.function->has_result();
601  fun_valence = vs.sym_val.function->get_fun_valence();
602  created = vs.sym_val.function->get_creation_time();
603  exec_properties = vs.sym_val.function->get_exec_properties();
604  break;
605 
606  case NC_OPERATOR:
607  fun_valence = vs.sym_val.function->get_fun_valence();
608  oper_valence = vs.sym_val.function->get_oper_valence();
609  created = vs.sym_val.function->get_creation_time();
610  exec_properties = vs.sym_val.function->get_exec_properties();
611  break;
612 
613  default: break;
614  }
615 
616  switch(mode)
617  {
618  case 1: // valences
619  new (dest + 0) IntCell(has_result ? 1 : 0);
620  new (dest + 1) IntCell(fun_valence);
621  new (dest + 2) IntCell(oper_valence);
622  break;
623 
624  case 2: // creation time
625  if (created == 0.0) // system function
626  {
627  new (dest + 0) IntCell(0);
628  new (dest + 1) IntCell(0);
629  new (dest + 2) IntCell(0);
630  new (dest + 3) IntCell(0);
631  new (dest + 4) IntCell(0);
632  new (dest + 5) IntCell(0);
633  new (dest + 6) IntCell(0);
634  }
635  else // user define function
636  {
637  tm tc;
638  const time_t time = created;
639  localtime_r(&time, &tc);
640 
641  new (dest + 0) IntCell(1900 + tc.tm_year);
642  new (dest + 1) IntCell( 1 + tc.tm_mon );
643  new (dest + 2) IntCell( tc.tm_mday);
644  new (dest + 3) IntCell( tc.tm_hour);
645  new (dest + 4) IntCell( tc.tm_min);
646  new (dest + 5) IntCell( tc.tm_sec);
647  new (dest + 6) IntCell((created - time) / 1000);
648  }
649  break;
650 
651  case 3: // execution properties
652  new (dest + 0) IntCell(exec_properties[0]);
653  new (dest + 1) IntCell(exec_properties[1]);
654  new (dest + 2) IntCell(exec_properties[2]);
655  new (dest + 3) IntCell(exec_properties[3]);
656  break;
657 
658  case 4: {
659  Value_P val = get_apl_value();
660  const CDR_type cdr_type = val->get_CDR_type();
661  const int brutto = val->total_size_brutto(cdr_type);
662  const int data = val->data_size(cdr_type);
663 
664  new (dest + 0) IntCell(brutto);
665  new (dest + 1) IntCell(data);
666  }
667  break;
668 
669  default: Assert(0 && "bad mode");
670  }
671 }
672 //-----------------------------------------------------------------------------
673 void
674 Symbol::resolve(Token & tok, bool left_sym)
675 {
676  Log(LOG_SYMBOL_resolve)
677  CERR << "resolve(" << left_sym << ") symbol " << get_name() << endl;
678 
679  Assert1(value_stack.size());
680 
681 const ValueStackItem & vs = value_stack.back();
682  switch(vs.name_class)
683  {
684  case NC_UNUSED_USER_NAME:
685  if (!left_sym) Error::throw_symbol_error(get_name(), LOC);
686  return; // leave symbol as is
687 
688  case NC_LABEL:
689  if (left_sym) SYNTAX_ERROR; // assignment to (read-only) label
690 
691  {
692  IntCell lab(vs.sym_val.label);
693  Value_P value(lab, LOC);
694  Token t(TOK_APL_VALUE1, value);
695  tok.move_1(t, LOC);
696  }
697  return;
698 
699  case NC_VARIABLE:
700  if (left_sym) return; // leave symbol as is
701 
702  // if we resolve a variable. the value is considered grouped.
703  {
704  Token t(TOK_APL_VALUE1, get_apl_value()->clone(LOC));
705  tok.move_1(t, LOC);
706  }
707  return;
708 
709  case NC_FUNCTION:
710  case NC_OPERATOR:
711  if (left_sym && vs.sym_val.function->is_lambda())
712  {
713  // lambda re-assign, e.g. SYMâ{ ... }
714  //
715  return;
716  }
717  tok.move_2(vs.sym_val.function->get_token(), LOC);
718  return;
719 
720  case NC_SHARED_VAR:
721  if (left_sym) return; // leave symbol as is
722  resolve_shared_variable(tok);
723  return;

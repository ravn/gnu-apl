167 void
168 Symbol::assign_indexed(Value_P X, Value_P B) // A[X] ← B
169 {
170  // this function is called for A[X}←B when X is one-dimensional, i.e.
171  // an index with no semicolons. If X contains semicolons, then
172  // assign_indexed(IndexExpr IX, ...) is called instead.
173  //
174 const APL_Integer qio = Workspace::get_IO();
175 
176 Value_P A = get_apl_value();
177 const ShapeItem max_idx = A->element_count();
178  if (!!X && X->is_scalar() && B->is_scalar() && A->get_rank() == 1)
179  {
180  const APL_Integer idx = X->get_ravel(0).get_near_int() - qio;
181  if (idx >= 0 && idx < max_idx)
182  {
183  Cell & cell = A->get_ravel(idx);
184  cell.release(LOC);
185  cell.init(B->get_ravel(0), A.getref(), LOC);
186  return;
187  }
188  }
189 
190  if (A->get_rank() != 1) RANK_ERROR;
191 
192  if (!X) // X[] ← B
193  {
194  const Cell & src = B->get_ravel(0);
195  loop(a, max_idx)
196  {
197  Cell & dest = A->get_ravel(a);
198  dest.release(LOC); // free sub-values etc (if any)
199  dest.init(src, A.getref(), LOC);
200  }
201  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
202  return;
203  }
204 
205 const ShapeItem ec_B = B->element_count();
206 const ShapeItem ec_X = X->element_count();
207 const int incr_B = (ec_B == 1) ? 0 : 1; // maybe scalar extend B
208 const Cell * cX = &X->get_ravel(0);
209 const Cell * cB = &B->get_ravel(0);
210 
211  if (ec_B != 1 && ec_B != ec_X) LENGTH_ERROR;
212 
213  loop(x, ec_X)
214  {
215  const ShapeItem idx = cX++->get_near_int() - qio;
216  if (idx < 0) INDEX_ERROR;
217  if (idx >= max_idx) INDEX_ERROR;
218  Cell & dest = A->get_ravel(idx);
219  dest.release(LOC); // free sub-values etc (if any)
220  dest.init(*cB, A.getref(), LOC);
221 
222  cB += incr_B;
223  }
224 
225  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
226 }
227 //-----------------------------------------------------------------------------
228 void
229 Symbol::assign_indexed(IndexExpr & IX, Value_P B) // A[IX;...] ← B
230 {
231  if (IX.value_count() == 1 && !!IX.values[0]) // one-dimensional index
232  {
233  assign_indexed(IX.values[0], B);
234  return;
235  }
236 
237  // see Value::index() for comments.
238 
239 Value_P A = get_apl_value();
240  if (A->get_rank() != IX.value_count()) RANK_ERROR; // ISO p. 159
241 
242  // B must either be a scalar (and is then scalar extended to the size
243  // of the updated area, or else have the shape of the concatenated index
244  // items for example:
245  //
246  // X: X1 ; X2 ; X3
247  // ⍴B: b1 b2 b3 b4 b5 b6
248  //
249  if (1 && !B->is_scalar())
250  {
251  // remove dimensions with len 1 from the shapes of X and B...
252  // if we see an empty Xn then we return.
253  //
254  Shape B1;
255  loop(b, B->get_rank())
256  {
257  const ShapeItem sb = B->get_shape_item(b);
258  if (sb != 1) B1.add_shape_item(sb);
259  }
260 
261  Shape IX1;
262  for (ShapeItem ix = IX.value_count() - 1; ix >= 0; --ix)
263  {
264  const Value * ix_val = IX.values[ix].get();
265  if (ix_val) // normal index
266  {
267  if (ix_val->element_count() == 0) return; // empty index
268  loop(xx, ix_val->get_rank())
269  {
270  const ShapeItem sxx = ix_val->get_shape_item(xx);
271  if (sxx != 1) IX1.add_shape_item(sxx);
272  }
273  }
274  else // elided index: add corresponding B dimenssion
275  {
276  if (ix >= A->get_rank()) RANK_ERROR;
277  const ShapeItem sbx =
278  A->get_shape_item(A->get_rank() - ix - 1);
279  if (sbx == 0) return; // empty index
280  if (sbx != 1) IX1.add_shape_item(sbx);
281  }
282  }
283 
284  if (B1 != IX1)
285  {
286  if (B1.get_rank() != IX1.get_rank()) RANK_ERROR;
287  LENGTH_ERROR;
288  }
289  }
290 
291 MultiIndexIterator mult(A->get_shape(), IX);
292 
293 const ShapeItem ec_B = B->element_count();
294 const Cell * cB = &B->get_ravel(0);
295 const int incr_B = (ec_B == 1) ? 0 : 1;
296 
297  while (mult.more())
298  {
299  const ShapeItem offset_A = mult++;
300  if (offset_A < 0) INDEX_ERROR;
301  if (offset_A >= A->element_count()) INDEX_ERROR;
302  Cell & dest = A->get_ravel(offset_A);
303  dest.release(LOC); // free sub-values etc (if any)
304  dest.init(*cB, A.getref(), LOC);
305  cB += incr_B;
306  }
307 
308  if (monitor_callback) monitor_callback(*this, SEV_ASSIGNED);
309 }

314 ValueStackItem & vs = value_stack.back();
315 UserFunction * ufun = lambda->get_ufun1();

318  Assert(uexec);

320  switch(vs.name_class)

322  case NC_FUNCTION:
323  case NC_OPERATOR:
324  {
325  Function * old_fun = vs.sym_val.function;
326  Assert(old_fun);
327  if (!old_fun->is_lambda()) SYNTAX_ERROR;

329  Assert(old_ufun);
330  for (StateIndicator * si = Workspace::SI_top();
331  si; si = si->get_parent())
332  {
333 // if (uexec == si->get_executable())
334  if (si->uses_function(old_ufun))
335  {

339  }

343  }

345  /* fall through */
346 
347  case NC_UNUSED_USER_NAME:
348  if (lambda->is_operator()) vs.name_class = NC_OPERATOR;
349  else vs.name_class = NC_FUNCTION;

351  vs.sym_val.function = ufun;

576 //-----------------------------------------------------------------------------
577 void
578 Symbol::get_attributes(int mode, Cell * dest) const
579 {
580 const ValueStackItem & vs = value_stack.back();
581 bool has_result = false;
582 int fun_valence = 0;
583 int oper_valence = 0;
584 double created = 0.0;
585 static int exec_prop[] = { 0, 0, 0, 0 };
586 const int * exec_properties = exec_prop;
587 
588  switch(vs.name_class)
589  {
590  case NC_LABEL:
591  case NC_VARIABLE:
592  has_result = true;
593  break;
594 
595  case NC_FUNCTION:
596  has_result = vs.sym_val.function->has_result();
597  fun_valence = vs.sym_val.function->get_fun_valence();
598  created = vs.sym_val.function->get_creation_time();
599  exec_properties = vs.sym_val.function->get_exec_properties();
600  break;
601 
602  case NC_OPERATOR:
603  fun_valence = vs.sym_val.function->get_fun_valence();
604  oper_valence = vs.sym_val.function->get_oper_valence();
605  created = vs.sym_val.function->get_creation_time();
606  exec_properties = vs.sym_val.function->get_exec_properties();
607  break;
608 
609  default: break;
610  }
611 
612  switch(mode)
613  {
614  case 1: // valences
615  new (dest + 0) IntCell(has_result ? 1 : 0);
616  new (dest + 1) IntCell(fun_valence);
617  new (dest + 2) IntCell(oper_valence);
618  break;
619 
620  case 2: // creation time
621  if (created == 0.0) // system function
622  {
623  new (dest + 0) IntCell(0);
624  new (dest + 1) IntCell(0);
625  new (dest + 2) IntCell(0);
626  new (dest + 3) IntCell(0);
627  new (dest + 4) IntCell(0);
628  new (dest + 5) IntCell(0);
629  new (dest + 6) IntCell(0);
630  }
631  else // user define function
632  {
633  tm tc;
634  const time_t time = created;
635  localtime_r(&time, &tc);
636 
637  new (dest + 0) IntCell(1900 + tc.tm_year);
638  new (dest + 1) IntCell( 1 + tc.tm_mon );
639  new (dest + 2) IntCell( tc.tm_mday);
640  new (dest + 3) IntCell( tc.tm_hour);
641  new (dest + 4) IntCell( tc.tm_min);
642  new (dest + 5) IntCell( tc.tm_sec);
643  new (dest + 6) IntCell((created - time) / 1000);
644  }
645  break;
646 
647  case 3: // execution properties
648  new (dest + 0) IntCell(exec_properties[0]);
649  new (dest + 1) IntCell(exec_properties[1]);
650  new (dest + 2) IntCell(exec_properties[2]);
651  new (dest + 3) IntCell(exec_properties[3]);
652  break;
653 
654  case 4: {
655  Value_P val = get_apl_value();
656  const CDR_type cdr_type = val->get_CDR_type();
657  const int brutto = val->total_size_brutto(cdr_type);
658  const int data = val->data_size(cdr_type);
659 
660  new (dest + 0) IntCell(brutto);
661  new (dest + 1) IntCell(data);
662  }
663  break;
664 
665  default: Assert(0 && "bad mode");
666  }
667 }
668 //-----------------------------------------------------------------------------
669 void
670 Symbol::resolve(Token & tok, bool left_sym)
671 {
672  Log(LOG_SYMBOL_resolve)
673  CERR << "resolve(" << left_sym << ") symbol " << get_name() << endl;
674 
675  Assert1(value_stack.size());
676 
677 const ValueStackItem & vs = value_stack.back();
678  switch(vs.name_class)
679  {
680  case NC_UNUSED_USER_NAME:
681  if (!left_sym) Error::throw_symbol_error(get_name(), LOC);
682  return; // leave symbol as is
683 
684  case NC_LABEL:
685  if (left_sym) SYNTAX_ERROR; // assignment to (read-only) label
686 
687  {
688  IntCell lab(vs.sym_val.label);
689  Value_P value(lab, LOC);
690  Token t(TOK_APL_VALUE1, value);
691  tok.move_1(t, LOC);
692  }
693  return;
694 
695  case NC_VARIABLE:
696  if (left_sym) return; // leave symbol as is
697 
698  // if we resolve a variable. the value is considered grouped.
699  {
700  Token t(TOK_APL_VALUE1, get_apl_value()->clone(LOC));
701  tok.move_1(t, LOC);
702  }
703  return;
704 
705  case NC_FUNCTION:
706  case NC_OPERATOR:
707  if (left_sym && vs.sym_val.function->is_lambda())
708  {
709  // lambda re-assign, e.g. SYM←{ ... }
710  //
711  return;
712  }
713  tok.move_2(vs.sym_val.function->get_token(), LOC);
714  return;
715 
716  case NC_SHARED_VAR:
717  if (left_sym) return; // leave symbol as is
718  resolve_shared_variable(tok);
719  return;
720

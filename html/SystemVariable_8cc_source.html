83 //-----------------------------------------------------------------------------
84 void
85 SystemVariable::get_attributes(int mode, Cell * dest) const
86 {
87  switch(mode)
88  {
89  case 1: // valences (always 1 0 0 for variables)
90  new (dest + 0) IntCell(1);
91  new (dest + 1) IntCell(0);
92  new (dest + 2) IntCell(0);
93  break;
94 
95  case 2: // creation time (always 7⍴0 for variables)
96  new (dest + 0) IntCell(0);
97  new (dest + 1) IntCell(0);
98  new (dest + 2) IntCell(0);
99  new (dest + 3) IntCell(0);
100  new (dest + 4) IntCell(0);
101  new (dest + 5) IntCell(0);
102  new (dest + 6) IntCell(0);
103  break;
104 
105  case 3: // execution properties (always 4⍴0 for variables)
106  new (dest + 0) IntCell(0);
107  new (dest + 1) IntCell(0);
108  new (dest + 2) IntCell(0);
109  new (dest + 3) IntCell(0);
110  break;
111 
112  case 4: {
113  Value_P val = get_apl_value();
114  const CDR_type cdr_type = val->get_CDR_type();
115  const int brutto = val->total_size_brutto(cdr_type);
116  const int data = val->data_size(cdr_type);
117 
118  new (dest + 0) IntCell(brutto);
119  new (dest + 1) IntCell(data);
120  }
121  break;
122 
123  default: Assert(0 && "bad mode");
124  }
125 
126 }

209 Quad_CT::assign(Value_P value, bool clone, const char * loc)
210 {
211  if (!value->is_scalar_or_len1_vector())
212  {
213  if (value->get_rank() > 1) RANK_ERROR;
214  else LENGTH_ERROR;
215  }
216 
217 const Cell & cell = value->get_ravel(0);
218  if (!cell.is_numeric()) DOMAIN_ERROR;
219  if (cell.get_imag_value() != 0.0) DOMAIN_ERROR;
220 
221 APL_Float val = cell.get_real_value();
222  if (val < 0.0) DOMAIN_ERROR;
223 
224  // APL2 "discourages" the use of ⎕CT > 1E¯9.
225  // We round down to MAX_Quad_CT (= 1E¯9) instead.
226  //
227  if (val > MAX_Quad_CT)
228  {
229  Symbol::assign(FloatScalar(MAX_Quad_CT, LOC), false, LOC);
230  }
231  else
232  {
233  Symbol::assign(value, clone, LOC);
234  }
235 }

237 Value_P
238 Quad_EM::get_apl_value() const
239 {
240 const Error * err = 0;
241 
242  for (StateIndicator * si = Workspace::SI_top(); si; si = si->get_parent())
243  {
244  if (StateIndicator::get_error(si).get_error_code() != E_NO_ERROR)
245  {
246  err = &StateIndicator::get_error(si);
247  break;
248  }
249  if (si->get_parse_mode() == PM_FUNCTION) break;
250  }
251 
252  if (err == 0) // no SI entry with non-zero error code
253  {
254  // return 3 0⍴' '
255  //
256  Shape sh(ShapeItem(3), ShapeItem(0));
257  Value_P Z(sh, LOC);
258  Z->set_proto_Spc();
259  Z->check_value(LOC);
260  return Z;
261  }
262 
263 PrintBuffer pb;
264  pb.append_ucs(UCS_string(UTF8_string(err->get_error_line_1())));
265  pb.append_ucs(UCS_string(UTF8_string(err->get_error_line_2())));
266  pb.append_ucs(err->get_error_line_3());
267  pb.pad_to_spaces(); // replace pad char by spaces
268 Value_P Z(pb, LOC);
269  Z->check_value(LOC);
270  return Z;
271 }

330 //-----------------------------------------------------------------------------
331 void
332 Quad_FC::assign(Value_P value, bool clone, const char * loc)
333 {
334  if (!value->is_scalar_or_vector()) RANK_ERROR;
335 
336 ShapeItem value_len = value->element_count();
337  if (value_len > 6) value_len = 6;
338 
339  loop(c, value_len)
340  if (!value->get_ravel(c).is_character_cell()) DOMAIN_ERROR;
341 
342  // new value is correct.
343  //
344 Unicode fc[6] = { UNI_ASCII_FULLSTOP, UNI_ASCII_COMMA, UNI_STAR_OPERATOR,
345  UNI_ASCII_0, UNI_ASCII_UNDERSCORE, UNI_OVERBAR };
346 
347  loop(c, 6) if (c < value_len)
348  fc[c] = value->get_ravel(c).get_char_value();
349 
350  // 0123456789,. are forbidden for ⎕FC[4 + ⎕IO]
351  //
352  if (Bif_F12_FORMAT::is_control_char(fc[4]))
353  fc[4] = UNI_ASCII_SPACE;
354 
355 UCS_string ucs(fc, 6);
356 Value_P new_val(ucs, LOC);
357  Symbol::assign(new_val, false, LOC);
358 }
359 //-----------------------------------------------------------------------------
360 void
361 Quad_FC::assign_indexed(IndexExpr & IX, Value_P value)
362 {
363  if (IX.value_count() != 1) INDEX_ERROR;
364 
365  // at this point we have a one dimensional index. It it were non-empty,
366  // then assign_indexed(Value) would have been called instead. Therefore
367  // IX must be an elided index (like in ⎕FC[] ← value)
368  //
369  Assert1(!IX.values[0]);
370  assign(value, true, LOC); // ⎕FC[]←value
371 }
372 //-----------------------------------------------------------------------------
373 void
374 Quad_FC::assign_indexed(Value_P X, Value_P value)
375 {
376  // we don't do scalar extension but require indices to match the value.
377  //
378 ShapeItem ec = X->element_count();
379  if (ec != value->element_count()) INDEX_ERROR;
380 
381  // ignore extra values.
382  //
383  if (ec > 6) ec = 6;
384 
385 const APL_Integer qio = Workspace::get_IO();
386 Unicode fc[6];
387  {
388  Value_P old = get_apl_value();
389  loop(e, 6) fc[e] = old->get_ravel(e).get_char_value();
390  }
391 
392  loop(e, ec)
393  {
394  const APL_Integer idx = X->get_ravel(e).get_near_int() - qio;
395  if (idx < 0) continue;
396  if (idx > 5) continue;
397 
398  fc[idx] = value->get_ravel(e).get_char_value();
399  }
400 
401  // 0123456789,. are forbidden for ⎕FC[4 + ⎕IO]
402  //
403  if (Bif_F12_FORMAT::is_control_char(fc[4]))
404  fc[4] = UNI_ASCII_SPACE;
405 
406 UCS_string ucs(fc, 6);
407 Value_P new_val(ucs, LOC);
408  Symbol::assign(new_val, false, LOC);
409 }

559 //-----------------------------------------------------------------------------
560 void
561 Quad_PS::assign(Value_P B, bool clone, const char * loc)
562 {
563 APL_Integer B_quot = 0;
564 APL_Integer B_style = 0;
565 
566  if (B->get_rank() > 1) RANK_ERROR;
567 
568  if (B->element_count() < 1) LENGTH_ERROR;
569  if (B->element_count() > 2) LENGTH_ERROR;
570 
571  if (!B->get_ravel(0).is_near_bool())
572  {
573  MORE_ERROR() << "Bad quot in ⎕PS←quot style: quot is not near bool";
574  DOMAIN_ERROR;
575  }
576 
577  if (B->element_count() == 1)
578  {
579  // for compatibility with old workspaces
580  //
581  B_style = B->get_ravel(0).get_near_int();
582  }
583  else
584  {
585  B_quot = B->get_ravel(0).get_near_bool();
586  B_style = B->get_ravel(1).get_near_int();
587  }
588 
589  switch(B_style) // boxing format
590  {
591  case -29:
592  case -25: case -24: case -23:
593  case -22: case -21: case -20:
594  case -9: case -8: case -7:
595  case -4: case -3: case -2:
596  case 0:
597  case 2: case 3: case 4:
598  case 7: case 8: case 9:
599  case 20: case 21: case 22:
600  case 23: case 24: case 25:
601  case 29: break; // OK
602 
603  default: MORE_ERROR() <<
604  "Invalid style in ⎕PS←quot style or ⎕PS←style:\n"
605  "style is not ± 0, 2-4, 7-9, 20-25, or 29";
606  DOMAIN_ERROR;
607  }
608 
609  // values in B are valid
610  //
611  Command::boxing_format = B_style;
612  print_quotients = B_quot;
613  style = B_style;
614 
615 Value_P B2(2, LOC);
616  new (B2->next_ravel()) IntCell(B_quot);
617  new (B2->next_ravel()) IntCell(B_style);
618  B2->check_value(LOC);
619 
620  Symbol::assign(B2, false, LOC);
621  return;
622 }
623 //-----------------------------------------------------------------------------
624 void
625 Quad_PS::assign_indexed(Value_P X, Value_P B)
626 {
627  if (!(X->is_int_scalar() || X->is_int_vector())) INDEX_ERROR;
628  if (!(B->is_int_scalar() || B->is_int_vector())) DOMAIN_ERROR;
629  if (X->element_count() != B->element_count()) LENGTH_ERROR;
630 
631 const ShapeItem ec = X->element_count();
632 const APL_Integer qio = Workspace::get_IO();
633 
634 APL_Integer Z_quot = print_quotients;
635 APL_Integer Z_style = style;
636  loop(e, ec)
637  {
638  const APL_Integer x = X->get_ravel(e).get_near_int() - qio;
639  const APL_Integer b = B->get_ravel(e).get_near_int();
640 
641  if (x == 0) // display quotients
642  {
643  if (b < 0) DOMAIN_ERROR;
644  if (b > 1) DOMAIN_ERROR;
645  Z_quot = b;
646  }
647  else if (x == 1) switch(b) // boxing format
648  {
649  case -29:
650  case -25: case -24: case -23:
651  case -22: case -21: case -20:
652  case -9: case -8: case -7:
653  case -4: case -3: case -2:
654  case 0:
655  case 2: case 3: case 4:
656  case 7: case 8: case 9:
657  case 20: case 21: case 22:
658  case 23: case 24: case 25:
659  case 29: Z_style = b;
660  Command::boxing_format = b;
661  break;
662 
663  default:
664  MORE_ERROR() <<
665  "Invalid print style (try 0, 2-4, 7-9, 20-25, or 29)";
666  DOMAIN_ERROR;
667  }
668  else
669  {
670  INDEX_ERROR;
671  }
672  }
673 
674 Value_P Z(2, LOC);
675  new (Z->next_ravel()) IntCell(Z_quot);
676  new (Z->next_ravel()) IntCell(Z_style);
677  Z->check_value(LOC);
678  Symbol::assign(Z, false, LOC);
679 }

898 //-----------------------------------------------------------------------------
899 void
900 Quad_SYL::assign_indexed(Value_P X, Value_P B)
901 {
902  // try to assign ⎕SYL[X;2]
903  //
904  if (!(X->is_int_scalar() || X->is_int_vector())) INDEX_ERROR;
905  if (!(B->is_int_scalar() || B->is_int_vector())) DOMAIN_ERROR;
906  if (X->element_count() != B->element_count()) LENGTH_ERROR;
907 
908 const ShapeItem ec = X->element_count();
909 const APL_Integer qio = Workspace::get_IO();
910 
911  loop(e, ec)
912  {
913  const APL_Integer x = X->get_ravel(e).get_near_int() - qio;
914  const APL_Integer b = B->get_ravel(e).get_near_int();
915 
916  if (x == SYL_SI_DEPTH_LIMIT) // SI depth limit
917  {
918  const int depth = Workspace::SI_entry_count();
919  if (b && (b < (depth + 4))) DOMAIN_ERROR; // limit too low
920  si_depth_limit = b;
921  }
922  else if (x == SYL_VALUE_COUNT_LIMIT) // value count limit
923  {
924  if (b && (b < APL_Integer(Value::value_count + 10))) // too low
925  DOMAIN_ERROR;
926  value_count_limit = b;
927  }
928  else if (x == SYL_RAVEL_BYTES_LIMIT) // ravel bytes limit
929  {
930  const int cells = b / sizeof(Cell);
931  if (cells && (cells < APL_Integer(Value::total_ravel_count + 1000)))
932  DOMAIN_ERROR;
933 
934  ravel_count_limit = cells;
935  }
936  else if (x == SYL_CURRENT_CORES) // number of cores
937  {
938 #if PARALLEL_ENABLED
939  if (CPU_pool::change_core_count(CoreCount(b), false))
940  {
941  MORE_ERROR() << "Bad core count " << b
942  << " in ⎕SYL[" << x << "] ";
943  DOMAIN_ERROR;
944  }
945 #else
946  MORE_ERROR() << "PARALLEL_ENABLED not set in ⎕SYL[" << x << "] ";
947  DOMAIN_ERROR;
948 #endif
949  }
950  else if (x == SYL_PRINT_LIMIT) // print length limit
951  {
952  if (b < 1000) DOMAIN_ERROR;
953  print_length_limit = b;
954  }
955  else if (x == SYL_WA_MARGIN) // ⎕WA safety margin
956  {
957  if (b < 1000000) DOMAIN_ERROR;
958  Quad_WA::WA_margin = b;
959  }
960  else if (x == SYL_WA_SCALE) // ⎕WA memory scale (45% - 200% %
961  {
962  if (b < 45) DOMAIN_ERROR;
963  if (b > 200) DOMAIN_ERROR;
964  Quad_WA::WA_scale = b;
965  }
966  else
967  {
968  MORE_ERROR() << "Bad ⎕SYL index " << x;
969  INDEX_ERROR;
970  }
971  }
972 }

1041 //-----------------------------------------------------------------------------
1042 int
1043 Quad_TZ::compute_offset()
1044 {
1045  // the GNU/Linux timezone variable conflicts with function timezone() on
1046  // other systems. timezone.tz_minuteswest in gettimeofday() does not
1047  // sound reliable either.
1048  //
1049  // We compute localtime() and gmtime() and take the difference (hoping
1050  // that localtime() and gmtime() are more portable.
1051  //
1052 
1053  // choose a reference point in time.
1054  //
1055 const time_t ref = now()/1000000;
1056 
1057 tm * local = localtime(&ref);
1058 const int local_minutes = local->tm_min + 60*local->tm_hour;
1059 const int local_days = local->tm_mday + 31*local->tm_mon + 12*31*local->tm_year;
1060 
1061 tm * gmean = gmtime(&ref);
1062 const int gm_minutes = gmean->tm_min + 60*gmean->tm_hour;
1063 const int gm_days = gmean->tm_mday + 31*gmean->tm_mon + 12*31*gmean->tm_year;
1064 int diff_minutes = local_minutes - gm_minutes;
1065 
1066  if (local_days < gm_days) diff_minutes -= 24*60; // local < gmean
1067  else if (local_days > gm_days) diff_minutes += 24*60; // local > gmean
1068 
1069  return 60*diff_minutes;
1070 }

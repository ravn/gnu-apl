74 //-----------------------------------------------------------------------------
75 void
76 Workspace::push_SI(Executable * fun, const char * loc)
77 {
78  Assert1(fun);
79 
80  if (Quad_SYL::si_depth_limit && SI_top() &&
81  Quad_SYL::si_depth_limit <= SI_top()->get_level())
82  {
83  MORE_ERROR() <<
84  "the system limit on SI depth (as set in âSYL) was reached\n"
85  "(and to avoid lock-up, the limit in âSYL was automatically cleared).";
86 
87  Quad_SYL::si_depth_limit = 0;
88  set_attention_raised(LOC);
89  set_interrupt_raised(LOC);
90  }
91 
92  if (Value::check_WS_FULL(__FUNCTION__, 1000, LOC))
93  {
94  if (SI_top() && SI_top()->get_executable() &&
95  the_workspace.SI_entry_count() > 100)
96  MORE_ERROR() <<
97  "Value::check_WS_FULL() complained when calling a defined function.\n"
98  "This was most likely caused by some infinite recursion involving "
99  << SI_top()->get_executable()->get_name();
100  else
101  MORE_ERROR() <<
102  "Value::check_WS_FULL() complained when calling a defined function";
103  WS_FULL;
104  }
105 
106  try
107  {
108  StateIndicator * old_top = SI_top();
109  the_workspace.top_SI = new StateIndicator(fun, old_top);
110  if (the_workspace.top_SI == 0)
111  {
112  MORE_ERROR() <<
113  "malloc() â 0 when calling a defined function";
114  the_workspace.top_SI = old_top;
115  WS_FULL;
116  }
117  }
118  catch (const std::bad_alloc & e)
119  {
120  if (SI_top() && SI_top()->get_executable() &&
121  the_workspace.SI_entry_count() > 100)
122  MORE_ERROR() <<
123  "std::bad_alloc exception when calling a defined function\n"
124  "This was most likely caused by some infinite recursion involving "
125  << SI_top()->get_executable()->get_name();
126  else
127  MORE_ERROR() <<
128  "std::bad_alloc exception when calling a defined function";
129  WS_FULL;
130  }
131 
132  Log(LOG_StateIndicator__push_pop)
133  {
134  CERR << "Push SI[" << SI_top()->get_level() << "] "
135  << "pmode=" << fun->get_parse_mode()
136  << " exec=" << fun << " "
137  << fun->get_name();
138 
139  CERR << " new SI is " << voidP(SI_top())
140  << " at " << loc << endl;
141  }
142 }

166  the_workspace.top_SI = del->get_parent();
167  delete del;

169 //-----------------------------------------------------------------------------
170 uint64_t
171 Workspace::get_RL(uint64_t mod)
172 {
173  // we discard random numbers >= max_rand in order to avoid a bias
174  // towards small numbers
175  //
176 const uint64_t max_rand = 0xFFFFFFFFFFFFFFFFULL - (0xFFFFFFFFFFFFFFFFULL % mod);
177 uint64_t rand = the_workspace.v_Quad_RL.get_random();
178 
179  do { rand ^= rand >> 11;
180  rand ^= rand << 29;
181  rand ^= rand >> 14;
182  } while (rand >= max_rand);
183 
184  return rand % mod;
185 }

259  catch (...)

261  CERR << "*** " << __FUNCTION__
262  << "() caught other exception ***" << endl;
263  IO_Files::apl_error(LOC);
264  if (exit_on_error) return Token(TOK_OFF);
265  }
266  }
267 
268  return Token(TOK_ESCAPE);
269 }
270 //-----------------------------------------------------------------------------
271 NamedObject *
272 Workspace::lookup_existing_name(const UCS_string & name)
273 {
274  if (name.size() == 0) return 0;
275 
276  if (Avec::is_quad(name[0])) // distinguished name
277  {
278  int len;
279  Token tok = get_quad(name, len);
280  if (len == 1) return 0;
281  if (name.size() != len) return 0;
282  if (tok.get_Class() == TC_SYMBOL) return tok.get_sym_ptr();
283  if (tok.get_Class() == TC_FUN0) return tok.get_function();
284  if (tok.get_Class() == TC_FUN1) return tok.get_function();
285  if (tok.get_Class() == TC_FUN2) return tok.get_function();
286 
287  Assert(0);
288  }
289 
290  // user defined variable or function
291  //
292 Symbol * sym = the_workspace.symbol_table.lookup_existing_symbol(name);
293  if (sym == 0) return 0;
294 
295  switch(sym->get_nc())
296  {
297  case NC_VARIABLE: return sym;
298 
299  case NC_FUNCTION:
300  case NC_OPERATOR: return sym->get_function();
301  default: return 0;
302  }
303 }
304 //-----------------------------------------------------------------------------
305 Symbol *
306 Workspace::lookup_existing_symbol(const UCS_string & symbol_name)
307 {
308  if (symbol_name.size() == 0) return 0;
309 
310  if (Avec::is_quad(symbol_name[0])) // distinguished name
311  {
312  int len;
313  Token tok = get_quad(symbol_name, len);
314  if (symbol_name.size() != len) return 0;
315  if (tok.get_Class() != TC_SYMBOL) return 0; // system function
316 
317  return tok.get_sym_ptr();
318  }
319 
320  return the_workspace.symbol_table.lookup_existing_symbol(symbol_name);
321 }
322 //-----------------------------------------------------------------------------
323 Token
324 Workspace::get_quad(const UCS_string & ucs, int & len)
325 {
326  if (ucs.size() == 0 || !Avec::is_quad(ucs[0]))
327  {
328  len = 0;
329  return Token();
330  }
331 
332 UCS_string name(UNI_Quad_Quad);
333  len = 1;
334 
335 SystemName * longest = 0;
336 
337  for (ShapeItem u = 1; u < ucs.size(); ++u)
338  {
339  Unicode uni = ucs[u];
340  if (!Avec::is_symbol_char(uni)) break;
341 
342  if (uni >= 'a' && uni <= 'z') uni = Unicode(uni - 0x20);
343  name.append(uni);
344  SystemName * dn =
345  the_workspace.distinguished_names.lookup_existing_symbol(name);
346 
347  if (dn) // ss is a valid distinguished name
348  {
349  longest = dn;
350  len = name.size();
351  }
352  }
353 
354  if (longest == 0) return Token(TOK_Quad_Quad, &the_workspace.v_Quad_Quad);
355 
356  if (longest->get_variable()) return longest->get_variable()->get_token();
357  else return longest->get_function()->get_token();
358 }

370 //-----------------------------------------------------------------------------
371 bool
372 Workspace::is_called(const UCS_string & funname)
373 {
374  // return true if the current-referent of funname is pendant
375  //
376 Symbol * current_referent = lookup_existing_symbol(funname);
377  if (current_referent == 0) return false; // no such symbol
378 
379  Assert(current_referent->get_Id() == ID_USER_SYMBOL);
380 
381 const NameClass nc = current_referent->get_nc();
382  if (nc != NC_FUNCTION && nc != NC_OPERATOR) return false;
383 
384 const Function * fun = current_referent->get_function();
385  Assert(fun);
386 
387 const UserFunction * ufun = fun->get_ufun1();
388  if (!ufun) return false; // not a defined function
389 
390  for (const StateIndicator * si = SI_top(); si; si = si->get_parent())
391  {
392  if (si->uses_function(ufun)) return true;
393  }
394 
395  return false;
396 }

399 Workspace::write_OUT(FILE * out, uint64_t & seq, const UCS_string_vector
400  & objects)
401 {
402  // if objects is empty then write all user define objects and some system
403  // variables
404  //
405  if (objects.size() == 0) // all objects plus some system variables
406  {
407  get_v_Quad_CT().write_OUT(out, seq);
408  get_v_Quad_FC().write_OUT(out, seq);
409  get_v_Quad_IO().write_OUT(out, seq);
410  get_v_Quad_LX().write_OUT(out, seq);

413  get_v_Quad_RL().write_OUT(out, seq);
414 
415  get_symbol_table().write_all_symbols(out, seq);
416  }
417  else // only specified objects
418  {
419  loop(o, objects.size())
420  {
421  NamedObject * obj = lookup_existing_name(objects[o]);
422  if (obj == 0) // not found
423  {
424  COUT << ")OUT: " << objects[o] << " NOT SAVED (not found)"
425  << endl;
426  continue;
427  }
428 
429  if (obj->get_Id() == ID_USER_SYMBOL) // user defined name
430  {
431  const Symbol * sym = lookup_existing_symbol(objects[o]);
432  Assert(sym);
433  sym->write_OUT(out, seq);
434  }
435  else // distinguished name
436  {
437  const Symbol * sym = obj->get_symbol();
438  if (sym == 0)
439  {
440  COUT << ")OUT: " << objects[o]
441  << " NOT SAVED (not a variable)" << endl;
442  continue;
443  }
444 
445  sym->write_OUT(out, seq);
446  }
447  }
448  }

666 //-----------------------------------------------------------------------------
667 bool
668 Workspace::backup_existing_file(const char * filename)
669 {
670  // 1. if file 'filename' does not exist then no backup is needed.
671  //
672  if (access(filename, F_OK) != 0) return false; // OK
673 
674 UTF8_string backup_filename = filename;
675  backup_filename.append_ASCII(".bak");
676 
677  // 2. if backup file exists then remove it...
678  //
679  if (access(backup_filename.c_str(), F_OK) == 0) // backup exists
680  {
681  const int err = unlink(backup_filename.c_str());
682  if (err)
683  {
684  CERR << "Could not remove backup file " << backup_filename
685  << ": " << strerror(errno) << endl;
686  return true; // error
687  }
688

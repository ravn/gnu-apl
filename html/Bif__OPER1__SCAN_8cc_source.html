36 Token
37 Bif_SCAN::expand(Value_P A, Value_P B, uAxis axis)
38 {
39  // turn scalar B into ,B
40  //
41 Shape shape_B = B->get_shape();
42  if (shape_B.get_rank() == 0)
43  {
44  shape_B.add_shape_item(1);
45  axis = 0;
46  }
47  if (axis >= shape_B.get_rank()) INDEX_ERROR;
48 
49  if (A->get_rank() > 1) RANK_ERROR;
50  if (shape_B.get_rank() <= axis) RANK_ERROR;
51 
52 const ShapeItem ec_A = A->element_count();
53 ShapeItem ones_A = 0;
54 std::vector<ShapeItem> rep_counts;
55  rep_counts.reserve(ec_A);
56  loop(a, ec_A)
57  {
58  APL_Integer rep_A = A->get_ravel(a).get_near_int();
59  rep_counts.push_back(rep_A);
60  if (rep_A == 0) ;
61  else if (rep_A == 1) ++ones_A;
62  else DOMAIN_ERROR;
63  }
64 
65 Shape shape_Z(shape_B);
66  shape_Z.set_shape_item(axis, ec_A);
67 Value_P Z(shape_Z, LOC);
68 
69  if (ec_A == 0) // (â³0)/B :
70  {
71  if (shape_B.get_shape_item(axis) > 1) LENGTH_ERROR;
72 
73  Z->set_default(*B.get(), LOC);
74  Z->check_value(LOC);
75  return Token(TOK_APL_VALUE1, Z);
76  }
77 
78 const Shape3 shape_Z3(shape_Z, axis);
79 
80 const Cell * cB = &B->get_ravel(0);
81 const bool lval = cB->is_lval_cell();
82 
83 ShapeItem inc_1 = shape_Z3.l(); // increment after result l items
84 ShapeItem inc_2 = 0; // increment after result m*l items
85 
86  if (B->is_scalar() || (shape_B.get_shape_item(axis) == 1
87  && (shape_Z3.l() != 1)))
88  {
89  inc_1 = 0;
90  inc_2 = shape_Z3.l();
91  }
92  else if (ones_A != shape_B.get_shape_item(axis)) LENGTH_ERROR;
93 
94  loop(h, shape_Z3.h())
95  {
96  const Cell * fill = cB;
97  loop(m, rep_counts.size())
98  {
99  if (rep_counts[m] == 1) // copy items from B
100  {
101  loop(l, shape_Z3.l())
102  Z->next_ravel()->init(cB[l], Z.getref(), LOC);
103  cB += inc_1;
104  }
105  else // init items
106  {
107  if (lval)
108  {
109  loop(l, shape_Z3.l())
110  new (Z->next_ravel()) LvalCell(0, 0);
111  }
112  else
113  {
114  loop(l, shape_Z3.l())
115  Z->next_ravel()->init_type(fill[l], Z.getref(), LOC);
116  }
117  }
118  }
119 
120  cB += inc_2;
121  }
122 
123  Z->set_default(*B.get(), LOC);
124 
125  Z->check_value(LOC);
126  return Token(TOK_APL_VALUE1, Z);
127 }

129 Token
130 Bif_SCAN::scan(Token & _LO, Value_P B, uAxis axis)
131 {
132 Function * LO = _LO.get_function();
133  Assert(LO);
134 
135  if (!LO->has_result()) DOMAIN_ERROR;
136 
137  if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, B->clone(LOC));
138 
139  if (axis >= B->get_rank()) INDEX_ERROR;
140 
141 const ShapeItem m_len = B->get_shape_item(axis);
142 
143  if (m_len == 0) return Token(TOK_APL_VALUE1, B->clone(LOC));
144 
145  if (m_len == 1)
146  {
147  const Shape shape_Z = B->get_shape().without_axis(axis);
148  return Bif_F12_RHO::do_reshape(shape_Z, *B);
149  }
150 
151 const Shape3 shape_Z3(B->get_shape(), axis);
152 
153 Value_P Z(B->get_shape(), LOC);
154 ErrorCode (Cell::*assoc_f2)(Cell *, const Cell *) const = LO->get_assoc();
155 
156  if (assoc_f2)
157  {
158  // LO is an associative primitive scalar function.
159  //
160  const Cell * cB = &B->get_ravel(0);
161  loop(h, shape_Z3.h())
162  loop(m, shape_Z3.m())
163  loop(l, shape_Z3.l())
164  {
165  Cell * cZ = Z->next_ravel();
166  if (m == 0)
167  {
168  cZ->init(*cB++, Z.getref(), LOC);
169  }
170  else
171  {
172 
173  Value_P AA(LOC);
174  AA->next_ravel()->init(cZ[-shape_Z3.l()], AA.getref(), LOC);
175  AA->set_complete();
176 
177  Value_P BB(LOC);
178  BB->next_ravel()->init(*cB++, BB.getref(), LOC);
179  BB->set_complete();
180 
181  Token tok = LO->eval_AB(AA, BB);
182  if (!tok.is_apl_val()) return tok;
183 
184  Value_P ZZ = tok.get_apl_val();
185  cZ->init(ZZ->get_ravel(0), ZZ.getref(), LOC);
186  ZZ->set_complete();
187  }
188  }
189  Z->check_value(LOC);
190  return Token(TOK_APL_VALUE1, Z);
191  }
192 
193  // non-trivial reduce (len > 1)
194  //
195 const Shape3 Z3(B->get_shape(), axis);
196  if (LO->may_push_SI()) // user defined LO
197  {
198  Value_P X4(4, LOC);
199  new (X4->next_ravel()) IntCell(axis + Workspace::get_IO());
200  new (X4->next_ravel()) IntCell(Z3.h());
201  new (X4->next_ravel()) IntCell(Z3.m());
202  new (X4->next_ravel()) IntCell(Z3.l());
203  X4->check_value(LOC);
204  return Macro::get_macro(Macro::MAC_Z__LO_SCAN_X4_B)
205  ->eval_LXB(_LO, X4, B);
206  }
207 
208  if (B->get_shape().is_empty()) return LO->eval_identity_fun(B, axis);
209 
210  return Bif_REDUCE::do_reduce(B->get_shape(), Z3, -1, LO, B, m_len);
211 }

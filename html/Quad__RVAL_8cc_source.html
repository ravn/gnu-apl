35 Quad_RVAL::Quad_RVAL()
36  : QuadFunction(TOK_Quad_RVAL),
37  N(8),
38  desired_maxdepth(4)
39 {
40  memset(state, 0, sizeof(state));
41  initstate_r(1, state, N, &rdata);
42 
43  while (desired_shape.get_rank() < MAX_RANK)
44  desired_shape.add_shape_item(1);
45 
46  desired_ranks.push_back(0);
47 
48  desired_types.push_back(0); // no chars
49  desired_types.push_back(1); // ints
50  desired_types.push_back(0); // no real
51  desired_types.push_back(0); // no complex
52  desired_types.push_back(0); // no nested

54 //-----------------------------------------------------------------------------
55 Value_P
56 Quad_RVAL::do_eval_B(const Value & B, int depth)
57 {
58  if (B.get_rank() != 1) RANK_ERROR;
59 const ShapeItem ec_B = B.element_count();
60  if (ec_B > 4)
61  {
62  MORE_ERROR() << "monadic ⎕RVAL B expects at most 4 properties "
63  "(rank, shape, type, and maxdepth)";
64  LENGTH_ERROR;
65  }
66 
67  // save properties values so that we can restore them
68  //
69 vector<int> old_desired_ranks = desired_ranks;
70 Shape old_desired_shape = desired_shape;
71 vector<int> old_desired_types = desired_types;
72 int old_desired_maxdepth = desired_maxdepth;
73 bool need_restore = false;
74 
75  try {
76  need_restore = true; // since something was changed
77 
78  if (ec_B >= 1) // rank: scalar or enclosed vector (distribution)
79  {
80  const Cell & cell = B.get_ravel(0);
81  if (cell.is_pointer_cell())
82  {
83  do_eval_AB(1, cell.get_pointer_value().getref());
84  }
85  else // rank as scalar
86  {
87  Value_P rank = IntScalar(cell.get_int_value(), LOC);
88  do_eval_AB(1, rank.getref());
89  }
90  }
91 
92  if (ec_B >= 2) // shape: scalar or enclosed vector
93  {
94  const Cell & cell = B.get_ravel(1);
95  if (cell.is_pointer_cell())
96  {
97  do_eval_AB(2, cell.get_pointer_value().getref());
98  }
99  else // rank as scalar
100  {
101  Value_P rank = IntScalar(cell.get_int_value(), LOC);
102  do_eval_AB(2, rank.getref());
103  }
104  }
105 
106  if (ec_B >= 3) // type: always enclosed vector (distribution)
107  do_eval_AB(3, B.get_ravel(2).get_pointer_value().getref());
108 
109  if (ec_B >= 4) // maxdepth: scalar or 1-element vector
110  {
111  const Cell & cell = B.get_ravel(3);
112  if (cell.is_pointer_cell()) // maxdepth as 1-element vector
113  {
114  do_eval_AB(4, cell.get_pointer_value().getref());
115  }
116  else // maxdepth as scalar
117  {
118  Value_P rank = IntScalar(cell.get_int_value(), LOC);
119  do_eval_AB(4, rank.getref());
120  }
121  }
122  }
123  catch (...)
124  {
125  desired_ranks = old_desired_ranks;
126  desired_shape = old_desired_shape;
127  desired_types = old_desired_types;
128  desired_maxdepth = old_desired_maxdepth;
129  throw;
130  }
131 
132 const Rank rank = choose_integer(desired_ranks);
133 Shape shape;
134 
135 
136  for (Rank r = MAX_RANK - rank; r < MAX_RANK; ++r)
137  {
138  vector<int> vsh_r; vsh_r.push_back(desired_shape.get_shape_item(r));
139  const int sh_r = choose_integer(vsh_r);
140  shape.add_shape_item(sh_r);
141  }
142 
143 Value_P Z(shape, LOC);
144 
145 const ShapeItem ec = Z->element_count();
146 
147  loop(z, ec)
148  {
149  int type_z; do { type_z = choose_integer(desired_types); }
150  while (depth == desired_maxdepth && type_z == 4);
151  Cell * cZ = Z->next_ravel();
152  switch(type_z)
153  {
154  case 0: random_character(cZ); continue;
155  case 1: random_integer(cZ); continue;
156  case 2: random_float(cZ); continue;
157  case 3: random_complex(cZ); continue;
158  case 4: random_nested(cZ, Z.getref(), B, depth); continue;
159  default: FIXME;
160  }
161  }
162 
163  if (need_restore)
164  {
165  desired_ranks = old_desired_ranks;
166  desired_shape = old_desired_shape;
167  desired_types = old_desired_types;
168  desired_maxdepth = old_desired_maxdepth;
169  }
170 
171  if (ec == 0) new (&Z->get_ravel(0)) IntCell(0);
172 
173  Z->check_value(LOC);
174  return Z;
175 }

178 Quad_RVAL::eval_AB(Value_P A, Value_P B)
179 {

181  if (!A->is_int_scalar()) DOMAIN_ERROR;

184  return Token(TOK_APL_VALUE1, Z);

186 //-----------------------------------------------------------------------------
187 Value_P
188 Quad_RVAL::do_eval_AB(int A, const Value & B)
189 {
190 const int function = A;
191 
192  switch(function)
193  {
194  case 0: return generator_state(B);
195  case 1: return result_rank(B);
196  case 2: return result_shape(B);
197  case 3: return result_type(B);
198  case 4: return result_maxdepth(B);
199  }
200 
201  MORE_ERROR() << "Bad function number A in A ⎕RVAL B";
202  DOMAIN_ERROR;
203 }
204 //-----------------------------------------------------------------------------
205 Value_P
206 Quad_RVAL::generator_state(const Value & B)
207 {
208  // expect an empty, 8, 16, 32, 64, 128, or 256 byte integer vector
209  if (B.get_rank() != 1) RANK_ERROR;
210 
211 const ShapeItem new_N = B.element_count();
212  if (new_N != 0 && new_N != 8 && new_N != 32 &&
213  new_N != 64 && new_N != 128 && new_N != 256) DOMAIN_ERROR;
214 
215  // always return the previous state
216  //
217 Value_P Z(N, LOC);
218  loop(n, N) new (Z->next_ravel()) IntCell(state[n] & 0xFF);
219  Z->check_value(LOC);
220 
221  if (new_N) // set generator state
222  {
223  // make sure that all values are bytes
224  //
225  loop(b, N)
226  {
227  const APL_Integer byte = B.get_ravel(b).get_int_value();
228  if ((byte < -256) || (byte > 255))
229  {
230  MORE_ERROR() << "Bad right argument B in 0 ⎕RVAL B,"
231  "expecting bytes (integers -256...255)";
232  DOMAIN_ERROR;
233  }
234  }
235  N = new_N;
236  loop(b, N)
237  {
238  state[b] = B.get_ravel(b).get_int_value();
239  }
240  setstate_r(state, &rdata);
241  }
242 
243  return Z;
244 }
245 //-----------------------------------------------------------------------------
246 Value_P
247 Quad_RVAL::result_rank(const Value & B)
248 {
249  // B is a single rank or a distribution of ranks 0, 1, ...
250  if (B.get_rank() > 1) RANK_ERROR;
251 
252  // always return the previous rank
253  //
254 Value_P Z(desired_ranks.size(), LOC);
255  loop(z, desired_ranks.size())
256  new (Z->next_ravel()) IntCell(desired_ranks[z]);
257  Z->check_value(LOC);
258 
259  if (B.is_scalar()) // single rank (fixed or equal distribution)
260  {
261  ShapeItem rk = B.get_ravel(0).get_int_value();
262 
263  if ((rk < -MAX_RANK) || (rk > MAX_RANK))
264  {
265  MORE_ERROR() << "a scalar right argument B of 1 ⎕RVAL B should "
266  "be an integer from ¯" << MAX_RANK << " to "
267  << MAX_RANK;
268  DOMAIN_ERROR;
269  }
270 
271  desired_ranks.clear();
272  desired_ranks.push_back(rk);
273  }
274  else if (B.element_count()) // distribution of ranks
275  {
276  vector<int>new_ranks;
277  loop(b, B.element_count())
278  {
279  const int rank_b = B.get_ravel(b).get_int_value();
280  if (rank_b < 0)
281  {
282  MORE_ERROR() << "a vector right argument B of 1 ⎕RVAL B "
283  "should be a distribution (integers ≥ 0)";
284  DOMAIN_ERROR;
285  }
286  new_ranks.push_back(rank_b);
287  }
288 
289  desired_ranks = new_ranks;
290  }
291 
292  return Z;
293 }
294 //-----------------------------------------------------------------------------
295 Value_P
296 Quad_RVAL::result_shape(const Value & B)
297 {
298  if (B.get_rank() > 1) RANK_ERROR;
299 
300 const ShapeItem len_B = B.element_count();
301  if (len_B > MAX_RANK) LENGTH_ERROR; // to many shape items
302 
303  // always return the previous shape
304  //
305 Value_P Z(MAX_RANK, LOC);
306  loop(r, MAX_RANK)
307  new (Z->next_ravel()) IntCell(desired_shape.get_shape_item(r));
308  Z->check_value(LOC);
309 
310  if (len_B) // set the current shape
311  {
312  Shape new_shape;
313 
314  if (B.is_scalar()) // scalar-extend B
315  {
316  const APL_Integer len = B.get_ravel(0).get_int_value();
317  loop(b, MAX_RANK) new_shape.add_shape_item(len);
318  }
319  else // vector B: prepend 1s as needed
320  {
321  // fill leading dimensions with 1
322  while (new_shape.get_rank() < MAX_RANK - len_B)
323  new_shape.add_shape_item(1);
324 
325  // fill lower dimensions with B
326  loop(b, len_B)
327  new_shape.add_shape_item(B.get_ravel(b).get_int_value());
328  }
329 
330  desired_shape = new_shape;
331  }
332 
333  return Z;
334 }
335 //-----------------------------------------------------------------------------
336 Value_P
337 Quad_RVAL::result_type(const Value & B)
338 {
339  // B is a distribution of cell types char (0), int (1), real (2),
340  // complex (3), // or nested (4).
341 
342  if (!B.is_vector())
343  {
344  MORE_ERROR() << "the right argument B of 3 ⎕RVAL B "
345  "should be a vector of up to 5 integers (the\nrelative "
346  "probabilities for types CHAR INT REAL COMPLEX or NESTED respectively)";
347  RANK_ERROR;
348  }
349  if (B.element_count() > 5)
350  {
351  MORE_ERROR() << "the right argument B of 3 ⎕RVAL B "
352  "should be a vector of up to 5 integers (the\nrelative "
353  "probabilities for types CHAR INT REAL COMPLEX or NESTED respectively)";
354  LENGTH_ERROR;
355  }
356 
357  // always return the previous types
358  //
359 Value_P Z(desired_types.size(), LOC);
360  loop(z, desired_types.size())
361  new (Z->next_ravel()) IntCell(desired_types[z]);
362  Z->check_value(LOC);
363 
364  if (B.element_count()) // distribution of depths
365  {
366  vector<int>new_types;
367  bool B_has_simple = false;
368  loop(b, B.element_count())
369  {
370  const int type_b = B.get_ravel(b).get_int_value();
371  if (type_b < 0)
372  {
373  MORE_ERROR() << "the right argument B of 3 ⎕RVAL B "
374  "should be a distribution (vector of integers ≥ 0)";
375  DOMAIN_ERROR;
376  }
377  if (type_b && b < 4) B_has_simple = true;
378  new_types.push_back(type_b);
379  }
380 
381  // if all simple types had a probability of 0 then random_nested()
382  // would loop forever, attempting to choose one. Do not allow that.
383  //
384  if (!B_has_simple)
385  {
386  MORE_ERROR() << "the right argument B of 3 ⎕RVAL B should contain "
387  "at least one simple type with a non-zero probability";
388  DOMAIN_ERROR;
389  }
390 
391  while (new_types.size() < 5) new_types.push_back(0); // make 5=⍴Z
392  desired_types = new_types;
393  }
394 
395  return Z;
396 }

401  if (B.get_rank() > 1) RANK_ERROR;

415 //-----------------------------------------------------------------------------
416 int
417 Quad_RVAL::choose_integer(const vector<int> & dist)
418 {
419 const int n = dist.size();
420  Assert(n > 0);
421 
422  if (n == 1) // fixed value or equal distribution 0, 1, ... n
423  {
424  const int desired = dist[0];
425  if (desired >= 0) return desired; // fixed value
426  const int rand = rand17();
427  return rand % (1 - desired); // = 0... desired
428  }
429 
430  // dist is a distribution
431  //
432 int sum = 0;
433  for (size_t d = 0; d < dist.size(); ++d) sum += dist[d];
434  const int rand = (rand17() & 0xFFFF) % sum;
435  sum = 0;
436  for (size_t d = 0; d < dist.size(); ++d)
437  {
438  sum += dist[d];
439  if (rand < sum) return d;
440  }
441 
442  FIXME; // not reached
443 }

451 //-----------------------------------------------------------------------------
452 void
453 Quad_RVAL::random_integer(Cell * cell)
454 {
455 const int64_t rnd = rand17()
456  ^ (rand17() << 16)
457  ^ (rand17() << 32)
458  ^ (rand17() << 48);
459  new (cell) IntCell(rnd);
460 }
461 //-----------------------------------------------------------------------------
462 void
463 Quad_RVAL::random_float(Cell * cell)
464 {
465 union { uint64_t i; double f; } u;
466  do {
467  u.i = rand17() | rand17() << 17 | rand17() << 34;
468  u.i &= 0x000FFFFFFFFFFFFFULL;
469  u.i |= 0x3FE0000000000000ULL;
470  } while (!isnormal(u.f));
471 
472  new (cell) FloatCell(u.f);
473 }
474 //-----------------------------------------------------------------------------
475 void
476 Quad_RVAL::random_complex(Cell * cell)
477 {
478 union { int64_t i; double f; } u1, u2;
479  do {
480  u1.i = rand17() | rand17() << 17 | rand17() << 34;
481  u1.i &= 0x000FFFFFFFFFFFFFULL;
482  u1.i |= 0x3FE0000000000000ULL;
483  } while (!isnormal(u1.f));
484 
485  do {
486  u2.i = rand17() | rand17() << 17 | rand17() << 34;
487  u2.i &= 0x000FFFFFFFFFFFFFULL;
488  u2.i |= 0x3FE0000000000000ULL;
489  } while (!isnormal(u2.f));
490  new (cell) ComplexCell(u1.f - 1.0, u2.f - 1.0);
491 }

498

505 uint64_t
506 Quad_RVAL::rand17()
507 {
508 int32_t rnd;
509  if (random_r(&rdata, &rnd)) FIXME;
510 
511  // the lower bits are less random, so we xor the upper 16 bits into
512  // the lower 16 bits and return them.
513  return (rnd ^ (rnd >> 16)) & 0x1FFFF;
514 }

60 Value_P
61 Bif_F12_PARTITION::enclose_with_axes(const Shape & shape_X, Value_P B)
62 {
63 Shape item_shape;
64 Shape it_weight;
65 Shape shape_Z;
66 Shape weight_Z;
67 
68 const Shape weight_B = B->get_shape().reverse_scan();
69 
70  // put the dimensions mentioned in X into item_shape and the
71  // others into shape_Z
72  //
73 
74  loop(r, B->get_rank()) // the axes not in shape_X
75  {
76  if (!shape_X.contains_axis(r))
77  {
78  shape_Z.add_shape_item(B->get_shape_item(r));
79  weight_Z.add_shape_item(weight_B.get_shape_item(r));
80  }
81  }
82 
83 int X_axes_used = 0;
84  loop(r, shape_X.get_rank()) // the axes in shape_X
85  {
86  const ShapeItem x_r = shape_X.get_shape_item(r);
87 
88  // check that X∈⍳⍴⍴B
89  //
90  if (x_r < 0) AXIS_ERROR;
91  if (x_r >= B->get_rank()) AXIS_ERROR;
92  if (X_axes_used & 1 << x_r)
93  {
94  MORE_ERROR() = "Duplicate axis";
95  AXIS_ERROR;
96  }
97  X_axes_used |= 1 << x_r;
98 
99  item_shape.add_shape_item(B->get_shape_item(x_r));
100  it_weight.add_shape_item(weight_B.get_shape_item(x_r));
101  }
102 
103  if (item_shape.get_rank() == 0) // empty axes
104  {
105  // ⊂[⍳0]B ←→ ⊂¨B
106  Token part(TOK_FUN1, Bif_F12_PARTITION::fun);
107  return Bif_OPER1_EACH::fun->eval_LB(part, B).get_apl_val();
108  }
109 
110 Value_P Z(shape_Z, LOC);
111  if (Z->is_empty())
112  {
113  Z->set_default(*B.get(), LOC);
114  Z->check_value(LOC);
115  return Z;
116  }
117 
118  for (ArrayIterator it_Z(shape_Z); it_Z.more(); ++it_Z)
119  {
120  const ShapeItem off_Z = it_Z.multiply(weight_Z); // offset in Z
121 
122  Value_P vZ(item_shape, LOC);
123  new (Z->next_ravel()) PointerCell(vZ.get(), Z.getref());
124 
125  if (item_shape.is_empty())
126  {
127  vZ->get_ravel(0).init(B->get_ravel(0), vZ.getref(), LOC);
128  }
129  else
130  {
131  for (ArrayIterator it_it(item_shape); it_it.more(); ++it_it)
132  {
133  const ShapeItem off_B = // offset in B
134  it_it.multiply(it_weight);
135  vZ->next_ravel()->init(B->get_ravel(off_Z + off_B),
136  vZ.getref(), LOC);
137  }
138  }
139  vZ->check_value(LOC);
140  }
141 
142  Z->check_value(LOC);
143  return Z;
144 }

146 Token
147 Bif_F12_PARTITION::partition(Value_P A, Value_P B, Axis axis)
148 {
149  if (A->get_rank() > 1) RANK_ERROR;
150  if (B->get_rank() == 0) RANK_ERROR;
151 
152  if (A->is_scalar())
153  {
154  APL_Integer val = A->get_ravel(0).get_near_int();
155  if (val == 0)
156  {
157  return Token(TOK_APL_VALUE1, Idx0(LOC));
158  }
159 
160  return eval_B(B);
161  }
162 
163  if (A->get_shape_item(0) != B->get_shape_item(axis)) LENGTH_ERROR;
164 
165  // determine the length of the partitioned dimension...
166  //
167 ShapeItem len_Z = 0;
168  {
169  ShapeItem prev = 0;
170  loop(l, A->get_shape_item(0))
171  {
172  const APL_Integer am = A->get_ravel(l).get_near_int();
173  if (am < 0) DOMAIN_ERROR;
174  if (am > prev) ++len_Z;
175  prev = am;
176  }
177  }
178 
179 Shape shape_Z(B->get_shape());
180  shape_Z.set_shape_item(axis, len_Z);
181 
182 Value_P Z(shape_Z, LOC);
183 
184 
185  if (Z->is_empty())
186  {
187  Z->set_default(*B.get(), LOC);
188 
189  Z->check_value(LOC);
190  return Token(TOK_APL_VALUE1, Z);
191  }
192 
193 const Shape3 shape_B3(B->get_shape(), axis);
194 Cell * cZ;
195 
196  loop(h, shape_B3.h())
197  loop(l, shape_B3.l())
198  {
199  ShapeItem from = -1;
200  ShapeItem prev_am = 0;
201  ShapeItem zm = 0;
202  cZ = &Z->get_ravel(h*len_Z*shape_B3.l() + l);
203 
204  loop(m, shape_B3.m())
205  {
206  const APL_Integer am = A->get_ravel(m).get_near_int();
207  Assert(am >= 0); // already verified above.
208 
209  if (am == 0) // skip element m (and following)
210  {
211  if (from != -1) // an old segment is pending
212  {
213  copy_segment(cZ + zm, Z.getref(), h,
214  from, m, shape_B3.m(),
215  l, shape_B3.l(), B);
216  zm += shape_B3.l();
217  }
218  from = -1;
219  }
220  else if (am > prev_am) // new segment
221  {
222  if (from != -1) // an old segment is pending
223  {
224  copy_segment(cZ + zm, Z.getref(), h, from, m,
225  shape_B3.m(), l, shape_B3.l(), B);
226  zm += shape_B3.l();
227  }
228  from = m;
229  }
230  prev_am = am;
231  }
232 
233  if (from != -1) // an old segment is pending
234  {
235  copy_segment(cZ + zm, Z.getref(), h, from, shape_B3.m(),
236  shape_B3.m(), l, shape_B3.l(), B);
237 
238  // zm += shape_B3.l(); not needed since we are done.
239  }
240  }
241 
242  Z->check_value(LOC);
243  return Token(TOK_APL_VALUE1, Z);
244 }

246 void
247 Bif_F12_PARTITION::copy_segment(Cell * dest, Value & dest_owner,
248  ShapeItem h, ShapeItem m_from,
249  ShapeItem m_to, ShapeItem m_len,
250  ShapeItem l, ShapeItem l_len, Value_P B)
251 {
252  Assert(m_from < m_to);
253  Assert(m_to <= m_len);
254  Assert(l < l_len);
255 
256 Value_P V(m_to - m_from, LOC);
257 
258 Cell * vv = &V->get_ravel(0);
259  for (ShapeItem m = m_from; m < m_to; ++m)
260  {
261  const Cell & cb = B->get_ravel(l + (m + h*m_len)*l_len);
262  vv++->init(cb, V.getref(), LOC);
263  }
264 
265  V->check_value(LOC);
266  new (dest) PointerCell(V.get(), dest_owner);
267 }

288 //-----------------------------------------------------------------------------
289 Token
290 Bif_F12_PICK::disclose(Value_P B, bool rank_tolerant)
291 {
292  if (B->is_simple_scalar()) return Token(TOK_APL_VALUE1, B);
293 
294 const Shape item_shape = compute_item_shape(B, rank_tolerant);
295 const Shape shape_Z = B->get_shape() + item_shape;
296 
297 const ShapeItem len_B = B->element_count();
298  if (len_B == 0)
299  {
300  Value_P first = Bif_F12_TAKE::first(B);
301  Token result = disclose(first, rank_tolerant);
302  if (result.get_Class() == TC_VALUE) // success
303  result.get_apl_val()->set_shape(shape_Z);
304  return result;
305  }
306 
307 Value_P Z(shape_Z, LOC);
308 
309 const ShapeItem llen = item_shape.get_volume();
310 
311  if (llen == 0) // empty enclosed value
312  {
313  const Cell & B0 = B->get_ravel(0);
314  if (B0.is_pointer_cell())
315  {
316  Value_P vB = B0.get_pointer_value();
317  Value_P B_proto = vB->prototype(LOC);
318  Z->get_ravel(0).init(B_proto->get_ravel(0), Z.getref(), LOC);
319  }
320  else
321  {
322  Z->get_ravel(0).init(B0, Z.getref(), LOC);
323  }
324 
325  Z->check_value(LOC);
326  return Token(TOK_APL_VALUE1, Z);
327  }
328 
329  loop(h, len_B)
330  {
331  const Cell & B_item = B->get_ravel(h);
332  Cell * Z_from = &Z->get_ravel(h*llen);
333  if (B_item.is_pointer_cell())
334  {
335  Value_P vB = B_item.get_pointer_value();
336  Bif_F12_TAKE::fill(item_shape, Z_from, Z.getref(), vB);
337  }
338  else if (B_item.is_lval_cell())
339  {
340  const Cell * pointee = B_item.get_lval_value();
341  if (pointee && pointee->is_pointer_cell()) // pointer to nested
342  {
343  Value_P vB = pointee->get_pointer_value();
344  Value_P ref_B = vB->get_cellrefs(LOC);
345  Bif_F12_TAKE::fill(item_shape, Z_from, Z.getref(), ref_B);
346  }
347  else // pointer to simple scalar
348  {
349  Z_from->init(B_item, Z.getref(), LOC);
350  for (ShapeItem c = 1; c < llen; ++c)
351  new (Z_from + c) LvalCell(0, 0);
352  }
353  }
354  else if (B_item.is_character_cell()) // simple char scalar
355  {
356  Z_from->init(B_item, Z.getref(), LOC);
357  for (ShapeItem c = 1; c < llen; ++c)
358  new (Z_from + c) CharCell(UNI_ASCII_SPACE);
359  }
360  else // simple numeric scalar
361  {
362  Z->get_ravel(h*llen).init(B_item, Z.getref(), LOC);
363  for (ShapeItem c = 1; c < llen; ++c)
364  new (Z_from + c) IntCell(0);
365  }
366  }
367 
368  Z->set_default(*B.get(), LOC);
369  Z->check_value(LOC);
370  return Token(TOK_APL_VALUE1, Z);
371 }
372 //-----------------------------------------------------------------------------
373 Token
374 Bif_F12_PICK::disclose_with_axis(const Shape & axes_X, Value_P B,
375  bool rank_tolerant)
376 {
377  // disclose with axis
378 
379  // all items of B must have the same rank. item_shape is the smallest
380  // shape that can contain each item of B.
381  //
382 const Shape item_shape = compute_item_shape(B, rank_tolerant);
383 
384  // the number of items in X must be the number of axes in item_shape
385  if (item_shape.get_rank() != axes_X.get_rank()) AXIS_ERROR;
386 
387  // distribute shape_B and item_shape into 2 shapes perm and shape_Z.
388  // if a dimension is mentioned in axes_X then it goes into shape_Z and
389  // otherwise into perm.
390  //
391 Shape perm;
392 Shape shape_Z;
393  {
394  ShapeItem B_idx = 0;
395  //
396  loop(z, B->get_rank() + item_shape.get_rank())
397  {
398  // check if z is in X, remembering its position if so.
399  //
400  bool z_in_X = false;
401  ShapeItem x_pos = -1;
402  loop(x, axes_X.get_rank())
403  {
404  if (axes_X.get_shape_item(x) == z)
405  {
406  z_in_X = true;
407  x_pos = x;
408  break;
409  }
410  }
411 
412  if (z_in_X) // z is an item dimension: put it in shape_Z
413  {
414  shape_Z.add_shape_item(item_shape.get_shape_item(x_pos));
415  }
416  else // z is a B dimension: put it in perm
417  {
418  if (B_idx >= B->get_rank()) INDEX_ERROR;
419  perm.add_shape_item(z);
420  shape_Z.add_shape_item(B->get_shape_item(B_idx++));
421  }
422  }
423 
424  // append X to perm with each X item reduced by the B items before it.
425  loop(x, axes_X.get_rank())
426  {
427  Rank before_x = 0; // items before X that are not in X
428  loop(x1, x - 1)
429  if (!axes_X.contains_axis(x1)) ++before_x;
430 
431  perm.add_shape_item(axes_X.get_shape_item(x) + before_x);
432  }
433  }
434 
435 Value_P Z(shape_Z, LOC);
436  if (Z->is_empty())
437  {
438  Z->set_default(*B.get(), LOC);
439  return Token(TOK_APL_VALUE1, Z);
440  }
441 
442  // loop over sources and place them in the result.
443  //
444 PermutedArrayIterator it_Z(shape_Z, perm);
445 
446  for (ArrayIterator it_B(B->get_shape()); it_B.more(); ++it_B)
447  {
448  const Cell & B_item = B->get_ravel(it_B());
449  if (B_item.is_pointer_cell())
450  {
451  Value_P vB = B_item.get_pointer_value();
452  ArrayIterator vB_it(vB->get_shape());
453  for (ArrayIterator it_it(item_shape); it_it.more(); ++it_it)
454  {
455  Cell * dest = &Z->get_ravel(it_Z());
456  if (vB->get_shape().contains(it_it.get_offsets()))
457  {
458  dest->init(vB->get_ravel(vB_it()), Z.getref(), LOC);
459  ++vB_it;
460  }
461  else if (vB->get_ravel(0).is_character_cell()) // char
462  new (dest) CharCell(UNI_ASCII_SPACE);
463  else // simple numeric
464  new (dest) IntCell(0);
465 
466  ++it_Z;
467  }
468  }
469  else
470  {
471  for (ArrayIterator it_it(item_shape); it_it.more(); ++it_it)
472  {
473  Cell * dest = &Z->get_ravel(it_Z());
474  if (it_it() == 0) // first element: use B_item
475  dest->init(B_item, Z.getref(), LOC);
476  else if (B_item.is_character_cell()) // simple char scalar
477  new (dest) CharCell(UNI_ASCII_SPACE);
478  else // simple numeric scalar
479  new (dest) IntCell(0);
480  ++it_Z;
481  }
482  }
483  }
484 
485  Z->set_default(*B.get(), LOC);
486 
487  Z->check_value(LOC);
488  return Token(TOK_APL_VALUE1, Z);
489 }
490 //-----------------------------------------------------------------------------
491 Shape
492 Bif_F12_PICK::compute_item_shape(Value_P B, bool rank_tolerant)
493 {
494  // all items of B are scalars or (nested) arrays of the same rank R.
495  // return the shape with rank R and the (per-dimension) max. of
496  // each shape item
497  //
498 const ShapeItem len_B = B->nz_element_count();
499 
500 Shape ret; // of the first non-scalar in B
501 
502  loop(b, len_B)
503  {
504  Value_P v;
505  {
506  const Cell & cB = B->get_ravel(b);
507  if (cB.is_pointer_cell())
508  {
509  v = cB.get_pointer_value();
510  }
511  else if (cB.is_lval_cell())
512  {
513  const Cell * pointee = cB.get_lval_value();
514  if (!pointee) continue; // ⊃ NOOP element
515  if (!pointee->is_pointer_cell()) continue; // ptr to scalar
516  v = pointee->get_pointer_value();
517  }
518  else
519  {
520  continue; // simple scalar
521  }
522  }
523 
524  if (ret.get_rank() == 0) // first non-scalar
525  {
526  ret = v->get_shape();
527  continue;
528  }
529 
530  // the items of B must have the same rank, unless we are rank_tolerant
531  //
532  if (ret.get_rank() != v->get_rank())
533  {
534  if (!rank_tolerant) RANK_ERROR;
535 
536  if (ret.get_rank() < v->get_rank())
537  ret.expand_rank(v->get_rank());
538 
539  // if ret.get_rank() > v->get_rank() then we are OK because
540  // only the dimensions present in v are expanded below.
541  }
542 
543  loop(r, v->get_rank())
544  {
545  if (ret.get_shape_item(r) < v->get_shape_item(r))
546  {
547  ret.set_shape_item(r, v->get_shape_item(r));
548  }
549  }
550  }
551 
552  return ret;
553 }
554 //-----------------------------------------------------------------------------
555 Value_P
556 Bif_F12_PICK::pick(const Cell * cA, ShapeItem len_A, Value_P B,
557  APL_Integer qio, Value * cell_owner)
558 {
559 ShapeItem c = 0;
560 
561  if (cA->is_pointer_cell()) // then B shall be a 1-dimensional array
562  {
563  Value_P A = cA->get_pointer_value();
564  if (A->get_rank() > 1) RANK_ERROR;
565 
566  const ShapeItem len_A = A->element_count();
567  if (B->get_rank() != len_A) RANK_ERROR;
568 
569  const Shape weight = B->get_shape().reverse_scan();
570  const Shape A_as_shape(A.get(), qio);
571 
572  loop(r, A->element_count())
573  {
574  const ShapeItem ar = A_as_shape.get_shape_item(r);
575  if (ar < 0) INDEX_ERROR;
576  if (ar >= B->get_shape_item(r)) INDEX_ERROR;
577  c += weight.get_shape_item(r) * ar;
578  }
579  }
580  else // A is a scalar, so B must be a vector.
581  {
582  if (B->get_rank() != 1) RANK_ERROR;
583  const APL_Integer a = cA->get_near_int() - qio;
584  if (a < 0) INDEX_ERROR;
585  if (a >= B->get_shape_item(0)) INDEX_ERROR;
586  c = a;
587  }
588 
589 const Cell * cB = &B->get_ravel(c);
590 
591  if (len_A > 1) // more levels coming.
592  {
593  if (cB->is_pointer_cell())
594  {
595  return pick(cA + 1, len_A - 1, cB->get_pointer_value(),
596  qio, cell_owner);
597  }
598 
599  if (cB->is_lval_cell())
600  {
601  Assert(cell_owner);
602  Cell & cell = *cB->get_lval_value();
603  if (!cell.is_pointer_cell()) DOMAIN_ERROR;
604 
605  Value_P subval = cell.get_pointer_value();
606  Value_P subrefs = subval->get_cellrefs(LOC);
607  Value * sub_cellowner = subrefs->get_lval_cellowner();
608  Assert(sub_cellowner);
609  return pick(cA + 1, len_A - 1, subrefs, qio, sub_cellowner);
610  }
611 
612  // at this point the depth implied by A is greater than the
613  // depth of B.
614  //
615  RANK_ERROR; // ISO p.166
616  }
617 
618  // at this point, cB is the cell in B pick'ed by A->
619  //
620  if (cB->is_pointer_cell())
621  {
622  Value_P Z = cB->get_pointer_value()->clone(LOC);
623  return Z;
624  }
625 
626  if (cB->is_lval_cell()) // e.g. (A⊃B) ← C
627  {
628  Assert(cell_owner);
629 
630  Cell * cell = cB->get_lval_value();
631  Assert(cell);
632 
633  Value_P Z(LOC);
634  if (cell->is_pointer_cell())
635  new (Z->next_ravel()) LvalCell_picked(cell, cell_owner);
636  else
637  new (Z->next_ravel()) LvalCell(cell, cell_owner);
638  return Z;
639  }
640  else
641  {
642  Value_P Z(LOC);
643  Z->next_ravel()->init(*cB, Z.getref(), LOC);
644  return Z;
645  }
646 }

52 Value_P
53 Bif_F12_PARTITION::enclose_with_axes(const Shape & shape_X, Value_P B)
54 {
55 Shape item_shape;
56 Shape it_weights;
57 Shape shape_Z;
58 Shape weights_Z;
59 
60 const Shape weights_B = B->get_shape().get_weights();
61 
62  // put the dimensions mentioned in X into item_shape and the
63  // others into shape_Z
64  //
65 
66  loop(r, B->get_rank()) // the axes not in shape_X
67  {
68  if (!shape_X.contains_axis(r))
69  {
70  shape_Z.add_shape_item(B->get_shape_item(r));
71  weights_Z.add_shape_item(weights_B.get_shape_item(r));
72  }
73  }
74 
75 int X_axes_used = 0;
76  loop(r, shape_X.get_rank()) // the axes in shape_X
77  {
78  const ShapeItem x_r = shape_X.get_shape_item(r);
79 
80  // check that X∈⍳⍴⍴B
81  //
82  if (x_r < 0) AXIS_ERROR;
83  if (x_r >= B->get_rank()) AXIS_ERROR;
84  if (X_axes_used & 1 << x_r)
85  {
86  MORE_ERROR() = "Duplicate axis";
87  AXIS_ERROR;
88  }
89  X_axes_used |= 1 << x_r;
90 
91  item_shape.add_shape_item(B->get_shape_item(x_r));
92  it_weights.add_shape_item(weights_B.get_shape_item(x_r));
93  }
94 
95  if (item_shape.get_rank() == 0) // empty axes
96  {
97  // ⊂[⍳0]B ←→ ⊂¨B
98  Token part(TOK_FUN1, Bif_F12_PARTITION::fun);
99  return Bif_OPER1_EACH::fun->eval_LB(part, B).get_apl_val();
100  }
101 
102 Value_P Z(shape_Z, LOC);
103  if (Z->is_empty())
104  {
105  Z->set_default(*B.get(), LOC);
106  Z->check_value(LOC);
107  return Z;
108  }
109 
110  for (ArrayIterator it_Z(shape_Z); it_Z.more(); ++it_Z)
111  {
112  const ShapeItem off_Z = it_Z.multiply(weights_Z); // offset in Z
113 
114  Value_P vZ(item_shape, LOC);
115  Z->next_ravel_Pointer(vZ.get());
116 
117  if (item_shape.is_empty())
118  {
119  vZ->get_wproto().init(B->get_cfirst(), vZ.getref(), LOC);
120  }
121  else
122  {
123  for (ArrayIterator it_it(item_shape); it_it.more(); ++it_it)
124  {
125  const ShapeItem off_B = // offset in B
126  it_it.multiply(it_weights);
127  vZ->next_ravel_Cell(B->get_cravel(off_Z + off_B));
128  }
129  }
130  vZ->check_value(LOC);
131  }
132 
133  Z->check_value(LOC);
134  return Z;
135 }
136 //-----------------------------------------------------------------------------
137 Token
138 Bif_F12_PARTITION::partition(Value_P A, Value_P B, Axis axis) const
139 {
140  // A must be a scalar or vector (of non-negative integers)
141  //
142  if (A->get_rank() > 1) RANK_ERROR;
143  if (B->get_rank() == 0) RANK_ERROR;
144 
145  // construct a vector of breakpoints from A
146  //
147 const ShapeItem len_A = A->get_shape_item(0);
148  if (len_A != B->get_shape_item(axis)) LENGTH_ERROR;
149 
150 vector<ShapeItem> breakpoints; // start position of a partition (including)
151 vector<ShapeItem> breakends; // end position of a partition (excluding)
152 
153  // A is scalar-extended as needed. The case where A is a scalar is, however
154  // identical to the case where all items in A are equal, and in both cases
155  // there is only one breakpoint (at B[0])
156  //
157 ShapeItem len_Zm = 0;
158 
159  if (A->is_scalar())
160  {
161  breakpoints.push_back(0);
162  breakends.push_back(len_A);
163  }
164  else
165  {
166  ShapeItem prev_A = 0;
167  bool in_partition = false;
168  loop(apos, len_A)
169  {
170  const APL_Integer aval = A->get_cravel(apos).get_near_int();
171  if (aval < 0) DOMAIN_ERROR;
172 
173  if (aval > prev_A) // new partition starting at apos
174  {
175  breakpoints.push_back(apos);
176  if (in_partition) breakends.push_back(apos);
177  in_partition = true;
178  ++len_Zm;
179  }
180  else if (in_partition && aval == 0)
181  {
182  breakends.push_back(apos);
183  in_partition = false;
184  }
185  prev_A = aval;
186  }
187 
188  if (in_partition) breakends.push_back(len_A);
189 
190  Assert(breakpoints.size() == breakends.size());
191  }
192 
193  if (len_Zm == 1) // single partition
194  {
195  if (A->get_cfirst().get_near_int()) // non-zero A0
196  {
197  Value_P ZZ = B->clone(LOC);
198  Value_P Z(LOC);
199  Z->next_ravel_Pointer(ZZ.get());
200  return Token(TOK_APL_VALUE1, Z);
201  }
202 
203  // 0 ⊂ B is empty
204  return Token(TOK_APL_VALUE1, Idx0(LOC));
205  }
206 
207 Shape shape_Z(B->get_shape());
208  shape_Z.set_shape_item(axis, len_Zm);
209 
210 Value_P Z(shape_Z, LOC);
211 
212  if (Z->is_empty())
213  {
214  Z->set_default(*B.get(), LOC);
215 
216  Z->check_value(LOC);
217  return Token(TOK_APL_VALUE1, Z);
218  }
219 
220 const Shape3 shape_B3(B->get_shape(), axis);
221 
222  loop(h, shape_B3.h())
223  loop(m, len_Zm)
224  loop(l, shape_B3.l())
225  {
226  const Cell * src_B = &B->get_cravel(l + shape_B3.l() *
227  (breakpoints[m] + h * shape_B3.m()));
228 
229  const ShapeItem partition_len = breakends[m] - breakpoints[m];
230 
231  Value_P ZZ(partition_len, LOC);
232  loop(p, partition_len)
233  {
234  ZZ->next_ravel_Cell(*src_B);
235  src_B += shape_B3.l();
236  }
237  ZZ->check_value(LOC);
238 
239  Z->next_ravel_Pointer(ZZ.get());
240  }
241 
242  Z->check_value(LOC);
243  return Token(TOK_APL_VALUE1, Z);
244 }

246 Token
247 Bif_F12_PICK::eval_AB(Value_P A, Value_P B) const
248 {
249  if (A->get_rank() > 1) RANK_ERROR;
250 
251 const ShapeItem ec_A = A->element_count();
252 
253  // if A is empty, return B
254  //
255  if (ec_A == 0) return Token(TOK_APL_VALUE1, B);
256 
257 const APL_Integer qio = Workspace::get_IO();
258 
259 Value_P Z = pick(&A->get_cfirst(), 0, ec_A, B, qio);
260 
261  Z->check_value(LOC);
262  return Token(TOK_APL_VALUE1, Z);
263 }
264 //-----------------------------------------------------------------------------
265 Value_P
266 Bif_F12_PICK::disclose(Value_P B, bool rank_tolerant)
267 {
268  // for simple scalars B: B ≡ ⊂ B and therefore B ≡ ⊃ B
269  //
270  if (B->is_simple_scalar()) return B;
271 
272  // compute item_shape, which is the smallest shape into which each
273  // item of B fits, and then ⍴Z ←→ (⍴B), item_shape
274 
275 const Shape item_shape = compute_item_shape(B, rank_tolerant);
276 const Shape shape_Z = B->get_shape() + item_shape;
277 
278 const ShapeItem len_B = B->element_count();
279  if (len_B == 0)
280  {
281  Value_P first = Bif_F12_TAKE::first(B);
282  Value_P result = disclose(first, rank_tolerant);
283  result->set_shape(shape_Z);
284  return result;
285  }
286 
287 Value_P Z(shape_Z, LOC);
288 
289 const ShapeItem item_len = item_shape.get_volume();
290 
291  if (item_len == 0) // empty enclosed value
292  {
293  const Cell & B0 = B->get_cproto();
294  if (B0.is_pointer_cell())
295  {
296  Value_P vB = B0.get_pointer_value();
297  Value_P B_proto = vB->prototype(LOC);
298  Z->get_wproto().init(B_proto->get_cproto(), Z.getref(), LOC);
299  }
300  else
301  {
302  Z->get_wproto().init(B0, Z.getref(), LOC);
303  }
304 
305  Z->check_value(LOC);
306  return Z;
307  }
308 
309  loop(b, len_B) // for all items in B...
310  {
311  const Cell & B_item = B->get_cravel(b);
312  disclose_item(Z.getref(), b, item_shape, item_len, B_item);
313  }
314 
315  Z->set_default(*B.get(), LOC);
316  Z->check_value(LOC);
317  return Z;
318 }
319 //-----------------------------------------------------------------------------
320 void
321 Bif_F12_PICK::disclose_item(Value & Z, ShapeItem b,
322  const Shape & item_shape, ShapeItem item_len,
323  const Cell & B_item)
324 {
325  if (B_item.is_pointer_cell())
326  {
327  // B_item points to a nested APL value. Take that value, expand
328  // it to item_shape, and store it in Z.
329  //
330  const Value & vB = B_item.get_pointer_value().getref();
331  Bif_F12_TAKE::fill(item_shape, Z, vB);
332  }
333  else if (B_item.is_lval_cell())
334  {
335  /* B_item is a left-value Cell (pointing to a Cell (target) that
336  shall be assigned at a later point in time. This is a litte
337  tricky since there are 3 cases:
338 
339  1. target == 0, which means that target was padded at an earlier
340  point in time (and then nothing shall be done), or
341 
342  2. target is a right-value (even though it is left of →. The
343  convention for cellrefs has it, that left hand PointerCells are
344  not expanded when their parents are (because they are subject to
345  be deleted by other primitives, and then the early conversion to
346  left-values would not pay of). We expand the target now, expand
347  it to item_shape, and store it in Z.
348 
349  3. target is a simple Cell, so B_item points a simple Cell. We
350  expand B_item to item_shape using LvalCells that point nowhere.
351 
352  In all 3 cases, since we are left of ←, the expansion needs to be
353  made with LvalCell(0, 0) pointing nowhere rather than with UNI_SPACE
354  or with 0.
355  */
356  Cell * target = B_item.get_lval_value();
357  if (target == 0) // case 1.
358  {
359  loop(c, item_len) Z.next_ravel_Lval(0, 0);
360  }
361  else if (target->is_pointer_cell()) // case 2.
362  {
363  Value_P vB = target->get_pointer_value();
364  const Value & cellref_B = vB->get_cellrefs(LOC).getref();
365  Bif_F12_TAKE::fill(item_shape, Z, cellref_B);
366  }
367  else // case 3.
368  {
369  Z.next_ravel_Cell(B_item);
370  for (ShapeItem c = 1; c < item_len; ++c)
371  Z.next_ravel_Lval(0, 0);
372  }
373  }
374  else if (B_item.is_character_cell()) // simple char scalar
375  {
376  // B_item is a character, so expand it to item_shape with UNI_SPACE,
377  // and store it in Z
378  //
379  Z.next_ravel_Cell(B_item);
380  for (ShapeItem c = 1; c < item_len; ++c) // the remaining items.
381  Z.next_ravel_Char(UNI_SPACE);
382  }
383  else // simple numeric scalar
384  {
385  // B_item is a number, so expand it to item_shape with 0, and store
386  // it in Z
387  //
388  Z.next_ravel_Cell(B_item);
389  for (ShapeItem c = 1; c < item_len; ++c)
390  Z.next_ravel_Int(0);
391  }
392 }
393 //-----------------------------------------------------------------------------
394 Value_P
395 Bif_F12_PICK::disclose_with_axis(const Shape & axes_X, Value_P B)
396 {
397  // disclose with axis: Z←⊃[Z] B
398  // implemented as: cB ← ⊃ B ◊ cX ← ((⍳⍴⍴cB)∼X),X ◊ Z←cX ⍉ B
399 
400  // Note: axes_X is already normalized to ⎕IO←0
401 
402 Value_P cB = disclose(B, true); // cB ← ⊃ B
403 
404 ShapeItem bmX = 0; // axes in axes_X with ⎕IO←0
405 
406  loop(x, axes_X.get_rank())
407  {
408  const ShapeItem ax = axes_X.get_shape_item(x);
409  if (bmX & 1 << ax)
410  {
411  MORE_ERROR() << "In ⊃[X]B: duplicated axis "
412  << (ax + Workspace::get_IO())
413  << " in X";
414  AXIS_ERROR;
415  }
416  bmX |= 1 << ax;
417  }
418 
419  /* ⍴Z is the axes of B that are not in X, followed by those which are.
420  That is, shape_Z is a permutation of ⍳⍴⍴B defined by X.
421 
422  We prepend the permutation axes_X (of each disclosed item) with the
423  axes not in X to obtain the entire permutation perm_cB of cB.
424  */
425 Shape perm_cB; // perm_cB is the permutation of cB, constructed from X
426  loop(x, cB->get_rank()) // axes not in X
427  {
428  if (bmX & 1 << x) continue; // axis x is in X
429  perm_cB.add_shape_item(x); // axis x is not in X
430  }
431 
432  loop(x, axes_X.get_rank()) // axes in X
433  {
434  perm_cB.add_shape_item(axes_X.get_shape_item(x));
435  }
436 
437  Assert(perm_cB.get_rank() == cB->get_rank());
438 
439  return Bif_F12_TRANSPOSE::transpose(perm_cB, cB);
440 }
441 //-----------------------------------------------------------------------------
442 Shape
443 Bif_F12_PICK::compute_item_shape(Value_P B, bool rank_tolerant)
444 {
445  /* The ravel cells of B are either simple or else PointerCells of nested
446  values with possibly different shapes.
447 
448  Return the smallest shape that is large enough to contain each value.
449 
450  If rank_tolerant is false, then all nested values have the same rank
451  (which is also the rank of the result). Otherwise: RANK_ERROR.
452 
453  If rank_tolerant is true, then the smaller ranks are extended to the
454  largest rank of all shapes by prepending 1s.
455  */
456 
457  // we use ret_rank and ret[MAX_RANK] instead of Shape to avoid unnecessary
458  // volume recomputations in class Shape.
459  //
460 ShapeItem ret_rank = 0;
461 ShapeItem ret[MAX_RANK];
462  loop(r, MAX_RANK) ret[r] = 0;
463 
464  loop(b, B->nz_element_count())
465  {
466  const Value * val;
467  {
468  const Cell & cB = B->get_cravel(b);
469  if (cB.is_pointer_cell())
470  {
471  val = cB.get_pointer_value().get();
472  }
473  else if (cB.is_lval_cell())
474  {
475  const Cell * target = cB.get_lval_value();
476 
477  /* pointee was created by a selective assignment and
478  there are 3 cases:
479 
480  1. pointee == 0 (e.g from over-take): do nothing
481  2. pointee is simple. do nothing
482  3. pointee is sub-value: use its value
483  */
484  if (!target) continue; // case 1.
485  if (!target->is_pointer_cell()) continue; // case 2.
486  val = target->get_pointer_value().get(); // case 3.
487  }
488  else
489  {
490  continue; // cB is simple, so it fits anywhere.
491  }
492  }
493 
494  // at this point val is a non-scalar value.
495  //
496  const Rank val_rank = val->get_rank();
497  if (ret_rank == 0) // val is the first non-scalar
498  {
499  ret_rank = val_rank;
500  loop(r, ret_rank) ret[r] = val->get_shape_item(r);
501  continue;
502  }
503 
504  // the items of B must have the same rank, unless we are rank_tolerant
505  //
506  if (ret_rank != val_rank)
507  {
508  if (!rank_tolerant) RANK_ERROR;
509 
510  if (ret_rank < val_rank) ret_rank = val_rank;
511 
512  // if ret_rank() > v->get_rank() then we are OK because
513  // only the dimensions present in v are expanded below.
514  }
515 
516  loop(r, val_rank)
517  {
518  if (ret[r] < val->get_shape_item(r))
519  {
520  ret[r] = val->get_shape_item(r);
521  }
522  }
523  }
524 
525  return Shape(ret_rank, ret);
526 }
527 //-----------------------------------------------------------------------------
528 ShapeItem
529 Bif_F12_PICK::pick_offset(const Cell * const A0, ShapeItem idx_A,
530  ShapeItem len_A, Value_P B, APL_Integer qio)
531 {
532 const Cell & cA = A0[idx_A];
533 
534  if (cA.is_pointer_cell()) // then B shall be a 1-dimensional array
535  {
536  /*
537  cA = A[idx_A] is nested, e.g.
538 
539  case i. (⊂"b") ⊃ A.b.c ← 'leaf-A.b.c' (structured variable A)
540 
541  case ii. (⊂1 1) ⊃ A←3 3⍴⍳9 (normal A)
542  */
543 
544  const Value * A = cA.get_pointer_value().get();
545 
546  if (B->is_member()) // case i. (structured B)
547  {
548  if (!A->is_char_string())
549  {
550  MORE_ERROR() << "member name expected for A⊃B (nested A["
551  << (idx_A + qio) << "])";
552  DOMAIN_ERROR;
553  }
554 
555  const UCS_string top_level("B");
556  const UCS_string member(*A);
557  vector<const UCS_string *> members;
558  members.push_back(&member);
559  members.push_back(&top_level); // dummy, must be last
560  Value * val_B = B.get();
561  Cell * Bsub = B->get_member(members, val_B, false, true);
562  return Bsub - &B->get_cfirst();
563  }
564  else // case ii. (normal B)
565  {
566  if (A->get_rank() > 1)
567  {
568  MORE_ERROR() << "rank A ≤ 1 expected for A⊃B (nested A["
569  << (idx_A + qio) << "])";
570  RANK_ERROR;
571  }
572 
573  const ShapeItem len_A = A->element_count();
574  if (B->get_rank() != len_A)
575  {
576  MORE_ERROR() << "⍴⍴B (" << B->get_rank()
577  << ") = ⍴,A (" << len_A
578  << ") expected for A⊃B (nested A["
579  << (idx_A + qio) << "])";
580  RANK_ERROR;
581  }
582 
583  const Shape weights_B = B->get_shape().get_weights();
584  const Shape A_as_shape(A, qio);
585  ShapeItem offset = 0;
586 
587  loop(r, A->element_count())
588  {
589  const ShapeItem ar = A_as_shape.get_shape_item(r);
590  if (ar < 0) INDEX_ERROR;
591  if (ar >= B->get_shape_item(r)) INDEX_ERROR;
592  offset += weights_B.get_shape_item(r) * ar;
593  }
594  return offset;
595  }
596  }
597  else // A is a scalar, so B must be a vector.
598  {
599  if (B->get_rank() != 1)
600  {
601  MORE_ERROR() << "⍴⍴B (" << B->get_rank()
602  << ") = 1 expected for A⊃B (with scalar A)";
603  RANK_ERROR;
604  }
605  const APL_Integer a = cA.get_near_int() - qio;
606  if (a < 0) INDEX_ERROR;
607  if (a >= B->get_shape_item(0)) INDEX_ERROR;
608  return a;
609  }
610 }
611 //-----------------------------------------------------------------------------
612 Value_P
613 Bif_F12_PICK::pick(const Cell * const A0, ShapeItem idx_A, ShapeItem len_A,
614  Value_P B, APL_Integer qio)
615 {
616  // A0 points to ↑A and we are at depth idx_A, which means that A0[idx_A] is
617  // the current index of B.
618  //
619 const ShapeItem offset = pick_offset(A0, idx_A, len_A, B, qio);
620 const Cell * cB = &B->get_cravel(offset);
621 
622  if (len_A > 1) // more levels coming.
623  {
624  if (cB->is_pointer_cell())
625  {
626  return pick(A0, idx_A+1, len_A-1, cB->get_pointer_value(), qio);
627  }
628 
629  if (cB->is_lval_cell())
630  {
631  // Note: this is a little tricky...
632 
633  // first of all, we need a pointer cell. Therefore the target of cB
634  // should be a PointerCell.
635  //
636  Cell & target = *cB->get_lval_value();
637  if (!target.is_pointer_cell()) DOMAIN_ERROR;
638 
639  // secondly, get_cellrefs() is not recursive, therefore target has
640  // not (yet) been converted to a left-value. We do that now.
641  //
642  Value_P subval = target.get_pointer_value(); // right-value
643  Value_P subrefs = subval->get_cellrefs(LOC); // left-value
644  return pick(A0, idx_A + 1, len_A - 1, subrefs, qio);
645  }
646 
647  // simple cell. This means that the depth of B does not suffice to pick
648  // the item selected by A or, in other words, A is too long).
649  //
650  RANK_ERROR; // ISO p.166 wants rank-error. LENGTH ERROR would be correct.
651  }
652 
653  // len_A == 1, i.e. the end of the recursion over A has been reached,
654  // cB is the cell in B that was pick'ed by A->
655  //
656  if (cB->is_pointer_cell())
657  {

34 Token
35 Bif_OPER2_INNER::eval_ALRB(Value_P A, Token & _LO, Token & _RO, Value_P B)
36 {
37  if (!_LO.is_function() || !_RO.is_function()) SYNTAX_ERROR;
38 
39 Function * LO = _LO.get_function();
40 Function * RO = _RO.get_function();
41  Assert1(LO);
42  Assert1(RO);
43 
44  if (LO->get_fun_valence() != 2 || RO->get_fun_valence() != 2) SYNTAX_ERROR;
45  if (!LO->has_result() || !RO->has_result()) DOMAIN_ERROR;
46 
47  if (!A->is_scalar_extensible() && !B->is_scalar_extensible() &&
48  A->get_rank() > 1 && B->get_rank() > 1 &&
49  A->get_shape().get_last_shape_item() !=
50  B->get_shape().get_shape_item(0)) LENGTH_ERROR;
51 
52 Shape shape_A1;
53 ShapeItem len_A = 1;
54  if (!A->is_scalar())
55  {
56  len_A = A->get_last_shape_item();
57  shape_A1 = A->get_shape().without_axis(A->get_rank() - 1);
58  }
59 
60 Shape shape_B1;
61 ShapeItem len_B = 1;
62  if (!B->is_scalar())
63  {
64  len_B = B->get_shape_item(0);
65  shape_B1 = B->get_shape().without_axis(0);
66  }
67 
68  // we do not check len_A == len_B here, since a non-scalar LO may
69  // accept different lengths of its left and right arguments
70 
71 const ShapeItem items_A = shape_A1.get_volume();
72 const ShapeItem items_B = shape_B1.get_volume();
73 
74  if (items_A == 0 || items_B == 0) // empty result
75  {
76  // the outer product portion of LO.RO is empty.
77  // Apply the fill function of RO
78  //
79  const Shape shape_Z = shape_A1 + shape_B1;
80  return fill(shape_Z, A, RO, B, LOC);
81  }
82 
83  if (LO->may_push_SI() || RO->may_push_SI()) // user defined LO or RO
84  {
85  // iso: if A1 and B1 are both vectors, return f/A1 g B1.
86  //
87  if (A->get_rank() <= 1 && B->get_rank() <= 1)
88  return Macro::get_macro(Macro::MAC_Z__vA_LO_INNER_RO_vB)
89  ->eval_ALRB(A, _LO, _RO, B);
90  else
91  return Macro::get_macro(Macro::MAC_Z__A_LO_INNER_RO_B)
92  ->eval_ALRB(A, _LO, _RO, B);
93  }
94 
95 Value_P Z(shape_A1 + shape_B1, LOC);
96 
97  // an important (and the most likely) special case is LO and RO being scalar
98  // functions. This case can be implemented in a far simpler fashion than
99  // the general case.
100  //
101  job.LO = LO->get_scalar_f2();
102  job.RO = RO->get_scalar_f2();
103  if (job.LO && job.RO && A->is_simple() && B->is_simple() && len_A)
104  {
105  job.incA = (A->element_count() == 1) ? 0 : 1;
106  job.incB = (B->element_count() == 1) ? 0 : 1;
107 
108  // len_A must be len_B, unless at least one length is 1
109  //
110  if (len_A != len_B && job.incA && job.incB) LENGTH_ERROR;
111 
112  job.cZ = &Z->get_ravel(0);
113  job.cA = &A->get_ravel(0);
114  job.ZAh = items_A;
115  job.LO_len = A->is_scalar() ? len_B : len_A;
116  job.cB = &B->get_ravel(0);
117  job.ZBl = items_B;
118  job.ec = E_NO_ERROR;
119 
120  scalar_inner_product();
121  if (job.ec != E_NO_ERROR) throw_apl_error(job.ec, LOC);
122 
123  Z->set_default(*B.get(), LOC);
124 
125  Z->check_value(LOC);
126  return Token(TOK_APL_VALUE1, Z);
127  }
128 
129 const bool A_enclosed = A->get_rank() > 1;
130 const bool B_enclosed = B->get_rank() > 1;
131 
132  // enclose last axis of A if necessary
133  //
134  if (A_enclosed)
135  {
136  const Shape last_axis(A->get_rank() - 1);
137  A = Bif_F12_PARTITION::enclose_with_axes(last_axis, A);
138  }
139 
140  // enclose first axis of B if necessary
141  //
142  if (B_enclosed)
143  {
144  const Shape first_axis(0);
145  B = Bif_F12_PARTITION::enclose_with_axes(first_axis, B);
146  }
147 
148  loop (a, items_A)
149  loop (b, items_B)
150  {
151  Value_P RO_A(A, LOC);
152  if (A_enclosed) RO_A = A->get_ravel(a).get_pointer_value();
153 
154  Value_P RO_B(B, LOC);
155  if (B_enclosed) RO_B = B->get_ravel(b).get_pointer_value();
156 
157  const Token T1 = RO->eval_AB(RO_A, RO_B);
158 
159  if (T1.get_tag() == TOK_ERROR) return T1;
160 
161  Value_P A_RO_B = T1.get_apl_val();
162 
163  Cell * cZ = Z->next_ravel();
164  if (A_RO_B->is_simple_scalar()) // A_RO_B is A RO B
165  {
166  // A RO B has returned a scalar, so LO/A_RO_B is A_RO_B
167  //
168  cZ->init(A_RO_B->get_ravel(0), Z.getref(), LOC);
169  }
170  else
171  {
172  // A RO B has returned a vector, so compute LO/A_RO_B
173  //
174  // new (cZ) PointerCell(A_RO_B.get(), Z.getref());
175 
176  const Token T2 = Bif_OPER1_REDUCE::fun->eval_LB(_LO, A_RO_B);
177  if (T2.get_tag() == TOK_ERROR) return T2;
178  cZ->init_from_value(T2.get_apl_val().get(), Z.getref(), LOC);
179  }
180  }
181 
182  Z->set_default(*B.get(), LOC);
183  Z->check_value(LOC);
184  return Token(TOK_APL_VALUE1, Z);
185 }

187 void
188 Bif_OPER2_INNER::scalar_inner_product() const
189 {
190 #ifdef PERFORMANCE_COUNTERS_WANTED
191 const uint64_t start_1 = cycle_counter();
192 #endif
193 
194  // the empty cases have been ruled out already in inner_product()
195 
196  job.ec = E_NO_ERROR;
197 
198 #if PARALLEL_ENABLED
199  if ( Parallel::run_parallel
200  && Thread_context::get_active_core_count() > 1
201  && job.ZAh * job.ZBl > get_dyadic_threshold())
202  {
203  job.cores = Thread_context::get_active_core_count();
204  Thread_context::do_work = PF_scalar_inner_product;
205  Thread_context::M_fork("scalar_inner_product"); // start pool
206  PF_scalar_inner_product(Thread_context::get_master());
207  Thread_context::M_join();
208  }
209  else
210 #endif // PARALLEL_ENABLED
211  {
212  job.cores = CCNT_1;
213  PF_scalar_inner_product(Thread_context::get_master());
214  }
215 
216 #ifdef PERFORMANCE_COUNTERS_WANTED
217 const uint64_t end_1 = cycle_counter();
218  Performance::fs_OPER2_INNER_AB.add_sample(end_1-start_1, job.ZAh * job.ZBl);
219 #endif
220 }

222 void
223 Bif_OPER2_INNER::PF_scalar_inner_product(Thread_context & tctx)
224 {
225 const ShapeItem Z_len = job.ZAh * job.ZBl;
226 
227 const ShapeItem slice_len = (Z_len + job.cores - 1)/job.cores;
228 ShapeItem z = tctx.get_N() * slice_len;
229 ShapeItem end_z = z + slice_len;
230  if (end_z > Z_len) end_z = Z_len;
231 
232  for (; z < end_z; ++z)
233  {
234  const ShapeItem zah = z/job.ZBl; // z row = A row
235  const ShapeItem zbl = z - zah*job.ZBl; // z column = B column
236  const Cell * row_A = job.cA + job.incA*((zah + 1) * job.LO_len);
237  const Cell * col_B = job.cB + job.incB*(zbl + job.LO_len*job.ZBl);
238 
239  // compute Z[z] ← LO / (row_A RO colB)
240  // e.g. Z[z] ← +/ (row_A × colB)
241  //
242  // we use Z[z] as accumulator for LO /
243  //
244  Cell * sum = job.cZ + z;
245  loop(l, job.LO_len)
246  {
247  row_A -= job.incA;
248  col_B -= job.incB*job.ZBl;
249 
250  if (l == 0) // store first product in Z[z]
251  {
252  job.ec = (col_B->*job.RO)(sum, row_A);
253  if (job.ec != E_NO_ERROR) return;
254  }
255  else // add subsequent product to Z[z]
256  {
257  Cell product; // the result of RO, e.g. of × in +/×
258  job.ec = (col_B->*job.RO)(&product, row_A);
259  if (job.ec != E_NO_ERROR) return;
260 
261  job.ec = (product.*job.LO)(sum, sum);
262  if (job.ec != E_NO_ERROR) return;
263  }
264  }
265  }
266 }

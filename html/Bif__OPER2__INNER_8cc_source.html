33 Token
34 Bif_OPER2_INNER::eval_ALRB(Value_P A, Token & _LO, Token & _RO, Value_P B)
35 {
36  if (!_LO.is_function() || !_RO.is_function()) SYNTAX_ERROR;
37 
38 Function * LO = _LO.get_function();
39 Function * RO = _RO.get_function();
40  Assert1(LO);
41  Assert1(RO);
42 
43  if (LO->get_fun_valence() != 2 || RO->get_fun_valence() != 2) SYNTAX_ERROR;
44  if (!LO->has_result() || !RO->has_result()) DOMAIN_ERROR;
45 
46  if (!A->is_scalar_extensible() && !B->is_scalar_extensible() &&
47  A->get_rank() > 1 && B->get_rank() > 1 &&
48  A->get_shape().get_last_shape_item() !=
49  B->get_shape().get_shape_item(0)) LENGTH_ERROR;
50 
51 Shape shape_A1;
52 ShapeItem len_A = 1;
53  if (!A->is_scalar())
54  {
55  len_A = A->get_last_shape_item();
56  shape_A1 = A->get_shape().without_axis(A->get_rank() - 1);
57  }
58 
59 Shape shape_B1;
60 ShapeItem len_B = 1;
61  if (!B->is_scalar())
62  {
63  len_B = B->get_shape_item(0);
64  shape_B1 = B->get_shape().without_axis(0);
65  }
66 
67  // we do not check len_A == len_B here, since a non-scalar LO may
68  // accept different lengths of its left and right arguments
69 
70 const ShapeItem items_A = shape_A1.get_volume();
71 const ShapeItem items_B = shape_B1.get_volume();
72 
73  if (items_A == 0 || items_B == 0) // empty result
74  {
75  // the outer product portion of LO.RO is empty.
76  // Apply the fill function of RO
77  //
78  const Shape shape_Z = shape_A1 + shape_B1;
79  return fill(shape_Z, A, RO, B, LOC);
80  }
81 
82  if (LO->may_push_SI() || RO->may_push_SI()) // user defined LO or RO
83  {
84  // iso: if A1 and B1 are both vectors, return f/A1 g B1.
85  //
86  if (A->get_rank() <= 1 && B->get_rank() <= 1)
87  return Macro::get_macro(Macro::MAC_Z__vA_LO_INNER_RO_vB)
88  ->eval_ALRB(A, _LO, _RO, B);
89  else
90  return Macro::get_macro(Macro::MAC_Z__A_LO_INNER_RO_B)
91  ->eval_ALRB(A, _LO, _RO, B);
92  }
93 
94 Value_P Z(shape_A1 + shape_B1, LOC);
95 
96  // an important (and the most likely) special case is LO and RO being scalar
97  // functions. This case can be implemented in a far simpler fashion than
98  // the general case.
99  //
100  job.LO = LO->get_scalar_f2();
101  job.RO = RO->get_scalar_f2();
102  if (job.LO && job.RO && A->is_simple() && B->is_simple() && len_A)
103  {
104  job.incA = (A->element_count() == 1) ? 0 : 1;
105  job.incB = (B->element_count() == 1) ? 0 : 1;
106 
107  // len_A must be len_B, unless at least one length is 1
108  //
109  if (len_A != len_B && job.incA && job.incB) LENGTH_ERROR;
110 
111  job.cZ = &Z->get_ravel(0);
112  job.cA = &A->get_ravel(0);
113  job.ZAh = items_A;
114  job.LO_len = A->is_scalar() ? len_B : len_A;
115  job.cB = &B->get_ravel(0);
116  job.ZBl = items_B;
117  job.ec = E_NO_ERROR;
118 
119  scalar_inner_product();
120  if (job.ec != E_NO_ERROR) throw_apl_error(job.ec, LOC);
121 
122  Z->set_default(*B.get(), LOC);
123 
124  Z->check_value(LOC);
125  return Token(TOK_APL_VALUE1, Z);
126  }
127 
128 const bool A_enclosed = A->get_rank() > 1;
129 const bool B_enclosed = B->get_rank() > 1;
130 
131  // enclose last axis of A if necessary
132  //
133  if (A_enclosed)
134  {
135  const Shape last_axis(A->get_rank() - 1);
136  A = Bif_F12_PARTITION::enclose_with_axes(last_axis, A);
137  }
138 
139  // enclose first axis of B if necessary
140  //
141  if (B_enclosed)
142  {
143  const Shape first_axis(0);
144  B = Bif_F12_PARTITION::enclose_with_axes(first_axis, B);
145  }
146 
147  loop (a, items_A)
148  loop (b, items_B)
149  {
150  Value_P RO_A(A, LOC);
151  if (A_enclosed) RO_A = A->get_ravel(a).get_pointer_value();
152 
153  Value_P RO_B(B, LOC);
154  if (B_enclosed) RO_B = B->get_ravel(b).get_pointer_value();
155 
156  const Token T1 = RO->eval_AB(RO_A, RO_B);
157 
158  if (T1.get_tag() == TOK_ERROR) return T1;
159 
160  Value_P A_RO_B = T1.get_apl_val();
161 
162  Cell * cZ = Z->next_ravel();
163  new (cZ) PointerCell(A_RO_B.get(), Z.getref());
164 
165  // at this point Z[z] is the result of a built-in function RO.
166  // Compute LO/Z[z].
167  //
168  Value_P ZZ = cZ->get_pointer_value();
169  const Token T2 = Bif_OPER1_REDUCE::fun->eval_LB(_LO, ZZ);
170  cZ->release(LOC);
171 
172  if (T2.get_tag() == TOK_ERROR) return T2;
173 
174  cZ->init_from_value(T2.get_apl_val().get(), Z.getref(), LOC);
175  }
176 
177  Z->set_default(*B.get(), LOC);
178  Z->check_value(LOC);
179  return Token(TOK_APL_VALUE1, Z);
180 }

182 void
183 Bif_OPER2_INNER::scalar_inner_product() const
184 {
185 #ifdef PERFORMANCE_COUNTERS_WANTED
186 const uint64_t start_1 = cycle_counter();
187 #endif
188 
189  // the empty cases have been ruled out already in inner_product()
190 
191  job.ec = E_NO_ERROR;
192 
193 #if PARALLEL_ENABLED
194  if ( Parallel::run_parallel
195  && Thread_context::get_active_core_count() > 1
196  && job.ZAh * job.ZBl > get_dyadic_threshold())
197  {
198  job.cores = Thread_context::get_active_core_count();
199  Thread_context::do_work = PF_scalar_inner_product;
200  Thread_context::M_fork("scalar_inner_product"); // start pool
201  PF_scalar_inner_product(Thread_context::get_master());
202  Thread_context::M_join();
203  }
204  else
205 #endif // PARALLEL_ENABLED
206  {
207  job.cores = CCNT_1;
208  PF_scalar_inner_product(Thread_context::get_master());
209  }
210 
211 #ifdef PERFORMANCE_COUNTERS_WANTED
212 const uint64_t end_1 = cycle_counter();
213  Performance::fs_OPER2_INNER_AB.add_sample(end_1-start_1, job.ZAh * job.ZBl);
214 #endif
215 }

217 void
218 Bif_OPER2_INNER::PF_scalar_inner_product(Thread_context & tctx)
219 {
220 const ShapeItem Z_len = job.ZAh * job.ZBl;
221 
222 const ShapeItem slice_len = (Z_len + job.cores - 1)/job.cores;
223 ShapeItem z = tctx.get_N() * slice_len;
224 ShapeItem end_z = z + slice_len;
225  if (end_z > Z_len) end_z = Z_len;
226 
227  for (; z < end_z; ++z)
228  {
229  const ShapeItem zah = z/job.ZBl; // z row = A row
230  const ShapeItem zbl = z - zah*job.ZBl; // z column = B column
231  const Cell * row_A = job.cA + job.incA*((zah + 1) * job.LO_len);
232  const Cell * col_B = job.cB + job.incB*(zbl + job.LO_len*job.ZBl);
233 
234  // compute Z[z] ← LO / (row_A RO colB)
235  // e.g. Z[z] ← +/ (row_A × colB)
236  //
237  // we use Z[z] as accumulator for LO /
238  //
239  Cell * sum = job.cZ + z;
240  loop(l, job.LO_len)
241  {
242  row_A -= job.incA;
243  col_B -= job.incB*job.ZBl;
244 
245  if (l == 0) // store first product in Z[z]
246  {
247  job.ec = (col_B->*job.RO)(sum, row_A);
248  if (job.ec != E_NO_ERROR) return;
249  }
250  else // add subsequent product to Z[z]
251  {
252  Cell product; // the result of RO, e.g. of × in +/×
253  job.ec = (col_B->*job.RO)(&product, row_A);
254  if (job.ec != E_NO_ERROR) return;
255 
256  job.ec = (product.*job.LO)(sum, sum);
257  if (job.ec != E_NO_ERROR) return;
258  }
259  }
260  }
261 }

161 bool
162 UserPreferences::is_APL_script(const char * filename)
163 {
164  /* according to man execve():
165 
166  An interpreter script is a text file that has execute permission
167  enabled and whose first line is of the form:
168 
169  #! interpreter [optional-arg]
170 
171  */
172 
173  if (access(filename, X_OK)) return false;
174 int fd = open(filename, O_RDONLY);
175  if (fd == -1) return false;
176 
177 char buf[2];
178 const size_t len = read(fd, buf, sizeof(buf));
179  close(fd);
180 
181  if (len != 2) return false;
182  if (buf[0] != '#') return false;
183  if (buf[1] != '!') return false;
184  return true;
185 }

735 //-----------------------------------------------------------------------------
766 void
767 UserPreferences::expand_argv(int argc, const char ** argv)
768 {
769  loop(a, argc)
770  {
771  original_argv.push_back(argv[a]);
772  expanded_argv.push_back(argv[a]);
773  }
774 
775  if (argc <= 1) // no args at all
776  {
777  return;
778  }
779 
780  if (is_APL_script(argv[1])) // case 2: script without argument
781  {
782  script_argc = 1;
783  return;
784  }
785 
786  if (!is_APL_script(argv[2])) return; // not run from a script
787 
788  // at this point, argv[1] is the optional argument of the interpreter
789  // from the first line of the script, argv[2] is the name of the script,
790  // and argv[3] ... are the arguments given by the user when starting the
791  // script.
792  //
793  // expand argv[1], stopping at --
794  //
795 const char * apl_args = argv[1]; // the args after e.g. /usr/bin/apl
796  script_argc = 2;
797  if (!strchr(apl_args, ' ')) // single option
798  {
799  return;
800  }
801 
802  expanded_argv.erase(expanded_argv.begin() + 1);
803  --script_argc;
804  for (int index = 1;;)
805  {
806  // skip leading whitespace
807  //
808  while (*apl_args && *apl_args <= ' ') ++apl_args;
809 
810  if (*apl_args == 0) break;
811 
812  if (!strcmp(apl_args, "--")) // "--" at end of apl_args
813  {
814  expanded_argv.insert(expanded_argv.begin() + index++, "--");
815  ++script_argc;
816  break; // done
817  }
818 
819  if (!strncmp(apl_args, "-- ", 2) &&
820  apl_args[2] <= ' ') // "--" somewhere in apl_args
821  {
822  expanded_argv.insert(expanded_argv.begin() + index++, "--");
823  ++script_argc;

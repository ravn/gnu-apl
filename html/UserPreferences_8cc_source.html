162 //-----------------------------------------------------------------------------
163 bool
164 UserPreferences::is_APL_script(const char * filename)
165 {
166  /* according to man execve():
167 
168  An interpreter script is a text file that has execute permission
169  enabled and whose first line is of the form:
170 
171  #! interpreter [optional-arg]
172 
173  */
174 
175  if (access(filename, X_OK)) return false;
176 int fd = open(filename, O_RDONLY);
177  if (fd == -1) return false;
178 
179 char buf[2];
180 const size_t len = read(fd, buf, sizeof(buf));
181  close(fd);
182 
183  if (len != 2) return false;
184  if (buf[0] != '#') return false;
185  if (buf[1] != '!') return false;
186  return true;
187 }

189 bool
190 UserPreferences::parse_argv_1()
191 {
192 bool log_startup = false;
193  for (size_t a = 1; a < expanded_argv.size(); )
194  {
195  const char * opt = expanded_argv[a++];
196  const char * val = (a < expanded_argv.size()) ? expanded_argv[a] : 0;
197 
198  if (!strcmp(opt, "-l"))
199  {
200  ++a;
201  if (!val)
202  {
203  CERR << "-l without log level" << endl;
204  exit(a);
205  }
206 
207  log_startup = (atoi(val) == LID_startup);
208  continue;
209  }
210 
211  if (!strcmp(opt, "-p"))
212  {
213  ++a;
214  if (!val)
215  {
216  CERR << "-p without profile number" << endl;
217  exit(a);
218  }
219 
220  user_profile = atoi(val);
221  continue;
222  }
223 
224  if (!strcmp(opt, "-C"))
225  {
226  ++a;
227  if (!val)
228  {
229  CERR << "-C without directory" << endl;
230  exit(a);
231  }
232 
233 #ifndef WINDOWS
234  if (chroot(val))
235  {
236  CERR << "chroot(" << val << ") failed: "
237  << strerror(errno) << endl;
238  exit(a);
239  }
240 #endif // WINDOWS
241 
242  if (chdir("/"))
243  {
244  CERR << "chdir(\"/\") failed: "
245  << strerror(errno) << endl;
246  exit(a);
247  }
248 
249  continue;
250  }
251 
252 #ifndef WINDOWS
253  if (!strcmp(opt, "-u"))
254  {
255  ++a;
256  if (!val)
257  {
258  CERR << "-u without user ID" << endl;
259  exit(a);
260  }
261  if (setuid(strtoll(val, 0, 10)))
262  {
263  CERR << "setuid(" << val << ") failed: "
264  << strerror(errno) << endl;
265  exit(a);
266  }
267  continue;
268  }
269 #endif // WINDOWS
270  }
271 
272  return log_startup;
273 }

866  }
867

869  apl_args[2] <= ' ') // "--" somewhere in apl_args
870  {
871  expanded_argv.insert(expanded_argv.begin() + index++, "--");
872  ++script_argc;
873  apl_args += 2; // skip "--"
874  continue;
875  }
876 
877  const char * arg_end = strchr(apl_args, ' ');
878  if (arg_end) // more arguments
879  {
880  const int arg_len = arg_end - apl_args;
881  const char * arg = strndup(apl_args, arg_len);
882  expanded_argv.insert(expanded_argv.begin() + index++, arg);
883  ++script_argc;
884  apl_args += arg_len;
885  }
886  else // last argument
887  {
888  const char * arg = strdup(apl_args);
889  expanded_argv.insert(expanded_argv.begin() + index++, arg);
890  ++script_argc;
891  break;
892  }
893  }

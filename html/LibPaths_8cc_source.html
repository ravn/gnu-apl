60 //-----------------------------------------------------------------------------
61 void
62 LibPaths::compute_bin_path(const char * argv0, bool logit)
63 {
64  // compute APL_bin_path from argv0
65  //
66  if (strchr(argv0, '/') == 0)
67  {
68  // if argv0 contains no / then realpath() seems to prepend the current
69  // directory to argv0 (which is wrong since argv0 may be in $PATH).
70  //
71  // we fix this by searching argv0 in $PATH
72  //
73  const char * path = getenv("PATH"); // must NOT be modified
74 
75  if (path)
76  {
77  logit && CERR << "initializing paths from $PATH = "
78  << path << endl;
79 
80  // we must not modify path, so we copy it to path1 and
81  // replace the semicolons in path1 by 0. That converts
82  // p1;p2; ... into a sequence of 0-terminated strings
83  // p1 p2 ... The variable next points to the start of each
84  // string.
85  //
86  const size_t plen = strlen(path);
87  std::string path1;
88  path1.reserve(plen + 1);
89  loop(p, (plen + 1)) path1 += path[p];
90  char * next = &path1[0];
91  for (;;)
92  {
93  char * semi = strchr(next, ':');
94  if (semi) *semi = 0;
95  UTF8_string filename;
96  for (const char * n = next; *n; ++n) filename += *n;
97  filename += '/';
98  for (const char * a = argv0; *a; ++a) filename += *a;
99 
100  if (access(filename.c_str(), X_OK) == 0)
101  {
102  strncpy(APL_bin_path, filename.c_str(),
103  sizeof(APL_bin_path) - 1);
104  APL_bin_path[sizeof(APL_bin_path) - 1] = 0;
105 
106  char * slash = strrchr(APL_bin_path, '/');
107  Assert(slash); // due to %s/%s above
108  *slash = 0;
109  APL_bin_name = slash + 1;
110  goto done;
111  }
112 
113  if (semi == 0) break;
114  next = semi + 1;
115  }
116  }
117  else
118  {
119  logit && CERR << "initializing paths from $PATH failed because "
120  "it was not set" << endl;
121  }
122  }
123 
124  unused = realpath(argv0, APL_bin_path);
125  APL_bin_path[APL_PATH_MAX] = 0;
126  {
127  char * slash = strrchr(APL_bin_path, '/');
128  if (slash) { *slash = 0; APL_bin_name = slash + 1; }
129  else { APL_bin_name = APL_bin_path; }
130  }
131 
132  // if we have a PWD and it is a prefix of APL_bin_path then replace PWD
133  // by './'
134  //
135  if (const char * PWD = getenv("PWD")) // we have a pwd
136  {
137  logit && CERR << "initializing paths from $PWD = " << PWD << endl;
138  const int PWD_len = strlen(PWD);
139  if (!strncmp(PWD, APL_bin_path, PWD_len) && PWD_len > 1)
140  {
141  strcpy(APL_bin_path + 1, APL_bin_path + PWD_len);
142  APL_bin_path[0] = '.';
143  }
144  }
145  else
146  {
147  logit && CERR << "initializing paths from $PWD failed because "
148  "it was not set" << endl;
149  }
150 
151 done:
152  logit && CERR << "APL_bin_path is: " << APL_bin_path << endl
153  << "APL_bin_name is: " << APL_bin_name << endl;
154 }

158 {

160 
161  snprintf(filename, sizeof(filename), "%s/workspaces", dir);
162  if (access(filename, F_OK)) return false;
163

165  if (access(filename, F_OK)) return false;
166

169 //-----------------------------------------------------------------------------
170 void
171 LibPaths::search_APL_lib_root()
172 {
173  APL_lib_root[0] = 0;
174 
175 const char * path = getenv("APL_LIB_ROOT");
176  if (path)
177  {
178  unused = realpath(path, APL_lib_root);
179  root_from_env = true;
180  return;
181  }
182 
183  root_from_pwd = true;
184 
185  // search from "." to "/" for a valid lib-root
186  //
187  unused = realpath(".", APL_lib_root);
188  while (strlen(APL_lib_root))
189  {
190  if (is_lib_root(APL_lib_root)) return; // lib-root found
191  if (char * s = strrchr(APL_lib_root, '/')) *s = 0;
192  else if (char * s = strrchr(APL_lib_root, '\\')) *s = 0;
193  else
194  {
195  CERR << "*** Cannot locate APL_lib_root: no / or \\ in "
196  << APL_lib_root << endl;
197  break;
198  }
199  }
200 
201  unused = realpath(".", APL_lib_root);
202 }

217 UTF8_string
218 LibPaths::get_lib_dir(LibRef libref)
219 {
220  switch(lib_dirs[libref].cfg_src)
221  {
222  case LibDir::CSRC_NONE: return UTF8_string();
223 
224  case LibDir::CSRC_ENV:
225  case LibDir::CSRC_PWD: break; // continue below
226 
227  case LibDir::CSRC_PREF_SYS:
228  case LibDir::CSRC_PREF_HOME:
229  case LibDir::CSRC_CMD: return lib_dirs[libref].dir_path;
230  }
231 
232 UTF8_string ret(APL_lib_root);
233  if (libref == LIB0) // workspaces
234  {
235  const UTF8_string subdir("/workspaces");
236  ret.append_UTF8(subdir);
237  }
238  else // wslibN
239  {
240  const UTF8_string subdir("/wslib");
241  ret.append_UTF8(subdir);
242  ret += libref + '0';
243  }
244 
245  return ret;
246 }

248 void
249 LibPaths::maybe_warn_ambiguous(int name_has_extension, const UTF8_string name,
250  const char * ext1, const char * ext2)
251 {
252  if (name_has_extension) return; // extension was provided
253  if (ext2 == 0) return; // no second extension
254 
255 UTF8_string filename_ext2 = name;
256  filename_ext2.append_ASCII(ext2);
257  if (access(filename_ext2.c_str(), F_OK)) return; // not existing
258 
259  CERR << endl
260  << "WARNING: filename " << name << endl
261  << " is ambiguous because another file" << endl << " "
262  << filename_ext2 << endl
263  << " exists as well. Using the first." << endl << endl;
264 }

266 UTF8_string
267 LibPaths::get_lib_filename(LibRef lib, const UTF8_string & name,
268  bool existing, const char * ext1, const char * ext2)
269 {
270  // check if name has one of the extensions ext1 or ext2 already.
271  //
272 int name_has_extension = 0; // assume name has neither extension ext1 nor ext2
273  if (name.ends_with(ext1)) name_has_extension = 1;
274  else if (name.ends_with(ext2)) name_has_extension = 2;
275 
276  if (name.starts_with("/") ||
277  name.starts_with("./") ||
278  name.starts_with("../"))
279  {
280  // paths from / or ./ are fallbacks for the case where the library
281  // path setup is wrong. So that the user can survive by using an
282  // explicit path
283  //
284  if (name_has_extension) return name;
285 
286  UTF8_string filename(name);
287  if (access(filename.c_str(), F_OK) == 0) return filename;
288 
289  if (ext1)
290  {
291  UTF8_string filename_ext1 = name;
292  filename_ext1.append_ASCII(ext1);
293  if (!access(filename_ext1.c_str(), F_OK))
294  {
295  // filename_ext1 exists, but filename_ext2 may exist as well.
296  // warn user unless an explicit extension was given.
297  //
298  maybe_warn_ambiguous(name_has_extension, name, ext1, ext2);
299  return filename_ext1;
300  }
301  }
302 
303  if (ext2)
304  {
305  UTF8_string filename_ext2 = name;
306  filename_ext2.append_ASCII(ext2);
307  if (!access(filename_ext2.c_str(), F_OK)) return filename_ext2;
308  }
309 
310  // neither ext1 nor ext2 worked: return original name
311  //
312  filename = name;
313  return filename;
314  }
315 
316 UTF8_string filename = get_lib_dir(lib);
317  filename += '/';
318  filename.append_UTF8(name);
319 
320  if (name_has_extension) return filename;
321 
322  if (existing)
323  {
324  // file ret is supposed to exist (and will be openend read-only).
325  // If it does return filename otherwise filename.extension.
326  //
327  if (access(filename.c_str(), F_OK) == 0) return filename;
328 
329  if (ext1)
330  {
331  UTF8_string filename_ext1 = filename;
332  filename_ext1.append_ASCII(ext1);
333  if (!access(filename_ext1.c_str(), F_OK))
334  {
335  maybe_warn_ambiguous(name_has_extension,
336  filename, ext1, ext2);
337  return filename_ext1;
338  }
339  }
340 
341  if (ext2)
342  {
343  UTF8_string filename_ext2 = filename;
344  filename_ext2.append_ASCII(ext2);
345  if (!access(filename_ext2.c_str(), F_OK)) return filename_ext2;
346  }
347 
348  return filename; // without ext
349  }
350  else
351  {
352  // file may or may not exist (and will be created if not).
353  // therefore checking the existence does not work.
354  // check that the file ends with ext1 or ext2 if provided
355  //
356  if (name_has_extension) return filename;
357 
358  if (ext1) filename.append_ASCII(ext1);
359  else if (ext2) filename.append_ASCII(ext2);
360  return filename;
361  }
362 }

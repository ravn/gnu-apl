29 Token
30 Bif_F12_INDEX_OF::eval_B(Value_P B)
31 {
32  if (B->get_rank() > 1) RANK_ERROR;
33 
34 const APL_Integer qio = Workspace::get_IO();
35 const ShapeItem ec = B->element_count();
36 
37  if (ec == 1)
38  {
39  // interval (standard ⍳N)
40  //
41  const APL_Integer len = B->get_ravel(0).get_near_int();
42  if (len < 0) DOMAIN_ERROR;
43 
44  Value_P Z(len, LOC);
45 
46  loop(z, len) new (Z->next_ravel()) IntCell(qio + z);
47 
48  Z->check_value(LOC);
49  return Token(TOK_APL_VALUE1, Z);
50  }
51 
52  // generalized ⍳B a la Dyalog APL...
53  //
54  if (ec == 0)
55  {
56  Value_P Z(LOC);
57  new (Z->next_ravel()) PointerCell(Idx0(LOC).get(), Z.getref());
58  Z->check_value(LOC);
59  return Token(TOK_APL_VALUE1, Z);
60  }
61 
62 Shape sh(B.get(), 0);
63  loop(b, ec)
64  {
65  if (sh.get_shape_item(b) < 0) DOMAIN_ERROR;
66  }
67 
68  // at this point sh is correct and ⍳ cannot fail.
69  //
70 Value_P Z(sh, LOC);
71  loop(z, Z->element_count())
72  {
73  Value_P ZZ(sh.get_rank(), LOC);
74  ShapeItem N = z;
75  loop(r, sh.get_rank())
76  {
77  const ShapeItem q = sh.get_shape_item(ec - r - 1);
78  new (&ZZ->get_ravel(ec - r - 1)) IntCell(N%q + qio);
79  N /= q;
80  }
81  ZZ->check_value(LOC);
82  new (Z->next_ravel()) PointerCell(ZZ.get(), Z.getref());
83  }
84 
85  if (Z->element_count() == 0) // empty result
86  {
87  Value_P ZZ(ec, LOC);
88  loop(r, ec) new (ZZ->next_ravel()) IntCell(0);
89  ZZ->check_value(LOC);
90  new (&Z->get_ravel(0)) PointerCell(ZZ.get(), Z.getref()); // prototype
91  }
92 
93  Z->check_value(LOC);
94  return Token(TOK_APL_VALUE1, Z);
95 }

98 Token
99 Bif_F12_INDEX_OF::eval_AB(Value_P A, Value_P B)
100 {
101  // A⍳B (aka. Index of)
102  //
103 const bool simple_result = A->is_scalar_or_vector();
104 const double qct = Workspace::get_CT();
105 const APL_Integer qio = Workspace::get_IO();
106 
107 const ShapeItem len_A = A->element_count();
108 const ShapeItem len_BZ = B->element_count();
109 
110 Value_P Z(B->get_shape(), LOC);
111 
112 #if 1
113  // ⎕RL←42 ◊ D←?100 100⍴100 ◊ T←⎕FIO ¯1 ◊ ⊣(⍳100) ⍳ D ◊ -T-⎕FIO ¯1
114 
115  if (len_A >= 64 && len_BZ > 5)
116  {
117  // array A is being searched len_BZ times. We therefore reduce the
118  // per-item search time from O(len_A) to O(log(len_A)). That costs us
119  // O(len_A × log(len_A)) for sorting A, but hopefully pays off if
120  // len_BZ > log(len_A).
121  //
122  // We don't do that for too small A though, as to compensate for the
123  // start-up cost of the sorting.
124  //
125  const ShapeItem * Idx_A = Cell::sorted_indices(&A->get_ravel(0), len_A,
126  SORT_ASCENDING, 1);
127  loop(bz, len_BZ)
128  {
129  const APL_Integer z = find_B_in_sorted_A(&A->get_ravel(0),
130  len_A, Idx_A,
131  B->get_ravel(bz), qct);
132 
133  if (simple_result) new (Z->next_ravel()) IntCell(qio + z);
134  else if (z == len_A) // not found: set result item to ⍬
135  {
136  Value_P zilde(ShapeItem(0), LOC);
137  new (Z->next_ravel()) PointerCell(zilde.get(), Z.getref());
138  }
139  else // element found (first at z (+⎕IO)
140  {
141  const Shape Sz = A->get_shape().offset_to_index(z, qio);
142  Value_P Vz(LOC, &Sz);
143  new (Z->next_ravel()) PointerCell(Vz.get(), Z.getref());
144  }
145  }
146  delete[] Idx_A;
147  }
148  else
149 #endif
150  {
151  loop(bz, len_BZ)
152  {
153  const APL_Integer z = find_B_in_A(&A->get_ravel(0), len_A,
154  B->get_ravel(bz), qct);
155 
156  if (simple_result) new (Z->next_ravel()) IntCell(qio + z);
157  else if (z == len_A) // not found: set result item to ⍬
158  {
159  Value_P zilde(ShapeItem(0), LOC);
160  new (Z->next_ravel()) PointerCell(zilde.get(), Z.getref());
161  }
162  else // element found (first at z (+⎕IO)
163  {
164  const Shape Sz = A->get_shape().offset_to_index(z, qio);
165  Value_P Vz(LOC, &Sz);
166  new (Z->next_ravel()) PointerCell(Vz.get(), Z.getref());
167  }
168  }
169  }
170 
171  Z->check_value(LOC);
172  return Token(TOK_APL_VALUE1, Z);
173 }

187 ShapeItem
188 Bif_F12_INDEX_OF::find_B_in_sorted_A(const Cell * ravel_A, ShapeItem len_A,
189  const ShapeItem * Idx_A,
190  const Cell & cell_B, double qct)
191 {
192 const ShapeItem * const posp = Heapsort<ShapeItem>::search<const Cell &>(
193  cell_B, Idx_A, len_A, &bs_cmp, ravel_A);
194  if (!posp) return len_A; // cell_B was not found in ravel A
195 
196 ShapeItem pos = Idx_A[posp - Idx_A]; // A[pos] = cell_B within qct
197  Assert(cell_B.equal(ravel_A[pos], qct));
198 
199  // A[pos] = cell_B, but there could be predecessors of pos that also
200  // satisfy A[pos] = cell_B. Search neighbor with smallest index in A.
201  //
202 ShapeItem ret = pos;
203  for (const ShapeItem * posp1 = posp - 1; posp1 >= Idx_A; --posp1)
204  {
205  ShapeItem pos1 = Idx_A[posp1 - Idx_A];
206  const Cell & C1 = ravel_A[pos1];
207  if (!cell_B.equal(C1, qct)) break;
208  if (ret > pos1) ret = pos1;
209  }
210 
211  for (const ShapeItem * posp2 = posp + 1; posp2 < (Idx_A + len_A); ++posp2)
212  {
213  ShapeItem pos2 = Idx_A[posp2 - Idx_A];
214  const Cell & C2 = ravel_A[pos2];
215  if (!cell_B.equal(C2, qct)) break;
216  if (ret > pos2) ret = pos2;
217  }
218 
219  return ret;
220 }

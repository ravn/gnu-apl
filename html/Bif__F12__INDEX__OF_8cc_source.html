29 Token
30 Bif_F12_INDEX_OF::eval_B(Value_P B) const
31 {
32  if (B->get_rank() > 1) RANK_ERROR;
33 
34 const APL_Integer qio = Workspace::get_IO();
35 const ShapeItem ec = B->element_count();
36 
37  if (ec == 1)
38  {
39  // interval (standard ⍳B with scalar or 1-element B)
40  //
41  const APL_Integer len = B->get_cfirst().get_near_int();
42  if (len < 0) DOMAIN_ERROR;
43 
44  Value_P Z(len, LOC);
45 
46  loop(z, len) Z->next_ravel_Int(qio + z);
47 
48  Z->check_value(LOC);
49  return Token(TOK_APL_VALUE1, Z);
50  }
51 
52  // generalized ⍳B a la Dyalog APL...
53  //
54  // ⍴B ←→ ⍴⍴Z, B = ⍴Z
55  if (ec == 0)
56  {
57  Value_P Z(LOC);
58  Z->next_ravel_Pointer(Idx0(LOC).get());
59  Z->check_value(LOC);
60  return Token(TOK_APL_VALUE1, Z);
61  }
62 
63 Shape sh_Z(B.get(), 0);
64  loop(b, ec) if (sh_Z.get_shape_item(b) < 0) DOMAIN_ERROR;
65 
66  // at this point sh is correct and ⍳ cannot fail.
67  //
68 Value_P Z(sh_Z, LOC);
69 const Rank rk_Z = Z->get_rank();
70  loop(z, Z->element_count())
71  {
72  Value_P ZZ(rk_Z, LOC);
73  ShapeItem N = z;
74  ShapeItem zz[rk_Z];
75  loop(r, rk_Z)
76  {
77  const ShapeItem q = sh_Z.get_shape_item(ec - r - 1);
78  zz[ec - r - 1] = N % q + qio;
79  N /= q;
80  }
81 
82  loop(r, rk_Z) ZZ->next_ravel_Int(zz[r]);
83 
84  ZZ->check_value(LOC);
85  Z->next_ravel_Pointer(ZZ.get());
86  }
87 
88  if (Z->element_count() == 0) // empty result
89  {
90  Value_P ZZ(rk_Z, LOC);
91  while (ZZ->more()) ZZ->next_ravel_Int(0);
92  ZZ->check_value(LOC);
93  new (&Z->get_wproto()) PointerCell(ZZ.get(), Z.getref()); // prototype
94  }
95 
96  Z->check_value(LOC);
97  return Token(TOK_APL_VALUE1, Z);
98 }

101 Token
102 Bif_F12_INDEX_OF::eval_AB(Value_P A, Value_P B) const
103 {
104  // A⍳B (aka. Index of)
105  //
106 const bool simple_result = A->is_scalar_or_vector();
107 const double qct = Workspace::get_CT();
108 const APL_Integer qio = Workspace::get_IO();
109 
110 const ShapeItem len_A = A->element_count();
111 const ShapeItem len_BZ = B->element_count();
112 
113 Value_P Z(B->get_shape(), LOC);
114 
115 #if 1
116  // ⎕RL←42 ◊ D←?100 100⍴100 ◊ T←⎕FIO ¯1 ◊ ⊣(⍳100) ⍳ D ◊ -T-⎕FIO ¯1
117 
118  if (len_A >= 64 && len_BZ > 5)
119  {
120  // array A is being searched len_BZ times. We therefore reduce the
121  // per-item search time from O(len_A) to O(log(len_A)). That costs us
122  // O(len_A × log(len_A)) for sorting A, but hopefully pays off if
123  // len_BZ > log(len_A).
124  //
125  // We don't do that for too small A though, as to compensate for the
126  // start-up cost of the sorting.
127  //
128  const ShapeItem * Idx_A = Cell::sorted_indices(&A->get_cfirst(), len_A,
129  SORT_ASCENDING, 1);
130  loop(bz, len_BZ)
131  {
132  const APL_Integer z = find_B_in_sorted_A(&A->get_cfirst(),
133  len_A, Idx_A,
134  B->get_cravel(bz), qct);
135 
136  if (simple_result) Z->next_ravel_Int(qio + z);
137  else if (z == len_A) // not found: set result item to ⍬
138  {
139  Value_P zilde(ShapeItem(0), LOC);
140  Z->next_ravel_Pointer(zilde.get());
141  }
142  else // element found (first at z (+⎕IO)
143  {
144  const Shape Sz = A->get_shape().offset_to_index(z, qio);
145  Value_P Vz(LOC, &Sz);
146  Z->next_ravel_Pointer(Vz.get());
147  }
148  }
149  delete[] Idx_A;
150  }
151  else
152 #endif
153  {
154  loop(bz, len_BZ)
155  {
156  const APL_Integer z = find_B_in_A(&A->get_cfirst(), len_A,
157  B->get_cravel(bz), qct);
158 
159  if (simple_result) Z->next_ravel_Int(qio + z);
160  else if (z == len_A) // not found: set result item to ⍬
161  {
162  Value_P zilde(ShapeItem(0), LOC);
163  Z->next_ravel_Pointer(zilde.get());
164  }
165  else // element found (first at z (+⎕IO)
166  {
167  const Shape Sz = A->get_shape().offset_to_index(z, qio);
168  Value_P Vz(LOC, &Sz);
169  Z->next_ravel_Pointer(Vz.get());
170  }
171  }
172  }
173 
174  Z->check_value(LOC);
175  return Token(TOK_APL_VALUE1, Z);
176 }

191 ShapeItem
192 Bif_F12_INDEX_OF::find_B_in_sorted_A(const Cell * ravel_A, ShapeItem len_A,
193  const ShapeItem * Idx_A,
194  const Cell & cell_B, double qct)
195 {
196 const ShapeItem * const posp = Heapsort<ShapeItem>::search<const Cell &>(
197  cell_B, Idx_A, len_A, &bs_cmp, ravel_A);
198  if (!posp) return len_A; // cell_B was not found in ravel A
199 
200 ShapeItem pos = Idx_A[posp - Idx_A]; // A[pos] = cell_B within qct
201  Assert(cell_B.equal(ravel_A[pos], qct));
202 
203  // A[pos] = cell_B, but there could be predecessors of pos that also
204  // satisfy A[pos] = cell_B. Search neighbor with smallest index in A.
205  //
206 ShapeItem ret = pos;
207  for (const ShapeItem * posp1 = posp - 1; posp1 >= Idx_A; --posp1)
208  {
209  ShapeItem pos1 = Idx_A[posp1 - Idx_A];
210  const Cell & C1 = ravel_A[pos1];
211  if (!cell_B.equal(C1, qct)) break;
212  if (ret > pos1) ret = pos1;
213  }
214 
215  for (const ShapeItem * posp2 = posp + 1; posp2 < (Idx_A + len_A); ++posp2)
216  {
217  ShapeItem pos2 = Idx_A[posp2 - Idx_A];
218  const Cell & C2 = ravel_A[pos2];
219  if (!cell_B.equal(C2, qct)) break;
220  if (ret > pos2) ret = pos2;
221  }
222 
223  return ret;
224 }

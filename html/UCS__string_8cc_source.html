75 UCS_string::UCS_string(const char * cstring)
76 {
77  // calling this constructor with and utf8-encoded C string is usually wrong.
78  //
79  // Instead of: UCS_string(const char * utf8_string)
80  // one should: UCS_string(UTF8_string(utf8_string))
81  //
82  // so that the constructor below takes care of the utf8-decoding.
83  //
84  // For ASCII C strings this constuctor is fine.
85  //
86  create(LOC);
87 
88  for (const char * str = cstring; *str; ++str)
89  {
90  if (0x80 & *cstring) // ASCII
91  {
92  CERR << "non-ASCII char in C-String '" << cstring << "'" << endl;
93  Assert(0 && "Bad C-string");
94  }
95  *this += Unicode(*str);
96  }
97 }
98 //-----------------------------------------------------------------------------
99 UCS_string::UCS_string(const UTF8_string & utf)
100 {
101  create(LOC);
102 
103  Log(LOG_char_conversion)
104  CERR << "UCS_string::UCS_string(): utf = " << utf << endl;
105 
106  for (size_t i = 0; i < utf.size();)
107  {
108  const uint32_t b0 = utf[i++];
109  uint32_t bx = b0;
110  uint32_t more;
111  if ((b0 & 0x80) == 0x00) { more = 0; }
112  else if ((b0 & 0xE0) == 0xC0) { more = 1; bx &= 0x1F; }
113  else if ((b0 & 0xF0) == 0xE0) { more = 2; bx &= 0x0F; }
114  else if ((b0 & 0xF8) == 0xF0) { more = 3; bx &= 0x0E; }
115  else if ((b0 & 0xFC) == 0xF8) { more = 4; bx &= 0x07; }
116  else if ((b0 & 0xFE) == 0xFC) { more = 5; bx &= 0x03; }
117  else
118  {
119  utf.dump_hex(CERR << "Bad UTF8 string: ", 40)
120  << " at " << LOC << endl;
121  Backtrace::show(__FILE__, __LINE__);
122  return;
123  }
124 
125  uint32_t uni = 0;
126  for (; more; --more)
127  {
128  if (i >= utf.size())
129  {
130  utf.dump_hex(CERR << "Truncated UTF8 string: ", 40)
131  << " len " << utf.size() << " at " << LOC << endl;
132  if (utf.size() >= 40)
133  {
134  const UTF8_string end(&utf[utf.size() - 10], 10);
135  end.dump_hex(CERR << endl << "(ending with : ", 20)
136  << ")" << endl;
137  }
138  return;
139  }
140 
141  const UTF8 subc = utf[i++];
142  if ((subc & 0xC0) != 0x80)
143  {
144  utf.dump_hex(CERR << "Bad UTF8 string: ", 40)
145  << " len " << utf.size() << " at " << LOC << endl;
146  if (utf.size() >= 40)
147  {
148  const UTF8_string end(&utf[utf.size() - 10], 10);
149  end.dump_hex(CERR << endl << "(ending with : ", 20)
150  << ")" << endl;
151  }
152  Backtrace::show(__FILE__, __LINE__);
153  return;
154  }
155 
156  bx <<= 6;
157  uni <<= 6;
158  uni |= subc & 0x3F;
159  }
160 
161  append(Unicode(bx | uni));
162  }
163 
164  Log(LOG_char_conversion)
165  CERR << "UCS_string::UCS_string(): ucs = " << *this << endl;
166 }
167 //-----------------------------------------------------------------------------
168 UCS_string::UCS_string(APL_Float value, bool & scaled,
169  const PrintContext & pctx)
170 {
171  create(LOC);
172 
173 int quad_pp = pctx.get_PP();
174  if (quad_pp > MAX_Quad_PP) quad_pp = MAX_Quad_PP;
175  if (quad_pp < MIN_Quad_PP) quad_pp = MIN_Quad_PP;
176 
177 const bool negative = (value < 0.0);
178  if (negative) value = -value;
179 
180 int expo = 0;
181 
182  if (value >= 10.0) // large number, positive exponent
183  {
184  if (value > BIG_FLOAT || !isnormal(value)) // something odd
185  {
186  if (isnormal(value) || isinf(value)) // rather large
187  {
188  if (negative) append_UTF8("Â¯â");
189  else append_UTF8("â");
190  FloatCell::map_FC(*this);
191  }
192  else
193  {
194  append_UTF8("-nan-");
195  }
196  return;
197  }
198 
199  while (value >= 1e16) { value *= 1e-16; expo += 16; }
200  while (value >= 1e4) { value *= 1e-4; expo += 4; }
201  while (value >= 1e1) { value *= 1e-1; ++expo; }
202  }
203  else if (value < 1.0) // small number, negative exponent
204  {
205  if (value < 1e-305) // very small number: make it 0
206  {
207  append(UNI_ASCII_0);
208  return;
209  }
210 
211  while (value < 1e-16) { value *= 1e16; expo -= 16; }
212  while (value < 1e-4) { value *= 1e4; expo -= 4; }
213  while (value < 1.0) { value *= 10.0; --expo; }
214  }
215 
216  // In theory, at this point, 1.0 â¤ value < 10.0. In reality value can
217  // be outside, though, due to rounding errors.
218 
219  // create a string with quad_pp + 1 significant digits.
220  // The last digit is used for rounding and then discarded.
221  //
222 UCS_string digits;
223  loop(d, (quad_pp + 2))
224  {
225  if (value >= 10.0)
226  {
227  // 10.0 or more is a rounding error from 9,999...
228  digits.append(Unicode(10 + '0'));
229  while (digits.size() < (quad_pp + 2)) digits.append(UNI_ASCII_0);
230  break;
231  }
232  else if (value < 0.0)
233  {
234  // less than 0.0 is a rounding error from 0.000...
235  while (digits.size() < (quad_pp + 2)) digits.append(UNI_ASCII_0);
236  break;
237  digits.append(UNI_ASCII_0);
238  value = 0.0;
239  }
240  else
241  {
242  const int dig = int(value);
243  value -= dig;
244  value *= 10.0;
245  digits.append(Unicode(dig + '0'));
246  }
247  }
248 
249  if (digits[0] != '0') digits.pop_back();
250 
251  // round last digit
252  //
253 const Unicode last = digits.back();
254  digits.pop_back();
255 
256  if (last >= '5') digits.back() = Unicode(digits.back() + 1);
257 
258  // adjust carries of 2nd to last digit
259  //
260  for (int d = digits.size() - 1; d > 0; --d) // all but first
261  {
262  if (digits[d] > '9')
263  {
264  digits[d] = Unicode(digits[d] - 10);
265  digits[d - 1] = Unicode(digits[d - 1] + 1);
266  }
267  }
268 
269  // adjust carry of 1st digit
270  //
271  if (digits[0] > '9')
272  {
273  digits[0] = Unicode(digits[0] - 10);
274  digits.insert(0, UNI_ASCII_1);
275  ++expo;
276  digits.pop_back();
277  }
278 
279  // remove trailing zeros
280  //
281  while (digits.size() > 1 && digits.back() == UNI_ASCII_0) digits.pop_back();
282 
283  // force scaled format if:
284  //
285  // value < .00001 ( value has â¥ 5 leading zeroes)
286  // value > 10âquad_pp ( integer larger than âPP)
287  //
288  if ((expo < -6) || (expo > quad_pp)) scaled = true;
289 
290  if (negative) append(UNI_OVERBAR);
291 
292  if (scaled)
293  {
294  append(digits[0]); // integer part
295  if (digits.size() > 1) // fractional part
296  {
297  append(UNI_ASCII_FULLSTOP);
298  loop(d, (digits.size() - 1)) append(digits[d + 1]);
299  }
300  if (expo < 0)
301  {
302  append(UNI_ASCII_E);
303  append(UNI_OVERBAR);
304  append_number(-expo);
305  }
306  else if (expo > 0)
307  {
308  append(UNI_ASCII_E);
309  append_number(expo);
310  }
311  else if (!(pctx.get_style() & PST_NO_EXPO_0)) // expo == 0
312  {
313  append(UNI_ASCII_E);
314  append(UNI_ASCII_0);
315  }
316  }
317  else
318  {
319  if (expo < 0) // 0.000...
320  {
321  append(UNI_ASCII_0);
322  append(UNI_ASCII_FULLSTOP);
323  loop(e, (-(expo + 1))) append(UNI_ASCII_0);
324  append(digits);
325  }
326  else // expo >= 0
327  {
328  loop(e, expo + 1)
329  {
330  if (e < digits.size()) append(digits[e]);
331  else append(UNI_ASCII_0);
332  }
333 
334  if ((expo + 1) < digits.size()) // there are fractional digits
335  {
336  append(UNI_ASCII_FULLSTOP);
337  for (int e = expo + 1; e < digits.size(); ++e)
338  {
339  if (e < digits.size()) append(digits[e]);
340  else break;
341  }
342  }
343  }
344  }
345 
346  FloatCell::map_FC(*this);
347 }
348 //-----------------------------------------------------------------------------
349 UCS_string::UCS_string(const PrintBuffer & pb, Rank rank, int quad_PW)
350 {
351  create(LOC);
352 
353  if (pb.get_height() == 0) return; // empty PrintBuffer
354 
355 const int total_width = pb.get_width(0);
356 
357 std::vector<int> breakpoints;
358  breakpoints.reserve(2*total_width/quad_PW);
359 
360  // print rows, breaking at breakpoints
361  //
362  loop(row, pb.get_height())
363  {
364  if (row) append(UNI_ASCII_LF); // end previous row
365  int col = 0;
366  int b = 0;
367 
368  while (col < total_width)
369  {
370  int chunk_len;
371  if (row == 0) // first row: set up breakpoints
372  {
373  chunk_len = pb.get_line(0).compute_chunk_length(quad_PW,col);
374  breakpoints.push_back(chunk_len);
375  }
376  else
377  {
378  chunk_len = breakpoints[b++];
379  }
380 
381  if (col) append_UTF8("\n ");
382  UCS_string trow(pb.get_line(row), col, chunk_len);
383  trow.remove_trailing_padchars();
384  append(trow);
385 
386  col += chunk_len;
387  }
388  }
389 
390  // replace pad chars with blanks.
391  //
392  loop(u, size())
393  {
394  if (is_iPAD_char(at(u))) at(u) = UNI_ASCII_SPACE;
395  }
396 }

423 //-----------------------------------------------------------------------------
424 int
425 UCS_string::compute_chunk_length(int quad_PW, int col) const
426 {
427 int chunk_len = quad_PW;
428 
429  if (col) chunk_len -= 6; // subsequent line inden
430 
431 int pos = col + chunk_len;
432  if (pos >= size()) return size() - col;
433 
434  while (--pos > col)
435  {
436  const Unicode uni = at(pos);
437  if (uni == UNI_iPAD_U2 || uni == UNI_iPAD_U3)
438  {
439  chunk_len = pos - col + 1;
440  break;
441  }
442  }
443 
444  return chunk_len;
445 }
446 //-----------------------------------------------------------------------------
447 void
448 UCS_string::remove_trailing_padchars()
449 {
450  // remove trailing pad chars from align() and append_string(),
451  // but leave other pad chars intact.
452  // But only if the line has no frame (vert).
453  //
454 
455  // If the line contains UNI_iPAD_L0 (higher dimension separator)
456  // then discard all chars.
457  //
458  loop(u, size())
459  {
460  if (at(u) == UNI_LINE_VERT) break;
461  if (at(u) == UNI_LINE_VERT2) break;
462  if (at(u) == UNI_iPAD_L0)
463  {
464  clear();
465  return;
466  }
467  }
468 
469  while (size())
470  {
471  const Unicode last = back();
472  if (last == UNI_iPAD_L0 ||
473  last == UNI_iPAD_L1 ||
474  last == UNI_iPAD_L2 ||
475  last == UNI_iPAD_L3 ||
476  last == UNI_iPAD_L4 ||
477  last == UNI_iPAD_U7)
478  pop_back();
479  else
480  break;
481  }
482 }

485 UCS_string::remove_trailing_whitespaces()
486 {
487  while (size() && back() <= UNI_ASCII_SPACE) pop_back();
488 }
489 //-----------------------------------------------------------------------------
490 void
491 UCS_string::remove_leading_whitespaces()
492 {
493 int count = 0;
494  loop(s, size())
495  {
496  if (at(s) <= UNI_ASCII_SPACE) ++count;
497  else break;
498  }
499 
500  if (count == 0) return; // no leading whitspaces
501  if (count == size()) clear(); // only whitespaces
502  else basic_string::erase(0, count);
503 }
504 //-----------------------------------------------------------------------------
505 void
506 UCS_string::split_ws(UCS_string & rest)
507 {
508  remove_leading_and_trailing_whitespaces();
509 
510  loop(clen, size())
511  {
512  if (Avec::is_white(at(clen))) // whilespace: end of command
513  {
514  ShapeItem arg = clen;
515  while (arg < size() && Avec::is_white(at(arg))) ++arg;
516  while (arg < size()) rest.append(at(arg++));
517  resize(clen);
518  return;
519  }
520  }
521 }

605 //-----------------------------------------------------------------------------
606 bool
607 UCS_string::starts_iwith(const char * prefix) const
608 {
609  loop(s, size())
610  {
611  char pc = *prefix++;
612  if (pc == 0) return true; // prefix matches this string.
613  if (pc >= 'a' && pc <= 'z') pc -= 'a' - 'A';
614 
615  int uni = at(s);
616  if (uni >= 'a' && uni <= 'z') uni -= 'a' - 'A';
617 
618  if (uni != Unicode(pc)) return false;
619  }
620 
621  return *prefix == 0;
622 }
623 //-----------------------------------------------------------------------------
624 bool
625 UCS_string::starts_iwith(const UCS_string & prefix) const
626 {
627  if (prefix.size() > size()) return false;
628 
629  loop(p, prefix.size())
630  {
631  int c1 = at(p);
632  int c2 = prefix[p];
633  if (c1 >= 'a' && c1 <= 'z') c1 -= 'a' - 'A';
634  if (c2 >= 'a' && c2 <= 'z') c2 -= 'a' - 'A';
635  if (c1 != c2) return false;
636  }
637 
638  return true;
639 }

699 //-----------------------------------------------------------------------------
700 ShapeItem
701 UCS_string::double_quote_count(bool in_quote2) const
702 {
703 ShapeItem count = 0;
704 bool in_quote1 = false;
705  loop(s, size())
706  {
707  const Unicode uni = at(s);
708  switch(uni)
709  {
710  case UNI_SINGLE_QUOTE:
711  if (!in_quote2) in_quote1 = ! in_quote1;
712  break;
713 
714  case UNI_ASCII_DOUBLE_QUOTE:
715  if (!in_quote1)
716  {
717  ++count;
718  in_quote2 = ! in_quote2;
719  }
720  break;
721 
722  case UNI_ASCII_BACKSLASH:
723  if (in_quote2) ++s; // ignore next char inside ""
724  break;
725 
726  case UNI_ASCII_NUMBER_SIGN:
727  case UNI_COMMENT:
728  if (!(in_quote1 || in_quote2)) return count;
729 
730  default: ;
731  }
732  }
733 
734  return count;
735 }
736 //-----------------------------------------------------------------------------
737 ShapeItem
738 UCS_string::double_quote_first() const
739 {
740 bool in_quote1 = false;
741 bool in_quote2 = true;
742  loop(s, size())
743  {
744  const Unicode uni = at(s);
745  switch(uni)
746  {
747  case UNI_SINGLE_QUOTE:
748  if (!in_quote2) in_quote1 = ! in_quote1;
749  break;
750 
751  case UNI_ASCII_DOUBLE_QUOTE:
752  if (!in_quote1) return s;
753  break;
754 
755  case UNI_ASCII_BACKSLASH:
756  if (in_quote2) ++s; // ignore next char inside ""
757  break;
758 
759  case UNI_ASCII_NUMBER_SIGN:

884 //-----------------------------------------------------------------------------
885 UCS_string
886 UCS_string::un_escape(bool double_quoted, bool keep_LF) const
887 {
888 const char * hex = "0123456789abcdef";
889 UCS_string ret;
890  ret.reserve(size());
891 
892  if (double_quoted)
893  {
894  loop(s, size())
895  {
896  const Unicode uni = at(s);
897  if (uni != UNI_ASCII_BACKSLASH) // normal char
898  {
899  ret.append(uni);
900  continue;
901  }
902 
903  if (s >= (size() - 1)) // \ at end of string
904  {
905  ret.append(UNI_ASCII_BACKSLASH);
906  break;
907  }
908 
909  const Unicode uni1 = at(++s);
910  switch(uni1)
911  {
912  case UNI_ASCII_a: ret << UNI_ASCII_BEL; continue;
913  case UNI_ASCII_b: ret << UNI_ASCII_BS; continue;
914  case UNI_ASCII_f: ret << UNI_ASCII_FF; continue;
915  case UNI_ASCII_n: if (keep_LF) break;
916  ret << UNI_ASCII_LF; continue;
917  case UNI_ASCII_r: ret << UNI_ASCII_CR; continue;
918  case UNI_ASCII_t: ret << UNI_ASCII_BS; continue;
919  case UNI_ASCII_v: ret << UNI_ASCII_VT; continue;
920  case UNI_ASCII_DOUBLE_QUOTE:
921  case UNI_ASCII_BACKSLASH:
922  ret << uni1; continue;
923  default: break;
924  }
925 
926  int max_len = 0;
927  if (uni1 == UNI_ASCII_u)
928  {
929  max_len = 4;
930  }
931  else if (uni1 == UNI_ASCII_x)
932  {
933  max_len = 2;
934  }
935  else // \n or \": keep them escaped
936  {
937  ret.append(uni);
938  ret.append(uni1);
939  continue;
940  }
941 
942  // \x or \u
943  //
944  int value = 0;
945  loop(m, max_len)
946  {
947  if (s >= (size() - 1)) break;
948  const int dig = at(s+1);
949  const char * pos = strchr(hex, dig);
950  if (pos == 0) break; // non-hex character
951 
952  value = value << 4 | (pos - hex);
953  ++s;
954  }
955  ret.append(Unicode(value));
956  }
957  }
958  else
959  {
960  bool got_quote = false;
961  loop(s, size())
962  {
963  const Unicode uni = at(s);
964  if (uni == UNI_SINGLE_QUOTE)
965  {
966  if (got_quote)
967  {
968  ret.append(UNI_SINGLE_QUOTE);
969  got_quote = false;
970  }
971  else
972  {
973  ret.append(UNI_SINGLE_QUOTE);
974  got_quote = true;
975  }
976  }
977  else
978  {
979  if (got_quote) ret.append(UNI_SINGLE_QUOTE); // mal-formed
980  ret.append(uni);
981  got_quote = false;
982  }
983  }
984  }
985 
986  return ret;
987 }

1353 //-----------------------------------------------------------------------------
1354 UCS_string
1355 UCS_string::from_double_fixed_prec(APL_Float v, int fract_digits)
1356 {
1357 UCS_string ret;
1358 
1359  if (v < 0.0) { ret.append(UNI_OVERBAR); v = - v; }
1360 
1361  // in the loop below, there could be rounding errors when casting float
1362  // to int. We therefore increase v slighly (by 0.3 of the rounded digit)
1363  // to avoid that.
1364  //
1365  v += 0.03 * pow(10.0, -fract_digits);
1366 
1367  ret.append(from_big(v)); // leaves fractional part of v in v
1368 
1369  ret.append(UNI_ASCII_FULLSTOP);
1370 
1371  loop(f, fract_digits + 1)
1372  {
1373  v = v * 10.0;
1374  const int vv = v; // subject to rounding errors!
1375  ret.append(Unicode(UNI_ASCII_0 + vv));
1376  v -= vv;
1377  }
1378 
1379  ret.round_last_digit();
1380  return ret;
1381 }
1382 //-----------------------------------------------------------------------------
1383 void
1384 UCS_string::round_last_digit()
1385 {
1386  Assert1(size() > 1);
1387  if (back() >= UNI_ASCII_5) // round up
1388  {
1389  for (int q = size() - 2; q >= 0; --q)
1390  {
1391  const Unicode cc = at(q);
1392  if (cc < UNI_ASCII_0) continue; // not a digit
1393  if (cc > UNI_ASCII_9) continue; // not a digit
1394 
1395  at(q) = Unicode(cc + 1); // round up
1396  if (cc != UNI_ASCII_9) break; // 0-8 rounded up: stop
1397 
1398  at(q) = UNI_ASCII_0; // 9 rounded up: say 0 and repeat
1399  if (q) continue; // not first difit
1400 
1401  // something like 9.xxx has been rounded up to, say, 0.xxx
1402  // but should be 10.xxx Fix it.
1403  //
1404  for (int d = size() - 1; d > 0; --d) at(d) = at(d - 1);
1405  at(0) = UNI_ASCII_1;
1406  }
1407  }
1408 
1409  pop_back();
1410  if (back() == UNI_ASCII_FULLSTOP) pop_back();
1411 }

79 bool
80 PointerCell::equal(const Cell & other, double qct) const
81 {
82  if (!other.is_pointer_cell()) return false;
83 
84 Value_P A = get_pointer_value();
85 Value_P B = other.get_pointer_value();
86 
87  if (!A->same_shape(*B)) return false;
88 
89 const ShapeItem count = A->nz_element_count();
90  loop(c, count)
91  if (!A->get_ravel(c).equal(B->get_ravel(c), qct)) return false;
92 
93  return true;
94 }

103 //-----------------------------------------------------------------------------
104 Comp_result
105 PointerCell::compare(const Cell & other) const
106 {
107  if (other.get_cell_type() & CT_SIMPLE) // nested > numeric > char
108  return COMP_GT;
109 
110  if (other.get_cell_type() != CT_POINTER) DOMAIN_ERROR;
111 
112  // at this point both cells are pointer cells.
113  //
114 Value_P v1 = get_pointer_value();
115 Value_P v2 = other.get_pointer_value();
116 
117  // compare ranks
118  //
119  if (v1->get_rank() != v2->get_rank()) // ranks differ
120  return v1->get_rank() > v2->get_rank() ? COMP_GT : COMP_LT;
121 
122  // same rank, compare shapes
123  //
124  loop(r, v1->get_rank())
125  {
126  const ShapeItem axis1 = v1->get_shape_item(r);
127  const ShapeItem axis2 = v2->get_shape_item(r);
128  if (axis1 != axis2) // axis r differs
129  return axis1 > axis2 ? COMP_GT : COMP_LT;
130  }
131 
132  // same rank and shape, compare ravel
133  //
134 const Cell * C1 = &v1->get_ravel(0);
135 const Cell * C2 = &v2->get_ravel(0);
136  loop(e, v1->nz_element_count())
137  {
138  if (const Comp_result comp = C1++->compare(*C2++)) return comp;
139  }
140 
141  // everthing equal
142  //
143  return COMP_EQ;
144 }

166 PrintBuffer
167 PointerCell::character_representation(const PrintContext & pctx) const
168 {
169 Value_P val = get_pointer_value();
170  Assert(!!val);
171 
172  if (pctx.get_style() & PST_QUOTE_CHARS)
173  {
174  if (val->is_char_vector()) return PrintBuffer(val.getref(), pctx, 0);
175  }
176 
177  if (pctx.get_style() == PR_APL_FUN) // APL function display
178  {
179  const ShapeItem ec = val->element_count();
180  UCS_string ucs;
181 
182  if (val->is_char_vector())
183  {
184  ucs.append(UNI_SINGLE_QUOTE);
185  loop(e, ec)
186  {
187  const Unicode uni = val->get_ravel(e).get_char_value();
188  ucs.append(uni);
189  if (uni == UNI_SINGLE_QUOTE) ucs.append(uni); // ' -> ''
190  }
191  ucs.append(UNI_SINGLE_QUOTE);
192  }
193  else
194  {
195  ucs.append(UNI_ASCII_L_PARENT);
196  loop(e, ec)
197  {
198  PrintBuffer pb = val->get_ravel(e).
199  character_representation(pctx);
200  ucs.append(UCS_string(pb, 0, Workspace::get_PW()));
201 
202  if (e < ec - 1) ucs.append(UNI_ASCII_SPACE);
203  }
204  ucs.append(UNI_ASCII_R_PARENT);
205  }
206 
207  ColInfo ci;
208  PrintBuffer ret(ucs, ci);
209  ret.get_info().int_len = ret.get_width(0);
210  ret.get_info().real_len = ret.get_width(0);
211  return ret;
212  }
213 
214 PrintBuffer ret(*val, pctx, 0);
215  ret.get_info().flags &= ~CT_MASK;
216  ret.get_info().flags |= CT_POINTER;
217 
218  if (pctx.get_style() & PST_CS_INNER)
219  {
220  const ShapeItem ec = val->element_count();
221  if (ec == 0) // empty value
222  {
223  int style = pctx.get_style();
224  loop(r, val->get_rank())
225  {
226  if (val->get_shape_item(r) == 0) // an empty axis
227  {
228  if (r == val->get_rank() - 1) style |= PST_EMPTY_LAST;
229  else style |= PST_EMPTY_NLAST;
230  }
231  }
232 
233  // do A←(1⌈⍴A)⍴⊂↑A
234  Value_P proto = val->prototype(LOC);
235 
236  // compute a shape like ⍴ val but 0 replaced with 1.
237  //
238  Shape sh(val->get_shape());
239  loop(r, sh.get_rank())
240  {
241  if (sh.get_shape_item(r) == 0) sh.set_shape_item(r, 1);
242  }
243 
244  if (sh.get_volume() == 111) // one prototype
245  {
246  ret = PrintBuffer(*proto, pctx, 0);
247  ret.add_frame(PrintStyle(style), proto->get_shape(),
248  proto->compute_depth());
249  }
250  else // several prototypes
251  {
252  Value_P proto_reshaped(sh, LOC);
253  Cell * c = &proto_reshaped->get_ravel(0);
254  const ShapeItem len = proto_reshaped->nz_element_count();
255 
256  // store proto in the first ravel item, and copies of proto in
257  // the subsequent ravel items
258  //
259  loop(rv, len)
260  if (proto->is_simple_scalar())
261  c++->init(proto->get_ravel(0),
262  proto_reshaped.getref(), LOC);
263  else
264  new (c++) PointerCell(proto->clone(LOC).get(),
265  proto_reshaped.getref());
266 
267  proto_reshaped->check_value(LOC);
268  ret = PrintBuffer(*proto_reshaped, pctx, 0);
269  ret.add_frame(PrintStyle(style), proto_reshaped->get_shape(),
270  proto_reshaped->compute_depth());
271  }
272  }
273  else
274  {
275  if (!(pctx.get_style() & PST_QUOTE_CHARS && val->is_char_array()))
276  ret.add_frame(pctx.get_style(), val->get_shape(),
277  val->compute_depth());
278  }
279  }
280 
281  ret.get_info().int_len = ret.get_width(0);
282  ret.get_info().real_len = ret.get_width(0);
283  return ret;
284 }

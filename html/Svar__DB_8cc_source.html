65 //-----------------------------------------------------------------------------
66 bool
67 Svar_DB::start_APserver(const char * server_sockname,
68  const char * bin_dir, bool logit)
69 {
70  // bin_dir is the directory where the apl interpreter binary lives.
71  // The APserver then lives in:
72  //
73  // bin_dir/ if apl was built and installed, or
74  // bin_dir/APs/ if apl was started from the src directory in the source tree
75  //
76  // set APserver_path to the case that applies.
77  //
78 char APserver_path[APL_PATH_MAX + 1];
79 const int slen = snprintf(APserver_path, APL_PATH_MAX, "%s/APserver", bin_dir);
80  if (slen >= APL_PATH_MAX) APserver_path[APL_PATH_MAX] = 0;
81  else APserver_path[slen] = 0;
82  if (access(APserver_path, X_OK) != 0) // no APserver in bin_dir
83  {
84  logit && get_CERR() << " Executable " << APserver_path
85  << " not found (this is OK when apl was started\n"
86  " from the src directory): " << strerror(errno) << endl;
87 
88  const int slen = snprintf(APserver_path, APL_PATH_MAX,
89  "%s/APs/APserver", bin_dir);
90  if (slen >= APL_PATH_MAX) APserver_path[APL_PATH_MAX] = 0;
91  else APserver_path[slen] = 0;
92  if (access(APserver_path, X_OK) != 0) // no APs/APserver either
93  {
94  get_CERR() << "Executable " << APserver_path << " not found.\n"
95 "This could mean that 'apl' was not installed ('make install') or that it\n"
96 "was started in a non-standard way. The expected location of APserver is \n"
97 "either the same directory as the binary 'apl' or the subdirectory 'APs' of\n"
98 "that directory (the directory should also be in $PATH)." << endl;
99 
100  return true; // error
101  }
102  }
103 
104  logit && get_CERR() << "Found " << APserver_path << endl;
105 
106 char popen_args[APL_PATH_MAX + 1];
107  {
108  int slen;
109  if (server_sockname)
110  slen = snprintf(popen_args, APL_PATH_MAX,
111  "%s --path %s --auto", APserver_path, server_sockname);
112  else
113  slen = snprintf(popen_args, APL_PATH_MAX,
114  "%s --port %u --auto", APserver_path, APserver_port);
115  if (slen >= APL_PATH_MAX) popen_args[APL_PATH_MAX] = 0;
116  }
117 
118  logit && get_CERR() << "Starting " << popen_args << "..." << endl;
119 
120 FILE * fp = popen(popen_args, "r");
121  if (fp == 0)
122  {
123  get_CERR() << "popen(" << popen_args << " failed: " << strerror(errno)
124  << endl;
125  return true; // error
126  }
127 
128  // wait until the parent process (= this process) in APserver returns.
129  // APserver does not really output anything (and we would see if it would),
130  // but the interesting part is the EOF of the parent process in APserfver.
131  //
132  for (int cc; (cc = getc(fp)) != EOF;)
133  {
134  logit && get_CERR() << char(cc);
135  }
136 
137 const int APserver_result = pclose(fp);
138 
139  logit && get_CERR() << endl;
140 
141  if (APserver_result)
142  {
143  get_CERR() << "pclose(APserver) returned error: "
144  << strerror(errno) << endl;
145  }
146 
147  return false; // success
148 }
149 //-----------------------------------------------------------------------------
150 TCP_socket
151 Svar_DB::connect_to_APserver(const char * bin_dir, const char * prog,
152  int retry_max, bool logit)
153 {
154 int sock = NO_TCP_SOCKET;
155 const char * server_sockname = APSERVER_PATH;
156 char peer[100];
157 
158  // we use AF_UNIX sockets if the platform supports it and unix_socket_name
159  // is provided. Otherwise fall back to TCP.
160  //
161 #if HAVE_SYS_UN_H && APSERVER_TRANSPORT != 0
162  {
163  logit && get_CERR() << prog
164  << ": Using AF_UNIX socket towards APserver..."
165  << endl;
166  sock = socket(AF_UNIX, SOCK_STREAM, 0);
167  if (sock == NO_TCP_SOCKET)
168  {
169  get_CERR() << prog
170  << ": socket(AF_UNIX, SOCK_STREAM, 0) failed at "
171  << LOC << endl;
172  return NO_TCP_SOCKET;
173  }
174 
175  const unsigned int slen = snprintf(peer, sizeof(peer),
176  "%s", server_sockname);
177  if (slen >= sizeof(peer)) peer[sizeof(peer) - 1] = 0;
178  }
179 #else // use TCP
180  {
181  server_sockname = 0;
182  logit && get_CERR() << "Using TCP socket towards APserver..."
183  << endl;
184  sock = TCP_socket(socket(AF_INET, SOCK_STREAM, 0));
185  if (sock == NO_TCP_SOCKET)
186  {
187  get_CERR() << "*** socket(AF_INET, SOCK_STREAM, 0) failed at "
188  << LOC << endl;
189  return NO_TCP_SOCKET;
190  }
191 
192  // disable nagle
193  {
194  const int ndelay = 1;
195  setsockopt(sock, 6, TCP_NODELAY,
196  reinterpret_cast<const char *>(&ndelay), sizeof(int));
197  }
198 
199  // bind local port to 127.0.0.1
200  //
201  SockAddr local;
202  memset(&local, 0, sizeof(SockAddr));
203  local.inet.sin_family = AF_INET;
204  local.inet.sin_addr.s_addr = htonl(0x7F000001);
205 
206  if (::bind(sock, &local.addr, sizeof(sockaddr_in)))
207  {
208  get_CERR() << "bind(127.0.0.1) failed: "
209  << strerror(errno) << endl;
210  ::close(sock);
211  return NO_TCP_SOCKET;
212  }
213 
214  const unsigned int slen = snprintf(peer, sizeof(peer),
215  "127.0.0.1 TCP port %d", APserver_port);
216  if (slen >= sizeof(peer)) peer[sizeof(peer) - 1] = 0;
217  }
218 #endif
219 
220  // We try to connect to the APserver. If that fails then no
221  // APserver is running; we fork one and try again.
222  //
223  // If emacs mode is enabled then we try a little longer since emacs starts
224  // up at the same time (and we know that emacs is slow).
225  //
226  loop(retry, retry_max)
227  {
228 #if HAVE_SYS_UN_H
229  if (server_sockname)
230  {
231  SockAddr remote;
232  memset(&remote, 0, sizeof(SockAddr));
233  remote.uNix.sun_family = AF_UNIX;
234  strcpy(remote.uNix.sun_path + ABSTRACT_OFFSET, server_sockname);
235 
236  if (::connect(sock, &remote.addr, sizeof(sockaddr_un)) == 0)
237  break; // success
238  }
239  else // TCP
240 #endif
241  {
242  SockAddr remote;
243  memset(&remote, 0, sizeof(sockaddr_in));
244  remote.inet.sin_family = AF_INET;
245  remote.inet.sin_port = htons(APserver_port);
246  remote.inet.sin_addr.s_addr = htonl(0x7F000001);
247 
248  if (::connect(sock, &remote.addr, sizeof(sockaddr_in)) == 0)
249  break; // success
250  }
251 
252  // ::connect() to APserver failed. If
253  // then most likely no APserver was started and we do it.
254  //
255  if (logit) get_CERR() << "connecting to " << peer
256  << " failed." << endl;
257 
258  if (retry == 0) // first attempt
259  {
260  // this was the first ::connect() that has failed.
261  // most likely no APserver was started yet and we do it now
262  //
263  if (logit) get_CERR() <<
264 " (the first ::connect() to APserver is expected to fail, unless\n" <<
265 " APserver was started manually)\n" <<
266 "Starting a new APserver listening on " << peer << endl;
267 
268  if (start_APserver(server_sockname, bin_dir, logit))
269  {
270  // starting of APserver failed: no point to try longer
271  //
272  ::close(sock);
273  return NO_TCP_SOCKET;
274  }
275 
276  usleep(300000); // give APserver some time to start up
277  continue;
278  }
279 
280  if ((retry == (retry_max - 1)) && bin_dir) // last attempt: give up
281  {
282  get_CERR() <<
283  "::connect() to supposedly existing APserver failed: "
284  << strerror(errno) << endl;
285 
286  ::close(sock);
287  return NO_TCP_SOCKET;
288  }
289 
290  if (logit) get_CERR() <<
291  " (::connect() should succeed eventually. This was attempt "
292  << retry << " of " << retry_max << ")" << endl;
293 
294  usleep(200000); // more time for APserver to start up
295  }
296 
297  // at this point sock is != NO_TCP_SOCKET and connected.
298  //
299  usleep(50000);
300  logit && get_CERR() << "connected to APserver, socket is " << sock << endl;
301 
302  return TCP_socket(sock);
303 }

321 Svar_record_P::Svar_record_P(SV_key key)
322 {
323  cache.clear();
324 
325  if (!Svar_DB::APserver_available()) return;
326 
327 const int sock = Svar_DB::get_DB_tcp();
328 
329  { READ_SVAR_RECORD_c request(sock, key); }
330 
331 char * del = 0;
332 char buffer[2*MAX_SIGNAL_CLASS_SIZE + sizeof(Svar_record)];
333 ostream * log = (LOG_startup != 0 || LOG_Svar_DB_signals != 0) ? & cerr : 0;
334 const char * err_loc = 0;
335 Signal_base * response = Signal_base::recv_TCP(sock, buffer, sizeof(buffer),
336  del, log, &err_loc);
337  if (response)
338  {
339  memcpy(static_cast<void *>(&cache),
340  response->get__SVAR_RECORD_IS__record().data(),
341  sizeof(Svar_record));
342 
343  delete response;
344  }
345  else get_CERR() << "Svar_record_P() failed at " << LOC << endl;
346  if (del) delete del;
347 }

349 void
350 Svar_DB::init(const char * bin_dir, const char * prog, int retry_max,
351  bool logit, bool do_svars)
352 {
353  if (!do_svars) // shared variables disabled
354  {
355  if (logit)
356  get_CERR() << "Not opening shared memory because command "
357  "line option --noSV (or similar) was given." << endl;
358  return;
359  }
360 
361  DB_tcp = connect_to_APserver(bin_dir, prog, retry_max, logit);
362  if (APserver_available())
363  {
364  if (logit) get_CERR() << "using Svar_DB on APserver!" << endl;
365  }
366 }

368 SV_key
369 Svar_DB::match_or_make(const uint32_t * UCS_varname, const AP_num3 & to,
370  const Svar_partner & from)
371 {
372 const TCP_socket tcp = get_Svar_DB_tcp(__FUNCTION__);
373  if (tcp == NO_TCP_SOCKET) return 0;
374 
375 
376 uint32_t vname1[MAX_SVAR_NAMELEN];
377  memset(vname1, 0, sizeof(vname1));
378  loop(v, MAX_SVAR_NAMELEN)
379  {
380  if (*UCS_varname) vname1[v] = *UCS_varname++;
381  else break;
382  }
383 
384 string vname(charP(vname1), MAX_SVAR_NAMELEN*sizeof(uint32_t));
385 
386 MATCH_OR_MAKE_c request(tcp, vname,
387  to.proc, to.parent, to.grand,
388  from.id.proc, from.id.parent, from.id.grand);
389 
390 char * del = 0;
391 char buffer[2*MAX_SIGNAL_CLASS_SIZE + 16];
392 const char * err_loc = 0;
393 Signal_base * response = Signal_base::recv_TCP(tcp, buffer, sizeof(buffer),
394  del, 0, &err_loc);
395 
396  if (response)
397  {
398  const SV_key ret = response->get__MATCH_OR_MAKE_RESULT__key();
399  delete response;
400  return ret;
401  }
402 
403  else return 0;
404 }

217 //-----------------------------------------------------------------------------
218 APL_value
219 assign_var(const unsigned int * var_name_ucs, int rank, uint64_t * shape)
220 {
221 Shape sh;
222  loop(r, rank) sh.add_shape_item(*shape++);
223 
224 Value_P Z(sh, LOC);
225  loop(z, Z->nz_element_count()) new (Z->next_ravel()) IntCell(0);
226  Z->check_value(LOC);
227 
228  if (var_name_ucs == 0)
229  {
230  // the caller wants only q value initialized to 0 without
231  // assigning it to a value. We have to increment the owner count
232  // and the caller is responsiblr fopr decrementing it when the
233  // value is no longer needed.
234  //
235  Z.get()->increment_owner_count(LOC); // keep value
236  return Z.get();
237  }
238 
239  if (!Avec::is_first_symbol_char(Unicode(*var_name_ucs)))
240  return 0;
241 
242 UCS_string var_name;
243  var_name.reserve(40);
244  while (*var_name_ucs)
245  {
246  const Unicode uni = Unicode(*var_name_ucs++);
247  if (!Avec::is_symbol_char(uni)) return 0;
248  var_name.append(uni);
249  }
250 
251 Symbol * symbol = Workspace::lookup_symbol(var_name);
252  if (!symbol) return 0;
253  if (!symbol->can_be_assigned()) return 0;
254 
255  symbol->assign(Z, false, LOC);
256 
257  // at this point owner_count shall be at least 2 so we can return a pointer
258  // to Z without Z gettting deleted when we return from this function;
259  //
260  if (Z->get_owner_count() < 2) return 0;
261  return Z.get();
262 }

327  if (cell->is_pointer_cell())
328  {
329  Value * v = cell->get_pointer_value().get();
330  v->decrement_owner_count(LOC);

333  if (new_value->is_simple_scalar()) // e.g. ⊂5 is 5

337  else if (new_value->is_scalar()) // e.g. ⊂⊂5 is ⊂5

340  if (!src.is_pointer_cell()) DOMAIN_ERROR;
341  Value_P sub = src.get_pointer_value()->clone(LOC);
342  new (cell) PointerCell(sub.get(), *val);
343  }
344  else
345  {
346  Value_P sub = new_value->clone(LOC);
347  new (cell) PointerCell(sub.get(), *val);
348  }
349 }

424 //-----------------------------------------------------------------------------
425 APL_function
426 get_function_ucs(const unsigned int * name, APL_function * L, APL_function * R)
427 {
428 UCS_string function_ucs;
429  function_ucs.reserve(40);
430  while (*name) function_ucs.append(Unicode(*name++));
431 
432  if (function_ucs.size() == 0) return 0; // empty name
433 
434 Tokenizer tokenizer(PM_EXECUTE, LOC, false);
435 Token_string tos;
436  if (tokenizer.tokenize(function_ucs, tos) != E_NO_ERROR) return 0;
437 
438  // resolve user defined names to user defined functions
439  //
440  for (int j = 0; j < int(tos.size()); ++j)
441  {
442  if (tos[j].get_ValueType() == TV_SYM) // user defined function
443  {
444  Symbol * sym = tos[j].get_sym_ptr();
445  if (sym == 0) return 0;
446  sym->resolve(tos[j], false);
447  }
448  }
449 
450  if (tos.size() == 1) // single function
451  {
452  if (L) return 0; // but fun and operator was requested
453 
454  switch(tos[0].get_Class())
455  {
456  case TC_FUN0:
457  case TC_FUN2:
458  case TC_OPER1:
459  case TC_OPER2: return tos[0].get_function();
460  default: return 0; // error
461  }
462  }
463  else if (tos.size() == 2) // got left function and operator
464  {
465  if (!L) return 0; // but only left fun was requested
466  if (R) return 0; // but right fun is missing
467 
468  switch(tos[1].get_Class())
469  {
470  case TC_OPER1: *L = tos[0].get_function(); break;
471  default: return 0; // error
472  }
473 
474  switch(tos[0].get_Class())
475  {
476  case TC_FUN12:
477  case TC_OPER1: return tos[1].get_function();
478  default: return 0;
479  }
480  }
481  else if (tos.size() == 2) // got left and right functions and operator
482  {
483  if (!L) return 0; // but only fun was requested
484  if (!R) return 0; // but only fun was requested
485 
486  switch(tos[0].get_Class())
487  {
488  case TC_FUN12:
489  case TC_OPER1: *R = tos[0].get_function(); break;
490  default: return 0; // error
491  }
492 
493  switch(tos[2].get_Class())
494  {
495  case TC_FUN12:
496  case TC_OPER1: *L = tos[0].get_function(); break;
497  default: return 0; // error
498  }
499 
500  switch(tos[1].get_Class())
501  {
502  case TC_OPER2: return tos[1].get_function();
503  default: return 0;
504  }
505  }
506 
507  return 0;
508 }

520 //-----------------------------------------------------------------------------
521 APL_value
522 get_var_value(const char * var_name, const char * loc)
523 {
524 UTF8_string var_name_utf8(var_name);
525 UCS_string var_name_ucs(var_name_utf8);
526 Symbol * symbol = Workspace::lookup_existing_symbol(var_name_ucs);
527  if (symbol == 0) return 0; // unknown name
528  if (symbol->get_nc() != NC_VARIABLE) return 0; // name is not a variable
529 
530 Value_P Z = symbol->get_value();
531  if (!Z) return 0;
532 
533  Z.get()->increment_owner_count(loc); // keep value
534  return Z.get();
535 }
536 //-----------------------------------------------------------------------------
537 int
538 set_var_value(const char * var_name, const APL_value new_value,
539  const char * loc)
540 {
541 UTF8_string var_name_utf8(var_name);
542 UCS_string var_name_ucs(var_name_utf8);
543 
544  // check name...
545  //
546  if (!Avec::is_quad(var_name_ucs[0]) &&
547  !Avec::is_first_symbol_char(var_name_ucs[0])) return 1;
548 
549  loop(s, var_name_ucs.size())
550  {
551  if (!Avec::is_symbol_char(var_name_ucs[s])) return 2;
552  }
553 
554 Symbol * symbol = Workspace::lookup_symbol(var_name_ucs);
555  if (symbol == 0) return 3; // unknown name
556 
557  if (new_value == 0) return 0; // only test var_name
558 
559  if (symbol->get_nc() != NC_VARIABLE &&
560  symbol->get_nc() != NC_UNUSED_USER_NAME) return 4;
561 
562 Value_P B(new_value, loc);
563  symbol->assign(B, true, loc);
564  return 0; // ok
565 }

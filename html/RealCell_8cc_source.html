31 ErrorCode
32 RealCell::bif_logarithm(Cell * Z, const Cell * A) const
33 {
34  // ISO p. 88
35  //
36  if (!A->is_numeric()) return E_DOMAIN_ERROR;
37 
38  if (get_real_value() == A->get_real_value() &&
39  A->get_imag_value() == 0.0) return IntCell::z1(Z);
40 
41  if (get_real_value() == 0.0) return E_DOMAIN_ERROR;
42  if (A->is_near_one()) return E_DOMAIN_ERROR;
43 
44  if (A->is_real_cell() &&
45  A->get_real_value() >= 0.0 &&
46  get_real_value() >= 0.0)
47  {
48  const APL_Float z = log(get_real_value()) / log(A->get_real_value());
49  if (!isfinite(z)) return E_DOMAIN_ERROR;
50  return FloatCell::zF(Z, z);
51  }
52 
53  // complex result (complex B or negative A)
54  //
55 const APL_Complex z = log(get_complex_value()) / log(A->get_complex_value());
56  if (!isfinite(z.real())) return E_DOMAIN_ERROR;
57  if (!isfinite(z.imag())) return E_DOMAIN_ERROR;
58  return ComplexCell::zC(Z, z);
59 }

108 ErrorCode
109 RealCell::do_bif_circle_fun(Cell * Z, int fun) const
110 {
111 const APL_Float b = get_real_value();
112 
113  switch(fun)
114  {
115  case -12:
116  return ComplexCell(0, b).bif_exponential(Z);
117 
118  case -11:
119  return ComplexCell::zC(Z, 0.0, b);
120 
121  case -10:
122  return FloatCell::zF(Z, b);
123 
124  case -9:
125  return FloatCell::zF(Z, b);
126 
127  case -8:
128  { const APL_Float square = -(b*b + 1.0); // (¯1 + R⋆2)
129  if (square < 0.0) // complex square root
130  {
131  const APL_Float root = sqrt(-square); // (¯1 + R⋆2)⋆0.5
132  if (b < 0.0) return ComplexCell::zC(Z, 0.0, -root);
133  else return ComplexCell::zC(Z, 0.0, root);
134  }
135  else // real square root
136  {
137  const APL_Float root = sqrt(square); // (¯1 + R⋆2)⋆0.5
138  if (b < 0.0) return FloatCell::zF(Z, -root);
139  else return FloatCell::zF(Z, root);
140  }
141  }
142 
143  case -7:
144  if (b > -1.0 && b < 1.0) return FloatCell::zF(Z, atanh(b));
145  if (b == -1.0 || b == 1.0) return E_DOMAIN_ERROR;
146  return ComplexCell::do_bif_circle_fun(Z, -7, APL_Complex(b));
147 
148  case -6:
149  if (b > 1.0) return FloatCell::zF(Z, acosh(b));
150  return ComplexCell::do_bif_circle_fun(Z, -6, APL_Complex(b));
151 
152  case -5:
153  return FloatCell::zF(Z, asinh(b));
154 
155  case -4:
156  { const double arg = b*b - 1.0;
157  if (arg >= 0.0) return FloatCell::zF(Z, sqrt(arg));
158  return ComplexCell::do_bif_circle_fun(Z, -4, APL_Complex(b));
159  }
160  case -3:
161  return FloatCell::zF(Z, atan (b));
162 
163  case -2:
164  if (b >= -1.0 && b <= 1.0) return FloatCell::zF(Z, acos (b));
165  return ComplexCell::do_bif_circle_fun(Z, -2, APL_Complex(b));
166 
167  case -1:
168  if (b >= -1.0 && b <= 1.0) return FloatCell::zF(Z, asin (b));
169  return ComplexCell::do_bif_circle_fun(Z, -1, APL_Complex(b));
170 
171  case 0:
172  {
173  const APL_Float b2 = 1.0 - b*b;
174  if (b2 >= 0.0) return FloatCell::zF(Z, sqrt(b2));
175  return ComplexCell::do_bif_circle_fun(Z, 0, APL_Complex(b));
176  }
177 
178  case 1:
179  return FloatCell::zF(Z, sin(b));
180 
181  case 2:
182  return FloatCell::zF(Z, cos(b));
183 
184  case 3:
185  return FloatCell::zF(Z, tan(b));
186 
187  case 4:
188  return FloatCell::zF(Z, sqrt(1 + b*b));
189 
190  case 5: return FloatCell::zF(Z, sinh(b));
191 
192  case 6:
193  return FloatCell::zF(Z, cosh(b));
194 
195  case 7:
196  return FloatCell::zF(Z, tanh(b));
197 
198  case 8:
199  {
200  const APL_Float square = -(b*b + 1.0); // (¯1 - R⋆2)
201  if (square < 0.0) // complex square root
202  {
203  const APL_Float root = sqrt(-square); // (¯1 + R⋆2)⋆0.5
204  if (b < 0.0) return ComplexCell::zC(Z, 0.0, root);
205  else return ComplexCell::zC(Z, 0.0, -root);
206  }
207  else // real square root
208  {
209  const APL_Float root = sqrt(square); // (¯1 + R⋆2)⋆0.5
210  if (b < 0.0) return FloatCell::zF(Z, root);
211  else return FloatCell::zF(Z, -root);
212  }
213  }
214 
215  case 9:
216  return FloatCell::zF(Z, b);
217 
218  case 10:
219  if (b < 0.0) return FloatCell::zF(Z, -b);
220  else return FloatCell::zF(Z, b);
221 
222  case 11:
223  return FloatCell::zF(Z, 0.0);
224 
225  case 12:
226  return FloatCell::zF(Z, (b < 0.0) ? M_PI : 0.0);
227  }
228 
229  // invalid fun
230  //
231  return E_DOMAIN_ERROR;
232 }

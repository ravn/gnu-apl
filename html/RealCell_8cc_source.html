32 RealCell::bif_logarithm(Cell * Z, const Cell * A) const
33 {
34  // ISO p. 88
35  //
36  if (!A->is_numeric()) return E_DOMAIN_ERROR;
37 
38  if (get_real_value() == A->get_real_value() &&
39  A->get_imag_value() == 0.0) return IntCell::z1(Z);
40 
41  if (get_real_value() == 0.0) return E_DOMAIN_ERROR;
42  if (A->is_near_one()) return E_DOMAIN_ERROR;
43 
44  if (A->is_real_cell() &&
45  A->get_real_value() >= 0.0 &&
46  get_real_value() >= 0.0)
47  {
48  const APL_Float z = log(get_real_value()) / log(A->get_real_value());
49  if (!isfinite(z)) return E_DOMAIN_ERROR;
50  return FloatCell::zv(Z, z);
51  }
52 
53  // complex result (complex B or negative A)
54  //
55 const APL_Complex z = log(get_complex_value()) / log(A->get_complex_value());
56  if (!isfinite(z.real())) return E_DOMAIN_ERROR;
57  if (!isfinite(z.imag())) return E_DOMAIN_ERROR;
58  return ComplexCell::zv(Z, z);
59 }

108 ErrorCode
109 RealCell::do_bif_circle_fun(Cell * Z, int fun) const
110 {
111 const APL_Float b = get_real_value();
112 const APL_Complex cb(b);
113 
114  switch(fun)
115  {
116  case -12: ComplexCell(0, b).bif_exponential(Z); return E_NO_ERROR;
117  case -11: return ComplexCell::zv(Z, 0.0, b );
118  case -10: return FloatCell::zv(Z, b );
119  case -9: return FloatCell::zv(Z, b );
120  case -8: { const APL_Float par = -(b*b + 1.0); // (¯1 + R⋆2)
121  if (par < 0.0) // complex square root
122  {
123  const APL_Float sq = sqrt(-par); // (¯1 + R⋆2)⋆0.5
124  if (b < 0.0) return ComplexCell::zv(Z, 0.0, -sq);
125  else return ComplexCell::zv(Z, 0.0, sq);
126  }
127  else // real square root
128  {
129  const APL_Float sq = sqrt(par); // (¯1 + R⋆2)⋆0.5
130  if (b < 0.0) return FloatCell::zv(Z, -sq);
131  else return FloatCell::zv(Z, sq);
132  }
133  }
134 
135  case -7: if (b > -1.0 && b < 1.0) return FloatCell::zv(Z, atanh(b));
136  if (b == -1.0 || b == 1.0) return E_DOMAIN_ERROR;
137  return ComplexCell::do_bif_circle_fun(Z, -7, cb);
138 
139  case -6: if (b > 1.0) return FloatCell::zv(Z, acosh(b));
140  return ComplexCell::do_bif_circle_fun(Z, -6, cb);
141 
142  case -5: return FloatCell::zv(Z, asinh(b));
143  case -4: if (b >= -1.0) return FloatCell::zv(Z, sqrt(b*b - 1.0));
144  return ComplexCell::do_bif_circle_fun(Z, -4, cb);
145  case -3: return FloatCell::zv(Z, atan (b));
146  case -2: if (b >= -1.0 && b <= 1.0) return FloatCell::zv(Z, acos (b));
147  return ComplexCell::do_bif_circle_fun(Z, -2, cb);
148  case -1: if (b >= -1.0 && b <= 1.0) return FloatCell::zv(Z, asin (b));
149  return ComplexCell::do_bif_circle_fun(Z, -1, cb);
150  case 0: if (b*b < 1.0) return FloatCell::zv(Z, sqrt (1 - b*b));
151  return ComplexCell::do_bif_circle_fun(Z, 0, cb);
152  case 1: return FloatCell::zv(Z, sin (b));
153  case 2: return FloatCell::zv(Z, cos (b));
154  case 3: return FloatCell::zv(Z, tan (b));
155  case 4: return FloatCell::zv(Z, sqrt (1 + b*b));
156  case 5: return FloatCell::zv(Z, sinh(b));
157  case 6: return FloatCell::zv(Z, cosh(b));
158  case 7: return FloatCell::zv(Z, tanh(b));
159  case 8: { const APL_Float par = -(b*b + 1.0); // (¯1 - R⋆2)
160  if (par < 0.0) // complex square root
161  {
162  const APL_Float sq = sqrt(-par); // (¯1 + R⋆2)⋆0.5
163  if (b < 0.0) return ComplexCell::zv(Z, 0.0, sq);
164  else return ComplexCell::zv(Z, 0.0, -sq);
165  }
166  else // real square root
167  {
168  const APL_Float sq = sqrt(par); // (¯1 + R⋆2)⋆0.5
169  if (b < 0.0) return FloatCell::zv(Z, sq);
170  else return FloatCell::zv(Z, -sq);
171  }
172  }
173  case 9: return FloatCell::zv(Z, b );
174  case 10: if (b < 0.0) return FloatCell::zv(Z, -b);
175  else return FloatCell::zv(Z, b);
176  case 11: return FloatCell::zv(Z, 0.0 );
177  case 12: return FloatCell::zv(Z, (b < 0.0) ? M_PI : 0.0);
178  }
179 
180  // invalid fun
181  //
182  return E_DOMAIN_ERROR;
183 }

93 //-----------------------------------------------------------------------------
94 bool
95 ESCmap::has_prefix(const char * seq, int seq_len) const
96 {
97  if (seq_len >= len) return false;
98  loop(s, seq_len)
99  {
100  if ((seq[s] != seqence[s]) && seqence[s]) return false;
101  }
102 
103  return true;
104 }

107 ESCmap::is_equal(const char * seq, int seq_len) const
108 {
109  if (len != seq_len) return false;
110  loop(s, seq_len)
111  {
112  if ((seq[s] != seqence[s]) && seqence[s]) return false;
113  }
114 
115  return true;
116 }

153 //-----------------------------------------------------------------------------
154 void
155 LineHistory::read_history(const char * filename)
156 {
157 FILE * hist = fopen(filename, "r");
158  if (hist == 0)
159  {
160  Log(LOG_get_line)
161  CERR << "Cannot open history file " << filename
162  << ": " << strerror(errno) << endl;
163  return;
164  }
165 
166 int count = 0;
167  for (;;)
168  {
169  char buffer[4000];
170  const char * s = fgets(buffer, sizeof(buffer) - 1, hist);
171  if (s == 0) break; // end of file
172  buffer[sizeof(buffer) - 1] = 0;
173 
174  int slen = strlen(buffer);
175  if (slen && (buffer[slen - 1] == '\n')) buffer[--slen] = 0;
176  if (slen && (buffer[slen - 1] == '\r')) buffer[--slen] = 0;
177  ++count;
178  UTF8_string utf(buffer);
179  UCS_string ucs(utf);
180  add_line(ucs);
181  }
182 
183  next();
184 }

237 //-----------------------------------------------------------------------------
238 void
239 LineHistory::add_line(const UCS_string & line)
240 {
241  if (max_lines == 0) return; // no history
242  if (!line.has_black()) return;
243 
244  if (int(hist_lines.size()) < max_lines) // append
245  {
246  hist_lines.push_back(line);
247  put = 0;
248  }
249  else // override
250  {
251  if (put >= int(hist_lines.size())) put = 0; // wrap
252  hist_lines[put++] = line;
253  }
254 
255  next(); // update current_line
256 }
257 //-----------------------------------------------------------------------------
258 void
259 LineHistory::replace_line(const UCS_string & line)
260 {
261  if (put > 0) hist_lines[put - 1] = line;
262  else hist_lines.back() = line;
263 }
264 //-----------------------------------------------------------------------------
265 const UCS_string *
266 LineHistory::up()
267 {
268  if (hist_lines.size() == 0) return 0; // no history
269 
270 int new_current_line = current_line - 1;
271  if (new_current_line < 0) new_current_line += hist_lines.size(); // wrap
272  if (new_current_line == put) return 0;
273 
274  return &hist_lines[current_line = new_current_line];
275 }
276 //-----------------------------------------------------------------------------
277 const UCS_string *
278 LineHistory::down()
279 {
280  if (hist_lines.size() == 0) return 0; // no history
281  if (current_line == put) return 0;
282 
283 int new_current_line = current_line + 1;
284  if (new_current_line >= int(hist_lines.size()))
285  new_current_line = 0; // wrap
286  current_line = new_current_line;
287  if (current_line == put) return 0;
288 
289  return &hist_lines[current_line];
290 }

327 //-----------------------------------------------------------------------------
328 void
329 LineEditContext::adjust_allocated_height()
330 {
331 const int rows = 1 + get_total_length() / screen_cols;
332 
333  if (allocated_height >= rows) return;
334 
335  // scroll some lines so that prior text is not overridden.
336  //
337  CIN.set_cursor(-1, 0);
338  loop(a, rows - allocated_height) CIN << endl;
339 
340  allocated_height = rows;
341 
342  // redraw screen from -allocated_height:0 onwards
343  //
344  if (CIN.can_clear_EOS())
345  {
346  CIN.set_cursor(-allocated_height, 0);
347  CIN.clear_EOS();
348  }
349  else
350  {
351  loop(a, allocated_height)
352  {
353  CIN.set_cursor(a - allocated_height, 0);
354  CIN.clear_EOL();
355  }
356  }
357 
358  refresh_all();
359 }
360 //-----------------------------------------------------------------------------
361 void
362 LineEditContext::refresh_from_cursor()
363 {
364 const int saved_uidx = uidx;
365 
366  adjust_allocated_height();
367  set_cursor();
368  for (; uidx < user_line.size(); ++uidx)
369  {
370  refresh_wrapped_cursor();
371  CIN << user_line[uidx];
372  }
373 
374  // clear from end of user_line
375  //
376  move_idx(user_line.size());
377  if (CIN.can_clear_EOS())
378  {
379  CIN.clear_EOS();
380  }
381  else
382  {
383  CIN.clear_EOL();
384 
385  // clear subsequent lines
386  //
387  for (int a = 1; a < allocated_height; ++a)
388  {
389  CIN.set_cursor(a - allocated_height, 0);
390  CIN.clear_EOL();
391  }
392  }
393 
394  move_idx(saved_uidx);
395 }

466 {

468

483 }

504 //-----------------------------------------------------------------------------
505 void
506 LineEditContext::tab_expansion(LineInputMode mode)
507 {
508  if (mode != LIM_ImmediateExecution) return;
509 
510 UCS_string line = user_line;
511 TabExpansion tab_exp(line);
512 const ExpandResult expand_result = tab_exp.expand_tab(line);
513 
514  switch(expand_result)
515  {
516  case ER_IGNORE: return;
517 
518  case ER_AGAIN:
519  // expand_tab has shown a list of options.
520  // Reset the input window and redisplay.
521  allocated_height = 1;
522  adjust_allocated_height();
523  refresh_all();
524  return;
525 
526  case ER_REPLACE:
527  user_line.clear();
528  user_line.append(line);
529  uidx = 0;
530  refresh_from_cursor();
531  move_idx(user_line.size());
532  return;
533 
534  default: FIXME;
535  }
536 }
537 //-----------------------------------------------------------------------------
538 void
539 LineEditContext::cursor_UP()
540 {
541  Log(LOG_get_line) history.info(CERR << "cursor_UP()") << endl;
542 
543 const UCS_string * ucs = history.up();
544  if (ucs == 0) // no line above
545  {
546  Log(LOG_get_line) CERR << "hit top of history()" << endl;
547  Log(LOG_get_line) history.info(CERR << "cursor_UP() done" << endl);
548  return;
549  }
550 
551  if (!history_entered) // not yet in history: remember user_line
552  {
553  user_line_before_history = user_line;
554  history_entered = true;
555  }
556 
557  user_line = *ucs;
558  adjust_allocated_height();
559 
560  uidx = 0;
561  refresh_from_cursor();
562  move_idx(user_line.size());
563  Log(LOG_get_line) history.info(CERR << "cursor_UP() done" << endl);
564 }
565 //-----------------------------------------------------------------------------
566 void
567 LineEditContext::cursor_DOWN()
568 {
569  Log(LOG_get_line) history.info(CERR << "cursor_DOWN()" << endl);
570 
571 const UCS_string * ucs = history.down();
572  if (ucs == 0) // no line below
573  {
574  Log(LOG_get_line) CERR << "hit bottom of history()" << endl;
575  // if inside history: restore user_line
576  //
577  if (history_entered) user_line = user_line_before_history;
578  history_entered = false;
579  goto refresh;
580  }
581 
582  if (!history_entered) // not yet in history: remember user_line
583  {
584  user_line_before_history = user_line;
585  history_entered = true;
586  }
587 
588  user_line = *ucs;
589  adjust_allocated_height();
590 
591 refresh:
592  uidx = 0;
593  refresh_from_cursor();
594  move_idx(user_line.size());
595  Log(LOG_get_line) history.info(CERR << "cursor_DOWN() done" << endl);
596 }

653 void
654 InputMux::get_line(LineInputMode mode, const UCS_string & prompt,
655  UCS_string & line, bool & eof, LineHistory & hist)
656 {
657  if (get_line_callback)
658  {
659  get_line_callback(mode, prompt, line, eof, hist);
660  return;
661  }
662 
663  if (InputFile::is_validating()) Quad_QUOTE::done(true, LOC);
664 
665  InputFile::increment_current_line_no();
666 
667  // check if we have input from a file. We do NOT use the file if the input
668  // is for â unless we are in a .tc testcase file.
669  //
670  if (InputFile::is_validating() ||
671  (mode != LIM_Quad_Quad && (mode != LIM_Quote_Quad)))
672  {
673  UTF8_string file_line;
674  bool file_eof = false;
675  IO_Files::get_file_line(file_line, file_eof);
676 
677  if (!file_eof)
678  {
679  line = UCS_string(file_line);
680 
681  switch(mode)
682  {
683  case LIM_ImmediateExecution:
684  case LIM_Quad_Quad:
685  case LIM_Quad_INP:
686  CIN << prompt << line << endl;
687  break;
688 
689  case LIM_Quote_Quad:
690  line = prompt;
691 
692  // for each leading backspace in line: discard last
693  // prompt character. This is for testing the user
694  // backspacing over the â prompt
695  //
696  while (line.size() &&
697  file_line.size() &&
698  file_line[0] == UNI_ASCII_BS)
699  {
700  file_line.erase(0);
701  line.pop_back();
702  }
703  line.append_UTF8(file_line.c_str());
704  break;
705 
706  case LIM_Nabla:
707  break;
708 
709  default: FIXME;
710  }
711 
712  return;
713  }
714  }
715 
716  // no (more) input from files: get line from terminal
717  //
718  if (uprefs.raw_cin)
719  {
720  Quad_QUOTE::done(mode != LIM_Quote_Quad, LOC);
721  CIN << '\r' << prompt;
722  char buffer[4000];
723  const APL_time_us from = now();
724  const char * s = fgets(buffer, sizeof(buffer) - 1, stdin);
725  Workspace::add_wait(now() - from);
726 
727  if (s == 0)
728  {
729  eof = true;
730  return;
731  }
732  buffer[sizeof(buffer) - 1] = 0;
733 
734  int slen = strlen(buffer);
735  if (slen && buffer[slen - 1] == '\n') buffer[--slen] = 0;
736  if (slen && buffer[slen - 1] == '\r') buffer[--slen] = 0;
737 
738  UTF8_string line_utf(buffer);
739  line = UCS_string(line_utf);
740  return;
741  }
742 
743  Quad_QUOTE::done(mode != LIM_Quote_Quad, LOC);
744 
745 const APL_time_us from = now();
746  if (start_input) (*start_input)();
747 
748 #if PARALLEL_ENABLED
749  CPU_pool::lock_pool(false);
750 #endif
751 
752  for (int control_D_count = 0;;)
753  {
754  bool _eof = false;
755  LineInput::get_terminal_line(mode, prompt, line, _eof, hist);
756  if (!_eof) break;
757 
758  ++control_D_count;
759 
760  // ^D or end of file
761  if (uprefs.control_Ds_to_exit) // there is a ^D limit
762  {
763  if (control_D_count >= uprefs.control_Ds_to_exit)
764  {
765  CIN << endl;
766 #if PARALLEL_ENABLED
767  Thread_context::kill_pool();

992 Unicode
993 LineInput::get_uni()
994 {
995 again:
996 
997 const int b0 = safe_fgetc();
998  if (b0 == EOF) return UNI_EOF;
999 
1000  if (b0 & 0x80) // non-ASCII unicode
1001  {
1002  int len;
1003  uint32_t bx = b0; // the "significant" bits in b0
1004  if ((b0 & 0xE0) == 0xC0) { len = 2; bx &= 0x1F; }
1005  else if ((b0 & 0xF0) == 0xE0) { len = 3; bx &= 0x0F; }
1006  else if ((b0 & 0xF8) == 0xF0) { len = 4; bx &= 0x0E; }
1007  else if ((b0 & 0xFC) == 0xF8) { len = 5; bx &= 0x0E; }
1008  else if ((b0 & 0xFE) == 0xFC) { len = 6; bx &= 0x0E; }
1009  else
1010  {
1011  CERR << "Bad UTF8 sequence start at " << LOC << endl;
1012  return Invalid_Unicode;
1013  }
1014 
1015  uint32_t uni = 0;
1016  loop(l, len - 1)
1017  {
1018  const UTF8 subc = safe_fgetc();
1019  if ((subc & 0xC0) != 0x80)
1020  {
1021  CERR << "Bad UTF8 sequence: " << HEX(b0)
1022  << "... at " LOC << endl;
1023  return Invalid_Unicode;
1024  }
1025 
1026  bx <<= 6;
1027  uni <<= 6;
1028  uni |= subc & 0x3F;
1029  }
1030 
1031  return Unicode(bx | uni);
1032  }
1033 
1034  if (b0 == UNI_ASCII_ESC)
1035  {
1036  char seq[Output::MAX_ESC_LEN]; seq[0] = UNI_ASCII_ESC;
1037  for (int s = 1; s < Output::MAX_ESC_LEN; ++s)
1038  {
1039  const int bs = safe_fgetc();
1040  if (bs == EOF) return UNI_EOF;
1041  seq[s] = bs;
1042 
1043  // check for exact match
1044  //
1045  loop(e, ESCmap_entry_count)
1046  {
1047  if (ESCmap::the_ESCmap[e].is_equal(seq, s + 1))
1048  return ESCmap::the_ESCmap[e].uni;
1049  }
1050 
1051  // check for prefix match
1052  //
1053  if (ESCmap::need_more(seq, s)) continue;
1054 
1055 // CERR << endl << "Unknown ESC sequence: ESC";
1056 // loop(ss, s) CERR << " " << HEX2(seq[ss + 1]);
1057 // CERR << endl;
1058 
1059  return Invalid_Unicode;
1060  }
1061  }
1062  else if (b0 < UNI_ASCII_SPACE) // ^something (except ESC)
1063  {
1064  switch(b0)
1065  {
1066  case UNI_ASCII_SOH: return UNI_CursorHome; // ^A
1067  case UNI_ASCII_STX: return UNI_CursorLeft; // ^B
1068  case UNI_ASCII_ETX: return UNI_ASCII_ETX; // ^C
1069  case UNI_ASCII_EOT: return UNI_ASCII_EOT; // ^D
1070  case UNI_ASCII_ENQ: return UNI_CursorEnd; // ^E
1071  case UNI_ASCII_ACK: return UNI_CursorRight; // ^F
1072  case UNI_ASCII_BS: return UNI_ASCII_BS; // ^H
1073  case UNI_ASCII_HT: return UNI_ASCII_HT; // ^I
1074  case UNI_ASCII_LF: return UNI_ASCII_LF; // ^J
1075  case UNI_ASCII_VT: return UNI_ASCII_VT; // ^K
1076  case UNI_ASCII_SO: return UNI_CursorDown; // ^N
1077  case UNI_ASCII_DLE: return UNI_CursorUp; // ^P
1078  case UNI_ASCII_EM: return UNI_ASCII_EM; // ^Y
1079  default: goto again;
1080  }
1081  }
1082  else if (b0 == UNI_ASCII_DELETE) return UNI_ASCII_BS;
1083 
1084  return Unicode(b0);
1085 }

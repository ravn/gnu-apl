154 void
155 LineHistory::read_history(const char * filename)
156 {
157 FILE * hist = fopen(filename, "r");
158  if (hist == 0)
159  {
160  Log(LOG_get_line)
161  CERR << "Cannot open history file " << filename
162  << ": " << strerror(errno) << endl;
163  return;
164  }
165 
166 int count = 0;
167  for (;;)
168  {
169  char buffer[4000];
170  const char * s = fgets(buffer, sizeof(buffer) - 1, hist);
171  if (s == 0) break; // end of file
172  buffer[sizeof(buffer) - 1] = 0;
173 
174  int slen = strlen(buffer);
175  if (slen && (buffer[slen - 1] == '\n')) buffer[--slen] = 0;
176  if (slen && (buffer[slen - 1] == '\r')) buffer[--slen] = 0;
177  ++count;
178  UTF8_string utf(buffer);
179  UCS_string ucs(utf);
180  add_line(ucs);
181  }
182 
183  next();
184 }

237 //-----------------------------------------------------------------------------
238 void
239 LineHistory::add_line(const UCS_string & line)
240 {
241  if (max_lines == 0) return; // no history
242  if (!line.has_black()) return;
243 
244  if (int(hist_lines.size()) < max_lines) // append
245  {
246  hist_lines.push_back(line);
247  put = 0;
248  }
249  else // override
250  {
251  if (put >= int(hist_lines.size())) put = 0; // wrap
252  hist_lines[put++] = line;
253  }
254 
255  next(); // update current_line
256 }

266 LineHistory::up()
267 {
268  if (hist_lines.size() == 0) return 0;
269

271  if (new_current_line < 0) new_current_line += hist_lines.size(); // wrap
272  if (new_current_line == put) return 0;
273

275 }

279 {

281  if (current_line == put) return 0;

284  if (new_current_line >= int(hist_lines.size()))
285  new_current_line = 0; // wrap
286  if (new_current_line == put) return 0;
287 
288  return &hist_lines[current_line = new_current_line];
289 }

504 void
505 LineEditContext::tab_expansion(LineInputMode mode)
506 {
507  if (mode != LIM_ImmediateExecution) return;
508 
509 UCS_string line = user_line;
510 TabExpansion tab_exp(line);
511 const ExpandResult expand_result = tab_exp.expand_tab(line);
512 
513  switch(expand_result)

527  user_line.append(line);
528  uidx = 0;
529  refresh_from_cursor();
530  move_idx(user_line.size());
531  return;
532 
533  default: FIXME;
534  }

536 //-----------------------------------------------------------------------------
537 void
538 LineEditContext::cursor_UP()
539 {
540 const UCS_string * ucs = history.up();
541  if (ucs == 0) // no line above
542  {
543  return;
544  }
545 
546  if (!history_entered) // not yet in history: remember user_line
547  {
548  user_line_before_history = user_line;
549  history_entered = true;
550  }
551 
552  user_line = *ucs;
553  adjust_allocated_height();
554 
555  uidx = 0;
556  refresh_from_cursor();
557  move_idx(user_line.size());
558 }
559 //-----------------------------------------------------------------------------
560 void
561 LineEditContext::cursor_DOWN()
562 {
563 const UCS_string * ucs = history.down();
564  if (ucs == 0) // no line below
565  {
566  // if inside history: restore user_line
567  //
568  if (history_entered) user_line = user_line_before_history;
569  history_entered = false;
570  goto refresh;
571  }
572 
573  if (!history_entered) // not yet in history: remember user_line
574  {
575  user_line_before_history = user_line;
576  history_entered = true;
577  }
578 
579  user_line = *ucs;
580  adjust_allocated_height();
581 
582 refresh:
583  uidx = 0;
584  refresh_from_cursor();
585  move_idx(user_line.size());
586 }

643 void
644 InputMux::get_line(LineInputMode mode, const UCS_string & prompt,
645  UCS_string & line, bool & eof, LineHistory & hist)
646 {
647  if (get_line_callback)
648  {
649  get_line_callback(mode, prompt, line, eof, hist);
650  return;
651  }
652 
653  if (InputFile::is_validating()) Quad_QUOTE::done(true, LOC);
654 
655  InputFile::increment_current_line_no();
656 
657  // check if we have input from a file. We do NOT use the file if the input
658  // is for â unless we are in a .tc testcase file.
659  //
660  if (InputFile::is_validating() ||
661  (mode != LIM_Quad_Quad && (mode != LIM_Quote_Quad)))
662  {
663  UTF8_string file_line;
664  bool file_eof = false;
665  IO_Files::get_file_line(file_line, file_eof);
666 
667  if (!file_eof)
668  {
669  line = UCS_string(file_line);
670 
671  switch(mode)
672  {
673  case LIM_ImmediateExecution:
674  case LIM_Quad_Quad:
675  case LIM_Quad_INP:
676  CIN << prompt << line << endl;
677  break;
678 
679  case LIM_Quote_Quad:
680  line = prompt;
681 
682  // for each leading backspace in line: discard last
683  // prompt character. This is for testing the user
684  // backspacing over the â prompt
685  //
686  while (line.size() &&
687  file_line.size() &&
688  file_line[0] == UNI_ASCII_BS)
689  {
690  file_line.erase(0);
691  line.pop_back();
692  }
693  line.append_UTF8(file_line.c_str());
694  break;
695 
696  case LIM_Nabla:
697  break;
698 
699  default: FIXME;
700  }
701 
702  return;
703  }
704  }
705 
706  // no (more) input from files: get line from terminal
707  //
708  if (uprefs.raw_cin)
709  {
710  Quad_QUOTE::done(mode != LIM_Quote_Quad, LOC);
711  CIN << '\r' << prompt;
712  char buffer[4000];
713  const APL_time_us from = now();
714  const char * s = fgets(buffer, sizeof(buffer) - 1, stdin);
715  Workspace::add_wait(now() - from);
716 
717  if (s == 0)
718  {
719  eof = true;
720  return;
721  }
722  buffer[sizeof(buffer) - 1] = 0;
723 
724  int slen = strlen(buffer);
725  if (slen && buffer[slen - 1] == '\n') buffer[--slen] = 0;
726  if (slen && buffer[slen - 1] == '\r') buffer[--slen] = 0;
727 
728  UTF8_string line_utf(buffer);
729  line = UCS_string(line_utf);
730  return;
731  }
732 
733  Quad_QUOTE::done(mode != LIM_Quote_Quad, LOC);
734 
735 const APL_time_us from = now();
736  if (start_input) (*start_input)();
737 
738 #if PARALLEL_ENABLED
739  Parallel::lock_pool(false);
740 #endif
741 
742  for (int control_D_count = 0;;)
743  {
744  bool _eof = false;
745  LineInput::get_terminal_line(mode, prompt, line, _eof, hist);
746  if (!_eof) break;
747 
748  ++control_D_count;
749 
750  // ^D or end of file
751  if (uprefs.control_Ds_to_exit) // there is a ^D limit
752  {
753  if (control_D_count >= uprefs.control_Ds_to_exit)
754  {
755  CIN << endl;
756 #if PARALLEL_ENABLED
757  Thread_context::kill_pool();
758 #endif // PARALLEL_ENABLED
759  uprefs.silent = true; // exit silently
760  Command::cmd_OFF(4); // exit()s
761  return; // not reached
762  }
763  }
764  else if (control_D_count < 5)
765  {
766  CIN << endl;
767  COUT << " ^D or end-of-input detected ("
768  << control_D_count << "). Use )OFF to leave APL!"
769  << endl;
770  }
771 
772  eof = true;
773 
774  if (control_D_count > 10 && (now() - from)/control_D_count < 10000)

982 Unicode
983 LineInput::get_uni()
984 {
985 again:
986 
987 const int b0 = safe_fgetc();
988  if (b0 == EOF) return UNI_EOF;
989 
990  if (b0 & 0x80) // non-ASCII unicode
991  {
992  int len;
993  uint32_t bx = b0; // the "significant" bits in b0
994  if ((b0 & 0xE0) == 0xC0) { len = 2; bx &= 0x1F; }
995  else if ((b0 & 0xF0) == 0xE0) { len = 3; bx &= 0x0F; }
996  else if ((b0 & 0xF8) == 0xF0) { len = 4; bx &= 0x0E; }
997  else if ((b0 & 0xFC) == 0xF8) { len = 5; bx &= 0x0E; }
998  else if ((b0 & 0xFE) == 0xFC) { len = 6; bx &= 0x0E; }
999  else
1000  {
1001  CERR << "Bad UTF8 sequence start at " << LOC << endl;
1002  return Invalid_Unicode;
1003  }
1004 
1005  uint32_t uni = 0;
1006  loop(l, len - 1)
1007  {
1008  const UTF8 subc = safe_fgetc();
1009  if ((subc & 0xC0) != 0x80)
1010  {
1011  CERR << "Bad UTF8 sequence: " << HEX(b0)
1012  << "... at " LOC << endl;
1013  return Invalid_Unicode;
1014  }
1015 
1016  bx <<= 6;
1017  uni <<= 6;
1018  uni |= subc & 0x3F;
1019  }
1020 
1021  return Unicode(bx | uni);
1022  }
1023 
1024  if (b0 == UNI_ASCII_ESC)
1025  {
1026  char seq[Output::MAX_ESC_LEN]; seq[0] = UNI_ASCII_ESC;
1027  for (int s = 1; s < Output::MAX_ESC_LEN; ++s)
1028  {
1029  const int bs = safe_fgetc();
1030  if (bs == EOF) return UNI_EOF;
1031  seq[s] = bs;
1032 
1033  // check for exact match
1034  //
1035  loop(e, ESCmap_entry_count)
1036  {
1037  if (ESCmap::the_ESCmap[e].is_equal(seq, s + 1))
1038  return ESCmap::the_ESCmap[e].uni;
1039  }
1040 
1041  // check for prefix match
1042  //
1043  if (ESCmap::need_more(seq, s)) continue;
1044 
1045 // CERR << endl << "Unknown ESC sequence: ESC";
1046 // loop(ss, s) CERR << " " << HEX2(seq[ss + 1]);
1047 // CERR << endl;
1048 
1049  return Invalid_Unicode;
1050  }
1051  }
1052  else if (b0 < UNI_ASCII_SPACE) // ^something (except ESC)
1053  {
1054  switch(b0)
1055  {
1056  case UNI_ASCII_SOH: return UNI_CursorHome; // ^A
1057  case UNI_ASCII_STX: return UNI_CursorLeft; // ^B
1058  case UNI_ASCII_ETX: return UNI_ASCII_ETX; // ^C
1059  case UNI_ASCII_EOT: return UNI_ASCII_EOT; // ^D
1060  case UNI_ASCII_ENQ: return UNI_CursorEnd; // ^E
1061  case UNI_ASCII_ACK: return UNI_CursorRight; // ^F
1062  case UNI_ASCII_BS: return UNI_ASCII_BS; // ^H
1063  case UNI_ASCII_HT: return UNI_ASCII_HT; // ^I
1064  case UNI_ASCII_LF: return UNI_ASCII_LF; // ^J
1065  case UNI_ASCII_VT: return UNI_ASCII_VT; // ^K
1066  case UNI_ASCII_SO: return UNI_CursorDown; // ^N
1067  case UNI_ASCII_DLE: return UNI_CursorUp; // ^P
1068  case UNI_ASCII_EM: return UNI_ASCII_EM; // ^Y
1069  default: goto again;
1070  }
1071  }
1072  else if (b0 == UNI_ASCII_DELETE) return UNI_ASCII_BS;
1073 
1074  return Unicode(b0);
1075 }
